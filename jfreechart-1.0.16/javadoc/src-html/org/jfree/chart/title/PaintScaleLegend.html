134        public PaintScaleLegend(PaintScale scale, ValueAxis axis) {
135            ParamChecks.nullNotPermitted(axis, "axis");
136            this.scale = scale;
137            this.axis = axis;
138            this.axis.addChangeListener(this);
139            this.axisLocation = AxisLocation.BOTTOM_OR_LEFT;
140            this.axisOffset = 0.0;
141            this.axis.setRange(scale.getLowerBound(), scale.getUpperBound());
142            this.stripWidth = 15.0;
143            this.stripOutlineVisible = true;
144            this.stripOutlinePaint = Color.gray;
145            this.stripOutlineStroke = new BasicStroke(0.5f);
146            this.backgroundPaint = Color.white;
147            this.subdivisions = 100;
148        }

410        /**
411         * Arranges the contents of the block, within the given constraints, and
412         * returns the block size.
413         *
414         * @param g2  the graphics device.
415         * @param constraint  the constraint (<code>null</code> not permitted).
416         *
417         * @return The block size (in Java2D units, never <code>null</code>).
418         */
419        public Size2D arrange(Graphics2D g2, RectangleConstraint constraint) {
420            RectangleConstraint cc = toContentConstraint(constraint);
421            LengthConstraintType w = cc.getWidthConstraintType();
422            LengthConstraintType h = cc.getHeightConstraintType();
423            Size2D contentSize = null;
424            if (w == LengthConstraintType.NONE) {
425                if (h == LengthConstraintType.NONE) {
426                    contentSize = new Size2D(getWidth(), getHeight());
427                }
428                else if (h == LengthConstraintType.RANGE) {
429                    throw new RuntimeException("Not yet implemented.");
430                }
431                else if (h == LengthConstraintType.FIXED) {
432                    throw new RuntimeException("Not yet implemented.");
433                }
434            }
435            else if (w == LengthConstraintType.RANGE) {
436                if (h == LengthConstraintType.NONE) {
437                    throw new RuntimeException("Not yet implemented.");
438                }
439                else if (h == LengthConstraintType.RANGE) {
440                    contentSize = arrangeRR(g2, cc.getWidthRange(),
441                            cc.getHeightRange());
442                }
443                else if (h == LengthConstraintType.FIXED) {
444                    throw new RuntimeException("Not yet implemented.");
445                }
446            }
447            else if (w == LengthConstraintType.FIXED) {
448                if (h == LengthConstraintType.NONE) {
449                    throw new RuntimeException("Not yet implemented.");
450                }
451                else if (h == LengthConstraintType.RANGE) {
452                    throw new RuntimeException("Not yet implemented.");
453                }
454                else if (h == LengthConstraintType.FIXED) {
455                    throw new RuntimeException("Not yet implemented.");
456                }
457            }
458            return new Size2D(calculateTotalWidth(contentSize.getWidth()),
459                    calculateTotalHeight(contentSize.getHeight()));
460        }

462        /**
463         * Returns the content size for the title.  This will reflect the fact that
464         * a text title positioned on the left or right of a chart will be rotated
465         * 90 degrees.
466         *
467         * @param g2  the graphics device.
468         * @param widthRange  the width range.
469         * @param heightRange  the height range.
470         *
471         * @return The content size.
472         */
473        protected Size2D arrangeRR(Graphics2D g2, Range widthRange,
474                Range heightRange) {
475    
476            RectangleEdge position = getPosition();
477            if (position == RectangleEdge.TOP || position == RectangleEdge.BOTTOM) {
478    
479    
480                float maxWidth = (float) widthRange.getUpperBound();
481    
482                // determine the space required for the axis
483                AxisSpace space = this.axis.reserveSpace(g2, null,
484                        new Rectangle2D.Double(0, 0, maxWidth, 100),
485                        RectangleEdge.BOTTOM, null);
486    
487                return new Size2D(maxWidth, this.stripWidth + this.axisOffset
488                        + space.getTop() + space.getBottom());
489            }
490            else if (position == RectangleEdge.LEFT || position
491                    == RectangleEdge.RIGHT) {
492                float maxHeight = (float) heightRange.getUpperBound();
493                AxisSpace space = this.axis.reserveSpace(g2, null,
494                        new Rectangle2D.Double(0, 0, 100, maxHeight),
495                        RectangleEdge.RIGHT, null);
496                return new Size2D(this.stripWidth + this.axisOffset
497                        + space.getLeft() + space.getRight(), maxHeight);
498            }
499            else {
500                throw new RuntimeException("Unrecognised position.");
501            }
502        }

514        /**
515         * Draws the legend within the specified area.
516         *
517         * @param g2  the graphics target (<code>null</code> not permitted).
518         * @param area  the drawing area (<code>null</code> not permitted).
519         * @param params  drawing parameters (ignored here).
520         *
521         * @return <code>null</code>.
522         */
523        public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
524    
525            Rectangle2D target = (Rectangle2D) area.clone();
526            target = trimMargin(target);
527            if (this.backgroundPaint != null) {
528                g2.setPaint(this.backgroundPaint);
529                g2.fill(target);
530            }
531            getFrame().draw(g2, target);
532            getFrame().getInsets().trim(target);
533            target = trimPadding(target);
534            double base = this.axis.getLowerBound();
535            double increment = this.axis.getRange().getLength() / this.subdivisions;
536            Rectangle2D r = new Rectangle2D.Double();
537    
538            if (RectangleEdge.isTopOrBottom(getPosition())) {
539                RectangleEdge axisEdge = Plot.resolveRangeAxisLocation(
540                        this.axisLocation, PlotOrientation.HORIZONTAL);
541                if (axisEdge == RectangleEdge.TOP) {
542                    for (int i = 0; i < this.subdivisions; i++) {
543                        double v = base + (i * increment);
544                        Paint p = this.scale.getPaint(v);
545                        double vv0 = this.axis.valueToJava2D(v, target,
546                                RectangleEdge.TOP);
547                        double vv1 = this.axis.valueToJava2D(v + increment, target,
548                                RectangleEdge.TOP);
549                        double ww = Math.abs(vv1 - vv0) + 1.0;
550                        r.setRect(Math.min(vv0, vv1), target.getMaxY()
551                                - this.stripWidth, ww, this.stripWidth);
552                        g2.setPaint(p);
553                        g2.fill(r);
554                    }
555                    if (isStripOutlineVisible()) {
556                        g2.setPaint(this.stripOutlinePaint);
557                        g2.setStroke(this.stripOutlineStroke);
558                        g2.draw(new Rectangle2D.Double(target.getMinX(),
559                                target.getMaxY() - this.stripWidth,
560                                target.getWidth(), this.stripWidth));
561                    }
562                    this.axis.draw(g2, target.getMaxY() - this.stripWidth
563                            - this.axisOffset, target, target, RectangleEdge.TOP,
564                            null);
565                }
566                else if (axisEdge == RectangleEdge.BOTTOM) {
567                    for (int i = 0; i < this.subdivisions; i++) {
568                        double v = base + (i * increment);
569                        Paint p = this.scale.getPaint(v);
570                        double vv0 = this.axis.valueToJava2D(v, target,
571                                RectangleEdge.BOTTOM);
572                        double vv1 = this.axis.valueToJava2D(v + increment, target,
573                                RectangleEdge.BOTTOM);
574                        double ww = Math.abs(vv1 - vv0) + 1.0;
575                        r.setRect(Math.min(vv0, vv1), target.getMinY(), ww,
576                                this.stripWidth);
577                        g2.setPaint(p);
578                        g2.fill(r);
579                    }
580                    if (isStripOutlineVisible()) {
581                        g2.setPaint(this.stripOutlinePaint);
582                        g2.setStroke(this.stripOutlineStroke);
583                        g2.draw(new Rectangle2D.Double(target.getMinX(),
584                                target.getMinY(), target.getWidth(),
585                                this.stripWidth));
586                    }
587                    this.axis.draw(g2, target.getMinY() + this.stripWidth
588                            + this.axisOffset, target, target,
589                            RectangleEdge.BOTTOM, null);
590                }
591            }
592            else {
593                RectangleEdge axisEdge = Plot.resolveRangeAxisLocation(
594                        this.axisLocation, PlotOrientation.VERTICAL);
595                if (axisEdge == RectangleEdge.LEFT) {
596                    for (int i = 0; i < this.subdivisions; i++) {
597                        double v = base + (i * increment);
598                        Paint p = this.scale.getPaint(v);
599                        double vv0 = this.axis.valueToJava2D(v, target,
600                                RectangleEdge.LEFT);
601                        double vv1 = this.axis.valueToJava2D(v + increment, target,
602                                RectangleEdge.LEFT);
603                        double hh = Math.abs(vv1 - vv0) + 1.0;
604                        r.setRect(target.getMaxX() - this.stripWidth,
605                                Math.min(vv0, vv1), this.stripWidth, hh);
606                        g2.setPaint(p);
607                        g2.fill(r);
608                    }
609                    if (isStripOutlineVisible()) {
610                        g2.setPaint(this.stripOutlinePaint);
611                        g2.setStroke(this.stripOutlineStroke);
612                        g2.draw(new Rectangle2D.Double(target.getMaxX()
613                                - this.stripWidth, target.getMinY(), 
614                                this.stripWidth, target.getHeight()));
615                    }
616                    this.axis.draw(g2, target.getMaxX() - this.stripWidth
617                            - this.axisOffset, target, target, RectangleEdge.LEFT,
618                            null);
619                }
620                else if (axisEdge == RectangleEdge.RIGHT) {
621                    for (int i = 0; i < this.subdivisions; i++) {
622                        double v = base + (i * increment);
623                        Paint p = this.scale.getPaint(v);
624                        double vv0 = this.axis.valueToJava2D(v, target,
625                                RectangleEdge.LEFT);
626                        double vv1 = this.axis.valueToJava2D(v + increment, target,
627                                RectangleEdge.LEFT);
628                        double hh = Math.abs(vv1 - vv0) + 1.0;
629                        r.setRect(target.getMinX(), Math.min(vv0, vv1),
630                                this.stripWidth, hh);
631                        g2.setPaint(p);
632                        g2.fill(r);
633                    }
634                    if (isStripOutlineVisible()) {
635                        g2.setPaint(this.stripOutlinePaint);
636                        g2.setStroke(this.stripOutlineStroke);
637                        g2.draw(new Rectangle2D.Double(target.getMinX(),
638                                target.getMinY(), this.stripWidth,
639                                target.getHeight()));
640                    }
641                    this.axis.draw(g2, target.getMinX() + this.stripWidth
642                            + this.axisOffset, target, target, RectangleEdge.RIGHT,
643                            null);
644                }
645            }
646            return null;
647        }

649        /**
650         * Tests this legend for equality with an arbitrary object.
651         *
652         * @param obj  the object (<code>null</code> permitted).
653         *
654         * @return A boolean.
655         */
656        public boolean equals(Object obj) {
657            if (!(obj instanceof PaintScaleLegend)) {
658                return false;
659            }
660            PaintScaleLegend that = (PaintScaleLegend) obj;
661            if (!this.scale.equals(that.scale)) {
662                return false;
663            }
664            if (!this.axis.equals(that.axis)) {
665                return false;
666            }
667            if (!this.axisLocation.equals(that.axisLocation)) {
668                return false;
669            }
670            if (this.axisOffset != that.axisOffset) {
671                return false;
672            }
673            if (this.stripWidth != that.stripWidth) {
674                return false;
675            }
676            if (this.stripOutlineVisible != that.stripOutlineVisible) {
677                return false;
678            }
679            if (!PaintUtilities.equal(this.stripOutlinePaint,
680                    that.stripOutlinePaint)) {
681                return false;
682            }
683            if (!this.stripOutlineStroke.equals(that.stripOutlineStroke)) {
684                return false;
685            }
686            if (!PaintUtilities.equal(this.backgroundPaint, that.backgroundPaint)) {
687                return false;
688            }
689            if (this.subdivisions != that.subdivisions) {
690                return false;
691            }
692            return super.equals(obj);
693        }

699         *
700         * @throws IOException  if there is an I/O error.

702        private void writeObject(ObjectOutputStream stream) throws IOException {
703            stream.defaultWriteObject();
704            SerialUtilities.writePaint(this.backgroundPaint, stream);
705            SerialUtilities.writePaint(this.stripOutlinePaint, stream);
706            SerialUtilities.writeStroke(this.stripOutlineStroke, stream);
707        }
