212         * @param paint  the title paint (<code>null</code> not permitted).

221                         RectangleEdge position,
222                         HorizontalAlignment horizontalAlignment,
223                         VerticalAlignment verticalAlignment,
224                         RectangleInsets padding) {
225    
226            super(position, horizontalAlignment, verticalAlignment, padding);
227    
228            if (text == null) {
229                throw new NullPointerException("Null 'text' argument.");
230            }
231            if (font == null) {
232                throw new NullPointerException("Null 'font' argument.");
233            }
234            if (paint == null) {
235                throw new NullPointerException("Null 'paint' argument.");
236            }
237            this.text = text;
238            this.font = font;
239            this.paint = paint;
240            // the textAlignment and the horizontalAlignment are separate things,
241            // but it makes sense for the default textAlignment to match the
242            // title's horizontal alignment...
243            this.textAlignment = horizontalAlignment;
244            this.backgroundPaint = null;
245            this.content = null;
246            this.toolTipText = null;
247            this.urlText = null;
248    
249        }

465        /**
466         * Arranges the contents of the block, within the given constraints, and
467         * returns the block size.
468         *
469         * @param g2  the graphics device.
470         * @param constraint  the constraint (<code>null</code> not permitted).
471         *
472         * @return The block size (in Java2D units, never <code>null</code>).
473         */
474        public Size2D arrange(Graphics2D g2, RectangleConstraint constraint) {
475            RectangleConstraint cc = toContentConstraint(constraint);
476            LengthConstraintType w = cc.getWidthConstraintType();
477            LengthConstraintType h = cc.getHeightConstraintType();
478            Size2D contentSize = null;
479            if (w == LengthConstraintType.NONE) {
480                if (h == LengthConstraintType.NONE) {
481                    contentSize = arrangeNN(g2);
482                }
483                else if (h == LengthConstraintType.RANGE) {
484                    throw new RuntimeException("Not yet implemented.");
485                }
486                else if (h == LengthConstraintType.FIXED) {
487                    throw new RuntimeException("Not yet implemented.");
488                }
489            }
490            else if (w == LengthConstraintType.RANGE) {
491                if (h == LengthConstraintType.NONE) {
492                    contentSize = arrangeRN(g2, cc.getWidthRange());
493                }
494                else if (h == LengthConstraintType.RANGE) {
495                    contentSize = arrangeRR(g2, cc.getWidthRange(),
496                            cc.getHeightRange());
497                }
498                else if (h == LengthConstraintType.FIXED) {
499                    throw new RuntimeException("Not yet implemented.");
500                }
501            }
502            else if (w == LengthConstraintType.FIXED) {
503                if (h == LengthConstraintType.NONE) {
504                    contentSize = arrangeFN(g2, cc.getWidth());
505                }
506                else if (h == LengthConstraintType.RANGE) {
507                    throw new RuntimeException("Not yet implemented.");
508                }
509                else if (h == LengthConstraintType.FIXED) {
510                    throw new RuntimeException("Not yet implemented.");
511                }
512            }
513            return new Size2D(calculateTotalWidth(contentSize.getWidth()),
514                    calculateTotalHeight(contentSize.getHeight()));
515        }

517        /**
518         * Arranges the content for this title assuming no bounds on the width
519         * or the height, and returns the required size.  This will reflect the
520         * fact that a text title positioned on the left or right of a chart will
521         * be rotated by 90 degrees.
522         *
523         * @param g2  the graphics target.
524         *
525         * @return The content size.
526         *
527         * @since 1.0.9
528         */
529        protected Size2D arrangeNN(Graphics2D g2) {
530            Range max = new Range(0.0, Float.MAX_VALUE);
531            return arrangeRR(g2, max, max);
532        }
533    
534        /**
535         * Arranges the content for this title assuming a fixed width and no bounds
536         * on the height, and returns the required size.  This will reflect the
537         * fact that a text title positioned on the left or right of a chart will
538         * be rotated by 90 degrees.
539         *
540         * @param g2  the graphics target.
541         * @param w  the width.
542         *
543         * @return The content size.
544         *
545         * @since 1.0.9
546         */
547        protected Size2D arrangeFN(Graphics2D g2, double w) {
548            RectangleEdge position = getPosition();
549            if (position == RectangleEdge.TOP || position == RectangleEdge.BOTTOM) {
550                float maxWidth = (float) w;
551                g2.setFont(this.font);
552                this.content = TextUtilities.createTextBlock(this.text, this.font,
553                        this.paint, maxWidth, this.maximumLinesToDisplay,
554                        new G2TextMeasurer(g2));
555                this.content.setLineAlignment(this.textAlignment);
556                Size2D contentSize = this.content.calculateDimensions(g2);
557                if (this.expandToFitSpace) {
558                    return new Size2D(maxWidth, contentSize.getHeight());
559                }
560                else {
561                    return contentSize;
562                }
563            }
564            else if (position == RectangleEdge.LEFT || position
565                    == RectangleEdge.RIGHT) {
566                float maxWidth = Float.MAX_VALUE;
567                g2.setFont(this.font);
568                this.content = TextUtilities.createTextBlock(this.text, this.font,
569                        this.paint, maxWidth, this.maximumLinesToDisplay,
570                        new G2TextMeasurer(g2));
571                this.content.setLineAlignment(this.textAlignment);
572                Size2D contentSize = this.content.calculateDimensions(g2);
573    
574                // transpose the dimensions, because the title is rotated
575                if (this.expandToFitSpace) {
576                    return new Size2D(contentSize.getHeight(), maxWidth);
577                }
578                else {
579                    return new Size2D(contentSize.height, contentSize.width);
580                }
581            }
582            else {
583                throw new RuntimeException("Unrecognised exception.");
584            }
585        }
586    
587        /**
588         * Arranges the content for this title assuming a range constraint for the
589         * width and no bounds on the height, and returns the required size.  This
590         * will reflect the fact that a text title positioned on the left or right
591         * of a chart will be rotated by 90 degrees.
592         *
593         * @param g2  the graphics target.
594         * @param widthRange  the range for the width.
595         *
596         * @return The content size.
597         *
598         * @since 1.0.9
599         */
600        protected Size2D arrangeRN(Graphics2D g2, Range widthRange) {
601            Size2D s = arrangeNN(g2);
602            if (widthRange.contains(s.getWidth())) {
603                return s;
604            }
605            double ww = widthRange.constrain(s.getWidth());
606            return arrangeFN(g2, ww);
607        }

672        /**
673         * Draws the block within the specified area.
674         *
675         * @param g2  the graphics device.
676         * @param area  the area.
677         * @param params  if this is an instance of {@link EntityBlockParams} it
678         *                is used to determine whether or not an
679         *                {@link EntityCollection} is returned by this method.
680         *
681         * @return An {@link EntityCollection} containing a chart entity for the
682         *         title, or <code>null</code>.
683         */
684        public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
685            if (this.content == null) {
686                return null;
687            }
688            area = trimMargin(area);
689            drawBorder(g2, area);
690            if (this.text.equals("")) {
691                return null;
692            }
693            ChartEntity entity = null;
694            if (params instanceof EntityBlockParams) {
695                EntityBlockParams p = (EntityBlockParams) params;
696                if (p.getGenerateEntities()) {
697                    entity = new TitleEntity(area, this, this.toolTipText,
698                            this.urlText);
699                }
700            }
701            area = trimBorder(area);
702            if (this.backgroundPaint != null) {
703                g2.setPaint(this.backgroundPaint);
704                g2.fill(area);
705            }
706            area = trimPadding(area);
707            RectangleEdge position = getPosition();
708            if (position == RectangleEdge.TOP || position == RectangleEdge.BOTTOM) {
709                drawHorizontal(g2, area);
710            }
711            else if (position == RectangleEdge.LEFT
712                     || position == RectangleEdge.RIGHT) {
713                drawVertical(g2, area);
714            }
715            BlockResult result = new BlockResult();
716            if (entity != null) {
717                StandardEntityCollection sec = new StandardEntityCollection();
718                sec.add(entity);
719                result.setEntityCollection(sec);
720            }
721            return result;
722        }
