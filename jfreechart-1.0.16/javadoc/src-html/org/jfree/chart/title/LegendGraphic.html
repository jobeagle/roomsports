150        /**
151         * Creates a new legend graphic.
152         *
153         * @param shape  the shape (<code>null</code> not permitted).
154         * @param fillPaint  the fill paint (<code>null</code> not permitted).
155         */
156        public LegendGraphic(Shape shape, Paint fillPaint) {
157            ParamChecks.nullNotPermitted(shape, "shape");
158            ParamChecks.nullNotPermitted(fillPaint, "fillPaint");
159            this.shapeVisible = true;
160            this.shape = shape;
161            this.shapeAnchor = RectangleAnchor.CENTER;
162            this.shapeLocation = RectangleAnchor.CENTER;
163            this.shapeFilled = true;
164            this.fillPaint = fillPaint;
165            this.fillPaintTransformer = new StandardGradientPaintTransformer();
166            setPadding(2.0, 2.0, 2.0, 2.0);
167        }

494        /**
495         * Arranges the contents of the block, within the given constraints, and
496         * returns the block size.
497         *
498         * @param g2  the graphics device.
499         * @param constraint  the constraint (<code>null</code> not permitted).
500         *
501         * @return The block size (in Java2D units, never <code>null</code>).
502         */
503        public Size2D arrange(Graphics2D g2, RectangleConstraint constraint) {
504            RectangleConstraint contentConstraint = toContentConstraint(constraint);
505            LengthConstraintType w = contentConstraint.getWidthConstraintType();
506            LengthConstraintType h = contentConstraint.getHeightConstraintType();
507            Size2D contentSize = null;
508            if (w == LengthConstraintType.NONE) {
509                if (h == LengthConstraintType.NONE) {
510                    contentSize = arrangeNN(g2);
511                }
512                else if (h == LengthConstraintType.RANGE) {
513                    throw new RuntimeException("Not yet implemented.");
514                }
515                else if (h == LengthConstraintType.FIXED) {
516                    throw new RuntimeException("Not yet implemented.");
517                }
518            }
519            else if (w == LengthConstraintType.RANGE) {
520                if (h == LengthConstraintType.NONE) {
521                    throw new RuntimeException("Not yet implemented.");
522                }
523                else if (h == LengthConstraintType.RANGE) {
524                    throw new RuntimeException("Not yet implemented.");
525                }
526                else if (h == LengthConstraintType.FIXED) {
527                    throw new RuntimeException("Not yet implemented.");
528                }
529            }
530            else if (w == LengthConstraintType.FIXED) {
531                if (h == LengthConstraintType.NONE) {
532                    throw new RuntimeException("Not yet implemented.");
533                }
534                else if (h == LengthConstraintType.RANGE) {
535                    throw new RuntimeException("Not yet implemented.");
536                }
537                else if (h == LengthConstraintType.FIXED) {
538                    contentSize = new Size2D(
539                        contentConstraint.getWidth(),
540                        contentConstraint.getHeight()
541                    );
542                }
543            }
544            return new Size2D(
545                calculateTotalWidth(contentSize.getWidth()),
546                calculateTotalHeight(contentSize.getHeight())
547            );
548        }
549    
550        /**
551         * Performs the layout with no constraint, so the content size is
552         * determined by the bounds of the shape and/or line drawn to represent
553         * the series.
554         *
555         * @param g2  the graphics device.
556         *
557         * @return  The content size.
558         */
559        protected Size2D arrangeNN(Graphics2D g2) {
560            Rectangle2D contentSize = new Rectangle2D.Double();
561            if (this.line != null) {
562                contentSize.setRect(this.line.getBounds2D());
563            }
564            if (this.shape != null) {
565                contentSize = contentSize.createUnion(this.shape.getBounds2D());
566            }
567            return new Size2D(contentSize.getWidth(), contentSize.getHeight());
568        }

570        /**
571         * Draws the graphic item within the specified area.
572         *
573         * @param g2  the graphics device.
574         * @param area  the area.
575         */
576        public void draw(Graphics2D g2, Rectangle2D area) {
577    
578            area = trimMargin(area);
579            drawBorder(g2, area);
580            area = trimBorder(area);
581            area = trimPadding(area);
582    
583            if (this.lineVisible) {
584                Point2D location = RectangleAnchor.coordinates(area,
585                        this.shapeLocation);
586                Shape aLine = ShapeUtilities.createTranslatedShape(getLine(),
587                        this.shapeAnchor, location.getX(), location.getY());
588                g2.setPaint(this.linePaint);
589                g2.setStroke(this.lineStroke);
590                g2.draw(aLine);
591            }
592    
593            if (this.shapeVisible) {
594                Point2D location = RectangleAnchor.coordinates(area,
595                        this.shapeLocation);
596    
597                Shape s = ShapeUtilities.createTranslatedShape(this.shape,
598                        this.shapeAnchor, location.getX(), location.getY());
599                if (this.shapeFilled) {
600                    Paint p = this.fillPaint;
601                    if (p instanceof GradientPaint) {
602                        GradientPaint gp = (GradientPaint) this.fillPaint;
603                        p = this.fillPaintTransformer.transform(gp, s);
604                    }
605                    g2.setPaint(p);
606                    g2.fill(s);
607                }
608                if (this.shapeOutlineVisible) {
609                    g2.setPaint(this.outlinePaint);
610                    g2.setStroke(this.outlineStroke);
611                    g2.draw(s);
612                }
613            }
614    
615        }

631        /**
632         * Tests this <code>LegendGraphic</code> instance for equality with an
633         * arbitrary object.
634         *
635         * @param obj  the object (<code>null</code> permitted).
636         *
637         * @return A boolean.
638         */
639        public boolean equals(Object obj) {
640            if (!(obj instanceof LegendGraphic)) {
641                return false;
642            }
643            LegendGraphic that = (LegendGraphic) obj;
644            if (this.shapeVisible != that.shapeVisible) {
645                return false;
646            }
647            if (!ShapeUtilities.equal(this.shape, that.shape)) {
648                return false;
649            }
650            if (this.shapeFilled != that.shapeFilled) {
651                return false;
652            }
653            if (!PaintUtilities.equal(this.fillPaint, that.fillPaint)) {
654                return false;
655            }
656            if (!ObjectUtilities.equal(this.fillPaintTransformer,
657                    that.fillPaintTransformer)) {
658                return false;
659            }
660            if (this.shapeOutlineVisible != that.shapeOutlineVisible) {
661                return false;
662            }
663            if (!PaintUtilities.equal(this.outlinePaint, that.outlinePaint)) {
664                return false;
665            }
666            if (!ObjectUtilities.equal(this.outlineStroke, that.outlineStroke)) {
667                return false;
668            }
669            if (this.shapeAnchor != that.shapeAnchor) {
670                return false;
671            }
672            if (this.shapeLocation != that.shapeLocation) {
673                return false;
674            }
675            if (this.lineVisible != that.lineVisible) {
676                return false;
677            }
678            if (!ShapeUtilities.equal(this.line, that.line)) {
679                return false;
680            }
681            if (!PaintUtilities.equal(this.linePaint, that.linePaint)) {
682                return false;
683            }
684            if (!ObjectUtilities.equal(this.lineStroke, that.lineStroke)) {
685                return false;
686            }
687            return super.equals(obj);
688        }
