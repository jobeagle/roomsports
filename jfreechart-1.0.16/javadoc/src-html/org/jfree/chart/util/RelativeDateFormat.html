161        /**
162         * Creates a new instance.
163         *
164         * @param baseMillis  the time zone (<code>null</code> not permitted).
165         */
166        public RelativeDateFormat(long baseMillis) {
167            super();
168            this.baseMillis = baseMillis;
169            this.showZeroDays = false;
170            this.showZeroHours = true;
171            this.positivePrefix = "";
172            this.dayFormatter = NumberFormat.getNumberInstance();
173            this.daySuffix = "d";
174            this.hourFormatter = NumberFormat.getNumberInstance();
175            this.hourSuffix = "h";
176            this.minuteFormatter = NumberFormat.getNumberInstance();
177            this.minuteSuffix = "m";
178            this.secondFormatter = NumberFormat.getNumberInstance();
179            this.secondFormatter.setMaximumFractionDigits(3);
180            this.secondFormatter.setMinimumFractionDigits(3);
181            this.secondSuffix = "s";
182    
183            // we don't use the calendar or numberFormat fields, but equals(Object)
184            // is failing without them being non-null
185            this.calendar = new GregorianCalendar();
186            this.numberFormat = new DecimalFormat("0");
187        }

433        /**
434         * Formats the given date as the amount of elapsed time (relative to the
435         * base date specified in the constructor).
436         *
437         * @param date  the date.
438         * @param toAppendTo  the string buffer.
439         * @param fieldPosition  the field position.
440         *
441         * @return The formatted date.
442         */
443        public StringBuffer format(Date date, StringBuffer toAppendTo,
444                                   FieldPosition fieldPosition) {
445            long currentMillis = date.getTime();
446            long elapsed = currentMillis - this.baseMillis;
447            String signPrefix;
448            if (elapsed < 0) {
449                elapsed *= -1L;
450                signPrefix = "-";
451            }
452            else {
453                signPrefix = this.positivePrefix;
454            }
455    
456            long days = elapsed / MILLISECONDS_IN_ONE_DAY;
457            elapsed = elapsed - (days * MILLISECONDS_IN_ONE_DAY);
458            long hours = elapsed / MILLISECONDS_IN_ONE_HOUR;
459            elapsed = elapsed - (hours * MILLISECONDS_IN_ONE_HOUR);
460            long minutes = elapsed / 60000L;
461            elapsed = elapsed - (minutes * 60000L);
462            double seconds = elapsed / 1000.0;
463    
464            toAppendTo.append(signPrefix);
465            if (days != 0 || this.showZeroDays) {
466                toAppendTo.append(this.dayFormatter.format(days))
467                        .append(getDaySuffix());
468            }
469            if (hours != 0 || this.showZeroHours) {
470                toAppendTo.append(this.hourFormatter.format(hours))
471                        .append(getHourSuffix());
472            }
473            toAppendTo.append(this.minuteFormatter.format(minutes))
474                    .append(getMinuteSuffix());
475            toAppendTo.append(this.secondFormatter.format(seconds))
476                    .append(getSecondSuffix());
477            return toAppendTo;
478        }

492        /**
493         * Tests this formatter for equality with an arbitrary object.
494         *
495         * @param obj  the object (<code>null</code> permitted).
496         *
497         * @return A boolean.
498         */
499        public boolean equals(Object obj) {
500            if (obj == this) {
501                return true;
502            }
503            if (!(obj instanceof RelativeDateFormat)) {
504                return false;
505            }
506            if (!super.equals(obj)) {
507                return false;
508            }
509            RelativeDateFormat that = (RelativeDateFormat) obj;
510            if (this.baseMillis != that.baseMillis) {
511                return false;
512            }
513            if (this.showZeroDays != that.showZeroDays) {
514                return false;
515            }
516            if (this.showZeroHours != that.showZeroHours) {
517                return false;
518            }
519            if (!this.positivePrefix.equals(that.positivePrefix)) {
520                return false;
521            }
522            if (!this.daySuffix.equals(that.daySuffix)) {
523                return false;
524            }
525            if (!this.hourSuffix.equals(that.hourSuffix)) {
526                return false;
527            }
528            if (!this.minuteSuffix.equals(that.minuteSuffix)) {
529                return false;
530            }
531            if (!this.secondSuffix.equals(that.secondSuffix)) {
532                return false;
533            }
534            if (!this.dayFormatter.equals(that.dayFormatter)) {
535                return false;
536            }
537            if (!this.hourFormatter.equals(that.hourFormatter)) {
538                return false;
539            }
540            if (!this.minuteFormatter.equals(that.minuteFormatter)) {
541                return false;
542            }
543            if (!this.secondFormatter.equals(that.secondFormatter)) {
544                return false;
545            }
546            return true;
547        }

549        /**
550         * Returns a hash code for this instance.
551         *
552         * @return A hash code.
553         */
554        public int hashCode() {
555            int result = 193;
556            result = 37 * result
557                    + (int) (this.baseMillis ^ (this.baseMillis >>> 32));
558            result = 37 * result + this.positivePrefix.hashCode();
559            result = 37 * result + this.daySuffix.hashCode();
560            result = 37 * result + this.hourSuffix.hashCode();
561            result = 37 * result + this.minuteSuffix.hashCode();
562            result = 37 * result + this.secondSuffix.hashCode();
563            result = 37 * result + this.secondFormatter.hashCode();
564            return result;
565        }

585            GregorianCalendar c0 = new GregorianCalendar(2006, 10, 1, 0, 0, 0);

588    
589            System.out.println("Default: ");
590            RelativeDateFormat rdf = new RelativeDateFormat(c0.getTime().getTime());
591            System.out.println(rdf.format(c1.getTime()));
592            System.out.println();
593    
594            System.out.println("Hide milliseconds: ");

596            System.out.println(rdf.format(c1.getTime()));

599            System.out.println("Show zero day output: ");
600            rdf.setShowZeroDays(true);
601            System.out.println(rdf.format(c1.getTime()));
602            System.out.println();
603    
604            System.out.println("Alternative suffixes: ");
605            rdf.setShowZeroDays(false);
606            rdf.setDaySuffix(":");
607            rdf.setHourSuffix(":");
608            rdf.setMinuteSuffix(":");
609            rdf.setSecondSuffix("");
610            System.out.println(rdf.format(c1.getTime()));
