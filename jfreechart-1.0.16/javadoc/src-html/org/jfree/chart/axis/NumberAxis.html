130    /**
131     * An axis for displaying numerical data.
132     * <P>
133     * If the axis is set up to automatically determine its range to fit the data,

136     * <code>true</code>.

138     * The <code>NumberAxis</code> class has a mechanism for automatically
139     * selecting a tick unit that is appropriate for the current axis range.  This
140     * mechanism is an adaptation of code suggested by Laurence Vanhelsuwe.

418        /**
419         * Rescales the axis to ensure that all data is visible.
420         */
421        protected void autoAdjustRange() {
422    
423            Plot plot = getPlot();
424            if (plot == null) {
425                return;  // no plot, no data
426            }
427    
428            if (plot instanceof ValueAxisPlot) {
429                ValueAxisPlot vap = (ValueAxisPlot) plot;
430    
431                Range r = vap.getDataRange(this);
432                if (r == null) {
433                    r = getDefaultAutoRange();
434                }
435    
436                double upper = r.getUpperBound();
437                double lower = r.getLowerBound();
438                if (this.rangeType == RangeType.POSITIVE) {
439                    lower = Math.max(0.0, lower);
440                    upper = Math.max(0.0, upper);
441                }
442                else if (this.rangeType == RangeType.NEGATIVE) {
443                    lower = Math.min(0.0, lower);
444                    upper = Math.min(0.0, upper);
445                }
446    
447                if (getAutoRangeIncludesZero()) {
448                    lower = Math.min(lower, 0.0);
449                    upper = Math.max(upper, 0.0);
450                }
451                double range = upper - lower;
452    
453                // if fixed auto range, then derive lower bound...
454                double fixedAutoRange = getFixedAutoRange();
455                if (fixedAutoRange > 0.0) {
456                    lower = upper - fixedAutoRange;
457                }
458                else {
459                    // ensure the autorange is at least <minRange> in size...
460                    double minRange = getAutoRangeMinimumSize();
461                    if (range < minRange) {
462                        double expand = (minRange - range) / 2;
463                        upper = upper + expand;
464                        lower = lower - expand;
465                        if (lower == upper) { // see bug report 1549218
466                            double adjust = Math.abs(lower) / 10.0;
467                            lower = lower - adjust;
468                            upper = upper + adjust;
469                        }
470                        if (this.rangeType == RangeType.POSITIVE) {
471                            if (lower < 0.0) {
472                                upper = upper - lower;
473                                lower = 0.0;
474                            }
475                        }
476                        else if (this.rangeType == RangeType.NEGATIVE) {
477                            if (upper > 0.0) {
478                                lower = lower - upper;
479                                upper = 0.0;
480                            }
481                        }
482                    }
483    
484                    if (getAutoRangeStickyZero()) {
485                        if (upper <= 0.0) {
486                            upper = Math.min(0.0, upper + getUpperMargin() * range);
487                        }
488                        else {
489                            upper = upper + getUpperMargin() * range;
490                        }
491                        if (lower >= 0.0) {
492                            lower = Math.max(0.0, lower - getLowerMargin() * range);
493                        }
494                        else {
495                            lower = lower - getLowerMargin() * range;
496                        }
497                    }
498                    else {
499                        upper = upper + getUpperMargin() * range;
500                        lower = lower - getLowerMargin() * range;
501                    }
502                }
503    
504                setRange(new Range(lower, upper), false, false);
505            }
506    
507        }

509        /**
510         * Converts a data value to a coordinate in Java2D space, assuming that the
511         * axis runs along one edge of the specified dataArea.
512         * <p>
513         * Note that it is possible for the coordinate to fall outside the plotArea.
514         *
515         * @param value  the data value.
516         * @param area  the area for plotting the data.
517         * @param edge  the axis location.
518         *
519         * @return The Java2D coordinate.
520         *
521         * @see #java2DToValue(double, Rectangle2D, RectangleEdge)
522         */
523        public double valueToJava2D(double value, Rectangle2D area,
524                RectangleEdge edge) {
525    
526            Range range = getRange();
527            double axisMin = range.getLowerBound();
528            double axisMax = range.getUpperBound();
529    
530            double min = 0.0;
531            double max = 0.0;
532            if (RectangleEdge.isTopOrBottom(edge)) {
533                min = area.getX();
534                max = area.getMaxX();
535            }
536            else if (RectangleEdge.isLeftOrRight(edge)) {
537                max = area.getMinY();
538                min = area.getMaxY();
539            }
540            if (isInverted()) {
541                return max
542                       - ((value - axisMin) / (axisMax - axisMin)) * (max - min);
543            }
544            else {
545                return min
546                       + ((value - axisMin) / (axisMax - axisMin)) * (max - min);
547            }
548    
549        }
550    
551        /**
552         * Converts a coordinate in Java2D space to the corresponding data value,
553         * assuming that the axis runs along one edge of the specified dataArea.
554         *
555         * @param java2DValue  the coordinate in Java2D space.
556         * @param area  the area in which the data is plotted.
557         * @param edge  the location.
558         *
559         * @return The data value.
560         *
561         * @see #valueToJava2D(double, Rectangle2D, RectangleEdge)
562         */
563        public double java2DToValue(double java2DValue, Rectangle2D area,
564                RectangleEdge edge) {
565    
566            Range range = getRange();
567            double axisMin = range.getLowerBound();
568            double axisMax = range.getUpperBound();
569    
570            double min = 0.0;
571            double max = 0.0;
572            if (RectangleEdge.isTopOrBottom(edge)) {
573                min = area.getX();
574                max = area.getMaxX();
575            }
576            else if (RectangleEdge.isLeftOrRight(edge)) {
577                min = area.getMaxY();
578                max = area.getY();
579            }
580            if (isInverted()) {
581                return axisMax
582                       - (java2DValue - min) / (max - min) * (axisMax - axisMin);
583            }
584            else {
585                return axisMin
586                       + (java2DValue - min) / (max - min) * (axisMax - axisMin);
587            }
588    
589        }

591        /**
592         * Calculates the value of the lowest visible tick on the axis.
593         *
594         * @return The value of the lowest visible tick on the axis.
595         *
596         * @see #calculateHighestVisibleTickValue()
597         */
598        protected double calculateLowestVisibleTickValue() {
599            double unit = getTickUnit().getSize();
600            double index = Math.ceil(getRange().getLowerBound() / unit);
601            return index * unit;
602        }

604        /**
605         * Calculates the value of the highest visible tick on the axis.
606         *
607         * @return The value of the highest visible tick on the axis.
608         *
609         * @see #calculateLowestVisibleTickValue()
610         */
611        protected double calculateHighestVisibleTickValue() {
612            double unit = getTickUnit().getSize();
613            double index = Math.floor(getRange().getUpperBound() / unit);
614            return index * unit;
615        }

617        /**
618         * Calculates the number of visible ticks.
619         *
620         * @return The number of visible ticks on the axis.
621         */
622        protected int calculateVisibleTickCount() {
623            double unit = getTickUnit().getSize();
624            Range range = getRange();
625            return (int) (Math.floor(range.getUpperBound() / unit)
626                          - Math.ceil(range.getLowerBound() / unit) + 1);
627        }

629        /**
630         * Draws the axis on a Java 2D graphics device (such as the screen or a
631         * printer).
632         *
633         * @param g2  the graphics device (<code>null</code> not permitted).
634         * @param cursor  the cursor location.
635         * @param plotArea  the area within which the axes and data should be drawn
636         *                  (<code>null</code> not permitted).
637         * @param dataArea  the area within which the data should be drawn
638         *                  (<code>null</code> not permitted).
639         * @param edge  the location of the axis (<code>null</code> not permitted).
640         * @param plotState  collects information about the plot
641         *                   (<code>null</code> permitted).
642         *
643         * @return The axis state (never <code>null</code>).
644         */
645        public AxisState draw(Graphics2D g2, double cursor, Rectangle2D plotArea,
646                Rectangle2D dataArea, RectangleEdge edge,
647                PlotRenderingInfo plotState) {
648    
649            AxisState state = null;
650            // if the axis is not visible, don't draw it...
651            if (!isVisible()) {
652                state = new AxisState(cursor);
653                // even though the axis is not visible, we need ticks for the
654                // gridlines...
655                List ticks = refreshTicks(g2, state, dataArea, edge);
656                state.setTicks(ticks);
657                return state;
658            }
659    
660            // draw the tick marks and labels...
661            state = drawTickMarksAndLabels(g2, cursor, plotArea, dataArea, edge);
662    
663            if (getAttributedLabel() != null) {
664                state = drawAttributedLabel(getAttributedLabel(), g2, plotArea, 
665                        dataArea, edge, state);
666                
667            } else {

968        /**
969         * Estimates the maximum width of the tick labels, assuming the specified
970         * tick unit is used.
971         * <P>
972         * Rather than computing the string bounds of every tick on the axis, we
973         * just look at two values: the lower bound and the upper bound for the
974         * axis.  These two values will usually be representative.
975         *
976         * @param g2  the graphics device.
977         * @param unit  the tick unit to use for calculation.
978         *
979         * @return The estimated maximum width of the tick labels.
980         */
981        protected double estimateMaximumTickLabelWidth(Graphics2D g2,
982                                                       TickUnit unit) {
983    
984            RectangleInsets tickLabelInsets = getTickLabelInsets();
985            double result = tickLabelInsets.getLeft() + tickLabelInsets.getRight();
986    
987            if (isVerticalTickLabels()) {
988                // all tick labels have the same width (equal to the height of the
989                // font)...
990                FontRenderContext frc = g2.getFontRenderContext();
991                LineMetrics lm = getTickLabelFont().getLineMetrics("0", frc);
992                result += lm.getHeight();
993            }
994            else {
995                // look at lower and upper bounds...
996                FontMetrics fm = g2.getFontMetrics(getTickLabelFont());
997                Range range = getRange();
998                double lower = range.getLowerBound();
999                double upper = range.getUpperBound();
1000                String lowerStr, upperStr;
1001                NumberFormat formatter = getNumberFormatOverride();
1002                if (formatter != null) {
1003                    lowerStr = formatter.format(lower);
1004                    upperStr = formatter.format(upper);
1005                }
1006                else {
1007                    lowerStr = unit.valueToString(lower);
1008                    upperStr = unit.valueToString(upper);
1009                }
1010                double w1 = fm.stringWidth(lowerStr);
1011                double w2 = fm.stringWidth(upperStr);
1012                result += Math.max(w1, w2);
1013            }
1014    
1015            return result;
1016    
1017        }

1049       protected void selectHorizontalAutoTickUnit(Graphics2D g2,

1053                    getTickUnit());

1058            double unit1Width = lengthToJava2D(unit1.getSize(), dataArea, edge);
1059    
1060            // then extrapolate...

1064                    guess);

1066

1068            if (tickLabelWidth > unit2Width) {
1069                unit2 = (NumberTickUnit) tickUnits.getLargerTickUnit(unit2);
1070            }

1142         * @param dataArea  the area in which the data should be drawn.
1143         * @param edge  the location of the axis.
1144         *
1145         * @return A list of ticks.
1146         */
1147        protected List refreshTicksHorizontal(Graphics2D g2,
1148                Rectangle2D dataArea, RectangleEdge edge) {
1149    
1150            List result = new java.util.ArrayList();
1151    
1152            Font tickLabelFont = getTickLabelFont();
1153            g2.setFont(tickLabelFont);
1154    
1155            if (isAutoTickUnitSelection()) {
1156                selectAutoTickUnit(g2, dataArea, edge);
1157            }
1158    
1159            TickUnit tu = getTickUnit();
1160            double size = tu.getSize();
1161            int count = calculateVisibleTickCount();
1162            double lowestTickValue = calculateLowestVisibleTickValue();
1163    
1164            if (count <= ValueAxis.MAXIMUM_TICK_COUNT) {
1165                int minorTickSpaces = getMinorTickCount();
1166                if (minorTickSpaces <= 0) {
1167                    minorTickSpaces = tu.getMinorTickCount();
1168                }
1169                for (int minorTick = 1; minorTick < minorTickSpaces; minorTick++) {
1170                    double minorTickValue = lowestTickValue 
1171                            - size * minorTick / minorTickSpaces;
1172                    if (getRange().contains(minorTickValue)) {
1173                        result.add(new NumberTick(TickType.MINOR, minorTickValue,
1174                                "", TextAnchor.TOP_CENTER, TextAnchor.CENTER,
1175                                0.0));
1176                    }
1177                }
1178                for (int i = 0; i < count; i++) {
1179                    double currentTickValue = lowestTickValue + (i * size);
1180                    String tickLabel;
1181                    NumberFormat formatter = getNumberFormatOverride();
1182                    if (formatter != null) {
1183                        tickLabel = formatter.format(currentTickValue);
1184                    }
1185                    else {
1186                        tickLabel = getTickUnit().valueToString(currentTickValue);
1187                    }
1188                    TextAnchor anchor, rotationAnchor;
1189                    double angle = 0.0;
1190                    if (isVerticalTickLabels()) {
1191                        anchor = TextAnchor.CENTER_RIGHT;
1192                        rotationAnchor = TextAnchor.CENTER_RIGHT;
1193                        if (edge == RectangleEdge.TOP) {
1194                            angle = Math.PI / 2.0;
1195                        }
1196                        else {
1197                            angle = -Math.PI / 2.0;
1198                        }
1199                    }
1200                    else {
1201                        if (edge == RectangleEdge.TOP) {
1202                            anchor = TextAnchor.BOTTOM_CENTER;
1203                            rotationAnchor = TextAnchor.BOTTOM_CENTER;
1204                        }
1205                        else {
1206                            anchor = TextAnchor.TOP_CENTER;
1207                            rotationAnchor = TextAnchor.TOP_CENTER;
1208                        }
1209                    }
1210    
1211                    Tick tick = new NumberTick(new Double(currentTickValue),
1212                            tickLabel, anchor, rotationAnchor, angle);
1213                    result.add(tick);
1214                    double nextTickValue = lowestTickValue + ((i + 1) * size);
1215                    for (int minorTick = 1; minorTick < minorTickSpaces;
1216                            minorTick++) {
1217                        double minorTickValue = currentTickValue
1218                                + (nextTickValue - currentTickValue)
1219                                * minorTick / minorTickSpaces;
1220                        if (getRange().contains(minorTickValue)) {
1221                            result.add(new NumberTick(TickType.MINOR,
1222                                    minorTickValue, "", TextAnchor.TOP_CENTER,
1223                                    TextAnchor.CENTER, 0.0));

1233         * Calculates the positions of the tick labels for the axis, storing the
1234         * results in the tick label list (ready for drawing).
1235         *
1236         * @param g2  the graphics device.
1237         * @param dataArea  the area in which the plot should be drawn.
1238         * @param edge  the location of the axis.
1239         *
1240         * @return A list of ticks.
1241         */
1242        protected List refreshTicksVertical(Graphics2D g2,
1243                Rectangle2D dataArea, RectangleEdge edge) {
1244    
1245            List result = new java.util.ArrayList();
1246            result.clear();
1247    
1248            Font tickLabelFont = getTickLabelFont();
1249            g2.setFont(tickLabelFont);
1250            if (isAutoTickUnitSelection()) {
1251                selectAutoTickUnit(g2, dataArea, edge);
1252            }
1253    
1254            TickUnit tu = getTickUnit();
1255            double size = tu.getSize();
1256            int count = calculateVisibleTickCount();
1257            double lowestTickValue = calculateLowestVisibleTickValue();
1258    
1259            if (count <= ValueAxis.MAXIMUM_TICK_COUNT) {
1260                int minorTickSpaces = getMinorTickCount();
1261                if (minorTickSpaces <= 0) {
1262                    minorTickSpaces = tu.getMinorTickCount();
1263                }
1264                for (int minorTick = 1; minorTick < minorTickSpaces; minorTick++) {
1265                    double minorTickValue = lowestTickValue
1266                            - size * minorTick / minorTickSpaces;
1267                    if (getRange().contains(minorTickValue)) {
1268                        result.add(new NumberTick(TickType.MINOR, minorTickValue,
1269                                "", TextAnchor.TOP_CENTER, TextAnchor.CENTER,
1270                                0.0));
1271                    }
1272                }
1273    
1274                for (int i = 0; i < count; i++) {
1275                    double currentTickValue = lowestTickValue + (i * size);
1276                    String tickLabel;
1277                    NumberFormat formatter = getNumberFormatOverride();
1278                    if (formatter != null) {
1279                        tickLabel = formatter.format(currentTickValue);
1280                    }
1281                    else {
1282                        tickLabel = getTickUnit().valueToString(currentTickValue);
1283                    }
1284    
1285                    TextAnchor anchor;
1286                    TextAnchor rotationAnchor;
1287                    double angle = 0.0;
1288                    if (isVerticalTickLabels()) {
1289                        if (edge == RectangleEdge.LEFT) {
1290                            anchor = TextAnchor.BOTTOM_CENTER;
1291                            rotationAnchor = TextAnchor.BOTTOM_CENTER;
1292                            angle = -Math.PI / 2.0;
1293                        }
1294                        else {
1295                            anchor = TextAnchor.BOTTOM_CENTER;
1296                            rotationAnchor = TextAnchor.BOTTOM_CENTER;
1297                            angle = Math.PI / 2.0;
1298                        }
1299                    }
1300                    else {
1301                        if (edge == RectangleEdge.LEFT) {
1302                            anchor = TextAnchor.CENTER_RIGHT;
1303                            rotationAnchor = TextAnchor.CENTER_RIGHT;
1304                        }
1305                        else {
1306                            anchor = TextAnchor.CENTER_LEFT;
1307                            rotationAnchor = TextAnchor.CENTER_LEFT;
1308                        }
1309                    }
1310    
1311                    Tick tick = new NumberTick(new Double(currentTickValue),
1312                            tickLabel, anchor, rotationAnchor, angle);
1313                    result.add(tick);
1314    
1315                    double nextTickValue = lowestTickValue + ((i + 1) * size);
1316                    for (int minorTick = 1; minorTick < minorTickSpaces;
1317                            minorTick++) {
1318                        double minorTickValue = currentTickValue
1319                                + (nextTickValue - currentTickValue)
1320                                * minorTick / minorTickSpaces;
1321                        if (getRange().contains(minorTickValue)) {
1322                            result.add(new NumberTick(TickType.MINOR,
1323                                    minorTickValue, "", TextAnchor.TOP_CENTER,
1324                                    TextAnchor.CENTER, 0.0));
1325                        }
1326                    }
