080    /**
081     * A specialised category axis that can display sub-categories.
082     */
083    public class SubCategoryAxis extends CategoryAxis
084            implements Cloneable, Serializable {

086        /** For serialization. */
087        private static final long serialVersionUID = -1279463299793228344L;

089        /** Storage for the sub-categories (these need to be set manually). */
090        private List subCategories;
091    
092        /** The font for the sub-category labels. */
093        private Font subLabelFont = new Font("SansSerif", Font.PLAIN, 10);
094    
095        /** The paint for the sub-category labels. */
096        private transient Paint subLabelPaint = Color.black;

098        /**
099         * Creates a new axis.
100         *
101         * @param label  the axis label.
102         */
103        public SubCategoryAxis(String label) {
104            super(label);

106        }

170        /**
171         * Estimates the space required for the axis, given a specific drawing area.
172         *
173         * @param g2  the graphics device (used to obtain font information).
174         * @param plot  the plot that the axis belongs to.
175         * @param plotArea  the area within which the axis should be drawn.
176         * @param edge  the axis location (top or bottom).
177         * @param space  the space already reserved.
178         *
179         * @return The space required to draw the axis.
180         */
181        public AxisSpace reserveSpace(Graphics2D g2, Plot plot, 
182                Rectangle2D plotArea, RectangleEdge edge, AxisSpace space) {
183    
184            // create a new space object if one wasn't supplied...
185            if (space == null) {
186                space = new AxisSpace();
187            }
188    
189            // if the axis is not visible, no additional space is required...
190            if (!isVisible()) {
191                return space;
192            }
193    
194            space = super.reserveSpace(g2, plot, plotArea, edge, space);
195            double maxdim = getMaxDim(g2, edge);
196            if (RectangleEdge.isTopOrBottom(edge)) {
197                space.add(maxdim, edge);
198            }
199            else if (RectangleEdge.isLeftOrRight(edge)) {
200                space.add(maxdim, edge);
201            }
202            return space;
203        }
204    
205        /**
206         * Returns the maximum of the relevant dimension (height or width) of the
207         * subcategory labels.
208         *
209         * @param g2  the graphics device.
210         * @param edge  the edge.
211         *
212         * @return The maximum dimension.
213         */
214        private double getMaxDim(Graphics2D g2, RectangleEdge edge) {
215            double result = 0.0;
216            g2.setFont(this.subLabelFont);
217            FontMetrics fm = g2.getFontMetrics();
218            Iterator iterator = this.subCategories.iterator();
219            while (iterator.hasNext()) {
220                Comparable subcategory = (Comparable) iterator.next();
221                String label = subcategory.toString();
222                Rectangle2D bounds = TextUtilities.getTextBounds(label, g2, fm);
223                double dim;
224                if (RectangleEdge.isLeftOrRight(edge)) {
225                    dim = bounds.getWidth();
226                }
227                else {  // must be top or bottom
228                    dim = bounds.getHeight();
229                }
230                result = Math.max(result, dim);
231            }
232            return result;
233        }

235        /**
236         * Draws the axis on a Java 2D graphics device (such as the screen or a
237         * printer).
238         *
239         * @param g2  the graphics device (<code>null</code> not permitted).
240         * @param cursor  the cursor location.
241         * @param plotArea  the area within which the axis should be drawn
242         *                  (<code>null</code> not permitted).
243         * @param dataArea  the area within which the plot is being drawn
244         *                  (<code>null</code> not permitted).
245         * @param edge  the location of the axis (<code>null</code> not permitted).
246         * @param plotState  collects information about the plot
247         *                   (<code>null</code> permitted).
248         *
249         * @return The axis state (never <code>null</code>).
250         */
251        public AxisState draw(Graphics2D g2, double cursor, Rectangle2D plotArea,
252                Rectangle2D dataArea, RectangleEdge edge, 
253                PlotRenderingInfo plotState) {
254    
255            // if the axis is not visible, don't draw it...
256            if (!isVisible()) {
257                return new AxisState(cursor);
258            }
259    
260            if (isAxisLineVisible()) {
261                drawAxisLine(g2, cursor, dataArea, edge);
262            }
263    
264            // draw the category labels and axis label
265            AxisState state = new AxisState(cursor);
266            state = drawSubCategoryLabels(g2, plotArea, dataArea, edge, state, 
267                    plotState);
268            state = drawCategoryLabels(g2, plotArea, dataArea, edge, state,
269                    plotState);
270            if (getAttributedLabel() != null) {
271                state = drawAttributedLabel(getAttributedLabel(), g2, plotArea, 
272                        dataArea, edge, state);
273            } else {
274                state = drawLabel(getLabel(), g2, plotArea, dataArea, edge, state);
275            } 
276            return state;
277    
278        }

280        /**
281         * Draws the category labels and returns the updated axis state.
282         *
283         * @param g2  the graphics device (<code>null</code> not permitted).
284         * @param plotArea  the plot area (<code>null</code> not permitted).
285         * @param dataArea  the area inside the axes (<code>null</code> not
286         *                  permitted).
287         * @param edge  the axis location (<code>null</code> not permitted).
288         * @param state  the axis state (<code>null</code> not permitted).
289         * @param plotState  collects information about the plot (<code>null</code>
290         *                   permitted).
291         *
292         * @return The updated axis state (never <code>null</code>).
293         */
294        protected AxisState drawSubCategoryLabels(Graphics2D g2,
295                Rectangle2D plotArea, Rectangle2D dataArea, RectangleEdge edge,
296                AxisState state, PlotRenderingInfo plotState) {
297    
298            ParamChecks.nullNotPermitted(state, "state");
299    
300            g2.setFont(this.subLabelFont);
301            g2.setPaint(this.subLabelPaint);
302            CategoryPlot plot = (CategoryPlot) getPlot();
303            int categoryCount = 0;
304            CategoryDataset dataset = plot.getDataset();
305            if (dataset != null) {
306                categoryCount = dataset.getColumnCount();
307            }
308    
309            double maxdim = getMaxDim(g2, edge);
310            for (int categoryIndex = 0; categoryIndex < categoryCount;
311                 categoryIndex++) {
312    
313                double x0 = 0.0;
314                double x1 = 0.0;
315                double y0 = 0.0;
316                double y1 = 0.0;
317                if (edge == RectangleEdge.TOP) {
318                    x0 = getCategoryStart(categoryIndex, categoryCount, dataArea,
319                            edge);
320                    x1 = getCategoryEnd(categoryIndex, categoryCount, dataArea,
321                            edge);
322                    y1 = state.getCursor();
323                    y0 = y1 - maxdim;
324                }
325                else if (edge == RectangleEdge.BOTTOM) {
326                    x0 = getCategoryStart(categoryIndex, categoryCount, dataArea,
327                            edge);
328                    x1 = getCategoryEnd(categoryIndex, categoryCount, dataArea,
329                            edge);
330                    y0 = state.getCursor();
331                    y1 = y0 + maxdim;
332                }
333                else if (edge == RectangleEdge.LEFT) {
334                    y0 = getCategoryStart(categoryIndex, categoryCount, dataArea,
335                            edge);
336                    y1 = getCategoryEnd(categoryIndex, categoryCount, dataArea,
337                            edge);
338                    x1 = state.getCursor();
339                    x0 = x1 - maxdim;
340                }
341                else if (edge == RectangleEdge.RIGHT) {
342                    y0 = getCategoryStart(categoryIndex, categoryCount, dataArea,
343                            edge);
344                    y1 = getCategoryEnd(categoryIndex, categoryCount, dataArea,
345                            edge);
346                    x0 = state.getCursor();
347                    x1 = x0 + maxdim;
348                }
349                Rectangle2D area = new Rectangle2D.Double(x0, y0, (x1 - x0),
350                        (y1 - y0));
351                int subCategoryCount = this.subCategories.size();
352                float width = (float) ((x1 - x0) / subCategoryCount);
353                float height = (float) ((y1 - y0) / subCategoryCount);
354                float xx, yy;
355                for (int i = 0; i < subCategoryCount; i++) {
356                    if (RectangleEdge.isTopOrBottom(edge)) {
357                        xx = (float) (x0 + (i + 0.5) * width);
358                        yy = (float) area.getCenterY();
359                    }
360                    else {
361                        xx = (float) area.getCenterX();
362                        yy = (float) (y0 + (i + 0.5) * height);
363                    }
364                    String label = this.subCategories.get(i).toString();
365                    TextUtilities.drawRotatedString(label, g2, xx, yy,
366                            TextAnchor.CENTER, 0.0, TextAnchor.CENTER);
367                }
368            }
369    
370            if (edge.equals(RectangleEdge.TOP)) {
371                double h = maxdim;
372                state.cursorUp(h);
373            }
374            else if (edge.equals(RectangleEdge.BOTTOM)) {
375                double h = maxdim;
376                state.cursorDown(h);
377            }
378            else if (edge == RectangleEdge.LEFT) {
379                double w = maxdim;
380                state.cursorLeft(w);
381            }
382            else if (edge == RectangleEdge.RIGHT) {
383                double w = maxdim;
384                state.cursorRight(w);
385            }
386            return state;
387        }

389        /**
390         * Tests the axis for equality with an arbitrary object.
391         *
392         * @param obj  the object (<code>null</code> permitted).
393         *
394         * @return A boolean.
395         */
396        public boolean equals(Object obj) {
397            if (obj == this) {
398                return true;
399            }
400            if (obj instanceof SubCategoryAxis && super.equals(obj)) {
401                SubCategoryAxis axis = (SubCategoryAxis) obj;
402                if (!this.subCategories.equals(axis.subCategories)) {
403                    return false;
404                }
405                if (!this.subLabelFont.equals(axis.subLabelFont)) {
406                    return false;
407                }
408                if (!this.subLabelPaint.equals(axis.subLabelPaint)) {
409                    return false;
410                }
411                return true;
412            }
413            return false;
414        }
