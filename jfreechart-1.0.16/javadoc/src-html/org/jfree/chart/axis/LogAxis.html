290        /**
291         * Converts a Java2D coordinate to an axis value, assuming that the
292         * axis covers the specified <code>edge</code> of the <code>area</code>.
293         *
294         * @param java2DValue  the Java2D coordinate.
295         * @param area  the area.
296         * @param edge  the edge that the axis belongs to.
297         *
298         * @return A value along the axis scale.
299         */
300        public double java2DToValue(double java2DValue, Rectangle2D area,
301                RectangleEdge edge) {
302    
303            Range range = getRange();
304            double axisMin = calculateLog(range.getLowerBound());
305            double axisMax = calculateLog(range.getUpperBound());
306    
307            double min = 0.0;
308            double max = 0.0;
309            if (RectangleEdge.isTopOrBottom(edge)) {
310                min = area.getX();
311                max = area.getMaxX();
312            }
313            else if (RectangleEdge.isLeftOrRight(edge)) {
314                min = area.getMaxY();
315                max = area.getY();
316            }
317            double log = 0.0;
318            if (isInverted()) {
319                log = axisMax - (java2DValue - min) / (max - min)
320                        * (axisMax - axisMin);
321            }
322            else {
323                log = axisMin + (java2DValue - min) / (max - min)
324                        * (axisMax - axisMin);
325            }
326            return calculateValue(log);
327        }

329        /**
330         * Converts a value on the axis scale to a Java2D coordinate relative to
331         * the given <code>area</code>, based on the axis running along the
332         * specified <code>edge</code>.
333         *
334         * @param value  the data value.
335         * @param area  the area.
336         * @param edge  the edge.
337         *
338         * @return The Java2D coordinate corresponding to <code>value</code>.
339         */
340        public double valueToJava2D(double value, Rectangle2D area,
341                RectangleEdge edge) {
342    
343            Range range = getRange();
344            double axisMin = calculateLog(range.getLowerBound());
345            double axisMax = calculateLog(range.getUpperBound());
346            value = calculateLog(value);
347    
348            double min = 0.0;
349            double max = 0.0;
350            if (RectangleEdge.isTopOrBottom(edge)) {
351                min = area.getX();
352                max = area.getMaxX();
353            }
354            else if (RectangleEdge.isLeftOrRight(edge)) {
355                max = area.getMinY();
356                min = area.getMaxY();
357            }
358            if (isInverted()) {
359                return max
360                       - ((value - axisMin) / (axisMax - axisMin)) * (max - min);
361            }
362            else {
363                return min
364                       + ((value - axisMin) / (axisMax - axisMin)) * (max - min);
365            }
366        }

378        /**
379         * Adjusts the axis range to match the data range that the axis is
380         * required to display.
381         */
382        protected void autoAdjustRange() {
383            Plot plot = getPlot();
384            if (plot == null) {
385                return;  // no plot, no data
386            }
387    
388            if (plot instanceof ValueAxisPlot) {
389                ValueAxisPlot vap = (ValueAxisPlot) plot;
390    
391                Range r = vap.getDataRange(this);
392                if (r == null) {
393                    r = getDefaultAutoRange();
394                }
395    
396                double upper = r.getUpperBound();
397                double lower = Math.max(r.getLowerBound(), this.smallestValue);
398                double range = upper - lower;
399    
400                // if fixed auto range, then derive lower bound...
401                double fixedAutoRange = getFixedAutoRange();
402                if (fixedAutoRange > 0.0) {
403                    lower = Math.max(upper - fixedAutoRange, this.smallestValue);
404                }
405                else {
406                    // ensure the autorange is at least <minRange> in size...
407                    double minRange = getAutoRangeMinimumSize();
408                    if (range < minRange) {
409                        double expand = (minRange - range) / 2;
410                        upper = upper + expand;
411                        lower = lower - expand;
412                    }
413    
414                    // apply the margins - these should apply to the exponent range
415                    double logUpper = calculateLog(upper);
416                    double logLower = calculateLog(lower);
417                    double logRange = logUpper - logLower;
418                    logUpper = logUpper + getUpperMargin() * logRange;
419                    logLower = logLower - getLowerMargin() * logRange;
420                    upper = calculateValue(logUpper);
421                    lower = calculateValue(logLower);
422                }
423    
424                setRange(new Range(lower, upper), false, false);
425            }
426    
427        }

429        /**
430         * Draws the axis on a Java 2D graphics device (such as the screen or a
431         * printer).
432         *
433         * @param g2  the graphics device (<code>null</code> not permitted).
434         * @param cursor  the cursor location (determines where to draw the axis).
435         * @param plotArea  the area within which the axes and plot should be drawn.
436         * @param dataArea  the area within which the data should be drawn.
437         * @param edge  the axis location (<code>null</code> not permitted).
438         * @param plotState  collects information about the plot
439         *                   (<code>null</code> permitted).
440         *
441         * @return The axis state (never <code>null</code>).
442         */
443        public AxisState draw(Graphics2D g2, double cursor, Rectangle2D plotArea,
444                Rectangle2D dataArea, RectangleEdge edge,
445                PlotRenderingInfo plotState) {
446    
447            AxisState state;
448            // if the axis is not visible, don't draw it...
449            if (!isVisible()) {
450                state = new AxisState(cursor);
451                // even though the axis is not visible, we need ticks for the
452                // gridlines...
453                List ticks = refreshTicks(g2, state, dataArea, edge);
454                state.setTicks(ticks);
455                return state;
456            }
457            state = drawTickMarksAndLabels(g2, cursor, plotArea, dataArea, edge);
458            if (getAttributedLabel() != null) {
459                state = drawAttributedLabel(getAttributedLabel(), g2, plotArea, 
460                        dataArea, edge, state);
461                
462            } else {
463                state = drawLabel(getLabel(), g2, plotArea, dataArea, edge, state);
464            }
465            createAndAddEntity(cursor, state, dataArea, edge, plotState);
466            return state;
467        }

469        /**
470         * Calculates the positions of the tick labels for the axis, storing the
471         * results in the tick label list (ready for drawing).
472         *
473         * @param g2  the graphics device.
474         * @param state  the axis state.
475         * @param dataArea  the area in which the plot should be drawn.
476         * @param edge  the location of the axis.
477         *
478         * @return A list of ticks.
479         *
480         */
481        public List refreshTicks(Graphics2D g2, AxisState state,
482                Rectangle2D dataArea, RectangleEdge edge) {
483            List result = new java.util.ArrayList();
484            if (RectangleEdge.isTopOrBottom(edge)) {
485                result = refreshTicksHorizontal(g2, dataArea, edge);
486            }
487            else if (RectangleEdge.isLeftOrRight(edge)) {
488                result = refreshTicksVertical(g2, dataArea, edge);
489            }
490            return result;
491        }

493        /**
494         * Returns a list of ticks for an axis at the top or bottom of the chart.
495         *
496         * @param g2  the graphics device.
497         * @param dataArea  the data area.
498         * @param edge  the edge.
499         *
500         * @return A list of ticks.
501         */
502        protected List refreshTicksHorizontal(Graphics2D g2, Rectangle2D dataArea,
503                RectangleEdge edge) {
504    
505            Range range = getRange();
506            List ticks = new ArrayList();
507            Font tickLabelFont = getTickLabelFont();
508            g2.setFont(tickLabelFont);
509            TextAnchor textAnchor;
510            if (edge == RectangleEdge.TOP) {
511                textAnchor = TextAnchor.BOTTOM_CENTER;
512            }
513            else {
514                textAnchor = TextAnchor.TOP_CENTER;
515            }
516    
517            if (isAutoTickUnitSelection()) {
518                selectAutoTickUnit(g2, dataArea, edge);
519            }
520            int minorTickCount = this.tickUnit.getMinorTickCount();
521            double start = Math.floor(calculateLog(getLowerBound()));
522            double end = Math.ceil(calculateLog(getUpperBound()));
523            double current = start;
524            boolean hasTicks = (this.tickUnit.getSize() > 0.0)
525                               && !Double.isInfinite(start);
526            while (hasTicks && current <= end) {
527                double v = calculateValue(current);
528                if (range.contains(v)) {
529                    ticks.add(new NumberTick(TickType.MAJOR, v, createTickLabel(v),
530                            textAnchor, TextAnchor.CENTER, 0.0));
531                }
532                // add minor ticks (for gridlines)
533                double next = Math.pow(this.base, current
534                        + this.tickUnit.getSize());
535                for (int i = 1; i < minorTickCount; i++) {
536                    double minorV = v + i * ((next - v) / minorTickCount);
537                    if (range.contains(minorV)) {
538                        ticks.add(new NumberTick(TickType.MINOR, minorV, "",
539                                textAnchor, TextAnchor.CENTER, 0.0));
540                    }
541                }
542                current = current + this.tickUnit.getSize();
543            }
544            return ticks;
545        }

616                selectHorizontalAutoTickUnit(g2, dataArea, edge);

619                selectVerticalAutoTickUnit(g2, dataArea, edge);
620            }

624        /**
625         * Selects an appropriate tick value for the axis.  The strategy is to
626         * display as many ticks as possible (selected from an array of 'standard'
627         * tick units) without the labels overlapping.
628         *
629         * @param g2  the graphics device.
630         * @param dataArea  the area defined by the axes.
631         * @param edge  the axis location.
632         *
633         * @since 1.0.7
634         */
635       protected void selectHorizontalAutoTickUnit(Graphics2D g2,
636               Rectangle2D dataArea, RectangleEdge edge) {
637    
638            double tickLabelWidth = estimateMaximumTickLabelWidth(g2,
639                    getTickUnit());
640    
641            // start with the current tick unit...
642            TickUnitSource tickUnits = getStandardTickUnits();
643            TickUnit unit1 = tickUnits.getCeilingTickUnit(getTickUnit());
644            double unit1Width = exponentLengthToJava2D(unit1.getSize(), dataArea,
645                    edge);
646    
647            // then extrapolate...
648            double guess = (tickLabelWidth / unit1Width) * unit1.getSize();
649    
650            NumberTickUnit unit2 = (NumberTickUnit)
651                    tickUnits.getCeilingTickUnit(guess);
652            double unit2Width = exponentLengthToJava2D(unit2.getSize(), dataArea,
653                    edge);
654    
655            tickLabelWidth = estimateMaximumTickLabelWidth(g2, unit2);
656            if (tickLabelWidth > unit2Width) {
657                unit2 = (NumberTickUnit) tickUnits.getLargerTickUnit(unit2);
658            }
659    
660            setTickUnit(unit2, false, false);
661    
662        }

664        /**
665         * Converts a length in data coordinates into the corresponding length in
666         * Java2D coordinates.
667         *
668         * @param length  the length.
669         * @param area  the plot area.
670         * @param edge  the edge along which the axis lies.
671         *
672         * @return The length in Java2D coordinates.
673         *
674         * @since 1.0.7
675         */
676        public double exponentLengthToJava2D(double length, Rectangle2D area,
677                                    RectangleEdge edge) {
678            double one = valueToJava2D(calculateValue(1.0), area, edge);
679            double l = valueToJava2D(calculateValue(length + 1.0), area, edge);
680            return Math.abs(l - one);
681        }

692         * @since 1.0.7
693         */
694        protected void selectVerticalAutoTickUnit(Graphics2D g2, 
695                Rectangle2D dataArea, RectangleEdge edge) {
696    
697            double tickLabelHeight = estimateMaximumTickLabelHeight(g2);
698    
699            // start with the current tick unit...
700            TickUnitSource tickUnits = getStandardTickUnits();
701            TickUnit unit1 = tickUnits.getCeilingTickUnit(getTickUnit());
702            double unitHeight = exponentLengthToJava2D(unit1.getSize(), dataArea,
703                    edge);
704    
705            // then extrapolate...
706            double guess = (tickLabelHeight / unitHeight) * unit1.getSize();
707    
708            NumberTickUnit unit2 = (NumberTickUnit)
709                    tickUnits.getCeilingTickUnit(guess);
710            double unit2Height = exponentLengthToJava2D(unit2.getSize(), dataArea,
711                    edge);
712    
713            tickLabelHeight = estimateMaximumTickLabelHeight(g2);
714            if (tickLabelHeight > unit2Height) {
715                unit2 = (NumberTickUnit) tickUnits.getLargerTickUnit(unit2);
716            }
717    
718            setTickUnit(unit2, false, false);
719    
720        }

723         * Estimates the maximum tick label height.
724         *
725         * @param g2  the graphics device.
726         *
727         * @return The maximum height.
728         *
729         * @since 1.0.7

731        protected double estimateMaximumTickLabelHeight(Graphics2D g2) {
732    
733            RectangleInsets tickLabelInsets = getTickLabelInsets();
734            double result = tickLabelInsets.getTop() + tickLabelInsets.getBottom();
735    
736            Font tickLabelFont = getTickLabelFont();

738            result += tickLabelFont.getLineMetrics("123", frc).getHeight();
739            return result;

743        /**
744         * Estimates the maximum width of the tick labels, assuming the specified
745         * tick unit is used.
746         * <P>
747         * Rather than computing the string bounds of every tick on the axis, we
748         * just look at two values: the lower bound and the upper bound for the
749         * axis.  These two values will usually be representative.
750         *
751         * @param g2  the graphics device.
752         * @param unit  the tick unit to use for calculation.
753         *
754         * @return The estimated maximum width of the tick labels.
755         *
756         * @since 1.0.7
757         */
758        protected double estimateMaximumTickLabelWidth(Graphics2D g2, 
759                TickUnit unit) {
760    
761            RectangleInsets tickLabelInsets = getTickLabelInsets();
762            double result = tickLabelInsets.getLeft() + tickLabelInsets.getRight();
763    
764            if (isVerticalTickLabels()) {
765                // all tick labels have the same width (equal to the height of the
766                // font)...
767                FontRenderContext frc = g2.getFontRenderContext();
768                LineMetrics lm = getTickLabelFont().getLineMetrics("0", frc);
769                result += lm.getHeight();
770            }
771            else {
772                // look at lower and upper bounds...
773                FontMetrics fm = g2.getFontMetrics(getTickLabelFont());
774                Range range = getRange();
775                double lower = range.getLowerBound();
776                double upper = range.getUpperBound();
777                String lowerStr, upperStr;
778                NumberFormat formatter = getNumberFormatOverride();
779                if (formatter != null) {
780                    lowerStr = formatter.format(lower);
781                    upperStr = formatter.format(upper);
782                }
783                else {
784                    lowerStr = unit.valueToString(lower);
785                    upperStr = unit.valueToString(upper);
786                }
787                double w1 = fm.stringWidth(lowerStr);
788                double w2 = fm.stringWidth(upperStr);
789                result += Math.max(w1, w2);
790            }
791    
792            return result;
793    
794        }

796        /**
797         * Zooms in on the current range.
798         *
799         * @param lowerPercent  the new lower bound.
800         * @param upperPercent  the new upper bound.
801         */
802        public void zoomRange(double lowerPercent, double upperPercent) {
803            Range range = getRange();
804            double start = range.getLowerBound();
805            double end = range.getUpperBound();
806            double log1 = calculateLog(start);
807            double log2 = calculateLog(end);
808            double length = log2 - log1;
809            Range adjusted;
810            if (isInverted()) {
811                double logA = log1 + length * (1 - upperPercent);
812                double logB = log1 + length * (1 - lowerPercent);
813                adjusted = new Range(calculateValue(logA), calculateValue(logB));
814            }
815            else {
816                double logA = log1 + length * lowerPercent;
817                double logB = log1 + length * upperPercent;
818                adjusted = new Range(calculateValue(logA), calculateValue(logB));
819            }
820            setRange(adjusted);
821        }

823        /**
824         * Slides the axis range by the specified percentage.
825         *
826         * @param percent  the percentage.
827         *
828         * @since 1.0.13
829         */
830        public void pan(double percent) {
831            Range range = getRange();
832            double lower = range.getLowerBound();
833            double upper = range.getUpperBound();
834            double log1 = calculateLog(lower);
835            double log2 = calculateLog(upper);
836            double length = log2 - log1;
837            double adj = length * percent;
838            log1 = log1 + adj;
839            log2 = log2 + adj;
840            setRange(calculateValue(log1), calculateValue(log2));
841        }

863         * Tests this axis for equality with an arbitrary object.
864         *
865         * @param obj  the object (<code>null</code> permitted).
866         *
867         * @return A boolean.
868         */
869        public boolean equals(Object obj) {
870            if (obj == this) {
871                return true;
872            }
873            if (!(obj instanceof LogAxis)) {
874                return false;
875            }
876            LogAxis that = (LogAxis) obj;
877            if (this.base != that.base) {
878                return false;
879            }
880            if (this.smallestValue != that.smallestValue) {
881                return false;
882            }
883            return super.equals(obj);
884        }

886        /**
887         * Returns a hash code for this instance.
888         *
889         * @return A hash code.
890         */
891        public int hashCode() {
892            int result = 193;
893            long temp = Double.doubleToLongBits(this.base);
894            result = 37 * result + (int) (temp ^ (temp >>> 32));
895            temp = Double.doubleToLongBits(this.smallestValue);
896            result = 37 * result + (int) (temp ^ (temp >>> 32));
897            if (this.numberFormatOverride != null) {
898                result = 37 * result + this.numberFormatOverride.hashCode();
899            }
900            result = 37 * result + this.tickUnit.hashCode();
901            return result;
902        }
