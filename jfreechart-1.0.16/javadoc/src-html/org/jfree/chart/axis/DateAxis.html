168    /**
169     * The base class for axes that display dates.  You will find it easier to
170     * understand how this axis works if you bear in mind that it really
171     * displays/measures integer (or long) data, where the integers are
172     * milliseconds since midnight, 1-Jan-1970.  When displaying tick labels, the
173     * millisecond values are converted back to dates using a
174     * <code>DateFormat</code> instance.
175     * <P>
176     * You can also create a {@link org.jfree.chart.axis.Timeline} and supply in
177     * the constructor to create an axis that only contains certain domain values.
178     * For example, this allows you to create a date axis that only contains
179     * working days.
180     */

621        /**
622         * Sets the minimum date visible on the axis and sends an
623         * {@link AxisChangeEvent} to all registered listeners.  If
624         * <code>date</code> is on or after the current maximum date for
625         * the axis, the maximum date will be shifted to preserve the current
626         * length of the axis.
627         *
628         * @param date  the date (<code>null</code> not permitted).
629         *
630         * @see #getMinimumDate()
631         * @see #setMaximumDate(Date)
632         */
633        public void setMinimumDate(Date date) {
634            ParamChecks.nullNotPermitted(date, "date");
635            // check the new minimum date relative to the current maximum date
636            Date maxDate = getMaximumDate();
637            long maxMillis = maxDate.getTime();
638            long newMinMillis = date.getTime();
639            if (maxMillis <= newMinMillis) {
640                Date oldMin = getMinimumDate();
641                long length = maxMillis - oldMin.getTime();
642                maxDate = new Date(newMinMillis + length);
643            }
644            setRange(new DateRange(date, maxDate), true, false);
645            fireChangeEvent();
646        }

741         * Translates the data value to the display coordinates (Java 2D User Space)
742         * of the chart.
743         *
744         * @param value  the date to be plotted.
745         * @param area  the rectangle (in Java2D space) where the data is to be
746         *              plotted.
747         * @param edge  the axis location.
748         *
749         * @return The coordinate corresponding to the supplied data value.
750         */
751        @Override
752        public double valueToJava2D(double value, Rectangle2D area,
753                RectangleEdge edge) {
754    
755            value = this.timeline.toTimelineValue((long) value);
756    
757            DateRange range = (DateRange) getRange();
758            double axisMin = this.timeline.toTimelineValue(range.getLowerMillis());
759            double axisMax = this.timeline.toTimelineValue(range.getUpperMillis());
760            double result = 0.0;
761            if (RectangleEdge.isTopOrBottom(edge)) {
762                double minX = area.getX();
763                double maxX = area.getMaxX();
764                if (isInverted()) {
765                    result = maxX + ((value - axisMin) / (axisMax - axisMin))
766                             * (minX - maxX);
767                }
768                else {
769                    result = minX + ((value - axisMin) / (axisMax - axisMin))
770                             * (maxX - minX);
771                }
772            }
773            else if (RectangleEdge.isLeftOrRight(edge)) {
774                double minY = area.getMinY();
775                double maxY = area.getMaxY();
776                if (isInverted()) {
777                    result = minY + (((value - axisMin) / (axisMax - axisMin))
778                             * (maxY - minY));
779                }
780                else {
781                    result = maxY - (((value - axisMin) / (axisMax - axisMin))
782                             * (maxY - minY));
783                }
784            }
785            return result;

824

872         * Returns the previous "standard" date, for a given date and tick unit.
873         *
874         * @param date  the reference date.
875         * @param unit  the tick unit.
876         *
877         * @return The previous "standard" date.
878         */
879        protected Date previousStandardDate(Date date, DateTickUnit unit) {
880    
881            int milliseconds;
882            int seconds;
883            int minutes;
884            int hours;
885            int days;
886            int months;
887            int years;
888    
889            Calendar calendar = Calendar.getInstance(this.timeZone, this.locale);
890            calendar.setTime(date);
891            int count = unit.getCount();
892            int current = calendar.get(unit.getCalendarField());
893            int value = count * (current / count);
894    
895            switch (unit.getUnit()) {
896    
897                case DateTickUnit.MILLISECOND :
898                    years = calendar.get(Calendar.YEAR);
899                    months = calendar.get(Calendar.MONTH);
900                    days = calendar.get(Calendar.DATE);
901                    hours = calendar.get(Calendar.HOUR_OF_DAY);
902                    minutes = calendar.get(Calendar.MINUTE);
903                    seconds = calendar.get(Calendar.SECOND);
904                    calendar.set(years, months, days, hours, minutes, seconds);
905                    calendar.set(Calendar.MILLISECOND, value);
906                    Date mm = calendar.getTime();
907                    if (mm.getTime() >= date.getTime()) {
908                        calendar.set(Calendar.MILLISECOND, value - 1);
909                        mm = calendar.getTime();
910                    }
911                    return mm;
912    
913                case DateTickUnit.SECOND :
914                    years = calendar.get(Calendar.YEAR);
915                    months = calendar.get(Calendar.MONTH);
916                    days = calendar.get(Calendar.DATE);
917                    hours = calendar.get(Calendar.HOUR_OF_DAY);
918                    minutes = calendar.get(Calendar.MINUTE);
919                    if (this.tickMarkPosition == DateTickMarkPosition.START) {
920                        milliseconds = 0;
921                    }
922                    else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
923                        milliseconds = 500;
924                    }
925                    else {
926                        milliseconds = 999;
927                    }
928                    calendar.set(Calendar.MILLISECOND, milliseconds);
929                    calendar.set(years, months, days, hours, minutes, value);
930                    Date dd = calendar.getTime();
931                    if (dd.getTime() >= date.getTime()) {
932                        calendar.set(Calendar.SECOND, value - 1);
933                        dd = calendar.getTime();
934                    }
935                    return dd;
936    
937                case DateTickUnit.MINUTE :
938                    years = calendar.get(Calendar.YEAR);
939                    months = calendar.get(Calendar.MONTH);
940                    days = calendar.get(Calendar.DATE);
941                    hours = calendar.get(Calendar.HOUR_OF_DAY);
942                    if (this.tickMarkPosition == DateTickMarkPosition.START) {
943                        seconds = 0;
944                    }
945                    else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
946                        seconds = 30;
947                    }
948                    else {
949                        seconds = 59;
950                    }
951                    calendar.clear(Calendar.MILLISECOND);
952                    calendar.set(years, months, days, hours, value, seconds);
953                    Date d0 = calendar.getTime();
954                    if (d0.getTime() >= date.getTime()) {
955                        calendar.set(Calendar.MINUTE, value - 1);
956                        d0 = calendar.getTime();
957                    }
958                    return d0;
959    
960                case DateTickUnit.HOUR :
961                    years = calendar.get(Calendar.YEAR);
962                    months = calendar.get(Calendar.MONTH);
963                    days = calendar.get(Calendar.DATE);
964                    if (this.tickMarkPosition == DateTickMarkPosition.START) {
965                        minutes = 0;
966                        seconds = 0;
967                    }
968                    else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
969                        minutes = 30;
970                        seconds = 0;
971                    }
972                    else {
973                        minutes = 59;
974                        seconds = 59;
975                    }
976                    calendar.clear(Calendar.MILLISECOND);
977                    calendar.set(years, months, days, value, minutes, seconds);
978                    Date d1 = calendar.getTime();
979                    if (d1.getTime() >= date.getTime()) {
980                        calendar.set(Calendar.HOUR_OF_DAY, value - 1);
981                        d1 = calendar.getTime();
982                    }
983                    return d1;
984    
985                case DateTickUnit.DAY :
986                    years = calendar.get(Calendar.YEAR);
987                    months = calendar.get(Calendar.MONTH);
988                    if (this.tickMarkPosition == DateTickMarkPosition.START) {
989                        hours = 0;
990                        minutes = 0;
991                        seconds = 0;
992                    }
993                    else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
994                        hours = 12;
995                        minutes = 0;
996                        seconds = 0;
997                    }
998                    else {
999                        hours = 23;
1000                        minutes = 59;
1001                        seconds = 59;
1002                    }
1003                    calendar.clear(Calendar.MILLISECOND);
1004                    calendar.set(years, months, value, hours, 0, 0);
1005                    // long result = calendar.getTimeInMillis();
1006                        // won't work with JDK 1.3
1007                    Date d2 = calendar.getTime();
1008                    if (d2.getTime() >= date.getTime()) {
1009                        calendar.set(Calendar.DATE, value - 1);
1010                        d2 = calendar.getTime();
1011                    }
1012                    return d2;
1013    
1014                case DateTickUnit.MONTH :
1015                    years = calendar.get(Calendar.YEAR);
1016                    calendar.clear(Calendar.MILLISECOND);
1017                    calendar.set(years, value, 1, 0, 0, 0);
1018                    Month month = new Month(calendar.getTime(), this.timeZone,
1019                            this.locale);
1020                    Date standardDate = calculateDateForPosition(
1021                            month, this.tickMarkPosition);
1022                    long millis = standardDate.getTime();
1023                    if (millis >= date.getTime()) {
1024                        month = (Month) month.previous();
1025                        // need to peg the month in case the time zone isn't the
1026                        // default - see bug 2078057
1027                        month.peg(Calendar.getInstance(this.timeZone));
1028                        standardDate = calculateDateForPosition(
1029                                month, this.tickMarkPosition);
1030                    }
1031                    return standardDate;
1032    
1033                case DateTickUnit.YEAR :
1034                    if (this.tickMarkPosition == DateTickMarkPosition.START) {
1035                        months = 0;
1036                        days = 1;
1037                    }
1038                    else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
1039                        months = 6;
1040                        days = 1;
1041                    }
1042                    else {
1043                        months = 11;
1044                        days = 31;
1045                    }
1046                    calendar.clear(Calendar.MILLISECOND);
1047                    calendar.set(value, months, days, 0, 0, 0);
1048                    Date d3 = calendar.getTime();
1049                    if (d3.getTime() >= date.getTime()) {
1050                        calendar.set(Calendar.YEAR, value - 1);
1051                        d3 = calendar.getTime();
1052                    }
1053                    return d3;
1054    
1055                default: return null;
1056    
1057            }
1058    
1059        }

1253        /**
1254         * Rescales the axis to ensure that all data is visible.
1255         */
1256        @Override
1257        protected void autoAdjustRange() {
1258    
1259            Plot plot = getPlot();
1260    
1261            if (plot == null) {
1262                return;  // no plot, no data
1263            }
1264    
1265            if (plot instanceof ValueAxisPlot) {
1266                ValueAxisPlot vap = (ValueAxisPlot) plot;
1267    
1268                Range r = vap.getDataRange(this);
1269                if (r == null) {
1270                    if (this.timeline instanceof SegmentedTimeline) {
1271                        //Timeline hasn't method getStartTime()
1272                        r = new DateRange((
1273                                (SegmentedTimeline) this.timeline).getStartTime(),
1274                                ((SegmentedTimeline) this.timeline).getStartTime()
1275                                + 1);
1276                    }
1277                    else {
1278                        r = new DateRange();
1279                    }
1280                }
1281    
1282                long upper = this.timeline.toTimelineValue(
1283                        (long) r.getUpperBound());
1284                long lower;
1285                long fixedAutoRange = (long) getFixedAutoRange();
1286                if (fixedAutoRange > 0.0) {
1287                    lower = upper - fixedAutoRange;
1288                }
1289                else {
1290                    lower = this.timeline.toTimelineValue((long) r.getLowerBound());
1291                    double range = upper - lower;
1292                    long minRange = (long) getAutoRangeMinimumSize();
1293                    if (range < minRange) {
1294                        long expand = (long) (minRange - range) / 2;
1295                        upper = upper + expand;
1296                        lower = lower - expand;
1297                    }
1298                    upper = upper + (long) (range * getUpperMargin());
1299                    lower = lower - (long) (range * getLowerMargin());

1859         * reaches the equivalent of one millisecond).

1862         * @param upperPercent  the new upper bound.

1864        @Override
1865        public void zoomRange(double lowerPercent, double upperPercent) {
1866            double start = this.timeline.toTimelineValue(
1867                    (long) getRange().getLowerBound());
1868            double end = this.timeline.toTimelineValue(
1869                    (long) getRange().getUpperBound());
1870            double length = end - start;
1871            Range adjusted;
1872            long adjStart, adjEnd;
1873            if (isInverted()) {
1874                adjStart = (long) (start + (length * (1 - upperPercent)));
1875                adjEnd = (long) (start + (length * (1 - lowerPercent)));
1876            }
1877            else {
1878                adjStart = (long) (start + length * lowerPercent);
1879                adjEnd = (long) (start + length * upperPercent);
1880            }
1881            // when zooming to sub-millisecond ranges, it can be the case that
1882            // adjEnd == adjStart...and we can't have an axis with zero length
1883            // so we apply this instead:
1884            if (adjEnd <= adjStart) {
1885                adjEnd = adjStart + 1L;
1886            } 
1887            adjusted = new DateRange(this.timeline.toMillisecond(adjStart),
1888                   this.timeline.toMillisecond(adjEnd));

1946            // 'dateTickUnit' is immutable : no need to clone
1947            if (this.dateFormatOverride != null) {
1948                clone.dateFormatOverride

1950            }
1951            // 'tickMarkPosition' is immutable : no need to clone
1952            return clone;
1953        }
