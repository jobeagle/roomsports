079    /**
080    This class extends NumberAxis and handles cycling.
081    
082    Traditional representation of data in the range x0..x1
083    <pre>
084    |-------------------------|
085    x0                       x1
086    </pre>
087    
088    Here, the range bounds are at the axis extremities.
089    With cyclic axis, however, the time is split in
090    "cycles", or "time frames", or the same duration : the period.
091    
092    A cycle axis cannot by definition handle a larger interval
093    than the period : <pre>x1 - x0 >= period</pre>. Thus, at most a full
094    period can be represented with such an axis.
095    
096    The cycle bound is the number between x0 and x1 which marks
097    the beginning of new time frame:
098    <pre>
099    |---------------------|----------------------------|
100    x0                   cb                           x1
101    <---previous cycle---><-------current cycle-------->
102    </pre>
103    
104    It is actually a multiple of the period, plus optionally
105    a start offset: <pre>cb = n * period + offset</pre>
106    
107    Thus, by definition, two consecutive cycle bounds
108    period apart, which is precisely why it is called a
109    period.
110    
111    The visual representation of a cyclic axis is like that:
112    <pre>
113    |----------------------------|---------------------|
114    cb                         x1|x0                  cb
115    <-------current cycle--------><---previous cycle--->
116    </pre>
117    
118    The cycle bound is at the axis ends, then current
119    cycle is shown, then the last cycle. When using
120    dynamic data, the visual effect is the current cycle
121    erases the last cycle as x grows. Then, the next cycle
122    bound is reached, and the process starts over, erasing
123    the previous cycle.
124    
125    A Cyclic item renderer is provided to do exactly this.
126    
127     */

348        /**
349         * A special Number tick that also hold information about the cycle bound
350         * mapping for this tick.  This is especially useful for having a tick at
351         * each axis end with the cycle bound value.  See also
352         * isBoundMappedToLastCycle()

354        protected static class CycleBoundTick extends NumberTick {
355    
356            /** Map to last cycle. */
357            public boolean mapToLastCycle;
358    
359            /**
360             * Creates a new tick.
361             *
362             * @param mapToLastCycle  map to last cycle?
363             * @param number  the number.
364             * @param label  the label.
365             * @param textAnchor  the text anchor.
366             * @param rotationAnchor  the rotation anchor.
367             * @param angle  the rotation angle.
368             */
369            public CycleBoundTick(boolean mapToLastCycle, Number number,
370                                  String label, TextAnchor textAnchor,
371                                  TextAnchor rotationAnchor, double angle) {
372                super(number, label, textAnchor, rotationAnchor, angle);
373                this.mapToLastCycle = mapToLastCycle;
374            }
375        }

377        /**
378         * Calculates the anchor point for a tick.
379         *
380         * @param tick  the tick.
381         * @param cursor  the cursor.
382         * @param dataArea  the data area.
383         * @param edge  the side on which the axis is displayed.
384         *
385         * @return The anchor point.
386         */
387        protected float[] calculateAnchorPoint(ValueTick tick, double cursor,
388                                               Rectangle2D dataArea,
389                                               RectangleEdge edge) {
390            if (tick instanceof CycleBoundTick) {
391                boolean mapsav = this.boundMappedToLastCycle;
392                this.boundMappedToLastCycle
393                    = ((CycleBoundTick) tick).mapToLastCycle;
394                float[] ret = super.calculateAnchorPoint(
395                    tick, cursor, dataArea, edge
396                );
397                this.boundMappedToLastCycle = mapsav;
398                return ret;
399            }
400            return super.calculateAnchorPoint(tick, cursor, dataArea, edge);
401        }

405        /**
406         * Builds a list of ticks for the axis.  This method is called when the
407         * axis is at the top or bottom of the chart (so the axis is "horizontal").
408         *
409         * @param g2  the graphics device.
410         * @param dataArea  the data area.
411         * @param edge  the edge.
412         *
413         * @return A list of ticks.
414         */
415        protected List refreshTicksHorizontal(Graphics2D g2, Rectangle2D dataArea,
416                RectangleEdge edge) {
417    
418            List result = new java.util.ArrayList();
419    
420            Font tickLabelFont = getTickLabelFont();
421            g2.setFont(tickLabelFont);
422    
423            if (isAutoTickUnitSelection()) {
424                selectAutoTickUnit(g2, dataArea, edge);
425            }
426    
427            double unit = getTickUnit().getSize();
428            double cycleBound = getCycleBound();
429            double currentTickValue = Math.ceil(cycleBound / unit) * unit;
430            double upperValue = getRange().getUpperBound();
431            boolean cycled = false;
432    
433            boolean boundMapping = this.boundMappedToLastCycle;
434            this.boundMappedToLastCycle = false;
435    
436            CycleBoundTick lastTick = null;
437            float lastX = 0.0f;
438    
439            if (upperValue == cycleBound) {
440                currentTickValue = calculateLowestVisibleTickValue();
441                cycled = true;
442                this.boundMappedToLastCycle = true;
443            }
444    
445            while (currentTickValue <= upperValue) {
446    
447                // Cycle when necessary
448                boolean cyclenow = false;
449                if ((currentTickValue + unit > upperValue) && !cycled) {
450                    cyclenow = true;
451                }
452    
453                double xx = valueToJava2D(currentTickValue, dataArea, edge);
454                String tickLabel;
455                NumberFormat formatter = getNumberFormatOverride();
456                if (formatter != null) {
457                    tickLabel = formatter.format(currentTickValue);
458                }
459                else {
460                    tickLabel = getTickUnit().valueToString(currentTickValue);
461                }
462                float x = (float) xx;
463                TextAnchor anchor;
464                TextAnchor rotationAnchor;
465                double angle = 0.0;
466                if (isVerticalTickLabels()) {
467                    if (edge == RectangleEdge.TOP) {
468                        angle = Math.PI / 2.0;
469                    }
470                    else {
471                        angle = -Math.PI / 2.0;
472                    }
473                    anchor = TextAnchor.CENTER_RIGHT;
474                    // If tick overlap when cycling, update last tick too
475                    if ((lastTick != null) && (lastX == x)
476                            && (currentTickValue != cycleBound)) {
477                        anchor = isInverted()
478                            ? TextAnchor.TOP_RIGHT : TextAnchor.BOTTOM_RIGHT;
479                        result.remove(result.size() - 1);
480                        result.add(new CycleBoundTick(
481                            this.boundMappedToLastCycle, lastTick.getNumber(),
482                            lastTick.getText(), anchor, anchor,
483                            lastTick.getAngle())
484                        );
485                        this.internalMarkerWhenTicksOverlap = true;
486                        anchor = isInverted()
487                            ? TextAnchor.BOTTOM_RIGHT : TextAnchor.TOP_RIGHT;
488                    }
489                    rotationAnchor = anchor;
490                }
491                else {
492                    if (edge == RectangleEdge.TOP) {
493                        anchor = TextAnchor.BOTTOM_CENTER;
494                        if ((lastTick != null) && (lastX == x)
495                                && (currentTickValue != cycleBound)) {
496                            anchor = isInverted()
497                                ? TextAnchor.BOTTOM_LEFT : TextAnchor.BOTTOM_RIGHT;
498                            result.remove(result.size() - 1);
499                            result.add(new CycleBoundTick(
500                                this.boundMappedToLastCycle, lastTick.getNumber(),
501                                lastTick.getText(), anchor, anchor,
502                                lastTick.getAngle())
503                            );
504                            this.internalMarkerWhenTicksOverlap = true;
505                            anchor = isInverted()
506                                ? TextAnchor.BOTTOM_RIGHT : TextAnchor.BOTTOM_LEFT;
507                        }
508                        rotationAnchor = anchor;
509                    }
510                    else {
511                        anchor = TextAnchor.TOP_CENTER;
512                        if ((lastTick != null) && (lastX == x)
513                                && (currentTickValue != cycleBound)) {
514                            anchor = isInverted()
515                                ? TextAnchor.TOP_LEFT : TextAnchor.TOP_RIGHT;
516                            result.remove(result.size() - 1);
517                            result.add(new CycleBoundTick(
518                                this.boundMappedToLastCycle, lastTick.getNumber(),
519                                lastTick.getText(), anchor, anchor,
520                                lastTick.getAngle())
521                            );
522                            this.internalMarkerWhenTicksOverlap = true;
523                            anchor = isInverted()
524                                ? TextAnchor.TOP_RIGHT : TextAnchor.TOP_LEFT;
525                        }
526                        rotationAnchor = anchor;
527                    }
528                }
529    
530                CycleBoundTick tick = new CycleBoundTick(
531                    this.boundMappedToLastCycle,
532                    new Double(currentTickValue), tickLabel, anchor,
533                    rotationAnchor, angle
534                );
535                if (currentTickValue == cycleBound) {
536                    this.internalMarkerCycleBoundTick = tick;
537                }
538                result.add(tick);
539                lastTick = tick;
540                lastX = x;
541    
542                currentTickValue += unit;
543    
544                if (cyclenow) {
545                    currentTickValue = calculateLowestVisibleTickValue();
546                    upperValue = cycleBound;
547                    cycled = true;
548                    this.boundMappedToLastCycle = true;
549                }
550    
551            }
552            this.boundMappedToLastCycle = boundMapping;
553            return result;
554    
555        }

642                        if ((lastTick != null) && (lastY == y)
643                                && (currentTickValue != cycleBound)) {
644                            anchor = isInverted()
645                                ? TextAnchor.BOTTOM_RIGHT : TextAnchor.BOTTOM_LEFT;
646                            result.remove(result.size() - 1);
647                            result.add(new CycleBoundTick(
648                                this.boundMappedToLastCycle, lastTick.getNumber(),
649                                lastTick.getText(), anchor, anchor,
650                                lastTick.getAngle())
651                            );
652                            this.internalMarkerWhenTicksOverlap = true;
653                            anchor = isInverted()
654                                ? TextAnchor.BOTTOM_LEFT : TextAnchor.BOTTOM_RIGHT;
655                        }
656                        rotationAnchor = anchor;
657                        angle = Math.PI / 2.0;
658                    }
659                }

664                                && (currentTickValue != cycleBound)) {

666                                ? TextAnchor.BOTTOM_RIGHT : TextAnchor.TOP_RIGHT;
667                            result.remove(result.size() - 1);
668                            result.add(new CycleBoundTick(
669                                this.boundMappedToLastCycle, lastTick.getNumber(),
670                                lastTick.getText(), anchor, anchor,
671                                lastTick.getAngle())

675                                ? TextAnchor.TOP_RIGHT : TextAnchor.BOTTOM_RIGHT;
676                        }
677                        rotationAnchor = anchor;
678                    }
679                    else {
680                        anchor = TextAnchor.CENTER_LEFT;
681                        if ((lastTick != null) && (lastY == y)
682                                && (currentTickValue != cycleBound)) {
683                            anchor = isInverted()
684                                ? TextAnchor.BOTTOM_LEFT : TextAnchor.TOP_LEFT;
685                            result.remove(result.size() - 1);
686                            result.add(new CycleBoundTick(
687                                this.boundMappedToLastCycle, lastTick.getNumber(),
688                                lastTick.getText(), anchor, anchor,
689                                lastTick.getAngle())
690                            );
691                            this.internalMarkerWhenTicksOverlap = true;
692                            anchor = isInverted()
693                                ? TextAnchor.TOP_LEFT : TextAnchor.BOTTOM_LEFT;
694                        }

696                    }

702                if (currentTickValue == cycleBound) {
703                    this.internalMarkerCycleBoundTick = tick;
704                }
705                result.add(tick);
706                lastTick = tick;
707                lastY = y;
708    
709                if (currentTickValue == cycleBound) {
710                    this.internalMarkerCycleBoundTick = tick;
711                }
712    
713                currentTickValue += unit;
714    
715                if (cyclenow) {
716                    currentTickValue = calculateLowestVisibleTickValue();
717                    upperValue = cycleBound;
718                    cycled = true;
719                    this.boundMappedToLastCycle = false;
720                }
721    
722            }
723            this.boundMappedToLastCycle = boundMapping;
724            return result;
725        }

727        /**
728         * Converts a coordinate from Java 2D space to data space.
729         *
730         * @param java2DValue  the coordinate in Java2D space.
731         * @param dataArea  the data area.
732         * @param edge  the edge.
733         *
734         * @return The data value.
735         */
736        public double java2DToValue(double java2DValue, Rectangle2D dataArea,
737                                    RectangleEdge edge) {
738            Range range = getRange();
739    
740            double vmax = range.getUpperBound();
741            double vp = getCycleBound();
742    
743            double jmin = 0.0;
744            double jmax = 0.0;
745            if (RectangleEdge.isTopOrBottom(edge)) {
746                jmin = dataArea.getMinX();
747                jmax = dataArea.getMaxX();
748            }
749            else if (RectangleEdge.isLeftOrRight(edge)) {
750                jmin = dataArea.getMaxY();
751                jmax = dataArea.getMinY();
752            }
753    
754            if (isInverted()) {
755                double jbreak = jmax - (vmax - vp) * (jmax - jmin) / this.period;
756                if (java2DValue >= jbreak) {
757                    return vp + (jmax - java2DValue) * this.period / (jmax - jmin);
758                }
759                else {
760                    return vp - (java2DValue - jmin) * this.period / (jmax - jmin);
761                }
762            }
763            else {
764                double jbreak = (vmax - vp) * (jmax - jmin) / this.period + jmin;
765                if (java2DValue <= jbreak) {
766                    return vp + (java2DValue - jmin) * this.period / (jmax - jmin);
767                }
768                else {
769                    return vp - (jmax - java2DValue) * this.period / (jmax - jmin);
770                }
771            }
772        }

774        /**
775         * Translates a value from data space to Java 2D space.
776         *
777         * @param value  the data value.
778         * @param dataArea  the data area.
779         * @param edge  the edge.
780         *
781         * @return The Java 2D value.
782         */
783        public double valueToJava2D(double value, Rectangle2D dataArea,
784                                    RectangleEdge edge) {
785            Range range = getRange();
786    
787            double vmin = range.getLowerBound();
788            double vmax = range.getUpperBound();
789            double vp = getCycleBound();
790    
791            if ((value < vmin) || (value > vmax)) {
792                return Double.NaN;
793            }
794    
795    
796            double jmin = 0.0;
797            double jmax = 0.0;
798            if (RectangleEdge.isTopOrBottom(edge)) {
799                jmin = dataArea.getMinX();
800                jmax = dataArea.getMaxX();
801            }
802            else if (RectangleEdge.isLeftOrRight(edge)) {
803                jmax = dataArea.getMinY();
804                jmin = dataArea.getMaxY();
805            }
806    
807            if (isInverted()) {
808                if (value == vp) {
809                    return this.boundMappedToLastCycle ? jmin : jmax;
810                }
811                else if (value > vp) {
812                    return jmax - (value - vp) * (jmax - jmin) / this.period;
813                }
814                else {
815                    return jmin + (vp - value) * (jmax - jmin) / this.period;
816                }
817            }
818            else {
819                if (value == vp) {
820                    return this.boundMappedToLastCycle ? jmax : jmin;
821                }
822                else if (value >= vp) {
823                    return jmin + (value - vp) * (jmax - jmin) / this.period;
824                }
825                else {
826                    return jmax - (vp - value) * (jmax - jmin) / this.period;
827                }
828            }
829        }

890        /**
891         * The cycle bound is defined as the higest value x such that
892         * "offset + period * i = x", with i and integer and x &lt;
893         * range.getUpperBound() This is the value which is at both ends of the
894         * axis :  x...up|low...x
895         * The values from x to up are the valued in the current cycle.
896         * The values from low to x are the valued in the previous cycle.
897         *
898         * @return The cycle bound.
899         */
900        public double getCycleBound() {
901            return Math.floor(
902                (getRange().getUpperBound() - this.offset) / this.period
903            ) * this.period + this.offset;
904        }

1191                return false;
