225        /**
226         * Creates a new axis.
227         *
228         * @param label  the axis label (<code>null</code> permitted).
229         * @param first  the first time period in the axis range
230         *               (<code>null</code> not permitted).
231         * @param last  the last time period in the axis range
232         *              (<code>null</code> not permitted).
233         * @param timeZone  the time zone (<code>null</code> not permitted).
234         * @param locale  the locale (<code>null</code> not permitted).
235         *
236         * @since 1.0.13
237         */
238        public PeriodAxis(String label, RegularTimePeriod first,
239                RegularTimePeriod last, TimeZone timeZone, Locale locale) {
240            super(label, null);
241            ParamChecks.nullNotPermitted(timeZone, "timeZone");
242            ParamChecks.nullNotPermitted(locale, "locale");
243            this.first = first;
244            this.last = last;
245            this.timeZone = timeZone;
246            this.locale = locale;
247            this.calendar = Calendar.getInstance(timeZone, locale);
248            this.first.peg(this.calendar);
249            this.last.peg(this.calendar);
250            this.autoRangeTimePeriodClass = first.getClass();
251            this.majorTickTimePeriodClass = first.getClass();
252            this.minorTickMarksVisible = false;
253            this.minorTickTimePeriodClass = RegularTimePeriod.downsize(
254                    this.majorTickTimePeriodClass);
255            setAutoRange(true);
256            this.labelInfo = new PeriodAxisLabelInfo[2];
257            SimpleDateFormat df0 = new SimpleDateFormat("MMM", locale);
258            df0.setTimeZone(timeZone);
259            this.labelInfo[0] = new PeriodAxisLabelInfo(Month.class, df0);
260            SimpleDateFormat df1 = new SimpleDateFormat("yyyy", locale);
261            df1.setTimeZone(timeZone);
262            this.labelInfo[1] = new PeriodAxisLabelInfo(Year.class, df1);
263        }

540        /**
541         * Sets the range for the axis, if requested, sends an
542         * {@link AxisChangeEvent} to all registered listeners.  As a side-effect,
543         * the auto-range flag is set to <code>false</code> (optional).
544         *
545         * @param range  the range (<code>null</code> not permitted).
546         * @param turnOffAutoRange  a flag that controls whether or not the auto
547         *                          range is turned off.
548         * @param notify  a flag that controls whether or not listeners are
549         *                notified.
550         */
551        public void setRange(Range range, boolean turnOffAutoRange, 
552                boolean notify) {
553            long upper = Math.round(range.getUpperBound());
554            long lower = Math.round(range.getLowerBound());
555            this.first = createInstance(this.autoRangeTimePeriodClass,
556                    new Date(lower), this.timeZone, this.locale);
557            this.last = createInstance(this.autoRangeTimePeriodClass,
558                    new Date(upper), this.timeZone, this.locale);
559            super.setRange(new Range(this.first.getFirstMillisecond(),
560                    this.last.getLastMillisecond() + 1.0), turnOffAutoRange,
561                    notify);
562        }

574        /**
575         * Estimates the space (height or width) required to draw the axis.
576         *
577         * @param g2  the graphics device.
578         * @param plot  the plot that the axis belongs to.
579         * @param plotArea  the area within which the plot (including axes) should
580         *                  be drawn.
581         * @param edge  the axis location.
582         * @param space  space already reserved.
583         *
584         * @return The space required to draw the axis (including pre-reserved
585         *         space).
586         */
587        public AxisSpace reserveSpace(Graphics2D g2, Plot plot, 
588                Rectangle2D plotArea, RectangleEdge edge, AxisSpace space) {
589            // create a new space object if one wasn't supplied...
590            if (space == null) {
591                space = new AxisSpace();
592            }
593    
594            // if the axis is not visible, no additional space is required...
595            if (!isVisible()) {
596                return space;
597            }
598    
599            // if the axis has a fixed dimension, return it...
600            double dimension = getFixedDimension();
601            if (dimension > 0.0) {
602                space.ensureAtLeast(dimension, edge);
603            }
604    
605            // get the axis label size and update the space object...
606            Rectangle2D labelEnclosure = getLabelEnclosure(g2, edge);
607            double labelHeight, labelWidth;
608            double tickLabelBandsDimension = 0.0;
609    
610            for (int i = 0; i < this.labelInfo.length; i++) {
611                PeriodAxisLabelInfo info = this.labelInfo[i];
612                FontMetrics fm = g2.getFontMetrics(info.getLabelFont());
613                tickLabelBandsDimension
614                    += info.getPadding().extendHeight(fm.getHeight());
615            }
616    
617            if (RectangleEdge.isTopOrBottom(edge)) {
618                labelHeight = labelEnclosure.getHeight();
619                space.add(labelHeight + tickLabelBandsDimension, edge);
620            }
621            else if (RectangleEdge.isLeftOrRight(edge)) {
622                labelWidth = labelEnclosure.getWidth();
623                space.add(labelWidth + tickLabelBandsDimension, edge);
624            }
625    
626            // add space for the outer tick labels, if any...
627            double tickMarkSpace = 0.0;
628            if (isTickMarksVisible()) {
629                tickMarkSpace = getTickMarkOutsideLength();
630            }
631            if (this.minorTickMarksVisible) {
632                tickMarkSpace = Math.max(tickMarkSpace,
633                        this.minorTickMarkOutsideLength);
634            }
635            space.add(tickMarkSpace, edge);
636            return space;
637        }

639        /**
640         * Draws the axis on a Java 2D graphics device (such as the screen or a
641         * printer).
642         *
643         * @param g2  the graphics device (<code>null</code> not permitted).
644         * @param cursor  the cursor location (determines where to draw the axis).
645         * @param plotArea  the area within which the axes and plot should be drawn.
646         * @param dataArea  the area within which the data should be drawn.
647         * @param edge  the axis location (<code>null</code> not permitted).
648         * @param plotState  collects information about the plot
649         *                   (<code>null</code> permitted).
650         *
651         * @return The axis state (never <code>null</code>).
652         */
653        public AxisState draw(Graphics2D g2, double cursor, Rectangle2D plotArea,
654                Rectangle2D dataArea, RectangleEdge edge,
655                PlotRenderingInfo plotState) {
656    
657            AxisState axisState = new AxisState(cursor);
658            if (isAxisLineVisible()) {
659                drawAxisLine(g2, cursor, dataArea, edge);
660            }
661            if (isTickMarksVisible()) {
662                drawTickMarks(g2, axisState, dataArea, edge);
663            }
664            if (isTickLabelsVisible()) {
665                for (int band = 0; band < this.labelInfo.length; band++) {
666                    axisState = drawTickLabels(band, g2, axisState, dataArea, edge);
667                }
668            }
669    
670            if (getAttributedLabel() != null) {
671                axisState = drawAttributedLabel(getAttributedLabel(), g2, plotArea, 
672                        dataArea, edge, axisState);
673            } else {
674                axisState = drawLabel(getLabel(), g2, plotArea, dataArea, edge, 
675                        axisState);
676            } 
677            return axisState;
678    
679        }

681        /**
682         * Draws the tick marks for the axis.
683         *
684         * @param g2  the graphics device.
685         * @param state  the axis state.
686         * @param dataArea  the data area.
687         * @param edge  the edge.
688         */
689        protected void drawTickMarks(Graphics2D g2, AxisState state, 
690                Rectangle2D dataArea, RectangleEdge edge) {
691            if (RectangleEdge.isTopOrBottom(edge)) {
692                drawTickMarksHorizontal(g2, state, dataArea, edge);
693            }
694            else if (RectangleEdge.isLeftOrRight(edge)) {
695                drawTickMarksVertical(g2, state, dataArea, edge);
696            }
697        }

700         * Draws the major and minor tick marks for an axis that lies at the top or
701         * bottom of the plot.
702         *
703         * @param g2  the graphics device.
704         * @param state  the axis state.
705         * @param dataArea  the data area.
706         * @param edge  the edge.
707         */
708        protected void drawTickMarksHorizontal(Graphics2D g2, AxisState state,
709                Rectangle2D dataArea, RectangleEdge edge) {
710            List ticks = new ArrayList();
711            double x0;
712            double y0 = state.getCursor();
713            double insideLength = getTickMarkInsideLength();
714            double outsideLength = getTickMarkOutsideLength();
715            RegularTimePeriod t = createInstance(this.majorTickTimePeriodClass, 
716                    this.first.getStart(), getTimeZone(), this.locale);
717            long t0 = t.getFirstMillisecond();
718            Line2D inside = null;
719            Line2D outside = null;
720            long firstOnAxis = getFirst().getFirstMillisecond();
721            long lastOnAxis = getLast().getLastMillisecond() + 1;
722            while (t0 <= lastOnAxis) {
723                ticks.add(new NumberTick(Double.valueOf(t0), "", TextAnchor.CENTER,
724                        TextAnchor.CENTER, 0.0));
725                x0 = valueToJava2D(t0, dataArea, edge);
726                if (edge == RectangleEdge.TOP) {
727                    inside = new Line2D.Double(x0, y0, x0, y0 + insideLength);
728                    outside = new Line2D.Double(x0, y0, x0, y0 - outsideLength);
729                }
730                else if (edge == RectangleEdge.BOTTOM) {
731                    inside = new Line2D.Double(x0, y0, x0, y0 - insideLength);
732                    outside = new Line2D.Double(x0, y0, x0, y0 + outsideLength);
733                }
734                if (t0 >= firstOnAxis) {
735                    g2.setPaint(getTickMarkPaint());
736                    g2.setStroke(getTickMarkStroke());
737                    g2.draw(inside);
738                    g2.draw(outside);
739                }
740                // draw minor tick marks
741                if (this.minorTickMarksVisible) {
742                    RegularTimePeriod tminor = createInstance(
743                            this.minorTickTimePeriodClass, new Date(t0),
744                            getTimeZone(), this.locale);
745                    long tt0 = tminor.getFirstMillisecond();
746                    while (tt0 < t.getLastMillisecond()
747                            && tt0 < lastOnAxis) {
748                        double xx0 = valueToJava2D(tt0, dataArea, edge);
749                        if (edge == RectangleEdge.TOP) {
750                            inside = new Line2D.Double(xx0, y0, xx0,
751                                    y0 + this.minorTickMarkInsideLength);
752                            outside = new Line2D.Double(xx0, y0, xx0,
753                                    y0 - this.minorTickMarkOutsideLength);
754                        }
755                        else if (edge == RectangleEdge.BOTTOM) {
756                            inside = new Line2D.Double(xx0, y0, xx0,
757                                    y0 - this.minorTickMarkInsideLength);
758                            outside = new Line2D.Double(xx0, y0, xx0,
759                                    y0 + this.minorTickMarkOutsideLength);
760                        }
761                        if (tt0 >= firstOnAxis) {
762                            g2.setPaint(this.minorTickMarkPaint);
763                            g2.setStroke(this.minorTickMarkStroke);
764                            g2.draw(inside);
765                            g2.draw(outside);
766                        }
767                        tminor = tminor.next();
768                        tminor.peg(this.calendar);
769                        tt0 = tminor.getFirstMillisecond();
770                    }
771                }
772                t = t.next();
773                t.peg(this.calendar);
774                t0 = t.getFirstMillisecond();
775            }
776            if (edge == RectangleEdge.TOP) {
777                state.cursorUp(Math.max(outsideLength,
778                        this.minorTickMarkOutsideLength));
779            }
780            else if (edge == RectangleEdge.BOTTOM) {
781                state.cursorDown(Math.max(outsideLength,
782                        this.minorTickMarkOutsideLength));
783            }
784            state.setTicks(ticks);
785        }

800        /**
801         * Draws the tick labels for one "band" of time periods.
802         *
803         * @param band  the band index (zero-based).
804         * @param g2  the graphics device.
805         * @param state  the axis state.
806         * @param dataArea  the data area.
807         * @param edge  the edge where the axis is located.
808         *
809         * @return The updated axis state.
810         */
811        protected AxisState drawTickLabels(int band, Graphics2D g2, AxisState state,
812                Rectangle2D dataArea, RectangleEdge edge) {
813    
814            // work out the initial gap
815            double delta1 = 0.0;
816            FontMetrics fm = g2.getFontMetrics(this.labelInfo[band].getLabelFont());
817            if (edge == RectangleEdge.BOTTOM) {
818                delta1 = this.labelInfo[band].getPadding().calculateTopOutset(
819                        fm.getHeight());
820            }
821            else if (edge == RectangleEdge.TOP) {
822                delta1 = this.labelInfo[band].getPadding().calculateBottomOutset(
823                        fm.getHeight());
824            }
825            state.moveCursor(delta1, edge);
826            long axisMin = this.first.getFirstMillisecond();
827            long axisMax = this.last.getLastMillisecond();
828            g2.setFont(this.labelInfo[band].getLabelFont());
829            g2.setPaint(this.labelInfo[band].getLabelPaint());
830    
831            // work out the number of periods to skip for labelling
832            RegularTimePeriod p1 = this.labelInfo[band].createInstance(
833                    new Date(axisMin), this.timeZone, this.locale);
834            RegularTimePeriod p2 = this.labelInfo[band].createInstance(
835                    new Date(axisMax), this.timeZone, this.locale);
836            DateFormat df = this.labelInfo[band].getDateFormat();
837            df.setTimeZone(this.timeZone);
838            String label1 = df.format(new Date(p1.getMiddleMillisecond()));
839            String label2 = df.format(new Date(p2.getMiddleMillisecond()));
840            Rectangle2D b1 = TextUtilities.getTextBounds(label1, g2,
841                    g2.getFontMetrics());
842            Rectangle2D b2 = TextUtilities.getTextBounds(label2, g2,
843                    g2.getFontMetrics());
844            double w = Math.max(b1.getWidth(), b2.getWidth());
845            long ww = Math.round(java2DToValue(dataArea.getX() + w + 5.0,
846                    dataArea, edge));
847            if (isInverted()) {
848                ww = axisMax - ww;
849            }
850            else {
851                ww = ww - axisMin;
852            }
853            long length = p1.getLastMillisecond()
854                          - p1.getFirstMillisecond();
855            int periods = (int) (ww / length) + 1;
856    
857            RegularTimePeriod p = this.labelInfo[band].createInstance(
858                    new Date(axisMin), this.timeZone, this.locale);
859            Rectangle2D b = null;
860            long lastXX = 0L;
861            float y = (float) (state.getCursor());
862            TextAnchor anchor = TextAnchor.TOP_CENTER;
863            float yDelta = (float) b1.getHeight();
864            if (edge == RectangleEdge.TOP) {
865                anchor = TextAnchor.BOTTOM_CENTER;
866                yDelta = -yDelta;
867            }
868            while (p.getFirstMillisecond() <= axisMax) {
869                float x = (float) valueToJava2D(p.getMiddleMillisecond(), dataArea,
870                        edge);
871                String label = df.format(new Date(p.getMiddleMillisecond()));
872                long first = p.getFirstMillisecond();
873                long last = p.getLastMillisecond();
874                if (last > axisMax) {
875                    // this is the last period, but it is only partially visible
876                    // so check that the label will fit before displaying it...
877                    Rectangle2D bb = TextUtilities.getTextBounds(label, g2,
878                            g2.getFontMetrics());
879                    if ((x + bb.getWidth() / 2) > dataArea.getMaxX()) {
880                        float xstart = (float) valueToJava2D(Math.max(first,
881                                axisMin), dataArea, edge);
882                        if (bb.getWidth() < (dataArea.getMaxX() - xstart)) {
883                            x = ((float) dataArea.getMaxX() + xstart) / 2.0f;
884                        }
885                        else {
886                            label = null;
887                        }
888                    }
889                }
890                if (first < axisMin) {
891                    // this is the first period, but it is only partially visible
892                    // so check that the label will fit before displaying it...
893                    Rectangle2D bb = TextUtilities.getTextBounds(label, g2,
894                            g2.getFontMetrics());
895                    if ((x - bb.getWidth() / 2) < dataArea.getX()) {
896                        float xlast = (float) valueToJava2D(Math.min(last,
897                                axisMax), dataArea, edge);
898                        if (bb.getWidth() < (xlast - dataArea.getX())) {
899                            x = (xlast + (float) dataArea.getX()) / 2.0f;
900                        }
901                        else {
902                            label = null;
903                        }
904                    }
905    
906                }
907                if (label != null) {
908                    g2.setPaint(this.labelInfo[band].getLabelPaint());
909                    b = TextUtilities.drawAlignedString(label, g2, x, y, anchor);
910                }
911                if (lastXX > 0L) {
912                    if (this.labelInfo[band].getDrawDividers()) {
913                        long nextXX = p.getFirstMillisecond();
914                        long mid = (lastXX + nextXX) / 2;
915                        float mid2d = (float) valueToJava2D(mid, dataArea, edge);
916                        g2.setStroke(this.labelInfo[band].getDividerStroke());
917                        g2.setPaint(this.labelInfo[band].getDividerPaint());
918                        g2.draw(new Line2D.Float(mid2d, y, mid2d, y + yDelta));
919                    }
920                }
921                lastXX = last;
922                for (int i = 0; i < periods; i++) {
923                    p = p.next();
924                }
925                p.peg(this.calendar);
926            }
927            double used = 0.0;
928            if (b != null) {
929                used = b.getHeight();
930                // work out the trailing gap
931                if (edge == RectangleEdge.BOTTOM) {
932                    used += this.labelInfo[band].getPadding().calculateBottomOutset(
933                            fm.getHeight());
934                }
935                else if (edge == RectangleEdge.TOP) {
936                    used += this.labelInfo[band].getPadding().calculateTopOutset(
937                            fm.getHeight());
938                }
939            }
940            state.moveCursor(used, edge);
941            return state;
942        }

960        /**
961         * Converts a data value to a coordinate in Java2D space, assuming that the
962         * axis runs along one edge of the specified dataArea.
963         * <p>
964         * Note that it is possible for the coordinate to fall outside the area.
965         *
966         * @param value  the data value.
967         * @param area  the area for plotting the data.
968         * @param edge  the edge along which the axis lies.
969         *
970         * @return The Java2D coordinate.
971         */
972        public double valueToJava2D(double value, Rectangle2D area,
973                RectangleEdge edge) {
974    
975            double result = Double.NaN;
976            double axisMin = this.first.getFirstMillisecond();
977            double axisMax = this.last.getLastMillisecond();
978            if (RectangleEdge.isTopOrBottom(edge)) {
979                double minX = area.getX();
980                double maxX = area.getMaxX();
981                if (isInverted()) {
982                    result = maxX + ((value - axisMin) / (axisMax - axisMin))
983                             * (minX - maxX);
984                }
985                else {
986                    result = minX + ((value - axisMin) / (axisMax - axisMin))
987                             * (maxX - minX);
988                }
989            }
990            else if (RectangleEdge.isLeftOrRight(edge)) {
991                double minY = area.getMinY();
992                double maxY = area.getMaxY();
993                if (isInverted()) {
994                    result = minY + (((value - axisMin) / (axisMax - axisMin))
995                             * (maxY - minY));
996                }
997                else {
998                    result = maxY - (((value - axisMin) / (axisMax - axisMin))
999                             * (maxY - minY));
1000                }
1001            }
1002            return result;
1003    
1004        }

1006        /**
1007         * Converts a coordinate in Java2D space to the corresponding data value,
1008         * assuming that the axis runs along one edge of the specified dataArea.
1009         *
1010         * @param java2DValue  the coordinate in Java2D space.
1011         * @param area  the area in which the data is plotted.
1012         * @param edge  the edge along which the axis lies.
1013         *
1014         * @return The data value.
1015         */
1016        public double java2DToValue(double java2DValue, Rectangle2D area,
1017                RectangleEdge edge) {
1018    
1019            double result;
1020            double min = 0.0;
1021            double max = 0.0;
1022            double axisMin = this.first.getFirstMillisecond();
1023            double axisMax = this.last.getLastMillisecond();
1024            if (RectangleEdge.isTopOrBottom(edge)) {
1025                min = area.getX();
1026                max = area.getMaxX();
1027            }
1028            else if (RectangleEdge.isLeftOrRight(edge)) {
1029                min = area.getMaxY();
1030                max = area.getY();
1031            }
1032            if (isInverted()) {
1033                 result = axisMax - ((java2DValue - min) / (max - min)
1034                          * (axisMax - axisMin));
1035            }
1036            else {
1037                 result = axisMin + ((java2DValue - min) / (max - min)
1038                          * (axisMax - axisMin));
1039            }
1040            return result;
1041        }

1043        /**
1044         * Rescales the axis to ensure that all data is visible.
1045         */
1046        protected void autoAdjustRange() {
1047    
1048            Plot plot = getPlot();
1049            if (plot == null) {
1050                return;  // no plot, no data
1051            }
1052    
1053            if (plot instanceof ValueAxisPlot) {
1054                ValueAxisPlot vap = (ValueAxisPlot) plot;
1055    
1056                Range r = vap.getDataRange(this);
1057                if (r == null) {
1058                    r = getDefaultAutoRange();
1059                }
1060    
1061                long upper = Math.round(r.getUpperBound());
1062                long lower = Math.round(r.getLowerBound());
1063                this.first = createInstance(this.autoRangeTimePeriodClass,
1064                        new Date(lower), this.timeZone, this.locale);
1065                this.last = createInstance(this.autoRangeTimePeriodClass,
1066                        new Date(upper), this.timeZone, this.locale);
1067                setRange(r, false, false);
1068            }
1069    
1070        }

1150        /**
1151         * A utility method used to create a particular subclass of the
1152         * {@link RegularTimePeriod} class that includes the specified millisecond,
1153         * assuming the specified time zone.
1154         *
1155         * @param periodClass  the class.
1156         * @param millisecond  the time.
1157         * @param zone  the time zone.
1158         * @param locale  the locale.
1159         *
1160         * @return The time period.
1161         */
1162        private RegularTimePeriod createInstance(Class periodClass, 
1163                Date millisecond, TimeZone zone, Locale locale) {
1164            RegularTimePeriod result = null;
1165            try {
1166                Constructor c = periodClass.getDeclaredConstructor(new Class[] {
1167                        Date.class, TimeZone.class, Locale.class});
1168                result = (RegularTimePeriod) c.newInstance(new Object[] {
1169                        millisecond, zone, locale});
1170            }
1171            catch (Exception e) {
1172                try {
1173                    Constructor c = periodClass.getDeclaredConstructor(new Class[] {
1174                            Date.class});
1175                    result = (RegularTimePeriod) c.newInstance(new Object[] {
1176                            millisecond});
1177                }
1178                catch (Exception e2) {
1179                    // do nothing
1180                }
1181            }
1182            return result;
1183        }
