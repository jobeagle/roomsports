284         * @param dataArea  the area within which the data should be drawn
285         *                  (<code>null</code> not permitted).

292        public AxisState draw(Graphics2D g2, double cursor, Rectangle2D plotArea,
293                Rectangle2D dataArea, RectangleEdge edge, 
294                PlotRenderingInfo plotState) {
295    
296            AxisState info = new AxisState(cursor);
297            if (isVisible()) {
298                info = super.draw(g2, cursor, plotArea, dataArea, edge, plotState);
299            }
300            if (this.gridBandsVisible) {
301                drawGridBands(g2, plotArea, dataArea, edge, info.getTicks());
302            }
303            return info;

307        /**
308         * Draws the grid bands.  Alternate bands are colored using
309         * <CODE>gridBandPaint</CODE> (<CODE>DEFAULT_GRID_BAND_PAINT</CODE> by
310         * default).
311         *
312         * @param g2  the graphics device.
313         * @param plotArea  the area within which the chart should be drawn.
314         * @param dataArea  the area within which the plot should be drawn (a
315         *                  subset of the drawArea).
316         * @param edge  the axis location.
317         * @param ticks  the ticks.
318         */
319        protected void drawGridBands(Graphics2D g2, Rectangle2D plotArea,
320                Rectangle2D dataArea, RectangleEdge edge, List ticks) {
321    
322            Shape savedClip = g2.getClip();
323            g2.clip(dataArea);
324            if (RectangleEdge.isTopOrBottom(edge)) {
325                drawGridBandsHorizontal(g2, plotArea, dataArea, true, ticks);
326            }
327            else if (RectangleEdge.isLeftOrRight(edge)) {
328                drawGridBandsVertical(g2, plotArea, dataArea, true, ticks);
329            }
330            g2.setClip(savedClip);
331    
332        }

334        /**
335         * Draws the grid bands for the axis when it is at the top or bottom of
336         * the plot.
337         *
338         * @param g2  the graphics device.
339         * @param plotArea  the area within which the chart should be drawn.
340         * @param dataArea  the area within which the plot should be drawn
341         *                  (a subset of the drawArea).
342         * @param firstGridBandIsDark  True: the first grid band takes the
343         *                             color of <CODE>gridBandPaint</CODE>.
344         *                             False: the second grid band takes the
345         *                             color of <CODE>gridBandPaint</CODE>.
346         * @param ticks  the ticks.
347         */
348        protected void drawGridBandsHorizontal(Graphics2D g2,
349                Rectangle2D plotArea, Rectangle2D dataArea, 
350                boolean firstGridBandIsDark, List ticks) {
351    
352            boolean currentGridBandIsDark = firstGridBandIsDark;
353            double yy = dataArea.getY();
354            double xx1, xx2;
355    
356            //gets the outline stroke width of the plot
357            double outlineStrokeWidth;
358            if (getPlot().getOutlineStroke() !=  null) {
359                outlineStrokeWidth
360                    = ((BasicStroke) getPlot().getOutlineStroke()).getLineWidth();
361            }
362            else {
363                outlineStrokeWidth = 1d;
364            }
365    
366            Iterator iterator = ticks.iterator();
367            ValueTick tick;
368            Rectangle2D band;
369            while (iterator.hasNext()) {
370                tick = (ValueTick) iterator.next();
371                xx1 = valueToJava2D(tick.getValue() - 0.5d, dataArea,
372                        RectangleEdge.BOTTOM);
373                xx2 = valueToJava2D(tick.getValue() + 0.5d, dataArea,
374                        RectangleEdge.BOTTOM);
375                if (currentGridBandIsDark) {
376                    g2.setPaint(this.gridBandPaint);
377                }
378                else {
379                    g2.setPaint(this.gridBandAlternatePaint);
380                }
381                band = new Rectangle2D.Double(xx1, yy + outlineStrokeWidth,
382                    xx2 - xx1, dataArea.getMaxY() - yy - outlineStrokeWidth);
383                g2.fill(band);
384                currentGridBandIsDark = !currentGridBandIsDark;
385            }
386            g2.setPaintMode();
387        }

389        /**
390         * Draws the grid bands for the axis when it is at the top or bottom of
391         * the plot.
392         *
393         * @param g2  the graphics device.
394         * @param drawArea  the area within which the chart should be drawn.
395         * @param plotArea  the area within which the plot should be drawn (a
396         *                  subset of the drawArea).
397         * @param firstGridBandIsDark  True: the first grid band takes the
398         *                             color of <CODE>gridBandPaint</CODE>.
399         *                             False: the second grid band takes the
400         *                             color of <CODE>gridBandPaint</CODE>.
401         * @param ticks  a list of ticks.
402         */
403        protected void drawGridBandsVertical(Graphics2D g2, Rectangle2D drawArea,
404                Rectangle2D plotArea, boolean firstGridBandIsDark, List ticks) {
405    
406            boolean currentGridBandIsDark = firstGridBandIsDark;
407            double xx = plotArea.getX();
408            double yy1, yy2;
409    
410            //gets the outline stroke width of the plot
411            double outlineStrokeWidth;
412            Stroke outlineStroke = getPlot().getOutlineStroke();
413            if (outlineStroke != null && outlineStroke instanceof BasicStroke) {
414                outlineStrokeWidth = ((BasicStroke) outlineStroke).getLineWidth();
415            }
416            else {
417                outlineStrokeWidth = 1d;
418            }
419    
420            Iterator iterator = ticks.iterator();
421            ValueTick tick;
422            Rectangle2D band;
423            while (iterator.hasNext()) {
424                tick = (ValueTick) iterator.next();
425                yy1 = valueToJava2D(tick.getValue() + 0.5d, plotArea,
426                        RectangleEdge.LEFT);
427                yy2 = valueToJava2D(tick.getValue() - 0.5d, plotArea,
428                        RectangleEdge.LEFT);
429                if (currentGridBandIsDark) {
430                    g2.setPaint(this.gridBandPaint);
431                }
432                else {
433                    g2.setPaint(this.gridBandAlternatePaint);
434                }
435                band = new Rectangle2D.Double(xx + outlineStrokeWidth, yy1,
436                        plotArea.getMaxX() - xx - outlineStrokeWidth, yy2 - yy1);
437                g2.fill(band);
438                currentGridBandIsDark = !currentGridBandIsDark;
439            }
440            g2.setPaintMode();
441        }

443        /**
444         * Rescales the axis to ensure that all data is visible.
445         */
446        protected void autoAdjustRange() {
447    
448            Plot plot = getPlot();
449            if (plot == null) {
450                return;  // no plot, no data
451            }
452    
453            if (plot instanceof ValueAxisPlot) {
454    
455                // ensure that all the symbols are displayed
456                double upper = this.symbols.size() - 1;
457                double lower = 0;
458                double range = upper - lower;
459    
460                // ensure the autorange is at least <minRange> in size...
461                double minRange = getAutoRangeMinimumSize();
462                if (range < minRange) {
463                    upper = (upper + lower + minRange) / 2;
464                    lower = (upper + lower - minRange) / 2;
465                }
466    
467                // this ensure that the grid bands will be displayed correctly.
468                double upperMargin = 0.5;
469                double lowerMargin = 0.5;
470    
471                if (getAutoRangeIncludesZero()) {
472                    if (getAutoRangeStickyZero()) {
473                        if (upper <= 0.0) {
474                            upper = 0.0;
475                        }
476                        else {
477                            upper = upper + upperMargin;
478                        }
479                        if (lower >= 0.0) {
480                            lower = 0.0;
481                        }
482                        else {
483                            lower = lower - lowerMargin;
484                        }
485                    }
486                    else {
487                        upper = Math.max(0.0, upper + upperMargin);
488                        lower = Math.min(0.0, lower - lowerMargin);
489                    }
490                }
491                else {
492                    if (getAutoRangeStickyZero()) {
493                        if (upper <= 0.0) {
494                            upper = Math.min(0.0, upper + upperMargin);
495                        }
496                        else {
497                            upper = upper + upperMargin * range;
498                        }
499                        if (lower >= 0.0) {
500                            lower = Math.max(0.0, lower - lowerMargin);
501                        }
502                        else {
503                            lower = lower - lowerMargin;
504                        }
505                    }
506                    else {
507                        upper = upper + upperMargin;
508                        lower = lower - lowerMargin;
509                    }
510                }
511    
512                setRange(new Range(lower, upper), false, false);
513    
514            }
515    
516        }

538            return ticks;

541        /**
542         * Calculates the positions of the tick labels for the axis, storing the
543         * results in the tick label list (ready for drawing).
544         *
545         * @param g2  the graphics device.
546         * @param dataArea  the area in which the data should be drawn.
547         * @param edge  the location of the axis.
548         *
549         * @return The ticks.
550         */
551        protected List refreshTicksHorizontal(Graphics2D g2, Rectangle2D dataArea,
552                RectangleEdge edge) {
553    
554            List ticks = new java.util.ArrayList();
555    
556            Font tickLabelFont = getTickLabelFont();
557            g2.setFont(tickLabelFont);
558    
559            double size = getTickUnit().getSize();
560            int count = calculateVisibleTickCount();
561            double lowestTickValue = calculateLowestVisibleTickValue();
562    
563            double previousDrawnTickLabelPos = 0.0;
564            double previousDrawnTickLabelLength = 0.0;
565    
566            if (count <= ValueAxis.MAXIMUM_TICK_COUNT) {
567                for (int i = 0; i < count; i++) {
568                    double currentTickValue = lowestTickValue + (i * size);
569                    double xx = valueToJava2D(currentTickValue, dataArea, edge);
570                    String tickLabel;
571                    NumberFormat formatter = getNumberFormatOverride();
572                    if (formatter != null) {
573                        tickLabel = formatter.format(currentTickValue);
574                    }
575                    else {
576                        tickLabel = valueToString(currentTickValue);
577                    }
578    
579                    // avoid to draw overlapping tick labels
580                    Rectangle2D bounds = TextUtilities.getTextBounds(tickLabel, g2,
581                            g2.getFontMetrics());
582                    double tickLabelLength = isVerticalTickLabels()
583                            ? bounds.getHeight() : bounds.getWidth();
584                    boolean tickLabelsOverlapping = false;
585                    if (i > 0) {
586                        double avgTickLabelLength = (previousDrawnTickLabelLength
587                                + tickLabelLength) / 2.0;
588                        if (Math.abs(xx - previousDrawnTickLabelPos)
589                                < avgTickLabelLength) {
590                            tickLabelsOverlapping = true;
591                        }
592                    }
593                    if (tickLabelsOverlapping) {
594                        tickLabel = ""; // don't draw this tick label
595                    }
596                    else {
597                        // remember these values for next comparison
598                        previousDrawnTickLabelPos = xx;
599                        previousDrawnTickLabelLength = tickLabelLength;
600                    }
601    
602                    TextAnchor anchor;
603                    TextAnchor rotationAnchor;
604                    double angle = 0.0;
605                    if (isVerticalTickLabels()) {
606                        anchor = TextAnchor.CENTER_RIGHT;
607                        rotationAnchor = TextAnchor.CENTER_RIGHT;
608                        if (edge == RectangleEdge.TOP) {
609                            angle = Math.PI / 2.0;
610                        }
611                        else {
612                            angle = -Math.PI / 2.0;
613                        }
614                    }
615                    else {
616                        if (edge == RectangleEdge.TOP) {
617                            anchor = TextAnchor.BOTTOM_CENTER;
618                            rotationAnchor = TextAnchor.BOTTOM_CENTER;
619                        }
620                        else {
621                            anchor = TextAnchor.TOP_CENTER;
622                            rotationAnchor = TextAnchor.TOP_CENTER;
623                        }
624                    }
625                    Tick tick = new NumberTick(new Double(currentTickValue),
626                            tickLabel, anchor, rotationAnchor, angle);
627                    ticks.add(tick);
628                }
629            }
630            return ticks;
631    
632        }

673                    Rectangle2D bounds = TextUtilities.getTextBounds(tickLabel, g2,
674                            g2.getFontMetrics());
675                    double tickLabelLength = isVerticalTickLabels()
676                        ? bounds.getWidth() : bounds.getHeight();
677                    boolean tickLabelsOverlapping = false;
678                    if (i > 0) {
679                        double avgTickLabelLength = (previousDrawnTickLabelLength
680                                + tickLabelLength) / 2.0;
681                        if (Math.abs(yy - previousDrawnTickLabelPos)
682                                < avgTickLabelLength) {
683                            tickLabelsOverlapping = true;
684                        }
685                    }
686                    if (tickLabelsOverlapping) {
687                        tickLabel = ""; // don't draw this tick label
688                    }
689                    else {
690                        // remember these values for next comparison
691                        previousDrawnTickLabelPos = yy;
692                        previousDrawnTickLabelLength = tickLabelLength;
693                    }
694    
695                    TextAnchor anchor = null;
696                    TextAnchor rotationAnchor = null;
697                    double angle = 0.0;
698                    if (isVerticalTickLabels()) {
699                        anchor = TextAnchor.BOTTOM_CENTER;
700                        rotationAnchor = TextAnchor.BOTTOM_CENTER;
701                        if (edge == RectangleEdge.LEFT) {
702                            angle = -Math.PI / 2.0;
703                        }
704                        else {
705                            angle = Math.PI / 2.0;
706                        }
707                    }
708                    else {
709                        if (edge == RectangleEdge.LEFT) {
710                            anchor = TextAnchor.CENTER_RIGHT;
711                            rotationAnchor = TextAnchor.CENTER_RIGHT;
712                        }
713                        else {
714                            anchor = TextAnchor.CENTER_LEFT;
715                            rotationAnchor = TextAnchor.CENTER_LEFT;
716                        }
717                    }
718                    Tick tick = new NumberTick(new Double(currentTickValue),
719                            tickLabel, anchor, rotationAnchor, angle);
720                    ticks.add(tick);
721                }
722            }
723            return ticks;
724    
725        }

727        /**
728         * Converts a value to a string, using the list of symbols.
729         *
730         * @param value  value to convert.
731         *
732         * @return The symbol.
733         */
734        public String valueToString(double value) {
735            String strToReturn;
736            try {
737                strToReturn = (String) this.symbols.get((int) value);
738            }
739            catch (IndexOutOfBoundsException  ex) {
740                strToReturn = "";
741            }
742            return strToReturn;
743        }

745        /**
746         * Tests this axis for equality with an arbitrary object.
747         *
748         * @param obj  the object (<code>null</code> permitted).
749         *
750         * @return A boolean.
751         */
752        public boolean equals(Object obj) {
753            if (obj == this) {
754                return true;
755            }
756            if (!(obj instanceof SymbolAxis)) {
757                return false;
758            }
759            SymbolAxis that = (SymbolAxis) obj;
760            if (!this.symbols.equals(that.symbols)) {
761                return false;
762            }
763            if (this.gridBandsVisible != that.gridBandsVisible) {
764                return false;
765            }
766            if (!PaintUtilities.equal(this.gridBandPaint, that.gridBandPaint)) {
767                return false;
768            }
769            if (!PaintUtilities.equal(this.gridBandAlternatePaint,
770                    that.gridBandAlternatePaint)) {
771                return false;
772            }
773            return super.equals(obj);
774        }
