044    package org.jfree.chart.imagemap;
045    
046    /**
047     * Generates tooltips using the Dynamic Drive DHTML Tip Message
048     * library (http://www.dynamicdrive.com).
049     */
050    public class DynamicDriveToolTipTagFragmentGenerator
051            implements ToolTipTagFragmentGenerator {
052    
053        /** The title, empty string not to display */
054        protected String title = "";
055    
056        /** The style number */
057        protected int style = 1;
058    
059        /**
060         * Blank constructor.
061         */
062        public DynamicDriveToolTipTagFragmentGenerator() {
063            super();
064        }
065    
066        /**
067         * Creates a new generator with specific title and style settings.
068         *
069         * @param title  title for use in all tooltips, use empty String not to
070         *               display a title.
071         * @param style  style number, see http://www.dynamicdrive.com for more
072         *               information.
073         */
074        public DynamicDriveToolTipTagFragmentGenerator(String title, int style) {
075            this.title = title;
076            this.style = style;
077        }
078    
079        /**
080         * Generates a tooltip string to go in an HTML image map.
081         *
082         * @param toolTipText  the tooltip.
083         *
084         * @return The formatted HTML area tag attribute(s).
085         */
086        public String generateToolTipFragment(String toolTipText) {
087            return " onMouseOver=\"return stm(['"
088                + ImageMapUtilities.javascriptEscape(this.title) + "','"
089                + ImageMapUtilities.javascriptEscape(toolTipText) + "'],Style["
090                + this.style + "]);\"" + " onMouseOut=\"return htm();\"";
091        }
092    
093    }
