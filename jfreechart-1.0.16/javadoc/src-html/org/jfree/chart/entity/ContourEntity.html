045    package org.jfree.chart.entity;
046    
047    import java.awt.Shape;
048    import java.io.Serializable;
049    
050    import org.jfree.chart.plot.XYPlot;
051    import org.jfree.chart.renderer.xy.XYBlockRenderer;
052    
053    /**
054     * Represents an item on a contour chart.
055     *
056     * @deprecated This class is no longer supported (as of version 1.0.4).  If
057     *     you are creating contour plots, please try to use {@link XYPlot} and
058     *     {@link XYBlockRenderer}.
059     */
060    public class ContourEntity extends ChartEntity
061                               implements Cloneable, Serializable {
062    
063        /** For serialization. */
064        private static final long serialVersionUID = 1249570520505992847L;
065    
066        /** Holds the index into the dataset for this entity. */
067        private int index = -1;
068    
069        /**
070         * Constructor for ContourEntity.
071         *
072         * @param area  the area.
073         * @param toolTipText  the tooltip text.
074         */
075        public ContourEntity(Shape area, String toolTipText) {
076            super(area, toolTipText);
077        }
078    
079        /**
080         * Constructor for ContourEntity.
081         *
082         * @param area  the area.
083         * @param toolTipText  the tooltip text.
084         * @param urlText  the URL text.
085         */
086        public ContourEntity(Shape area, String toolTipText, String urlText) {
087            super(area, toolTipText, urlText);
088        }
089    
090        /**
091         * Returns the index.
092         *
093         * @return The index.
094         */
095        public int getIndex() {
096            return this.index;
097        }
098    
099        /**
100         * Sets the index.
101         *
102         * @param index  the index.
103         */
104        public void setIndex(int index) {
105            this.index = index;
106        }
107    
108        /**
109         * Tests the entity for equality with an arbitrary object.
110         *
111         * @param obj  the object (<code>null</code> permitted).
112         *
113         * @return A boolean.
114         */
115        public boolean equals(Object obj) {
116            if (obj == this) {
117                return true;
118            }
119            if (obj instanceof ContourEntity && super.equals(obj)) {
120                ContourEntity ce = (ContourEntity) obj;
121                if (this.index != ce.index) {
122                    return false;
123                }
124                return true;
125            }
126            return false;
127        }
128    
129        /**
130         * Returns a clone of the entity.
131         *
132         * @return A clone.
133         *
134         * @throws CloneNotSupportedException if cloning is not supported.
135         */
136        public Object clone() throws CloneNotSupportedException {
137            return super.clone();
138        }
139    
140    }
