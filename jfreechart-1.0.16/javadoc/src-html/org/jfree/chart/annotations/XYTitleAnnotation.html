075    /**
076     * An annotation that allows any {@link Title} to be placed at a location on
077     * an {@link XYPlot}.
078     *
079     * @since 1.0.11
080     */
081    public class XYTitleAnnotation extends AbstractXYAnnotation
082            implements Cloneable, PublicCloneable, Serializable {

088        private XYCoordinateType coordinateType;
089

091        private double x;
092    
093        /** The y-coordinate (in data space). */
094        private double y;
095    
096        /** The maximum width. */
097        private double maxWidth;
098    
099        /** The maximum height. */
100        private double maxHeight;
101    
102        /** The title. */
103        private Title title;

106         * The title anchor point.

110        /**
111         * Creates a new annotation to be displayed at the specified (x, y)
112         * location.
113         *
114         * @param x  the x-coordinate (in data space).
115         * @param y  the y-coordinate (in data space).
116         * @param title  the title (<code>null</code> not permitted).
117         */
118        public XYTitleAnnotation(double x, double y, Title title) {
119            this(x, y, title, RectangleAnchor.CENTER);
120        }
121    
122        /**
123         * Creates a new annotation to be displayed at the specified (x, y)
124         * location.
125         *
126         * @param x  the x-coordinate (in data space).
127         * @param y  the y-coordinate (in data space).
128         * @param title  the title (<code>null</code> not permitted).
129         * @param anchor  the title anchor (<code>null</code> not permitted).
130         */
131        public XYTitleAnnotation(double x, double y, Title title,
132                RectangleAnchor anchor) {
133            super();
134            ParamChecks.nullNotPermitted(title, "title");
135            ParamChecks.nullNotPermitted(anchor, "anchor");
136            this.coordinateType = XYCoordinateType.RELATIVE;
137            this.x = x;
138            this.y = y;
139            this.maxWidth = 0.0;
140            this.maxHeight = 0.0;
141            this.title = title;
142            this.anchor = anchor;
143        }

230        /**
231         * Draws the annotation.  This method is called by the drawing code in the
232         * {@link XYPlot} class, you don't normally need to call this method
233         * directly.
234         *
235         * @param g2  the graphics device.
236         * @param plot  the plot.
237         * @param dataArea  the data area.
238         * @param domainAxis  the domain axis.
239         * @param rangeAxis  the range axis.
240         * @param rendererIndex  the renderer index.
241         * @param info  if supplied, this info object will be populated with
242         *              entity information.
243         */
244        public void draw(Graphics2D g2, XYPlot plot, Rectangle2D dataArea,
245                         ValueAxis domainAxis, ValueAxis rangeAxis,
246                         int rendererIndex, PlotRenderingInfo info) {
247    
248            PlotOrientation orientation = plot.getOrientation();
249            AxisLocation domainAxisLocation = plot.getDomainAxisLocation();
250            AxisLocation rangeAxisLocation = plot.getRangeAxisLocation();
251            RectangleEdge domainEdge = Plot.resolveDomainAxisLocation(
252                    domainAxisLocation, orientation);
253            RectangleEdge rangeEdge = Plot.resolveRangeAxisLocation(
254                    rangeAxisLocation, orientation);
255            Range xRange = domainAxis.getRange();
256            Range yRange = rangeAxis.getRange();
257            double anchorX, anchorY;
258            if (this.coordinateType == XYCoordinateType.RELATIVE) {
259                anchorX = xRange.getLowerBound() + (this.x * xRange.getLength());
260                anchorY = yRange.getLowerBound() + (this.y * yRange.getLength());
261            }
262            else {
263                anchorX = domainAxis.valueToJava2D(this.x, dataArea, domainEdge);
264                anchorY = rangeAxis.valueToJava2D(this.y, dataArea, rangeEdge);
265            }
266    
267            float j2DX = (float) domainAxis.valueToJava2D(anchorX, dataArea,
268                    domainEdge);
269            float j2DY = (float) rangeAxis.valueToJava2D(anchorY, dataArea,
270                    rangeEdge);
271            float xx = 0.0f;
272            float yy = 0.0f;
273            if (orientation == PlotOrientation.HORIZONTAL) {
274                xx = j2DY;
275                yy = j2DX;
276            }
277            else if (orientation == PlotOrientation.VERTICAL) {
278                xx = j2DX;
279                yy = j2DY;
280            }
281    
282            double maxW = dataArea.getWidth();
283            double maxH = dataArea.getHeight();
284            if (this.coordinateType == XYCoordinateType.RELATIVE) {
285                if (this.maxWidth > 0.0) {
286                    maxW = maxW * this.maxWidth;
287                }
288                if (this.maxHeight > 0.0) {
289                    maxH = maxH * this.maxHeight;
290                }
291            }
292            if (this.coordinateType == XYCoordinateType.DATA) {
293                maxW = this.maxWidth;
294                maxH = this.maxHeight;
295            }
296            RectangleConstraint rc = new RectangleConstraint(
297                    new Range(0, maxW), new Range(0, maxH));
298    
299            Size2D size = this.title.arrange(g2, rc);
300            Rectangle2D titleRect = new Rectangle2D.Double(0, 0, size.width,
301                    size.height);
302            Point2D anchorPoint = RectangleAnchor.coordinates(titleRect,
303                    this.anchor);
304            xx = xx - (float) anchorPoint.getX();
305            yy = yy - (float) anchorPoint.getY();
306            titleRect.setRect(xx, yy, titleRect.getWidth(), titleRect.getHeight());
307            BlockParams p = new BlockParams();
308            if (info != null) {
309                if (info.getOwner().getEntityCollection() != null) {
310                    p.setGenerateEntities(true);
311                }
312            }
313            Object result = this.title.draw(g2, titleRect, p);
314            if (info != null) {
315                if (result instanceof EntityBlockResult) {
316                    EntityBlockResult ebr = (EntityBlockResult) result;
317                    info.getOwner().getEntityCollection().addAll(
318                            ebr.getEntityCollection());
319                }
320                String toolTip = getToolTipText();
321                String url = getURL();
322                if (toolTip != null || url != null) {
323                    addEntity(info, new Rectangle2D.Float(xx, yy,
324                            (float) size.width, (float) size.height),
325                            rendererIndex, toolTip, url);
326                }
327            }
328        }

330        /**
331         * Tests this object for equality with an arbitrary object.
332         *
333         * @param obj  the object (<code>null</code> permitted).
334         *
335         * @return A boolean.
336         */
337        public boolean equals(Object obj) {
338            if (obj == this) {
339                return true;
340            }
341            if (!(obj instanceof XYTitleAnnotation)) {
342                return false;
343            }
344            XYTitleAnnotation that = (XYTitleAnnotation) obj;
345            if (this.coordinateType != that.coordinateType) {
346                return false;
347            }
348            if (this.x != that.x) {
349                return false;
350            }
351            if (this.y != that.y) {
352                return false;
353            }
354            if (this.maxWidth != that.maxWidth) {
355                return false;
356            }
357            if (this.maxHeight != that.maxHeight) {
358                return false;
359            }
360            if (!ObjectUtilities.equal(this.title, that.title)) {
361                return false;
362            }
363            if (!this.anchor.equals(that.anchor)) {
364                return false;
365            }
366            return super.equals(obj);
367        }

369        /**
370         * Returns a hash code for this object.
371         *
372         * @return A hash code.
373         */
374        public int hashCode() {
375            int result = 193;
376            result = HashUtilities.hashCode(result, this.anchor);
377            result = HashUtilities.hashCode(result, this.coordinateType);
378            result = HashUtilities.hashCode(result, this.x);
379            result = HashUtilities.hashCode(result, this.y);
380            result = HashUtilities.hashCode(result, this.maxWidth);
381            result = HashUtilities.hashCode(result, this.maxHeight);
382            result = HashUtilities.hashCode(result, this.title);
383            return result;
384        }
