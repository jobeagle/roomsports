103        /**
104         * Creates a new annotation that draws a line between (category1, value1)
105         * and (category2, value2).
106         *
107         * @param category1  the category (<code>null</code> not permitted).
108         * @param value1  the value.
109         * @param category2  the category (<code>null</code> not permitted).
110         * @param value2  the value.
111         * @param paint  the line color (<code>null</code> not permitted).
112         * @param stroke  the line stroke (<code>null</code> not permitted).
113         */
114        public CategoryLineAnnotation(Comparable category1, double value1,
115                                      Comparable category2, double value2,
116                                      Paint paint, Stroke stroke) {
117            super();
118            ParamChecks.nullNotPermitted(category1, "category1");
119            ParamChecks.nullNotPermitted(category2, "category2");
120            ParamChecks.nullNotPermitted(paint, "paint");
121            ParamChecks.nullNotPermitted(stroke, "stroke");
122            this.category1 = category1;
123            this.value1 = value1;
124            this.category2 = category2;
125            this.value2 = value2;
126            this.paint = paint;
127            this.stroke = stroke;
128        }

278        /**
279         * Draws the annotation.
280         *
281         * @param g2  the graphics device.
282         * @param plot  the plot.
283         * @param dataArea  the data area.
284         * @param domainAxis  the domain axis.
285         * @param rangeAxis  the range axis.
286         */
287        public void draw(Graphics2D g2, CategoryPlot plot, Rectangle2D dataArea,
288                         CategoryAxis domainAxis, ValueAxis rangeAxis) {
289    
290            CategoryDataset dataset = plot.getDataset();
291            int catIndex1 = dataset.getColumnIndex(this.category1);
292            int catIndex2 = dataset.getColumnIndex(this.category2);
293            int catCount = dataset.getColumnCount();
294    
295            double lineX1 = 0.0f;
296            double lineY1 = 0.0f;
297            double lineX2 = 0.0f;
298            double lineY2 = 0.0f;
299            PlotOrientation orientation = plot.getOrientation();
300            RectangleEdge domainEdge = Plot.resolveDomainAxisLocation(
301                plot.getDomainAxisLocation(), orientation);
302            RectangleEdge rangeEdge = Plot.resolveRangeAxisLocation(
303                plot.getRangeAxisLocation(), orientation);
304    
305            if (orientation == PlotOrientation.HORIZONTAL) {
306                lineY1 = domainAxis.getCategoryJava2DCoordinate(
307                    CategoryAnchor.MIDDLE, catIndex1, catCount, dataArea,
308                    domainEdge);
309                lineX1 = rangeAxis.valueToJava2D(this.value1, dataArea, rangeEdge);
310                lineY2 = domainAxis.getCategoryJava2DCoordinate(
311                    CategoryAnchor.MIDDLE, catIndex2, catCount, dataArea,
312                    domainEdge);
313                lineX2 = rangeAxis.valueToJava2D(this.value2, dataArea, rangeEdge);
314            }
315            else if (orientation == PlotOrientation.VERTICAL) {
316                lineX1 = domainAxis.getCategoryJava2DCoordinate(
317                    CategoryAnchor.MIDDLE, catIndex1, catCount, dataArea,
318                    domainEdge);
319                lineY1 = rangeAxis.valueToJava2D(this.value1, dataArea, rangeEdge);
320                lineX2 = domainAxis.getCategoryJava2DCoordinate(
321                    CategoryAnchor.MIDDLE, catIndex2, catCount, dataArea,
322                    domainEdge);
323                lineY2 = rangeAxis.valueToJava2D(this.value2, dataArea, rangeEdge);
324            }
325            g2.setPaint(this.paint);
326            g2.setStroke(this.stroke);
327            g2.drawLine((int) lineX1, (int) lineY1, (int) lineX2, (int) lineY2);
328        }

330        /**
331         * Tests this object for equality with another.
332         *
333         * @param obj  the object (<code>null</code> permitted).
334         *
335         * @return <code>true</code> or <code>false</code>.
336         */
337        public boolean equals(Object obj) {
338            if (obj == this) {
339                return true;
340            }
341            if (!(obj instanceof CategoryLineAnnotation)) {
342                return false;
343            }
344            CategoryLineAnnotation that = (CategoryLineAnnotation) obj;
345            if (!this.category1.equals(that.getCategory1())) {
346                return false;
347            }
348            if (this.value1 != that.getValue1()) {
349                return false;
350            }
351            if (!this.category2.equals(that.getCategory2())) {
352                return false;
353            }
354            if (this.value2 != that.getValue2()) {
355                return false;
356            }
357            if (!PaintUtilities.equal(this.paint, that.paint)) {
358                return false;
359            }
360            if (!ObjectUtilities.equal(this.stroke, that.stroke)) {
361                return false;
362            }
363            return true;
364        }

366        /**
367         * Returns a hash code for this instance.
368         *
369         * @return A hash code.
370         */
371        public int hashCode() {
372            int result = 193;
373            result = 37 * result + this.category1.hashCode();
374            long temp = Double.doubleToLongBits(this.value1);
375            result = 37 * result + (int) (temp ^ (temp >>> 32));
376            result = 37 * result + this.category2.hashCode();
377            temp = Double.doubleToLongBits(this.value2);
378            result = 37 * result + (int) (temp ^ (temp >>> 32));
379            result = 37 * result + HashUtilities.hashCodeForPaint(this.paint);
380            result = 37 * result + this.stroke.hashCode();
381            return result;
382        }
