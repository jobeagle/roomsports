041    package org.jfree.chart.annotations;
042    
043    import org.jfree.chart.event.AnnotationChangeEvent;
044    import org.jfree.chart.event.AnnotationChangeListener;
045    
046    /**
047     * The base interface for annotations.  All annotations should support the
048     * {@link AnnotationChangeEvent} mechanism by allowing listeners to register
049     * and receive notification of any changes to the annotation.
050     * 
051     * @since 1.0.14
052     */
053    public interface Annotation {
054    
055        /**
056         * Registers an object for notification of changes to the annotation.
057         *
058         * @param listener  the object to register.
059         */
060        public void addChangeListener(AnnotationChangeListener listener);
061    
062        /**
063         * Deregisters an object for notification of changes to the annotation.
064         *
065         * @param listener  the object to deregister.
066         */
067        public void removeChangeListener(AnnotationChangeListener listener);
068    
069    }
