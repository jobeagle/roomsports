164        /**
165         * Creates a new annotation to be displayed at the given coordinates.  The
166         * coordinates are specified in data space (they will be converted to
167         * Java2D space for display).
168         *
169         * @param text  the text (<code>null</code> not permitted).
170         * @param x  the x-coordinate (in data space).
171         * @param y  the y-coordinate (in data space).
172         */
173        public XYTextAnnotation(String text, double x, double y) {
174            super();
175            ParamChecks.nullNotPermitted(text, "text");
176            this.text = text;
177            this.font = DEFAULT_FONT;
178            this.paint = DEFAULT_PAINT;
179            this.x = x;
180            this.y = y;
181            this.textAnchor = DEFAULT_TEXT_ANCHOR;
182            this.rotationAnchor = DEFAULT_ROTATION_ANCHOR;
183            this.rotationAngle = DEFAULT_ROTATION_ANGLE;
184    
185            // by default the outline and background won't be visible
186            this.backgroundPaint = null;
187            this.outlineVisible = false;
188            this.outlinePaint = Color.black;
189            this.outlineStroke = new BasicStroke(0.5f);
190        }

503        /**
504         * Draws the annotation.
505         *
506         * @param g2  the graphics device.
507         * @param plot  the plot.
508         * @param dataArea  the data area.
509         * @param domainAxis  the domain axis.
510         * @param rangeAxis  the range axis.
511         * @param rendererIndex  the renderer index.
512         * @param info  an optional info object that will be populated with
513         *              entity information.
514         */
515        public void draw(Graphics2D g2, XYPlot plot, Rectangle2D dataArea,
516                         ValueAxis domainAxis, ValueAxis rangeAxis,
517                         int rendererIndex, PlotRenderingInfo info) {
518    
519            PlotOrientation orientation = plot.getOrientation();
520            RectangleEdge domainEdge = Plot.resolveDomainAxisLocation(
521                    plot.getDomainAxisLocation(), orientation);
522            RectangleEdge rangeEdge = Plot.resolveRangeAxisLocation(
523                    plot.getRangeAxisLocation(), orientation);
524    
525            float anchorX = (float) domainAxis.valueToJava2D(
526                    this.x, dataArea, domainEdge);
527            float anchorY = (float) rangeAxis.valueToJava2D(
528                    this.y, dataArea, rangeEdge);
529    
530            if (orientation == PlotOrientation.HORIZONTAL) {
531                float tempAnchor = anchorX;
532                anchorX = anchorY;
533                anchorY = tempAnchor;
534            }
535    
536            g2.setFont(getFont());
537            Shape hotspot = TextUtilities.calculateRotatedStringBounds(
538                    getText(), g2, anchorX, anchorY, getTextAnchor(),
539                    getRotationAngle(), getRotationAnchor());
540            if (this.backgroundPaint != null) {
541                g2.setPaint(this.backgroundPaint);
542                g2.fill(hotspot);
543            }
544            g2.setPaint(getPaint());
545            TextUtilities.drawRotatedString(getText(), g2, anchorX, anchorY,
546                    getTextAnchor(), getRotationAngle(), getRotationAnchor());
547            if (this.outlineVisible) {
548                g2.setStroke(this.outlineStroke);
549                g2.setPaint(this.outlinePaint);
550                g2.draw(hotspot);
551            }
552    
553            String toolTip = getToolTipText();
554            String url = getURL();
555            if (toolTip != null || url != null) {
556                addEntity(info, hotspot, rendererIndex, toolTip, url);
557            }
558    
559        }

561        /**
562         * Tests this annotation for equality with an arbitrary object.
563         *
564         * @param obj  the object (<code>null</code> permitted).
565         *
566         * @return A boolean.
567         */
568        public boolean equals(Object obj) {
569            if (obj == this) {
570                return true;
571            }
572            if (!(obj instanceof XYTextAnnotation)) {
573                return false;
574            }
575            XYTextAnnotation that = (XYTextAnnotation) obj;
576            if (!this.text.equals(that.text)) {
577                return false;
578            }
579            if (this.x != that.x) {
580                return false;
581            }
582            if (this.y != that.y) {
583                return false;
584            }
585            if (!this.font.equals(that.font)) {
586                return false;
587            }
588            if (!PaintUtilities.equal(this.paint, that.paint)) {
589                return false;
590            }
591            if (!this.rotationAnchor.equals(that.rotationAnchor)) {
592                return false;
593            }
594            if (this.rotationAngle != that.rotationAngle) {
595                return false;
596            }
597            if (!this.textAnchor.equals(that.textAnchor)) {
598                return false;
599            }
600            if (this.outlineVisible != that.outlineVisible) {
601                return false;
602            }
603            if (!PaintUtilities.equal(this.backgroundPaint, that.backgroundPaint)) {
604                return false;
605            }
606            if (!PaintUtilities.equal(this.outlinePaint, that.outlinePaint)) {
607                return false;
608            }
609            if (!(this.outlineStroke.equals(that.outlineStroke))) {
610                return false;
611            }
612            return super.equals(obj);
613        }

615        /**
616         * Returns a hash code for the object.
617         *
618         * @return A hash code.
619         */
620        public int hashCode() {
621            int result = 193;
622            result = 37 * result + this.text.hashCode();
623            result = 37 * result + this.font.hashCode();
624            result = 37 * result + HashUtilities.hashCodeForPaint(this.paint);
625            long temp = Double.doubleToLongBits(this.x);
626            result = 37 * result + (int) (temp ^ (temp >>> 32));
627            temp = Double.doubleToLongBits(this.y);
628            result = 37 * result + (int) (temp ^ (temp >>> 32));
629            result = 37 * result + this.textAnchor.hashCode();
630            result = 37 * result + this.rotationAnchor.hashCode();
631            temp = Double.doubleToLongBits(this.rotationAngle);
632            result = 37 * result + (int) (temp ^ (temp >>> 32));
633            return result;
634        }
