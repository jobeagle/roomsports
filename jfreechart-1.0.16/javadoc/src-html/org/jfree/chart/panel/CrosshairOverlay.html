097        /**
098         * Adds a crosshair against the domain axis and sends an
099         * {@link OverlayChangeEvent} to all registered listeners.
100         *
101         * @param crosshair  the crosshair (<code>null</code> not permitted).
102         *
103         * @see #removeDomainCrosshair(org.jfree.chart.plot.Crosshair)
104         * @see #addRangeCrosshair(org.jfree.chart.plot.Crosshair)
105         */
106        public void addDomainCrosshair(Crosshair crosshair) {
107            ParamChecks.nullNotPermitted(crosshair, "crosshair");
108            this.xCrosshairs.add(crosshair);
109            crosshair.addPropertyChangeListener(this);
110            fireOverlayChanged();
111        }

114         * Removes a domain axis crosshair and sends an {@link OverlayChangeEvent}
115         * to all registered listeners.

118         *
119         * @see #addDomainCrosshair(org.jfree.chart.plot.Crosshair)
120         */
121        public void removeDomainCrosshair(Crosshair crosshair) {
122            ParamChecks.nullNotPermitted(crosshair, "crosshair");
123            if (this.xCrosshairs.remove(crosshair)) {
124                crosshair.removePropertyChangeListener(this);
125                fireOverlayChanged();
126            }

129        /**
130         * Clears all the domain crosshairs from the overlay and sends an
131         * {@link OverlayChangeEvent} to all registered listeners.
132         */
133        public void clearDomainCrosshairs() {
134            if (this.xCrosshairs.isEmpty()) {
135                return;  // nothing to do
136            }
137            List crosshairs = getDomainCrosshairs();
138            for (int i = 0; i < crosshairs.size(); i++) {
139                Crosshair c = (Crosshair) crosshairs.get(i);
140                this.xCrosshairs.remove(c);
141                c.removePropertyChangeListener(this);
142            }
143            fireOverlayChanged();
144        }

220        /**
221         * Paints the crosshairs in the layer.
222         *
223         * @param g2  the graphics target.
224         * @param chartPanel  the chart panel.
225         */
226        public void paintOverlay(Graphics2D g2, ChartPanel chartPanel) {
227            Shape savedClip = g2.getClip();
228            Rectangle2D dataArea = chartPanel.getScreenDataArea();
229            g2.clip(dataArea);
230            JFreeChart chart = chartPanel.getChart();
231            XYPlot plot = (XYPlot) chart.getPlot();
232            ValueAxis xAxis = plot.getDomainAxis();
233            RectangleEdge xAxisEdge = plot.getDomainAxisEdge();
234            Iterator iterator = this.xCrosshairs.iterator();
235            while (iterator.hasNext()) {
236                Crosshair ch = (Crosshair) iterator.next();
237                if (ch.isVisible()) {
238                    double x = ch.getValue();
239                    double xx = xAxis.valueToJava2D(x, dataArea, xAxisEdge);
240                    if (plot.getOrientation() == PlotOrientation.VERTICAL) {
241                        drawVerticalCrosshair(g2, dataArea, xx, ch);
242                    }
243                    else {
244                        drawHorizontalCrosshair(g2, dataArea, xx, ch);
245                    }
246                }
247            }
248            ValueAxis yAxis = plot.getRangeAxis();
249            RectangleEdge yAxisEdge = plot.getRangeAxisEdge();
250            iterator = this.yCrosshairs.iterator();
251            while (iterator.hasNext()) {
252                Crosshair ch = (Crosshair) iterator.next();
253                if (ch.isVisible()) {
254                    double y = ch.getValue();
255                    double yy = yAxis.valueToJava2D(y, dataArea, yAxisEdge);
256                    if (plot.getOrientation() == PlotOrientation.VERTICAL) {
257                        drawHorizontalCrosshair(g2, dataArea, yy, ch);
258                    }
259                    else {
260                        drawVerticalCrosshair(g2, dataArea, yy, ch);
261                    }
262                }
263            }
264            g2.setClip(savedClip);
265        }

267        /**
268         * Draws a crosshair horizontally across the plot.
269         *
270         * @param g2  the graphics target.
271         * @param dataArea  the data area.
272         * @param y  the y-value in Java2D space.
273         * @param crosshair  the crosshair.
274         */
275        protected void drawHorizontalCrosshair(Graphics2D g2, Rectangle2D dataArea,
276                double y, Crosshair crosshair) {
277    
278            if (y >= dataArea.getMinY() && y <= dataArea.getMaxY()) {
279                Line2D line = new Line2D.Double(dataArea.getMinX(), y,
280                        dataArea.getMaxX(), y);
281                Paint savedPaint = g2.getPaint();
282                Stroke savedStroke = g2.getStroke();
283                g2.setPaint(crosshair.getPaint());
284                g2.setStroke(crosshair.getStroke());
285                g2.draw(line);
286                if (crosshair.isLabelVisible()) {
287                    String label = crosshair.getLabelGenerator().generateLabel(
288                            crosshair);
289                    RectangleAnchor anchor = crosshair.getLabelAnchor();
290                    Point2D pt = calculateLabelPoint(line, anchor, 5, 5);
291                    float xx = (float) pt.getX();
292                    float yy = (float) pt.getY();
293                    TextAnchor alignPt = textAlignPtForLabelAnchorH(anchor);
294                    Shape hotspot = TextUtilities.calculateRotatedStringBounds(
295                            label, g2, xx, yy, alignPt, 0.0, TextAnchor.CENTER);
296                    if (!dataArea.contains(hotspot.getBounds2D())) {
297                        anchor = flipAnchorV(anchor);
298                        pt = calculateLabelPoint(line, anchor, 5, 5);
299                        xx = (float) pt.getX();
300                        yy = (float) pt.getY();
301                        alignPt = textAlignPtForLabelAnchorH(anchor);
302                        hotspot = TextUtilities.calculateRotatedStringBounds(
303                               label, g2, xx, yy, alignPt, 0.0, TextAnchor.CENTER);
304                    }
305    
306                    g2.setPaint(crosshair.getLabelBackgroundPaint());
307                    g2.fill(hotspot);
308                    g2.setPaint(crosshair.getLabelOutlinePaint());
309                    g2.draw(hotspot);
310                    TextUtilities.drawAlignedString(label, g2, xx, yy, alignPt);
311                }
312                g2.setPaint(savedPaint);
313                g2.setStroke(savedStroke);
314            }
315        }

366        /**
367         * Calculates the anchor point for a label.
368         *
369         * @param line  the line for the crosshair.
370         * @param anchor  the anchor point.
371         * @param deltaX  the x-offset.
372         * @param deltaY  the y-offset.
373         *
374         * @return The anchor point.
375         */
376        private Point2D calculateLabelPoint(Line2D line, RectangleAnchor anchor,
377                double deltaX, double deltaY) {
378            double x, y;
379            boolean left = (anchor == RectangleAnchor.BOTTOM_LEFT 
380                    || anchor == RectangleAnchor.LEFT 
381                    || anchor == RectangleAnchor.TOP_LEFT);
382            boolean right = (anchor == RectangleAnchor.BOTTOM_RIGHT 
383                    || anchor == RectangleAnchor.RIGHT 
384                    || anchor == RectangleAnchor.TOP_RIGHT);
385            boolean top = (anchor == RectangleAnchor.TOP_LEFT 
386                    || anchor == RectangleAnchor.TOP 
387                    || anchor == RectangleAnchor.TOP_RIGHT);
388            boolean bottom = (anchor == RectangleAnchor.BOTTOM_LEFT
389                    || anchor == RectangleAnchor.BOTTOM
390                    || anchor == RectangleAnchor.BOTTOM_RIGHT);
391            Rectangle rect = line.getBounds();
392            
393            // we expect the line to be vertical or horizontal
394            if (line.getX1() == line.getX2()) {  // vertical
395                x = line.getX1();
396                y = (line.getY1() + line.getY2()) / 2.0;
397                if (left) {
398                    x = x - deltaX;
399                }
400                if (right) {
401                    x = x + deltaX;
402                }
403                if (top) {
404                    y = Math.min(line.getY1(), line.getY2()) + deltaY;
405                }
406                if (bottom) {
407                    y = Math.max(line.getY1(), line.getY2()) - deltaY;
408                }
409            }
410            else {  // horizontal
411                x = (line.getX1() + line.getX2()) / 2.0;
412                y = line.getY1();
413                if (left) {
414                    x = Math.min(line.getX1(), line.getX2()) + deltaX;
415                }
416                if (right) {
417                    x = Math.max(line.getX1(), line.getX2()) - deltaX;
418                }
419                if (top) {
420                    y = y - deltaY;
421                }
422                if (bottom) {
423                    y = y + deltaY;
424                }
425            }
426            return new Point2D.Double(x, y);
427        }

429        /**
430         * Returns the text anchor that is used to align a label to its anchor 
431         * point.
432         * 
433         * @param anchor  the anchor.
434         * 
435         * @return The text alignment point.
436         */
437        private TextAnchor textAlignPtForLabelAnchorV(RectangleAnchor anchor) {
438            TextAnchor result = TextAnchor.CENTER;
439            if (anchor.equals(RectangleAnchor.TOP_LEFT)) {
440                result = TextAnchor.TOP_RIGHT;
441            }
442            else if (anchor.equals(RectangleAnchor.TOP)) {
443                result = TextAnchor.TOP_CENTER;
444            }
445            else if (anchor.equals(RectangleAnchor.TOP_RIGHT)) {
446                result = TextAnchor.TOP_LEFT;
447            }
448            else if (anchor.equals(RectangleAnchor.LEFT)) {
449                result = TextAnchor.HALF_ASCENT_RIGHT;
450            }
451            else if (anchor.equals(RectangleAnchor.RIGHT)) {
452                result = TextAnchor.HALF_ASCENT_LEFT;
453            }
454            else if (anchor.equals(RectangleAnchor.BOTTOM_LEFT)) {
455                result = TextAnchor.BOTTOM_RIGHT;
456            }
457            else if (anchor.equals(RectangleAnchor.BOTTOM)) {
458                result = TextAnchor.BOTTOM_CENTER;
459            }
460            else if (anchor.equals(RectangleAnchor.BOTTOM_RIGHT)) {
461                result = TextAnchor.BOTTOM_LEFT;
462            }
463            return result;
464        }

493            }
494            else if (anchor.equals(RectangleAnchor.BOTTOM)) {
495                result = TextAnchor.TOP_CENTER;

503        private RectangleAnchor flipAnchorH(RectangleAnchor anchor) {
504            RectangleAnchor result = anchor;
505            if (anchor.equals(RectangleAnchor.TOP_LEFT)) {
506                result = RectangleAnchor.TOP_RIGHT;
507            }
508            else if (anchor.equals(RectangleAnchor.TOP_RIGHT)) {
509                result = RectangleAnchor.TOP_LEFT;
510            }
511            else if (anchor.equals(RectangleAnchor.LEFT)) {
512                result = RectangleAnchor.RIGHT;
513            }
514            else if (anchor.equals(RectangleAnchor.RIGHT)) {
515                result = RectangleAnchor.LEFT;
516            }
517            else if (anchor.equals(RectangleAnchor.BOTTOM_LEFT)) {
518                result = RectangleAnchor.BOTTOM_RIGHT;
519            }
520            else if (anchor.equals(RectangleAnchor.BOTTOM_RIGHT)) {
521                result = RectangleAnchor.BOTTOM_LEFT;
522            }
523            return result;
524        }
525    
526        private RectangleAnchor flipAnchorV(RectangleAnchor anchor) {
527            RectangleAnchor result = anchor;
528            if (anchor.equals(RectangleAnchor.TOP_LEFT)) {
529                result = RectangleAnchor.BOTTOM_LEFT;
530            }
531            else if (anchor.equals(RectangleAnchor.TOP_RIGHT)) {
532                result = RectangleAnchor.BOTTOM_RIGHT;
533            }
534            else if (anchor.equals(RectangleAnchor.TOP)) {
535                result = RectangleAnchor.BOTTOM;
536            }
537            else if (anchor.equals(RectangleAnchor.BOTTOM)) {
538                result = RectangleAnchor.TOP;
539            }
540            else if (anchor.equals(RectangleAnchor.BOTTOM_LEFT)) {
541                result = RectangleAnchor.TOP_LEFT;
542            }
543            else if (anchor.equals(RectangleAnchor.BOTTOM_RIGHT)) {
544                result = RectangleAnchor.TOP_RIGHT;
545            }
546            return result;
547        }

549        /**
550         * Tests this overlay for equality with an arbitrary object.
551         *
552         * @param obj  the object (<code>null</code> permitted).
553         *
554         * @return A boolean.
555         */
556        public boolean equals(Object obj) {
557            if (obj == this) {
558                return true;
559            }
560            if (!(obj instanceof CrosshairOverlay)) {
561                return false;
562            }
563            CrosshairOverlay that = (CrosshairOverlay) obj;
564            if (!this.xCrosshairs.equals(that.xCrosshairs)) {
565                return false;
566            }
567            if (!this.yCrosshairs.equals(that.yCrosshairs)) {
568                return false;
569            }
570            return true;
571        }

573        /**
574         * Returns a clone of this instance.
575         *
576         * @return A clone of this instance.
577         *
578         * @throws java.lang.CloneNotSupportedException if there is some problem
579         *     with the cloning.
580         */
581        public Object clone() throws CloneNotSupportedException {
582            CrosshairOverlay clone = (CrosshairOverlay) super.clone();
583            clone.xCrosshairs = (List) ObjectUtilities.deepClone(this.xCrosshairs);
584            clone.yCrosshairs = (List) ObjectUtilities.deepClone(this.yCrosshairs);
585            return clone;
586        }
