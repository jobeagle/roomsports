501        /**
502         * Returns a string containing the characters from the given iterator.
503         *
504         * @param iterator  the iterator (<code>null</code> not permitted).
505         *
506         * @return A string.
507         */
508        private String characterIteratorToString(CharacterIterator iterator) {
509            int endIndex = iterator.getEndIndex();
510            int beginIndex = iterator.getBeginIndex();
511            int count = endIndex - beginIndex;
512            if (count <= 0) {
513                return "";
514            }
515            char[] chars = new char[count];
516            int i = 0;
517            char c = iterator.first();
518            while (c != CharacterIterator.DONE) {
519                chars[i] = c;
520                i++;
521                c = iterator.next();
522            }
523            return new String(chars);
524        }

1003         * @return A boolean.

1005        public boolean equals(Object obj) {
1006            if (obj == this) {
1007                return true;
1008            }
1009            if (!(obj instanceof LegendItem)) {
1010                return false;
1011            }
1012            LegendItem that = (LegendItem) obj;
1013            if (this.datasetIndex != that.datasetIndex) {
1014                return false;
1015            }
1016            if (this.series != that.series) {
1017                return false;
1018            }
1019            if (!this.label.equals(that.label)) {
1020                return false;
1021            }
1022            if (!AttributedStringUtilities.equal(this.attributedLabel,
1023                    that.attributedLabel)) {
1024                return false;
1025            }
1026            if (!ObjectUtilities.equal(this.description, that.description)) {
1027                return false;
1028            }
1029            if (this.shapeVisible != that.shapeVisible) {
1030                return false;
1031            }
1032            if (!ShapeUtilities.equal(this.shape, that.shape)) {
1033                return false;
1034            }
1035            if (this.shapeFilled != that.shapeFilled) {
1036                return false;
1037            }
1038            if (!PaintUtilities.equal(this.fillPaint, that.fillPaint)) {
1039                return false;
1040            }
1041            if (!ObjectUtilities.equal(this.fillPaintTransformer,
1042                    that.fillPaintTransformer)) {
1043                return false;
1044            }
1045            if (this.shapeOutlineVisible != that.shapeOutlineVisible) {
1046                return false;
1047            }
1048            if (!this.outlineStroke.equals(that.outlineStroke)) {
1049                return false;
1050            }

1052                return false;
1053            }
1054            if (!this.lineVisible == that.lineVisible) {
1055                return false;
1056            }
