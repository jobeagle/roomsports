150        /**
151         * Creates a new crosshair value with the specified value and line style.
152         *
153         * @param value  the value.
154         * @param paint  the line paint (<code>null</code> not permitted).
155         * @param stroke  the line stroke (<code>null</code> not permitted).
156         */
157        public Crosshair(double value, Paint paint, Stroke stroke) {
158            ParamChecks.nullNotPermitted(paint, "paint");
159            ParamChecks.nullNotPermitted(stroke, "stroke");
160            this.visible = true;
161            this.value = value;
162            this.paint = paint;
163            this.stroke = stroke;
164            this.labelVisible = false;
165            this.labelGenerator = new StandardCrosshairLabelGenerator();
166            this.labelAnchor = RectangleAnchor.BOTTOM_LEFT;
167            this.labelXOffset = 3.0;
168            this.labelYOffset = 3.0;
169            this.labelFont = new Font("Tahoma", Font.PLAIN, 12);
170            this.labelPaint = Color.black;
171            this.labelBackgroundPaint = new Color(0, 0, 255, 63);
172            this.labelOutlineVisible = true;
173            this.labelOutlinePaint = Color.black;
174            this.labelOutlineStroke = new BasicStroke(0.5f);
175            this.pcs = new PropertyChangeSupport(this);
176        }

563        /**
564         * Tests this crosshair for equality with an arbitrary object.
565         *
566         * @param obj  the object (<code>null</code> permitted).
567         *
568         * @return A boolean.
569         */
570        public boolean equals(Object obj) {
571            if (obj == this) {
572                return true;
573            }
574            if (!(obj instanceof Crosshair)) {
575                return false;
576            }
577            Crosshair that = (Crosshair) obj;
578            if (this.visible != that.visible) {
579                return false;
580            }
581            if (this.value != that.value) {
582                return false;
583            }
584            if (!PaintUtilities.equal(this.paint, that.paint)) {
585                return false;
586            }
587            if (!this.stroke.equals(that.stroke)) {
588                return false;
589            }
590            if (this.labelVisible != that.labelVisible) {
591                return false;
592            }
593            if (!this.labelGenerator.equals(that.labelGenerator)) {
594                return false;
595            }
596            if (!this.labelAnchor.equals(that.labelAnchor)) {
597                return false;
598            }
599            if (this.labelXOffset != that.labelXOffset) {
600                return false;
601            }
602            if (this.labelYOffset != that.labelYOffset) {
603                return false;
604            }
605            if (!this.labelFont.equals(that.labelFont)) {
606                return false;
607            }
608            if (!PaintUtilities.equal(this.labelPaint, that.labelPaint)) {
609                return false;
610            }
611            if (!PaintUtilities.equal(this.labelBackgroundPaint,
612                    that.labelBackgroundPaint)) {
613                return false;
614            }
615            if (this.labelOutlineVisible != that.labelOutlineVisible) {
616                return false;
617            }
618            if (!PaintUtilities.equal(this.labelOutlinePaint,
619                    that.labelOutlinePaint)) {
620                return false;
621            }
622            if (!this.labelOutlineStroke.equals(that.labelOutlineStroke)) {
623                return false;
624            }
625            return true;  // can't find any difference
626        }

689         *
690         * @param stream  the output stream.
691         *
692         * @throws IOException  if there is an I/O error.
693         */
694        private void writeObject(ObjectOutputStream stream) throws IOException {
695            stream.defaultWriteObject();
696            SerialUtilities.writePaint(this.paint, stream);
697            SerialUtilities.writeStroke(this.stroke, stream);
698            SerialUtilities.writePaint(this.labelPaint, stream);
699            SerialUtilities.writePaint(this.labelBackgroundPaint, stream);
700            SerialUtilities.writeStroke(this.labelOutlineStroke, stream);
701            SerialUtilities.writePaint(this.labelOutlinePaint, stream);
702        }

704        /**
705         * Provides serialization support.
706         *
707         * @param stream  the input stream.
708         *
709         * @throws IOException  if there is an I/O error.
710         * @throws ClassNotFoundException  if there is a classpath problem.
711         */
712        private void readObject(ObjectInputStream stream)
713                throws IOException, ClassNotFoundException {
714            stream.defaultReadObject();
715            this.paint = SerialUtilities.readPaint(stream);
716            this.stroke = SerialUtilities.readStroke(stream);
717            this.labelPaint = SerialUtilities.readPaint(stream);
718            this.labelBackgroundPaint = SerialUtilities.readPaint(stream);
719            this.labelOutlineStroke = SerialUtilities.readStroke(stream);
720            this.labelOutlinePaint = SerialUtilities.readPaint(stream);
721            this.pcs = new PropertyChangeSupport(this);
722        }
