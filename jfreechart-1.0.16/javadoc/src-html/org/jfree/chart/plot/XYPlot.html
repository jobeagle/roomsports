1520        /**
1521         * This method is used to perform argument checking on the list of
1522         * axis indices passed to mapDatasetToDomainAxes() and
1523         * mapDatasetToRangeAxes().
1524         *
1525         * @param indices  the list of indices (<code>null</code> permitted).
1526         */
1527        private void checkAxisIndices(List indices) {
1528            // axisIndices can be:
1529            // 1.  null;
1530            // 2.  non-empty, containing only Integer objects that are unique.
1531            if (indices == null) {
1532                return;  // OK
1533            }
1534            int count = indices.size();
1535            if (count == 0) {
1536                throw new IllegalArgumentException("Empty list not permitted.");
1537            }
1538            HashSet set = new HashSet();
1539            for (int i = 0; i < count; i++) {
1540                Object item = indices.get(i);
1541                if (!(item instanceof Integer)) {
1542                    throw new IllegalArgumentException(
1543                            "Indices must be Integer instances.");
1544                }
1545                if (set.contains(item)) {
1546                    throw new IllegalArgumentException("Indices must be unique.");
1547                }
1548                set.add(item);
1549            }
1550        }

4446        @Override

4448    
4449            Range result = null;
4450            List mappedDatasets = new ArrayList();
4451            List includedAnnotations = new ArrayList();
4452            boolean isDomainAxis = true;
4453    
4454            // is it a domain axis?
4455            int domainIndex = getDomainAxisIndex(axis);
4456            if (domainIndex >= 0) {
4457                isDomainAxis = true;
4458                mappedDatasets.addAll(getDatasetsMappedToDomainAxis(
4459                        new Integer(domainIndex)));
4460                if (domainIndex == 0) {
4461                    // grab the plot's annotations
4462                    Iterator iterator = this.annotations.iterator();
4463                    while (iterator.hasNext()) {
4464                        XYAnnotation annotation = (XYAnnotation) iterator.next();
4465                        if (annotation instanceof XYAnnotationBoundsInfo) {
4466                            includedAnnotations.add(annotation);
4467                        }
4468                    }
4469                }
4470            }
4471    
4472            // or is it a range axis?
4473            int rangeIndex = getRangeAxisIndex(axis);
4474            if (rangeIndex >= 0) {
4475                isDomainAxis = false;
4476                mappedDatasets.addAll(getDatasetsMappedToRangeAxis(
4477                        new Integer(rangeIndex)));
4478                if (rangeIndex == 0) {
4479                    Iterator iterator = this.annotations.iterator();
4480                    while (iterator.hasNext()) {
4481                        XYAnnotation annotation = (XYAnnotation) iterator.next();
4482                        if (annotation instanceof XYAnnotationBoundsInfo) {
4483                            includedAnnotations.add(annotation);
4484                        }
4485                    }
4486                }
4487            }
4488    
4489            // iterate through the datasets that map to the axis and get the union
4490            // of the ranges.
4491            Iterator iterator = mappedDatasets.iterator();
4492            while (iterator.hasNext()) {
4493                XYDataset d = (XYDataset) iterator.next();
4494                if (d != null) {
4495                    XYItemRenderer r = getRendererForDataset(d);
4496                    if (isDomainAxis) {
4497                        if (r != null) {
4498                            result = Range.combine(result, r.findDomainBounds(d));
4499                        }
4500                        else {
4501                            result = Range.combine(result,
4502                                    DatasetUtilities.findDomainBounds(d));
4503                        }
4504                    }
4505                    else {
4506                        if (r != null) {
4507                            result = Range.combine(result, r.findRangeBounds(d));
4508                        }
4509                        else {
4510                            result = Range.combine(result,
4511                                    DatasetUtilities.findRangeBounds(d));
4512                        }
4513                    }
4514                    // FIXME: the XYItemRenderer interface doesn't specify the
4515                    // getAnnotations() method but it should
4516                    if (r instanceof AbstractXYItemRenderer) {
4517                        AbstractXYItemRenderer rr = (AbstractXYItemRenderer) r;
4518                        Collection c = rr.getAnnotations();
4519                        Iterator i = c.iterator();
4520                        while (i.hasNext()) {
4521                            XYAnnotation a = (XYAnnotation) i.next();
4522                            if (a instanceof XYAnnotationBoundsInfo) {
4523                                includedAnnotations.add(a);
4524                            }
4525                        }
4526                    }
4527                }
4528            }
4529    
4530            Iterator it = includedAnnotations.iterator();
4531            while (it.hasNext()) {
4532                XYAnnotationBoundsInfo xyabi = (XYAnnotationBoundsInfo) it.next();
4533                if (xyabi.getIncludeInDataBounds()) {
4534                    if (isDomainAxis) {
4535                        result = Range.combine(result, xyabi.getXRange());
4536                    }
4537                    else {

4539                    }

4541            }
