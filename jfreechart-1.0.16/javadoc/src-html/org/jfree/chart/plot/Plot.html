190    /**
191     * The base class for all plots in JFreeChart.  The {@link JFreeChart} class
192     * delegates the drawing of axes and data to the plot.  This base class
193     * provides facilities common to most plot types.
194     */
195    public abstract class Plot implements AxisChangeListener,
196            DatasetChangeListener, AnnotationChangeListener, MarkerChangeListener,
197            LegendItemSource, PublicCloneable, Cloneable, Serializable {

556         * Sets the background color of the plot area and sends a
557         * {@link PlotChangeEvent} to all registered listeners.
558         *
559         * @param paint  the paint (<code>null</code> permitted).
560         *
561         * @see #getBackgroundPaint()
562         */
563        public void setBackgroundPaint(Paint paint) {
564    
565            if (paint == null) {
566                if (this.backgroundPaint != null) {
567                    this.backgroundPaint = null;
568                    fireChangeEvent();
569                }
570            }
571            else {
572                if (this.backgroundPaint != null) {
573                    if (this.backgroundPaint.equals(paint)) {
574                        return;  // nothing to do
575                    }
576                }
577                this.backgroundPaint = paint;
578                fireChangeEvent();
579            }
580    
581        }

957         * Notifies all registered listeners that the plot has been modified.
958         *
959         * @param event  information about the change event.
960         */
961        public void notifyListeners(PlotChangeEvent event) {
962            // if the 'notify' flag has been switched to false, we don't notify
963            // the listeners
964            if (!this.notify) {
965                return;
966            }
967            Object[] listeners = this.listenerList.getListenerList();
968            for (int i = listeners.length - 2; i >= 0; i -= 2) {
969                if (listeners[i] == PlotChangeListener.class) {
970                    ((PlotChangeListener) listeners[i + 1]).plotChanged(event);
971                }
972            }
973        }

1035        /**
1036         * Fills the specified area with the background paint.  If the background
1037         * paint is an instance of <code>GradientPaint</code>, the gradient will
1038         * run in the direction suggested by the plot's orientation.
1039         *
1040         * @param g2  the graphics target.
1041         * @param area  the plot area.
1042         * @param orientation  the plot orientation (<code>null</code> not
1043         *         permitted).
1044         *
1045         * @since 1.0.6
1046         */
1047        protected void fillBackground(Graphics2D g2, Rectangle2D area,
1048                PlotOrientation orientation) {
1049            ParamChecks.nullNotPermitted(orientation, "orientation");
1050            if (this.backgroundPaint == null) {
1051                return;
1052            }
1053            Paint p = this.backgroundPaint;
1054            if (p instanceof GradientPaint) {
1055                GradientPaint gp = (GradientPaint) p;
1056                if (orientation == PlotOrientation.VERTICAL) {
1057                    p = new GradientPaint((float) area.getCenterX(),
1058                            (float) area.getMaxY(), gp.getColor1(),
1059                            (float) area.getCenterX(), (float) area.getMinY(),
1060                            gp.getColor2());
1061                }
1062                else if (orientation == PlotOrientation.HORIZONTAL) {
1063                    p = new GradientPaint((float) area.getMinX(),
1064                            (float) area.getCenterY(), gp.getColor1(),
1065                            (float) area.getMaxX(), (float) area.getCenterY(),
1066                            gp.getColor2());
1067                }
1068            }
1069            Composite originalComposite = g2.getComposite();
1070            g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
1071                    this.backgroundAlpha));
1072            g2.setPaint(p);
1073            g2.fill(area);
1074            g2.setComposite(originalComposite);
1075        }

1078         * Draws the background image (if there is one) aligned within the
1079         * specified area.
1080         *
1081         * @param g2  the graphics device.
1082         * @param area  the area.
1083         *
1084         * @see #getBackgroundImage()
1085         * @see #getBackgroundImageAlignment()
1086         * @see #getBackgroundImageAlpha()
1087         */
1088        public void drawBackgroundImage(Graphics2D g2, Rectangle2D area) {
1089            if (this.backgroundImage == null) {
1090                return;  // nothing to do
1091            }
1092            Composite savedComposite = g2.getComposite();
1093            g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
1094                    this.backgroundImageAlpha));
1095            Rectangle2D dest = new Rectangle2D.Double(0.0, 0.0,
1096                    this.backgroundImage.getWidth(null),
1097                    this.backgroundImage.getHeight(null));
1098            Align.align(dest, area, this.backgroundImageAlignment);
1099            Shape savedClip = g2.getClip();
1100            g2.clip(area);
1101            g2.drawImage(this.backgroundImage, (int) dest.getX(),
1102                    (int) dest.getY(), (int) dest.getWidth() + 1,
1103                    (int) dest.getHeight() + 1, null);
1104            g2.setClip(savedClip);
1105            g2.setComposite(savedComposite);
1106        }

1310                return false;

1315                return false;
1316            }
1317            if (!PaintUtilities.equal(this.noDataMessagePaint,
1318                    that.noDataMessagePaint)) {
1319                return false;
1320            }
1321            if (!ObjectUtilities.equal(this.insets, that.insets)) {
1322                return false;
1323            }
1324            if (this.outlineVisible != that.outlineVisible) {
1325                return false;
1326            }
1327            if (!ObjectUtilities.equal(this.outlineStroke, that.outlineStroke)) {
1328                return false;
1329            }
1330            if (!PaintUtilities.equal(this.outlinePaint, that.outlinePaint)) {
1331                return false;
1332            }
1333            if (!PaintUtilities.equal(this.backgroundPaint, that.backgroundPaint)) {
1334                return false;
1335            }
1336            if (!ObjectUtilities.equal(this.backgroundImage,
1337                    that.backgroundImage)) {
1338                return false;
1339            }
1340            if (this.backgroundImageAlignment != that.backgroundImageAlignment) {
1341                return false;
1342            }
1343            if (this.backgroundImageAlpha != that.backgroundImageAlpha) {
1344                return false;
1345            }
1346            if (this.foregroundAlpha != that.foregroundAlpha) {
1347                return false;
1348            }
1349            if (this.backgroundAlpha != that.backgroundAlpha) {
1350                return false;
1351            }
1352            if (!this.drawingSupplier.equals(that.drawingSupplier)) {
1353                return false;
1354            }

1356                return false;

1361        /**

1363         *
1364         * @return A clone.
1365         *
1366         * @throws CloneNotSupportedException if some component of the plot does not
1367         *         support cloning.
1368         */
1369        public Object clone() throws CloneNotSupportedException {
1370    
1371            Plot clone = (Plot) super.clone();
1372            // private Plot parent <-- don't clone the parent plot, but take care
1373            // childs in combined plots instead
1374            if (this.datasetGroup != null) {
1375                clone.datasetGroup
1376                    = (DatasetGroup) ObjectUtilities.clone(this.datasetGroup);
1377            }

1426         * @param orientation  the orientation (<code>null</code> not permitted).
1427         *
1428         * @return The edge (never <code>null</code>).
1429         */
1430        public static RectangleEdge resolveDomainAxisLocation(
1431                AxisLocation location, PlotOrientation orientation) {
1432    
1433            ParamChecks.nullNotPermitted(location, "location");
1434            ParamChecks.nullNotPermitted(orientation, "orientation");
1435    
1436            RectangleEdge result = null;
1437            if (location == AxisLocation.TOP_OR_RIGHT) {
1438                if (orientation == PlotOrientation.HORIZONTAL) {
1439                    result = RectangleEdge.RIGHT;
1440                }
1441                else if (orientation == PlotOrientation.VERTICAL) {
1442                    result = RectangleEdge.TOP;
1443                }
1444            }
1445            else if (location == AxisLocation.TOP_OR_LEFT) {
1446                if (orientation == PlotOrientation.HORIZONTAL) {
1447                    result = RectangleEdge.LEFT;
1448                }
1449                else if (orientation == PlotOrientation.VERTICAL) {
1450                    result = RectangleEdge.TOP;
1451                }
1452            }
1453            else if (location == AxisLocation.BOTTOM_OR_RIGHT) {
1454                if (orientation == PlotOrientation.HORIZONTAL) {
1455                    result = RectangleEdge.RIGHT;
1456                }
1457                else if (orientation == PlotOrientation.VERTICAL) {
1458                    result = RectangleEdge.BOTTOM;
1459                }
1460            }
1461            else if (location == AxisLocation.BOTTOM_OR_LEFT) {
1462                if (orientation == PlotOrientation.HORIZONTAL) {
1463                    result = RectangleEdge.LEFT;
1464                }
1465                else if (orientation == PlotOrientation.VERTICAL) {
1466                    result = RectangleEdge.BOTTOM;
1467                }
1468            }
1469            // the above should cover all the options...
1470            if (result == null) {
1471                throw new IllegalStateException("resolveDomainAxisLocation()");
1472            }
1473            return result;
