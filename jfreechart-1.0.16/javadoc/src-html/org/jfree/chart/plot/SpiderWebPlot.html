282        /**
283         * Creates a new spider web plot with the given dataset.
284         *
285         * @param dataset  the dataset.
286         * @param extract  controls how data is extracted ({@link TableOrder#BY_ROW}
287         *                 or {@link TableOrder#BY_COLUMN}).
288         */
289        public SpiderWebPlot(CategoryDataset dataset, TableOrder extract) {
290            super();
291            ParamChecks.nullNotPermitted(extract, "extract");
292            this.dataset = dataset;
293            if (dataset != null) {
294                dataset.addChangeListener(this);
295            }
296    
297            this.dataExtractOrder = extract;
298            this.headPercent = DEFAULT_HEAD;
299            this.axisLabelGap = DEFAULT_AXIS_LABEL_GAP;
300            this.axisLinePaint = Color.black;
301            this.axisLineStroke = new BasicStroke(1.0f);
302    
303            this.interiorGap = DEFAULT_INTERIOR_GAP;
304            this.startAngle = DEFAULT_START_ANGLE;
305            this.direction = Rotation.CLOCKWISE;
306            this.maxValue = DEFAULT_MAX_VALUE;
307    
308            this.seriesPaint = null;
309            this.seriesPaintList = new PaintList();
310            this.baseSeriesPaint = null;
311    
312            this.seriesOutlinePaint = null;
313            this.seriesOutlinePaintList = new PaintList();
314            this.baseSeriesOutlinePaint = DEFAULT_OUTLINE_PAINT;
315    
316            this.seriesOutlineStroke = null;
317            this.seriesOutlineStrokeList = new StrokeList();
318            this.baseSeriesOutlineStroke = DEFAULT_OUTLINE_STROKE;
319    
320            this.labelFont = DEFAULT_LABEL_FONT;
321            this.labelPaint = DEFAULT_LABEL_PAINT;
322            this.labelGenerator = new StandardCategoryItemLabelGenerator();
323    
324            this.legendItemShape = DEFAULT_LEGEND_ITEM_CIRCLE;
325        }

348        /**
349         * Sets the dataset used by the plot and sends a {@link PlotChangeEvent}
350         * to all registered listeners.
351         *
352         * @param dataset  the dataset (<code>null</code> permitted).
353         *
354         * @see #getDataset()
355         */
356        public void setDataset(CategoryDataset dataset) {
357            // if there is an existing dataset, remove the plot from the list of
358            // change listeners...
359            if (this.dataset != null) {
360                this.dataset.removeChangeListener(this);
361            }
362    
363            // set the new dataset, and register the chart as a change listener...
364            this.dataset = dataset;
365            if (dataset != null) {
366                setDatasetGroup(dataset.getGroup());
367                dataset.addChangeListener(this);
368            }
369    
370            // send a dataset change event to self to trigger plot change event
371            datasetChanged(new DatasetChangeEvent(this, dataset));
372        }

670        /**
671         * Returns the paint for the specified series.
672         *
673         * @param series  the series index (zero-based).
674         *
675         * @return The paint (never <code>null</code>).
676         *
677         * @see #setSeriesPaint(int, Paint)
678         */
679        public Paint getSeriesPaint(int series) {
680    
681            // return the override, if there is one...
682            if (this.seriesPaint != null) {
683                return this.seriesPaint;
684            }
685    
686            // otherwise look up the paint list
687            Paint result = this.seriesPaintList.getPaint(series);
688            if (result == null) {
689                DrawingSupplier supplier = getDrawingSupplier();
690                if (supplier != null) {
691                    Paint p = supplier.getNextPaint();
692                    this.seriesPaintList.setPaint(series, p);
693                    result = p;
694                }
695                else {
696                    result = this.baseSeriesPaint;
697                }
698            }
699            return result;
700    
701        }

1052        /**
1053         * Returns a collection of legend items for the spider web chart.
1054         *
1055         * @return The legend items (never <code>null</code>).
1056         */
1057        public LegendItemCollection getLegendItems() {
1058            LegendItemCollection result = new LegendItemCollection();
1059            if (getDataset() == null) {
1060                return result;
1061            }
1062            List keys = null;
1063            if (this.dataExtractOrder == TableOrder.BY_ROW) {
1064                keys = this.dataset.getRowKeys();
1065            }
1066            else if (this.dataExtractOrder == TableOrder.BY_COLUMN) {
1067                keys = this.dataset.getColumnKeys();
1068            }
1069            if (keys == null) {
1070                return result;
1071            }
1072    
1073            int series = 0;
1074            Iterator iterator = keys.iterator();
1075            Shape shape = getLegendItemShape();
1076            while (iterator.hasNext()) {
1077                Comparable key = (Comparable) iterator.next();
1078                String label = key.toString();
1079                String description = label;
1080                Paint paint = getSeriesPaint(series);
1081                Paint outlinePaint = getSeriesOutlinePaint(series);
1082                Stroke stroke = getSeriesOutlineStroke(series);
1083                LegendItem item = new LegendItem(label, description,
1084                        null, null, shape, paint, stroke, outlinePaint);
1085                item.setDataset(getDataset());
1086                item.setSeriesKey(key);
1087                item.setSeriesIndex(series);
1088                result.add(item);
1089                series++;
1090            }
1091            return result;
1092        }

1094        /**
1095         * Returns a cartesian point from a polar angle, length and bounding box
1096         *
1097         * @param bounds  the area inside which the point needs to be.
1098         * @param angle  the polar angle, in degrees.
1099         * @param length  the relative length. Given in percent of maximum extend.
1100         *
1101         * @return The cartesian point.
1102         */
1103        protected Point2D getWebPoint(Rectangle2D bounds,
1104                                      double angle, double length) {
1105    
1106            double angrad = Math.toRadians(angle);
1107            double x = Math.cos(angrad) * length * bounds.getWidth() / 2;
1108            double y = -Math.sin(angrad) * length * bounds.getHeight() / 2;
1109    
1110            return new Point2D.Double(bounds.getX() + x + bounds.getWidth() / 2,
1111                    bounds.getY() + y + bounds.getHeight() / 2);
1112        }

1114        /**
1115         * Draws the plot on a Java 2D graphics device (such as the screen or a
1116         * printer).
1117         *
1118         * @param g2  the graphics device.
1119         * @param area  the area within which the plot should be drawn.
1120         * @param anchor  the anchor point (<code>null</code> permitted).
1121         * @param parentState  the state from the parent plot, if there is one.
1122         * @param info  collects info about the drawing.
1123         */
1124        public void draw(Graphics2D g2, Rectangle2D area, Point2D anchor,
1125                PlotState parentState, PlotRenderingInfo info) {
1126    
1127            // adjust for insets...
1128            RectangleInsets insets = getInsets();
1129            insets.trim(area);
1130    
1131            if (info != null) {
1132                info.setPlotArea(area);
1133                info.setDataArea(area);
1134            }
1135    
1136            drawBackground(g2, area);
1137            drawOutline(g2, area);
1138    
1139            Shape savedClip = g2.getClip();
1140    
1141            g2.clip(area);
1142            Composite originalComposite = g2.getComposite();
1143            g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
1144                    getForegroundAlpha()));
1145    
1146            if (!DatasetUtilities.isEmptyOrNull(this.dataset)) {
1147                int seriesCount = 0, catCount = 0;
1148    
1149                if (this.dataExtractOrder == TableOrder.BY_ROW) {
1150                    seriesCount = this.dataset.getRowCount();
1151                    catCount = this.dataset.getColumnCount();
1152                }
1153                else {
1154                    seriesCount = this.dataset.getColumnCount();
1155                    catCount = this.dataset.getRowCount();
1156                }
1157    
1158                // ensure we have a maximum value to use on the axes
1159                if (this.maxValue == DEFAULT_MAX_VALUE) {
1160                    calculateMaxValue(seriesCount, catCount);
1161                }
1162    
1163                // Next, setup the plot area
1164    
1165                // adjust the plot area by the interior spacing value
1166    
1167                double gapHorizontal = area.getWidth() * getInteriorGap();
1168                double gapVertical = area.getHeight() * getInteriorGap();
1169    
1170                double X = area.getX() + gapHorizontal / 2;
1171                double Y = area.getY() + gapVertical / 2;
1172                double W = area.getWidth() - gapHorizontal;
1173                double H = area.getHeight() - gapVertical;
1174    
1175                double headW = area.getWidth() * this.headPercent;
1176                double headH = area.getHeight() * this.headPercent;
1177    
1178                // make the chart area a square
1179                double min = Math.min(W, H) / 2;
1180                X = (X + X + W) / 2 - min;
1181                Y = (Y + Y + H) / 2 - min;
1182                W = 2 * min;
1183                H = 2 * min;
1184    
1185                Point2D  centre = new Point2D.Double(X + W / 2, Y + H / 2);
1186                Rectangle2D radarArea = new Rectangle2D.Double(X, Y, W, H);
1187    
1188                // draw the axis and category label
1189                for (int cat = 0; cat < catCount; cat++) {
1190                    double angle = getStartAngle()
1191                            + (getDirection().getFactor() * cat * 360 / catCount);
1192    
1193                    Point2D endPoint = getWebPoint(radarArea, angle, 1);
1194                                                         // 1 = end of axis
1195                    Line2D  line = new Line2D.Double(centre, endPoint);
1196                    g2.setPaint(this.axisLinePaint);
1197                    g2.setStroke(this.axisLineStroke);
1198                    g2.draw(line);
1199                    drawLabel(g2, radarArea, 0.0, cat, angle, 360.0 / catCount);
1200                }
1201    
1202                // Now actually plot each of the series polygons..
1203                for (int series = 0; series < seriesCount; series++) {
1204                    drawRadarPoly(g2, radarArea, centre, info, series, catCount,
1205                            headH, headW);
1206                }
1207            }
1208            else {
1209                drawNoDataMessage(g2, area);
1210            }
1211            g2.setClip(savedClip);
1212            g2.setComposite(originalComposite);
1213            drawOutline(g2, area);
1214        }

1216        /**
1217         * loop through each of the series to get the maximum value
1218         * on each category axis
1219         *
1220         * @param seriesCount  the number of series
1221         * @param catCount  the number of categories
1222         */
1223        private void calculateMaxValue(int seriesCount, int catCount) {
1224            double v = 0;
1225            Number nV = null;
1226    
1227            for (int seriesIndex = 0; seriesIndex < seriesCount; seriesIndex++) {
1228                for (int catIndex = 0; catIndex < catCount; catIndex++) {
1229                    nV = getPlotValue(seriesIndex, catIndex);
1230                    if (nV != null) {
1231                        v = nV.doubleValue();
1232                        if (v > this.maxValue) {
1233                            this.maxValue = v;
1234                        }
1235                    }
1236                }
1237            }
1238        }

1240        /**
1241         * Draws a radar plot polygon.
1242         *
1243         * @param g2 the graphics device.
1244         * @param plotArea the area we are plotting in (already adjusted).
1245         * @param centre the centre point of the radar axes
1246         * @param info chart rendering info.
1247         * @param series the series within the dataset we are plotting
1248         * @param catCount the number of categories per radar plot
1249         * @param headH the data point height
1250         * @param headW the data point width
1251         */
1252        protected void drawRadarPoly(Graphics2D g2,
1253                                     Rectangle2D plotArea,
1254                                     Point2D centre,
1255                                     PlotRenderingInfo info,
1256                                     int series, int catCount,
1257                                     double headH, double headW) {
1258    
1259            Polygon polygon = new Polygon();
1260    
1261            EntityCollection entities = null;
1262            if (info != null) {
1263                entities = info.getOwner().getEntityCollection();
1264            }
1265    
1266            // plot the data...
1267            for (int cat = 0; cat < catCount; cat++) {
1268    
1269                Number dataValue = getPlotValue(series, cat);
1270    
1271                if (dataValue != null) {
1272                    double value = dataValue.doubleValue();
1273    
1274                    if (value >= 0) { // draw the polygon series...
1275    
1276                        // Finds our starting angle from the centre for this axis
1277    
1278                        double angle = getStartAngle()
1279                            + (getDirection().getFactor() * cat * 360 / catCount);
1280    
1281                        // The following angle calc will ensure there isn't a top
1282                        // vertical axis - this may be useful if you don't want any
1283                        // given criteria to 'appear' move important than the
1284                        // others..
1285                        //  + (getDirection().getFactor()
1286                        //        * (cat + 0.5) * 360 / catCount);
1287    
1288                        // find the point at the appropriate distance end point
1289                        // along the axis/angle identified above and add it to the
1290                        // polygon
1291    
1292                        Point2D point = getWebPoint(plotArea, angle,
1293                                value / this.maxValue);
1294                        polygon.addPoint((int) point.getX(), (int) point.getY());
1295    
1296                        // put an elipse at the point being plotted..
1297    
1298                        Paint paint = getSeriesPaint(series);
1299                        Paint outlinePaint = getSeriesOutlinePaint(series);
1300                        Stroke outlineStroke = getSeriesOutlineStroke(series);
1301    
1302                        Ellipse2D head = new Ellipse2D.Double(point.getX()
1303                                - headW / 2, point.getY() - headH / 2, headW,
1304                                headH);
1305                        g2.setPaint(paint);
1306                        g2.fill(head);
1307                        g2.setStroke(outlineStroke);
1308                        g2.setPaint(outlinePaint);
1309                        g2.draw(head);
1310    
1311                        if (entities != null) {
1312                            int row = 0; int col = 0;
1313                            if (this.dataExtractOrder == TableOrder.BY_ROW) {
1314                                row = series;
1315                                col = cat;
1316                            }
1317                            else {
1318                                row = cat;
1319                                col = series;
1320                            }
1321                            String tip = null;
1322                            if (this.toolTipGenerator != null) {
1323                                tip = this.toolTipGenerator.generateToolTip(
1324                                        this.dataset, row, col);
1325                            }
1326    
1327                            String url = null;
1328                            if (this.urlGenerator != null) {
1329                                url = this.urlGenerator.generateURL(this.dataset,
1330                                       row, col);
1331                            }
1332    
1333                            Shape area = new Rectangle(
1334                                    (int) (point.getX() - headW),
1335                                    (int) (point.getY() - headH),
1336                                    (int) (headW * 2), (int) (headH * 2));
1337                            CategoryItemEntity entity = new CategoryItemEntity(
1338                                    area, tip, url, this.dataset,
1339                                    this.dataset.getRowKey(row),
1340                                    this.dataset.getColumnKey(col));
1341                            entities.add(entity);
1342                        }
1343    
1344                    }
1345                }
1346            }
1347            // Plot the polygon
1348    
1349            Paint paint = getSeriesPaint(series);
1350            g2.setPaint(paint);
1351            g2.setStroke(getSeriesOutlineStroke(series));
1352            g2.draw(polygon);
1353    
1354            // Lastly, fill the web polygon if this is required
1355    
1356            if (this.webFilled) {
1357                g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
1358                        0.1f));
1359                g2.fill(polygon);
1360                g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
1361                        getForegroundAlpha()));
1362            }
1363        }

1433         * Returns the location for a label
1434         *
1435         * @param labelBounds the label bounds.
1436         * @param ascent the ascent (height of font).
1437         * @param plotArea the plot area
1438         * @param startAngle the start angle for the pie series.
1439         *
1440         * @return The location for a label.
1441         */
1442        protected Point2D calculateLabelLocation(Rectangle2D labelBounds,
1443                                                 double ascent,
1444                                                 Rectangle2D plotArea,
1445                                                 double startAngle)
1446        {
1447            Arc2D arc1 = new Arc2D.Double(plotArea, startAngle, 0, Arc2D.OPEN);
1448            Point2D point1 = arc1.getEndPoint();
1449    
1450            double deltaX = -(point1.getX() - plotArea.getCenterX())
1451                            * this.axisLabelGap;
1452            double deltaY = -(point1.getY() - plotArea.getCenterY())
1453                            * this.axisLabelGap;
1454    
1455            double labelX = point1.getX() - deltaX;
1456            double labelY = point1.getY() - deltaY;
1457    
1458            if (labelX < plotArea.getCenterX()) {
1459                labelX -= labelBounds.getWidth();
1460            }
1461    
1462            if (labelX == plotArea.getCenterX()) {
1463                labelX -= labelBounds.getWidth() / 2;
1464            }
1465    
1466            if (labelY > plotArea.getCenterY()) {
1467                labelY += ascent;
1468            }
1469    
1470            return new Point2D.Double(labelX, labelY);
1471        }
