155        /**
156         * Creates a new plot.
157         *
158         * @param dataset  the dataset (<code>null</code> permitted).
159         */
160        public MultiplePiePlot(CategoryDataset dataset) {
161            super();
162            setDataset(dataset);
163            PiePlot piePlot = new PiePlot(null);
164            piePlot.setIgnoreNullValues(true);
165            this.pieChart = new JFreeChart(piePlot);
166            this.pieChart.removeLegend();
167            this.dataExtractOrder = TableOrder.BY_COLUMN;
168            this.pieChart.setBackgroundPaint(null);
169            TextTitle seriesTitle = new TextTitle("Series Title",
170                    new Font("SansSerif", Font.BOLD, 12));
171            seriesTitle.setPosition(RectangleEdge.BOTTOM);
172            this.pieChart.setTitle(seriesTitle);
173            this.aggregatedItemsKey = "Other";
174            this.aggregatedItemsPaint = Color.lightGray;
175            this.sectionPaints = new HashMap();
176            this.legendItemShape = new Ellipse2D.Double(-4.0, -4.0, 8.0, 8.0);
177        }

188        /**
189         * Sets the dataset used by the plot and sends a {@link PlotChangeEvent}
190         * to all registered listeners.
191         *
192         * @param dataset  the dataset (<code>null</code> permitted).
193         */
194        public void setDataset(CategoryDataset dataset) {
195            // if there is an existing dataset, remove the plot from the list of
196            // change listeners...
197            if (this.dataset != null) {
198                this.dataset.removeChangeListener(this);
199            }
200    
201            // set the new dataset, and register the chart as a change listener...
202            this.dataset = dataset;
203            if (dataset != null) {
204                setDatasetGroup(dataset.getGroup());
205                dataset.addChangeListener(this);
206            }
207    
208            // send a dataset change event to self to trigger plot change event
209            datasetChanged(new DatasetChangeEvent(this, dataset));
210        }

376        /**
377         * Draws the plot on a Java 2D graphics device (such as the screen or a
378         * printer).
379         *
380         * @param g2  the graphics device.
381         * @param area  the area within which the plot should be drawn.
382         * @param anchor  the anchor point (<code>null</code> permitted).
383         * @param parentState  the state from the parent plot, if there is one.
384         * @param info  collects info about the drawing.
385         */
386        public void draw(Graphics2D g2,
387                         Rectangle2D area,
388                         Point2D anchor,
389                         PlotState parentState,
390                         PlotRenderingInfo info) {
391    
392    
393            // adjust the drawing area for the plot insets (if any)...
394            RectangleInsets insets = getInsets();
395            insets.trim(area);
396            drawBackground(g2, area);
397            drawOutline(g2, area);
398    
399            // check that there is some data to display...
400            if (DatasetUtilities.isEmptyOrNull(this.dataset)) {
401                drawNoDataMessage(g2, area);
402                return;
403            }
404    
405            int pieCount = 0;
406            if (this.dataExtractOrder == TableOrder.BY_ROW) {
407                pieCount = this.dataset.getRowCount();
408            }
409            else {
410                pieCount = this.dataset.getColumnCount();
411            }
412    
413            // the columns variable is always >= rows
414            int displayCols = (int) Math.ceil(Math.sqrt(pieCount));
415            int displayRows
416                = (int) Math.ceil((double) pieCount / (double) displayCols);
417    
418            // swap rows and columns to match plotArea shape
419            if (displayCols > displayRows && area.getWidth() < area.getHeight()) {
420                int temp = displayCols;
421                displayCols = displayRows;
422                displayRows = temp;
423            }
424    
425            prefetchSectionPaints();
426    
427            int x = (int) area.getX();
428            int y = (int) area.getY();
429            int width = ((int) area.getWidth()) / displayCols;
430            int height = ((int) area.getHeight()) / displayRows;
431            int row = 0;
432            int column = 0;
433            int diff = (displayRows * displayCols) - pieCount;
434            int xoffset = 0;
435            Rectangle rect = new Rectangle();
436    
437            for (int pieIndex = 0; pieIndex < pieCount; pieIndex++) {
438                rect.setBounds(x + xoffset + (width * column), y + (height * row),
439                        width, height);
440    
441                String title = null;
442                if (this.dataExtractOrder == TableOrder.BY_ROW) {
443                    title = this.dataset.getRowKey(pieIndex).toString();
444                }
445                else {
446                    title = this.dataset.getColumnKey(pieIndex).toString();
447                }
448                this.pieChart.setTitle(title);
449    
450                PieDataset piedataset = null;
451                PieDataset dd = new CategoryToPieDataset(this.dataset,
452                        this.dataExtractOrder, pieIndex);
453                if (this.limit > 0.0) {
454                    piedataset = DatasetUtilities.createConsolidatedPieDataset(
455                            dd, this.aggregatedItemsKey, this.limit);
456                }
457                else {
458                    piedataset = dd;
459                }
460                PiePlot piePlot = (PiePlot) this.pieChart.getPlot();
461                piePlot.setDataset(piedataset);
462                piePlot.setPieIndex(pieIndex);
463    
464                // update the section colors to match the global colors...
465                for (int i = 0; i < piedataset.getItemCount(); i++) {
466                    Comparable key = piedataset.getKey(i);
467                    Paint p;
468                    if (key.equals(this.aggregatedItemsKey)) {
469                        p = this.aggregatedItemsPaint;
470                    }
471                    else {
472                        p = (Paint) this.sectionPaints.get(key);
473                    }
474                    piePlot.setSectionPaint(key, p);
475                }
476    
477                ChartRenderingInfo subinfo = null;
478                if (info != null) {
479                    subinfo = new ChartRenderingInfo();
480                }
481                this.pieChart.draw(g2, rect, subinfo);
482                if (info != null) {
483                    info.getOwner().getEntityCollection().addAll(
484                            subinfo.getEntityCollection());
485                    info.addSubplotInfo(subinfo.getPlotInfo());
486                }
487    
488                ++column;
489                if (column == displayCols) {
490                    column = 0;
491                    ++row;
492    
493                    if (row == displayRows - 1 && diff != 0) {
494                        xoffset = (diff * width) / 2;
495                    }
496                }
497            }
498    
499        }

501        /**
502         * For each key in the dataset, check the <code>sectionPaints</code>
503         * cache to see if a paint is associated with that key and, if not,
504         * fetch one from the drawing supplier.  These colors are cached so that
505         * the legend and all the subplots use consistent colors.
506         */
507        private void prefetchSectionPaints() {
508    
509            // pre-fetch the colors for each key...this is because the subplots
510            // may not display every key, but we need the coloring to be
511            // consistent...
512    
513            PiePlot piePlot = (PiePlot) getPieChart().getPlot();
514    
515            if (this.dataExtractOrder == TableOrder.BY_ROW) {
516                // column keys provide potential keys for individual pies
517                for (int c = 0; c < this.dataset.getColumnCount(); c++) {
518                    Comparable key = this.dataset.getColumnKey(c);
519                    Paint p = piePlot.getSectionPaint(key);
520                    if (p == null) {
521                        p = (Paint) this.sectionPaints.get(key);
522                        if (p == null) {
523                            p = getDrawingSupplier().getNextPaint();
524                        }
525                    }
526                    this.sectionPaints.put(key, p);
527                }
528            }
529            else {
530                // row keys provide potential keys for individual pies
531                for (int r = 0; r < this.dataset.getRowCount(); r++) {
532                    Comparable key = this.dataset.getRowKey(r);
533                    Paint p = piePlot.getSectionPaint(key);
534                    if (p == null) {
535                        p = (Paint) this.sectionPaints.get(key);
536                        if (p == null) {
537                            p = getDrawingSupplier().getNextPaint();
538                        }
539                    }
540                    this.sectionPaints.put(key, p);
541                }
542            }
543    
544        }

546        /**
547         * Returns a collection of legend items for the pie chart.
548         *
549         * @return The legend items.
550         */
551        public LegendItemCollection getLegendItems() {
552    
553            LegendItemCollection result = new LegendItemCollection();
554            if (this.dataset == null) {
555                return result;
556            }
557    
558            List keys = null;
559            prefetchSectionPaints();
560            if (this.dataExtractOrder == TableOrder.BY_ROW) {
561                keys = this.dataset.getColumnKeys();
562            }
563            else if (this.dataExtractOrder == TableOrder.BY_COLUMN) {
564                keys = this.dataset.getRowKeys();
565            }
566            if (keys == null) {
567                return result;
568            }
569            int section = 0;
570            Iterator iterator = keys.iterator();
571            while (iterator.hasNext()) {
572                Comparable key = (Comparable) iterator.next();
573                String label = key.toString();  // TODO: use a generator here
574                String description = label;
575                Paint paint = (Paint) this.sectionPaints.get(key);
576                LegendItem item = new LegendItem(label, description, null,
577                        null, getLegendItemShape(), paint,
578                        Plot.DEFAULT_OUTLINE_STROKE, paint);
579                item.setSeriesKey(key);
580                item.setSeriesIndex(section);
581                item.setDataset(getDataset());
582                result.add(item);
583                section++;
584            }
585            if (this.limit > 0.0) {
586                LegendItem a = new LegendItem(this.aggregatedItemsKey.toString(),
587                        this.aggregatedItemsKey.toString(), null, null,
588                        getLegendItemShape(), this.aggregatedItemsPaint,
589                        Plot.DEFAULT_OUTLINE_STROKE, this.aggregatedItemsPaint);
590                result.add(a);
591            }
592            return result;
593        }

595        /**
596         * Tests this plot for equality with an arbitrary object.  Note that the
597         * plot's dataset is not considered in the equality test.
598         *
599         * @param obj  the object (<code>null</code> permitted).
600         *
601         * @return <code>true</code> if this plot is equal to <code>obj</code>, and
602         *     <code>false</code> otherwise.
603         */
604        public boolean equals(Object obj) {
605            if (obj == this) {
606                return true;
607            }
608            if (!(obj instanceof MultiplePiePlot)) {
609                return false;
610            }
611            MultiplePiePlot that = (MultiplePiePlot) obj;
612            if (this.dataExtractOrder != that.dataExtractOrder) {
613                return false;
614            }
615            if (this.limit != that.limit) {
616                return false;
617            }
618            if (!this.aggregatedItemsKey.equals(that.aggregatedItemsKey)) {
619                return false;
620            }
621            if (!PaintUtilities.equal(this.aggregatedItemsPaint,
622                    that.aggregatedItemsPaint)) {
623                return false;
624            }
625            if (!ObjectUtilities.equal(this.pieChart, that.pieChart)) {
626                return false;
627            }
628            if (!ShapeUtilities.equal(this.legendItemShape, that.legendItemShape)) {
629                return false;
630            }
631            if (!super.equals(obj)) {
632                return false;
633            }
634            return true;
635        }
