041    package org.jfree.chart.plot.dial;
042    
043    import java.util.EventListener;
044    
045    /**
046     * The interface via which an object is notified of changes to a
047     * {@link DialLayer}.  The {@link DialPlot} class listens for changes to its
048     * layers in this way.
049     *
050     * @since 1.0.7
051     */
052    public interface DialLayerChangeListener extends EventListener {
053    
054        /**
055         * A call-back method for receiving notification of a change to a
056         * {@link DialLayer}.
057         *
058         * @param event  the event.
059         */
060        public void dialLayerChanged(DialLayerChangeEvent event);
061    
062    }
