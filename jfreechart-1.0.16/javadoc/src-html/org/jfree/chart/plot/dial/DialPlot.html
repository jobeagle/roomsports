516        /**
517         * Draws the plot.  This method is usually called by the {@link JFreeChart}
518         * instance that manages the plot.
519         *
520         * @param g2  the graphics target.
521         * @param area  the area in which the plot should be drawn.
522         * @param anchor  the anchor point (typically the last point that the
523         *     mouse clicked on, <code>null</code> is permitted).
524         * @param parentState  the state for the parent plot (if any).
525         * @param info  used to collect plot rendering info (<code>null</code>
526         *     permitted).
527         */
528        public void draw(Graphics2D g2, Rectangle2D area, Point2D anchor,
529                PlotState parentState, PlotRenderingInfo info) {
530    
531            Shape origClip = g2.getClip();
532            g2.setClip(area);
533    
534            // first, expand the viewing area into a drawing frame
535            Rectangle2D frame = viewToFrame(area);
536    
537            // draw the background if there is one...
538            if (this.background != null && this.background.isVisible()) {
539                if (this.background.isClippedToWindow()) {
540                    Shape savedClip = g2.getClip();
541                    g2.clip(this.dialFrame.getWindow(frame));
542                    this.background.draw(g2, this, frame, area);
543                    g2.setClip(savedClip);
544                }
545                else {
546                    this.background.draw(g2, this, frame, area);
547                }
548            }
549    
550            Iterator iterator = this.layers.iterator();
551            while (iterator.hasNext()) {
552                DialLayer current = (DialLayer) iterator.next();
553                if (current.isVisible()) {
554                    if (current.isClippedToWindow()) {
555                        Shape savedClip = g2.getClip();
556                        g2.clip(this.dialFrame.getWindow(frame));
557                        current.draw(g2, this, frame, area);
558                        g2.setClip(savedClip);
559                    }
560                    else {
561                        current.draw(g2, this, frame, area);
562                    }
563                }
564            }
565    
566            // draw the pointers
567            iterator = this.pointers.iterator();
568            while (iterator.hasNext()) {
569                DialPointer current = (DialPointer) iterator.next();
570                if (current.isVisible()) {
571                    if (current.isClippedToWindow()) {
572                        Shape savedClip = g2.getClip();
573                        g2.clip(this.dialFrame.getWindow(frame));
574                        current.draw(g2, this, frame, area);
575                        g2.setClip(savedClip);
576                    }
577                    else {
578                        current.draw(g2, this, frame, area);
579                    }
580                }
581            }
582    
583            // draw the cap if there is one...
584            if (this.cap != null && this.cap.isVisible()) {
585                if (this.cap.isClippedToWindow()) {
586                    Shape savedClip = g2.getClip();
587                    g2.clip(this.dialFrame.getWindow(frame));
588                    this.cap.draw(g2, this, frame, area);
589                    g2.setClip(savedClip);
590                }
591                else {
592                    this.cap.draw(g2, this, frame, area);
593                }
594            }
595    
596            if (this.dialFrame.isVisible()) {
597                this.dialFrame.draw(g2, this, frame, area);
598            }
599    
600            g2.setClip(origClip);
601    
602        }

604        /**
605         * Returns the frame surrounding the specified view rectangle.
606         *
607         * @param view  the view rectangle (<code>null</code> not permitted).
608         *
609         * @return The frame rectangle.
610         */
611        private Rectangle2D viewToFrame(Rectangle2D view) {
612            double width = view.getWidth() / this.viewW;
613            double height = view.getHeight() / this.viewH;
614            double x = view.getX() - (width * this.viewX);
615            double y = view.getY() - (height * this.viewY);
616            return new Rectangle2D.Double(x, y, width, height);
617        }

699        /**
700         * A utility method that computes a rectangle using relative radius values.
701         *
702         * @param rect  the reference rectangle (<code>null</code> not permitted).
703         * @param radiusW  the width radius (must be > 0.0)
704         * @param radiusH  the height radius.
705         *
706         * @return A new rectangle.
707         */
708        public static Rectangle2D rectangleByRadius(Rectangle2D rect,
709                double radiusW, double radiusH) {
710            ParamChecks.nullNotPermitted(rect, "rect");
711            double x = rect.getCenterX();
712            double y = rect.getCenterY();
713            double w = rect.getWidth() * radiusW;
714            double h = rect.getHeight() * radiusH;
715            return new Rectangle2D.Double(x - w / 2.0, y - h / 2.0, w, h);
716        }

728        /**
729         * Tests this <code>DialPlot</code> instance for equality with an
730         * arbitrary object.  The plot's dataset(s) is (are) not included in
731         * the test.
732         *
733         * @param obj  the object (<code>null</code> permitted).
734         *
735         * @return A boolean.
736         */
737        public boolean equals(Object obj) {
738            if (obj == this) {
739                return true;
740            }
741            if (!(obj instanceof DialPlot)) {
742                return false;
743            }
744            DialPlot that = (DialPlot) obj;
745            if (!ObjectUtilities.equal(this.background, that.background)) {
746                return false;
747            }
748            if (!ObjectUtilities.equal(this.cap, that.cap)) {
749                return false;
750            }
751            if (!this.dialFrame.equals(that.dialFrame)) {
752                return false;
753            }
754            if (this.viewX != that.viewX) {
755                return false;
756            }
757            if (this.viewY != that.viewY) {
758                return false;
759            }
760            if (this.viewW != that.viewW) {
761                return false;
762            }
763            if (this.viewH != that.viewH) {
764                return false;
765            }
766            if (!this.layers.equals(that.layers)) {
