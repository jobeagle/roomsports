555        /**
556         * Draws the background to the specified graphics device.  If the dial
557         * frame specifies a window, the clipping region will already have been
558         * set to this window before this method is called.
559         *
560         * @param g2  the graphics device (<code>null</code> not permitted).
561         * @param plot  the plot (ignored here).
562         * @param frame  the dial frame (ignored here).
563         * @param view  the view rectangle (<code>null</code> not permitted).
564         */
565        public void draw(Graphics2D g2, DialPlot plot, Rectangle2D frame,
566                Rectangle2D view) {
567    
568            // work out the anchor point
569            Rectangle2D f = DialPlot.rectangleByRadius(frame, this.radius,
570                    this.radius);
571            Arc2D arc = new Arc2D.Double(f, this.angle, 0.0, Arc2D.OPEN);
572            Point2D pt = arc.getStartPoint();
573    
574            // the indicator bounds is calculated from the templateValue (which
575            // determines the minimum size), the maxTemplateValue (which, if
576            // specified, provides a maximum size) and the actual value
577            FontMetrics fm = g2.getFontMetrics(this.font);
578            double value = plot.getValue(this.datasetIndex);
579            String valueStr = this.formatter.format(value);
580            Rectangle2D valueBounds = TextUtilities.getTextBounds(valueStr, g2, fm);
581    
582            // calculate the bounds of the template value
583            String s = this.formatter.format(this.templateValue);
584            Rectangle2D tb = TextUtilities.getTextBounds(s, g2, fm);
585            double minW = tb.getWidth();
586            double minH = tb.getHeight();
587    
588            double maxW = Double.MAX_VALUE;
589            double maxH = Double.MAX_VALUE;
590            if (this.maxTemplateValue != null) {
591                s = this.formatter.format(this.maxTemplateValue);
592                tb = TextUtilities.getTextBounds(s, g2, fm);
593                maxW = Math.max(tb.getWidth(), minW);
594                maxH = Math.max(tb.getHeight(), minH);
595            }
596            double w = fixToRange(valueBounds.getWidth(), minW, maxW);
597            double h = fixToRange(valueBounds.getHeight(), minH, maxH);
598    
599            // align this rectangle to the frameAnchor
600            Rectangle2D bounds = RectangleAnchor.createRectangle(new Size2D(w, h),
601                    pt.getX(), pt.getY(), this.frameAnchor);
602    
603            // add the insets
604            Rectangle2D fb = this.insets.createOutsetRectangle(bounds);
605    
606            // draw the background
607            g2.setPaint(this.backgroundPaint);
608            g2.fill(fb);
609    
610            // draw the border
611            g2.setStroke(this.outlineStroke);
612            g2.setPaint(this.outlinePaint);
613            g2.draw(fb);
614    
615            // now find the text anchor point
616            Shape savedClip = g2.getClip();
617            g2.clip(fb);
618    
619            Point2D pt2 = RectangleAnchor.coordinates(bounds, this.valueAnchor);
620            g2.setPaint(this.paint);
621            g2.setFont(this.font);
622            TextUtilities.drawAlignedString(valueStr, g2, (float) pt2.getX(),
623                    (float) pt2.getY(), this.textAnchor);
624            g2.setClip(savedClip);
625    
626        }

628        /**
629         * A utility method that adjusts a value, if necessary, to be within a 
630         * specified range.
631         * 
632         * @param x  the value.
633         * @param minX  the minimum value in the range.
634         * @param maxX  the maximum value in the range.
635         * 
636         * @return The adjusted value.
637         */
638        private double fixToRange(double x, double minX, double maxX) {
639            if (minX > maxX) {
640                throw new IllegalArgumentException("Requires 'minX' <= 'maxX'.");
641            }
642            if (x < minX) {
643                return minX;
644            }
645            else if (x > maxX) {
646                return maxX;
647            }
648            else {
649                return x;
650            }
651        }

653        /**
654         * Tests this instance for equality with an arbitrary object.
655         *
656         * @param obj  the object (<code>null</code> permitted).
657         *
658         * @return A boolean.
659         */
660        public boolean equals(Object obj) {
661            if (obj == this) {
662                return true;
663            }
664            if (!(obj instanceof DialValueIndicator)) {
665                return false;
666            }
667            DialValueIndicator that = (DialValueIndicator) obj;
668            if (this.datasetIndex != that.datasetIndex) {
669                return false;
670            }
671            if (this.angle != that.angle) {
672                return false;
673            }
674            if (this.radius != that.radius) {
675                return false;
676            }
677            if (!this.frameAnchor.equals(that.frameAnchor)) {
678                return false;
679            }
680            if (!this.templateValue.equals(that.templateValue)) {
681                return false;
682            }
683            if (!ObjectUtilities.equal(this.maxTemplateValue,
684                    that.maxTemplateValue)) {
685                return false;
686            }
687            if (!this.font.equals(that.font)) {
688                return false;
689            }
690            if (!PaintUtilities.equal(this.paint, that.paint)) {
691                return false;
692            }
693            if (!PaintUtilities.equal(this.backgroundPaint, that.backgroundPaint)) {
694                return false;
695            }
696            if (!this.outlineStroke.equals(that.outlineStroke)) {
697                return false;
698            }
699            if (!PaintUtilities.equal(this.outlinePaint, that.outlinePaint)) {
700                return false;
701            }
702            if (!this.insets.equals(that.insets)) {
703                return false;
704            }
705            if (!this.valueAnchor.equals(that.valueAnchor)) {
706                return false;
707            }
708            if (!this.textAnchor.equals(that.textAnchor)) {
709                return false;
710            }
711            return super.equals(obj);
712        }
