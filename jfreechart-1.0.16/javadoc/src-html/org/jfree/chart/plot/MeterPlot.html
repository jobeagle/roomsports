236        /**
237         * Creates a new plot that displays the value from the supplied dataset.
238         *
239         * @param dataset  the dataset (<code>null</code> permitted).
240         */
241        public MeterPlot(ValueDataset dataset) {
242            super();
243            this.shape = DialShape.CIRCLE;
244            this.meterAngle = DEFAULT_METER_ANGLE;
245            this.range = new Range(0.0, 100.0);
246            this.tickSize = 10.0;
247            this.tickPaint = Color.white;
248            this.units = "Units";
249            this.needlePaint = MeterPlot.DEFAULT_NEEDLE_PAINT;
250            this.tickLabelsVisible = true;
251            this.tickLabelFont = MeterPlot.DEFAULT_LABEL_FONT;
252            this.tickLabelPaint = Color.black;
253            this.tickLabelFormat = NumberFormat.getInstance();
254            this.valueFont = MeterPlot.DEFAULT_VALUE_FONT;
255            this.valuePaint = MeterPlot.DEFAULT_VALUE_PAINT;
256            this.dialBackgroundPaint = MeterPlot.DEFAULT_DIAL_BACKGROUND_PAINT;
257            this.intervals = new java.util.ArrayList();
258            setDataset(dataset);
259        }

688        /**
689         * Sets the dataset for the plot, replacing the existing dataset if there
690         * is one, and triggers a {@link PlotChangeEvent}.
691         *
692         * @param dataset  the dataset (<code>null</code> permitted).
693         *
694         * @see #getDataset()
695         */
696        public void setDataset(ValueDataset dataset) {
697    
698            // if there is an existing dataset, remove the plot from the list of
699            // change listeners...
700            ValueDataset existing = this.dataset;
701            if (existing != null) {
702                existing.removeChangeListener(this);
703            }
704    
705            // set the new dataset, and register the chart as a change listener...
706            this.dataset = dataset;
707            if (dataset != null) {
708                setDatasetGroup(dataset.getGroup());
709                dataset.addChangeListener(this);
710            }
711    
712            // send a dataset change event to self...
713            DatasetChangeEvent event = new DatasetChangeEvent(this, dataset);
714            datasetChanged(event);
715    
716        }

778        /**
779         * Draws the plot on a Java 2D graphics device (such as the screen or a
780         * printer).
781         *
782         * @param g2  the graphics device.
783         * @param area  the area within which the plot should be drawn.
784         * @param anchor  the anchor point (<code>null</code> permitted).
785         * @param parentState  the state from the parent plot, if there is one.
786         * @param info  collects info about the drawing.
787         */
788        public void draw(Graphics2D g2, Rectangle2D area, Point2D anchor,
789                         PlotState parentState,
790                         PlotRenderingInfo info) {
791    
792            if (info != null) {
793                info.setPlotArea(area);
794            }
795    
796            // adjust for insets...
797            RectangleInsets insets = getInsets();
798            insets.trim(area);
799    
800            area.setRect(area.getX() + 4, area.getY() + 4, area.getWidth() - 8,
801                    area.getHeight() - 8);
802    
803            // draw the background
804            if (this.drawBorder) {
805                drawBackground(g2, area);
806            }
807    
808            // adjust the plot area by the interior spacing value
809            double gapHorizontal = (2 * DEFAULT_BORDER_SIZE);
810            double gapVertical = (2 * DEFAULT_BORDER_SIZE);
811            double meterX = area.getX() + gapHorizontal / 2;
812            double meterY = area.getY() + gapVertical / 2;
813            double meterW = area.getWidth() - gapHorizontal;
814            double meterH = area.getHeight() - gapVertical
815                    + ((this.meterAngle <= 180) && (this.shape != DialShape.CIRCLE)
816                    ? area.getHeight() / 1.25 : 0);
817    
818            double min = Math.min(meterW, meterH) / 2;
819            meterX = (meterX + meterX + meterW) / 2 - min;
820            meterY = (meterY + meterY + meterH) / 2 - min;
821            meterW = 2 * min;
822            meterH = 2 * min;
823    
824            Rectangle2D meterArea = new Rectangle2D.Double(meterX, meterY, meterW,
825                    meterH);
826    
827            Rectangle2D.Double originalArea = new Rectangle2D.Double(
828                    meterArea.getX() - 4, meterArea.getY() - 4,
829                    meterArea.getWidth() + 8, meterArea.getHeight() + 8);
830    
831            double meterMiddleX = meterArea.getCenterX();
832            double meterMiddleY = meterArea.getCenterY();
833    
834            // plot the data (unless the dataset is null)...
835            ValueDataset data = getDataset();
836            if (data != null) {
837                double dataMin = this.range.getLowerBound();
838                double dataMax = this.range.getUpperBound();
839    
840                Shape savedClip = g2.getClip();
841                g2.clip(originalArea);
842                Composite originalComposite = g2.getComposite();
843                g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
844                        getForegroundAlpha()));
845    
846                if (this.dialBackgroundPaint != null) {
847                    fillArc(g2, originalArea, dataMin, dataMax,
848                            this.dialBackgroundPaint, true);
849                }
850                drawTicks(g2, meterArea, dataMin, dataMax);
851                drawArcForInterval(g2, meterArea, new MeterInterval("", this.range,
852                        this.dialOutlinePaint, new BasicStroke(1.0f), null));
853    
854                Iterator iterator = this.intervals.iterator();
855                while (iterator.hasNext()) {
856                    MeterInterval interval = (MeterInterval) iterator.next();
857                    drawArcForInterval(g2, meterArea, interval);
858                }
859    
860                Number n = data.getValue();
861                if (n != null) {
862                    double value = n.doubleValue();
863                    drawValueLabel(g2, meterArea);
864    
865                    if (this.range.contains(value)) {
866                        g2.setPaint(this.needlePaint);
867                        g2.setStroke(new BasicStroke(2.0f));
868    
869                        double radius = (meterArea.getWidth() / 2)
870                                        + DEFAULT_BORDER_SIZE + 15;
871                        double valueAngle = valueToAngle(value);
872                        double valueP1 = meterMiddleX
873                                + (radius * Math.cos(Math.PI * (valueAngle / 180)));
874                        double valueP2 = meterMiddleY
875                                - (radius * Math.sin(Math.PI * (valueAngle / 180)));
876    
877                        Polygon arrow = new Polygon();
878                        if ((valueAngle > 135 && valueAngle < 225)
879                            || (valueAngle < 45 && valueAngle > -45)) {
880    
881                            double valueP3 = (meterMiddleY
882                                    - DEFAULT_CIRCLE_SIZE / 4);
883                            double valueP4 = (meterMiddleY
884                                    + DEFAULT_CIRCLE_SIZE / 4);
885                            arrow.addPoint((int) meterMiddleX, (int) valueP3);
886                            arrow.addPoint((int) meterMiddleX, (int) valueP4);
887    
888                        }
889                        else {
890                            arrow.addPoint((int) (meterMiddleX
891                                    - DEFAULT_CIRCLE_SIZE / 4), (int) meterMiddleY);
892                            arrow.addPoint((int) (meterMiddleX
893                                    + DEFAULT_CIRCLE_SIZE / 4), (int) meterMiddleY);
894                        }
895                        arrow.addPoint((int) valueP1, (int) valueP2);
896                        g2.fill(arrow);
897    
898                        Ellipse2D circle = new Ellipse2D.Double(meterMiddleX
899                                - DEFAULT_CIRCLE_SIZE / 2, meterMiddleY
900                                - DEFAULT_CIRCLE_SIZE / 2, DEFAULT_CIRCLE_SIZE,
901                                DEFAULT_CIRCLE_SIZE);
902                        g2.fill(circle);
903                    }
904                }
905    
906                g2.setClip(savedClip);
907                g2.setComposite(originalComposite);
908    
909            }
910            if (this.drawBorder) {
911                drawOutline(g2, area);
912            }
913    
914        }

917         * Draws the arc to represent an interval.
918         *
919         * @param g2  the graphics device.
920         * @param meterArea  the drawing area.
921         * @param interval  the interval.
922         */
923        protected void drawArcForInterval(Graphics2D g2, Rectangle2D meterArea,
924                                          MeterInterval interval) {
925    
926            double minValue = interval.getRange().getLowerBound();
927            double maxValue = interval.getRange().getUpperBound();
928            Paint outlinePaint = interval.getOutlinePaint();
929            Stroke outlineStroke = interval.getOutlineStroke();
930            Paint backgroundPaint = interval.getBackgroundPaint();
931    
932            if (backgroundPaint != null) {
933                fillArc(g2, meterArea, minValue, maxValue, backgroundPaint, false);
934            }
935            if (outlinePaint != null) {
936                if (outlineStroke != null) {
937                    drawArc(g2, meterArea, minValue, maxValue, outlinePaint,
938                            outlineStroke);
939                }
940                drawTick(g2, meterArea, minValue, true);
941                drawTick(g2, meterArea, maxValue, true);
942            }
943        }

946         * Draws an arc.
947         *
948         * @param g2  the graphics device.
949         * @param area  the plot area.
950         * @param minValue  the minimum value.
951         * @param maxValue  the maximum value.
952         * @param paint  the paint.
953         * @param stroke  the stroke.
954         */
955        protected void drawArc(Graphics2D g2, Rectangle2D area, double minValue,
956                               double maxValue, Paint paint, Stroke stroke) {
957    
958            double startAngle = valueToAngle(maxValue);
959            double endAngle = valueToAngle(minValue);
960            double extent = endAngle - startAngle;
961    
962            double x = area.getX();
963            double y = area.getY();
964            double w = area.getWidth();
965            double h = area.getHeight();
966            g2.setPaint(paint);
967            g2.setStroke(stroke);
968    
969            if (paint != null && stroke != null) {
970                Arc2D.Double arc = new Arc2D.Double(x, y, w, h, startAngle,
971                        extent, Arc2D.OPEN);
972                g2.setPaint(paint);
973                g2.setStroke(stroke);
974                g2.draw(arc);
975            }
976    
977        }

979        /**
980         * Fills an arc on the dial between the given values.
981         *
982         * @param g2  the graphics device.
983         * @param area  the plot area.
984         * @param minValue  the minimum data value.
985         * @param maxValue  the maximum data value.
986         * @param paint  the background paint (<code>null</code> not permitted).
987         * @param dial  a flag that indicates whether the arc represents the whole
988         *              dial.
989         */
990        protected void fillArc(Graphics2D g2, Rectangle2D area,
991                double minValue, double maxValue, Paint paint, boolean dial) {
992    
993            ParamChecks.nullNotPermitted(paint, "paint");
994            double startAngle = valueToAngle(maxValue);
995            double endAngle = valueToAngle(minValue);
996            double extent = endAngle - startAngle;
997    
998            double x = area.getX();
999            double y = area.getY();
1000            double w = area.getWidth();
1001            double h = area.getHeight();
1002            int joinType = Arc2D.OPEN;
1003            if (this.shape == DialShape.PIE) {
1004                joinType = Arc2D.PIE;
1005            }
1006            else if (this.shape == DialShape.CHORD) {
1007                if (dial && this.meterAngle > 180) {
1008                    joinType = Arc2D.CHORD;
1009                }
1010                else {
1011                    joinType = Arc2D.PIE;
1012                }
1013            }
1014            else if (this.shape == DialShape.CIRCLE) {
1015                joinType = Arc2D.PIE;
1016                if (dial) {
1017                    extent = 360;
1018                }
1019            }
1020            else {
1021                throw new IllegalStateException("DialShape not recognised.");
1022            }
1023    
1024            g2.setPaint(paint);
1025            Arc2D.Double arc = new Arc2D.Double(x, y, w, h, startAngle, extent,
1026                    joinType);
1027            g2.fill(arc);
1028        }

1030        /**
1031         * Translates a data value to an angle on the dial.
1032         *
1033         * @param value  the value.
1034         *
1035         * @return The angle on the dial.
1036         */
1037        public double valueToAngle(double value) {
1038            value = value - this.range.getLowerBound();
1039            double baseAngle = 180 + ((this.meterAngle - 180) / 2);
1040            return baseAngle - ((value / this.range.getLength()) * this.meterAngle);
1041        }

1070        /**
1071         * Draws a tick on the dial.
1072         *
1073         * @param g2  the graphics device.
1074         * @param meterArea  the meter area.
1075         * @param value  the tick value.
1076         * @param label  a flag that controls whether or not a value label is drawn.
1077         */
1078        protected void drawTick(Graphics2D g2, Rectangle2D meterArea,
1079                                double value, boolean label) {
1080    
1081            double valueAngle = valueToAngle(value);
1082    
1083            double meterMiddleX = meterArea.getCenterX();
1084            double meterMiddleY = meterArea.getCenterY();
1085    
1086            g2.setPaint(this.tickPaint);
1087            g2.setStroke(new BasicStroke(2.0f));
1088    
1089            double valueP2X = 0;
1090            double valueP2Y = 0;
1091    
1092            double radius = (meterArea.getWidth() / 2) + DEFAULT_BORDER_SIZE;
1093            double radius1 = radius - 15;
1094    
1095            double valueP1X = meterMiddleX
1096                    + (radius * Math.cos(Math.PI * (valueAngle / 180)));
1097            double valueP1Y = meterMiddleY
1098                    - (radius * Math.sin(Math.PI * (valueAngle / 180)));
1099    
1100            valueP2X = meterMiddleX
1101                    + (radius1 * Math.cos(Math.PI * (valueAngle / 180)));
1102            valueP2Y = meterMiddleY
1103                    - (radius1 * Math.sin(Math.PI * (valueAngle / 180)));
1104    
1105            Line2D.Double line = new Line2D.Double(valueP1X, valueP1Y, valueP2X,
1106                    valueP2Y);
1107            g2.draw(line);
1108    
1109            if (this.tickLabelsVisible && label) {
1110    
1111                String tickLabel =  this.tickLabelFormat.format(value);
1112                g2.setFont(this.tickLabelFont);
1113                g2.setPaint(this.tickLabelPaint);
1114    
1115                FontMetrics fm = g2.getFontMetrics();
1116                Rectangle2D tickLabelBounds
1117                    = TextUtilities.getTextBounds(tickLabel, g2, fm);
1118    
1119                double x = valueP2X;
1120                double y = valueP2Y;
1121                if (valueAngle == 90 || valueAngle == 270) {
1122                    x = x - tickLabelBounds.getWidth() / 2;
1123                }
1124                else if (valueAngle < 90 || valueAngle > 270) {
1125                    x = x - tickLabelBounds.getWidth();
1126                }
1127                if ((valueAngle > 135 && valueAngle < 225)
1128                        || valueAngle > 315 || valueAngle < 45) {
1129                    y = y - tickLabelBounds.getHeight() / 2;
1130                }
1131                else {
1132                    y = y + tickLabelBounds.getHeight() / 2;
1133                }
1134                g2.drawString(tickLabel, (float) x, (float) y);
1135            }
1136        }
