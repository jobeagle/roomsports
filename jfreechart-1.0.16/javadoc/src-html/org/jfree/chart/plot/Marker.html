158        /**
159         * Constructs a new marker.
160         *
161         * @param paint  the paint (<code>null</code> not permitted).
162         * @param stroke  the stroke (<code>null</code> not permitted).
163         * @param outlinePaint  the outline paint (<code>null</code> permitted).
164         * @param outlineStroke  the outline stroke (<code>null</code> permitted).
165         * @param alpha  the alpha transparency (must be in the range 0.0f to
166         *     1.0f).
167         *
168         * @throws IllegalArgumentException if <code>paint</code> or
169         *     <code>stroke</code> is <code>null</code>, or <code>alpha</code> is
170         *     not in the specified range.
171         */
172        protected Marker(Paint paint, Stroke stroke, Paint outlinePaint, 
173                Stroke outlineStroke, float alpha) {
174    
175            ParamChecks.nullNotPermitted(paint, "paint");
176            ParamChecks.nullNotPermitted(stroke, "stroke");
177            if (alpha < 0.0f || alpha > 1.0f) {
178                throw new IllegalArgumentException(
179                        "The 'alpha' value must be in the range 0.0f to 1.0f");
180            }
181    
182            this.paint = paint;
183            this.stroke = stroke;
184            this.outlinePaint = outlinePaint;
185            this.outlineStroke = outlineStroke;
186            this.alpha = alpha;
187    
188            this.labelFont = new Font("SansSerif", Font.PLAIN, 9);
189            this.labelPaint = Color.black;
190            this.labelAnchor = RectangleAnchor.TOP_LEFT;
191            this.labelOffset = new RectangleInsets(3.0, 3.0, 3.0, 3.0);
192            this.labelOffsetType = LengthAdjustmentType.CONTRACT;
193            this.labelTextAnchor = TextAnchor.CENTER;
194    
195            this.listenerList = new EventListenerList();
196        }

532        /**
533         * Notifies all registered listeners that the marker has been modified.
534         *
535         * @param event  information about the change event.
536         *
537         * @since 1.0.3
538         */
539        public void notifyListeners(MarkerChangeEvent event) {
540    
541            Object[] listeners = this.listenerList.getListenerList();
542            for (int i = listeners.length - 2; i >= 0; i -= 2) {
543                if (listeners[i] == MarkerChangeListener.class) {
544                    ((MarkerChangeListener) listeners[i + 1]).markerChanged(event);
545                }
546            }
547    
548        }

563        /**
564         * Tests the marker for equality with an arbitrary object.
565         *
566         * @param obj  the object (<code>null</code> permitted).
567         *
568         * @return A boolean.
569         */
570        public boolean equals(Object obj) {
571            if (obj == this) {
572                return true;
573            }
574            if (!(obj instanceof Marker)) {
575                return false;
576            }
577            Marker that = (Marker) obj;
578            if (!PaintUtilities.equal(this.paint, that.paint)) {
579                return false;
580            }
581            if (!ObjectUtilities.equal(this.stroke, that.stroke)) {
582                return false;
583            }
584            if (!PaintUtilities.equal(this.outlinePaint, that.outlinePaint)) {
585                return false;
586            }
587            if (!ObjectUtilities.equal(this.outlineStroke, that.outlineStroke)) {
588                return false;
589            }
590            if (this.alpha != that.alpha) {
591                return false;
592            }
593            if (!ObjectUtilities.equal(this.label, that.label)) {
594                return false;
595            }
596            if (!ObjectUtilities.equal(this.labelFont, that.labelFont)) {
597                return false;
598            }
599            if (!PaintUtilities.equal(this.labelPaint, that.labelPaint)) {
600                return false;
601            }
602            if (this.labelAnchor != that.labelAnchor) {
603                return false;
604            }
605            if (this.labelTextAnchor != that.labelTextAnchor) {
606                return false;
607            }
608            if (!ObjectUtilities.equal(this.labelOffset, that.labelOffset)) {
609                return false;
610            }
611            if (!this.labelOffsetType.equals(that.labelOffsetType)) {
612                return false;
613            }
614            return true;
615        }

628        /**
629         * Provides serialization support.
630         *
631         * @param stream  the output stream.
632         *
633         * @throws IOException  if there is an I/O error.
634         */
635        private void writeObject(ObjectOutputStream stream) throws IOException {
636            stream.defaultWriteObject();
637            SerialUtilities.writePaint(this.paint, stream);
638            SerialUtilities.writeStroke(this.stroke, stream);
639            SerialUtilities.writePaint(this.outlinePaint, stream);
640            SerialUtilities.writeStroke(this.outlineStroke, stream);
641            SerialUtilities.writePaint(this.labelPaint, stream);
642        }
643    
644        /**
645         * Provides serialization support.
646         *
647         * @param stream  the input stream.
648         *
649         * @throws IOException  if there is an I/O error.
650         * @throws ClassNotFoundException  if there is a classpath problem.
651         */
652        private void readObject(ObjectInputStream stream)
653            throws IOException, ClassNotFoundException {
654            stream.defaultReadObject();
655            this.paint = SerialUtilities.readPaint(stream);
656            this.stroke = SerialUtilities.readStroke(stream);
657            this.outlinePaint = SerialUtilities.readPaint(stream);
658            this.outlineStroke = SerialUtilities.readStroke(stream);
659            this.labelPaint = SerialUtilities.readPaint(stream);
660            this.listenerList = new EventListenerList();
661        }
