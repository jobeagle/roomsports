137         * @param text  the text for the label (<code>null</code> not permitted).
138         * @param font  the font (<code>null</code> not permitted).
139         */
140        public LabelBlock(String text, Font font) {

142        }

144        /**
145         * Creates a new label block.
146         *
147         * @param text  the text for the label (<code>null</code> not permitted).
148         * @param font  the font (<code>null</code> not permitted).
149         * @param paint the paint (<code>null</code> not permitted).
150         */
151        public LabelBlock(String text, Font font, Paint paint) {
152            this.text = text;
153            this.paint = paint;
154            this.label = TextUtilities.createTextBlock(text, font, this.paint);
155            this.font = font;
156            this.toolTipText = null;
157            this.urlText = null;
158            this.contentAlignmentPoint = TextBlockAnchor.CENTER;
159            this.textAnchor = RectangleAnchor.CENTER;
160        }

301        /**
302         * Arranges the contents of the block, within the given constraints, and
303         * returns the block size.
304         *
305         * @param g2  the graphics device.
306         * @param constraint  the constraint (<code>null</code> not permitted).
307         *
308         * @return The block size (in Java2D units, never <code>null</code>).
309         */
310        public Size2D arrange(Graphics2D g2, RectangleConstraint constraint) {
311            g2.setFont(this.font);
312            Size2D s = this.label.calculateDimensions(g2);
313            return new Size2D(calculateTotalWidth(s.getWidth()),
314                    calculateTotalHeight(s.getHeight()));
315        }

317        /**
318         * Draws the block.
319         *
320         * @param g2  the graphics device.
321         * @param area  the area.
322         */
323        public void draw(Graphics2D g2, Rectangle2D area) {
324            draw(g2, area, null);
325        }
326    
327        /**
328         * Draws the block within the specified area.
329         *
330         * @param g2  the graphics device.
331         * @param area  the area.
332         * @param params  ignored (<code>null</code> permitted).
333         *
334         * @return Always <code>null</code>.
335         */
336        public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
337            area = trimMargin(area);
338            drawBorder(g2, area);
339            area = trimBorder(area);
340            area = trimPadding(area);
341    
342            // check if we need to collect chart entities from the container
343            EntityBlockParams ebp = null;
344            StandardEntityCollection sec = null;
345            Shape entityArea = null;
346            if (params instanceof EntityBlockParams) {
347                ebp = (EntityBlockParams) params;
348                if (ebp.getGenerateEntities()) {
349                    sec = new StandardEntityCollection();
350                    entityArea = (Shape) area.clone();
351                }
352            }
353            g2.setPaint(this.paint);
354            g2.setFont(this.font);
355            Point2D pt = RectangleAnchor.coordinates(area, this.textAnchor);
356            this.label.draw(g2, (float) pt.getX(), (float) pt.getY(),
357                    this.contentAlignmentPoint);
358            BlockResult result = null;
359            if (ebp != null && sec != null) {
360                if (this.toolTipText != null || this.urlText != null) {
361                    ChartEntity entity = new ChartEntity(entityArea,
362                            this.toolTipText, this.urlText);
363                    sec.add(entity);
364                    result = new BlockResult();
365                    result.setEntityCollection(sec);
366                }
367            }
368            return result;
369        }
370    
371        /**
372         * Tests this <code>LabelBlock</code> for equality with an arbitrary
373         * object.
374         *
375         * @param obj  the object (<code>null</code> permitted).
376         *
377         * @return A boolean.
378         */
379        public boolean equals(Object obj) {
380            if (!(obj instanceof LabelBlock)) {
381                return false;
382            }
383            LabelBlock that = (LabelBlock) obj;
384            if (!this.text.equals(that.text)) {
385                return false;
386            }
387            if (!this.font.equals(that.font)) {
388                return false;
389            }
390            if (!PaintUtilities.equal(this.paint, that.paint)) {
391                return false;
392            }
393            if (!ObjectUtilities.equal(this.toolTipText, that.toolTipText)) {
394                return false;
395            }
396            if (!ObjectUtilities.equal(this.urlText, that.urlText)) {
397                return false;
398            }
399            if (!this.contentAlignmentPoint.equals(that.contentAlignmentPoint)) {
400                return false;
401            }
402            if (!this.textAnchor.equals(that.textAnchor)) {
403                return false;
404            }
405            return super.equals(obj);
406        }

442            this.paint = SerialUtilities.readPaint(stream);
