052    package org.jfree.chart.urls;
053    
054    import java.io.Serializable;
055    import org.jfree.chart.util.ParamChecks;
056    
057    import org.jfree.data.general.PieDataset;
058    import org.jfree.util.ObjectUtilities;
059    
060    /**
061     * A URL generator for pie charts.  Instances of this class are immutable.
062     */
063    public class StandardPieURLGenerator implements PieURLGenerator, Serializable {
064    
065        /** For serialization. */
066        private static final long serialVersionUID = 1626966402065883419L;
067    
068        /** The prefix. */
069        private String prefix = "index.html";
070    
071        /** The category parameter name. */
072        private String categoryParameterName = "category";
073    
074        /** The pie index parameter name. */
075        private String indexParameterName = "pieIndex";
076    
077        /**
078         * Default constructor.
079         */
080        public StandardPieURLGenerator() {
081            this("index.html");
082        }
083    
084        /**
085         * Creates a new generator.
086         *
087         * @param prefix  the prefix (<code>null</code> not permitted).
088         */
089        public StandardPieURLGenerator(String prefix) {
090            this(prefix, "category");
091        }
092    
093        /**
094         * Creates a new generator.
095         *
096         * @param prefix  the prefix (<code>null</code> not permitted).
097         * @param categoryParameterName  the category parameter name
098         *     (<code>null</code> not permitted).
099         */
100        public StandardPieURLGenerator(String prefix,
101                                       String categoryParameterName) {
102            this(prefix, categoryParameterName, "pieIndex");
103        }
104    
105        /**
106         * Creates a new generator.
107         *
108         * @param prefix  the prefix (<code>null</code> not permitted).
109         * @param categoryParameterName  the category parameter name
110         *     (<code>null</code> not permitted).
111         * @param indexParameterName  the index parameter name (<code>null</code>
112         *     permitted).
113         */
114        public StandardPieURLGenerator(String prefix, String categoryParameterName,
115                String indexParameterName) {
116            ParamChecks.nullNotPermitted(prefix, "prefix");
117            ParamChecks.nullNotPermitted(categoryParameterName, "categoryParameterName");
118            this.prefix = prefix;
119            this.categoryParameterName = categoryParameterName;
120            this.indexParameterName = indexParameterName;
121        }
122    
123        /**
124         * Generates a URL.
125         *
126         * @param dataset  the dataset (ignored).
127         * @param key  the item key (<code>null</code> not permitted).
128         * @param pieIndex  the pie index.
129         *
130         * @return A string containing the generated URL.
131         */
132        public String generateURL(PieDataset dataset, Comparable key,
133                int pieIndex) {
134            String url = this.prefix;
135            if (url.indexOf("?") > -1) {
136                url += "&amp;" + this.categoryParameterName + "="
137                        + URLUtilities.encode(key.toString(), "UTF-8");
138            }
139            else {
140                url += "?" + this.categoryParameterName + "="
141                        + URLUtilities.encode(key.toString(), "UTF-8");
142            }
143            if (this.indexParameterName != null) {
144                url += "&amp;" + this.indexParameterName + "="
145                       + String.valueOf(pieIndex);
146            }
147            return url;
148        }
149    
150        /**
151         * Tests if this object is equal to another.
152         *
153         * @param obj  the object (<code>null</code> permitted).
154         *
155         * @return A boolean.
156         */
157        public boolean equals(Object obj) {
158            if (obj == this) {
159                return true;
160            }
161            if (!(obj instanceof StandardPieURLGenerator)) {
162                return false;
163            }
164            StandardPieURLGenerator that = (StandardPieURLGenerator) obj;
165            if (!this.prefix.equals(that.prefix)) {
166                return false;
167            }
168            if (!this.categoryParameterName.equals(that.categoryParameterName)) {
169                return false;
170            }
171            if (!ObjectUtilities.equal(this.indexParameterName,
172                    that.indexParameterName)) {
173                return false;
174            }
175            return true;
176        }
177    }
