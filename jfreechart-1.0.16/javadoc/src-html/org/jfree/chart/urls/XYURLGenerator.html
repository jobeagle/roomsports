046    package org.jfree.chart.urls;
047    
048    import org.jfree.data.xy.XYDataset;
049    
050    /**
051     * Interface for a URL generator for plots that uses data from an
052     * {@link XYDataset}.  Classes that implement this interface are responsible
053     * for correctly escaping any text that is derived from the dataset, as this
054     * may be user-specified and could pose a security risk.
055     */
056    public interface XYURLGenerator {
057    
058        /**
059         * Generates a URL for a particular item within a series. As a guideline,
060         * the URL should be valid within the context of an XHTML 1.0 document.
061         *
062         * @param dataset  the dataset (<code>null</code> not permitted).
063         * @param series  the series index (zero-based).
064         * @param item  the item index (zero-based).
065         *
066         * @return A string containing the generated URL (possibly
067         *         <code>null</code>).
068         */
069        public String generateURL(XYDataset dataset, int series, int item);
070    
071    }
