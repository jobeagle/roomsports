258        public static void setChartTheme(ChartTheme theme) {
259            ParamChecks.nullNotPermitted(theme, "theme");
260            currentTheme = theme;
261    
262            // here we do a check to see if the user is installing the "Legacy"
263            // theme, and reset the bar painters in that case...
264            if (theme instanceof StandardChartTheme) {
265                StandardChartTheme sct = (StandardChartTheme) theme;
266                if (sct.getName().equals("Legacy")) {
267                    BarRenderer.setDefaultBarPainter(new StandardBarPainter());
268                    XYBarRenderer.setDefaultBarPainter(new StandardXYBarPainter());
269                }
270                else {
271                    BarRenderer.setDefaultBarPainter(new GradientBarPainter());
272                    XYBarRenderer.setDefaultBarPainter(new GradientXYBarPainter());
273                }
274            }

358        /**
359         * Creates a pie chart with default settings that compares 2 datasets.
360         * The colour of each section will be determined by the move from the value
361         * for the same key in <code>previousDataset</code>. ie if value1 > value2
362         * then the section will be in green (unless <code>greenForIncrease</code>
363         * is <code>false</code>, in which case it would be <code>red</code>).
364         * Each section can have a shade of red or green as the difference can be
365         * tailored between 0% (black) and percentDiffForMaxScale% (bright
366         * red/green).
367         * <p>
368         * For instance if <code>percentDiffForMaxScale</code> is 10 (10%), a
369         * difference of 5% will have a half shade of red/green, a difference of
370         * 10% or more will have a maximum shade/brightness of red/green.
371         * <P>
372         * The chart object returned by this method uses a {@link PiePlot} instance
373         * as the plot.
374         * <p>
375         * Written by <a href="mailto:opensource@objectlab.co.uk">Benoit
376         * Xhenseval</a>.
377         *
378         * @param title  the chart title (<code>null</code> permitted).
379         * @param dataset  the dataset for the chart (<code>null</code> permitted).
380         * @param previousDataset  the dataset for the last run, this will be used
381         *                         to compare each key in the dataset
382         * @param percentDiffForMaxScale scale goes from bright red/green to black,
383         *                               percentDiffForMaxScale indicate the change
384         *                               required to reach top scale.
385         * @param greenForIncrease  an increase since previousDataset will be
386         *                          displayed in green (decrease red) if true.
387         * @param legend  a flag specifying whether or not a legend is required.
388         * @param tooltips  configure chart to generate tool tips?
389         * @param locale  the locale (<code>null</code> not permitted).
390         * @param subTitle displays a subtitle with colour scheme if true
391         * @param showDifference  create a new dataset that will show the %
392         *                        difference between the two datasets.
393         *
394         * @return A pie chart.
395         *
396         * @since 1.0.7
397         */
398        public static JFreeChart createPieChart(String title, PieDataset dataset,
399                PieDataset previousDataset, int percentDiffForMaxScale,
400                boolean greenForIncrease, boolean legend, boolean tooltips,
401                Locale locale, boolean subTitle, boolean showDifference) {
402    
403            PiePlot plot = new PiePlot(dataset);
404            plot.setLabelGenerator(new StandardPieSectionLabelGenerator(locale));
405            plot.setInsets(new RectangleInsets(0.0, 5.0, 5.0, 5.0));
406    
407            if (tooltips) {
408                plot.setToolTipGenerator(new StandardPieToolTipGenerator(locale));
409            }
410    
411            List keys = dataset.getKeys();
412            DefaultPieDataset series = null;
413            if (showDifference) {
414                series = new DefaultPieDataset();
415            }
416    
417            double colorPerPercent = 255.0 / percentDiffForMaxScale;
418            for (Iterator it = keys.iterator(); it.hasNext();) {
419                Comparable key = (Comparable) it.next();
420                Number newValue = dataset.getValue(key);
421                Number oldValue = previousDataset.getValue(key);
422    
423                if (oldValue == null) {
424                    if (greenForIncrease) {
425                        plot.setSectionPaint(key, Color.green);
426                    }
427                    else {
428                        plot.setSectionPaint(key, Color.red);
429                    }
430                    if (showDifference) {
431                        series.setValue(key + " (+100%)", newValue);
432                    }
433                }
434                else {
435                    double percentChange = (newValue.doubleValue()
436                            / oldValue.doubleValue() - 1.0) * 100.0;
437                    double shade
438                        = (Math.abs(percentChange) >= percentDiffForMaxScale ? 255
439                        : Math.abs(percentChange) * colorPerPercent);
440                    if (greenForIncrease
441                            && newValue.doubleValue() > oldValue.doubleValue()
442                            || !greenForIncrease && newValue.doubleValue()
443                            < oldValue.doubleValue()) {
444                        plot.setSectionPaint(key, new Color(0, (int) shade, 0));
445                    }
446                    else {
447                        plot.setSectionPaint(key, new Color((int) shade, 0, 0));
448                    }
449                    if (showDifference) {
450                        series.setValue(key + " (" + (percentChange >= 0 ? "+" : "")
451                                + NumberFormat.getPercentInstance().format(
452                                percentChange / 100.0) + ")", newValue);
453                    }
454                }
455            }
456    
457            if (showDifference) {
458                plot.setDataset(series);
459            }
460    
461            JFreeChart chart =  new JFreeChart(title,
462                    JFreeChart.DEFAULT_TITLE_FONT, plot, legend);
463    
464            if (subTitle) {
465                TextTitle subtitle = new TextTitle("Bright " + (greenForIncrease 
466                        ? "red" : "green") + "=change >=-" + percentDiffForMaxScale
467                        + "%, Bright " + (!greenForIncrease ? "red" : "green")
468                        + "=change >=+" + percentDiffForMaxScale + "%",
469                        new Font("SansSerif", Font.PLAIN, 10));
470                chart.addSubtitle(subtitle);
471            }
472            currentTheme.apply(chart);
473            return chart;
474        }

539                Number newValue = dataset.getValue(key);
540                Number oldValue = previousDataset.getValue(key);
541    
542                if (oldValue == null) {
543                    if (greenForIncrease) {
544                        plot.setSectionPaint(key, Color.green);
545                    }
546                    else {
547                        plot.setSectionPaint(key, Color.red);
548                    }
549                    if (showDifference) {
550                        series.setValue(key + " (+100%)", newValue);
551                    }
552                }
553                else {
554                    double percentChange = (newValue.doubleValue()
555                            / oldValue.doubleValue() - 1.0) * 100.0;
556                    double shade
557                        = (Math.abs(percentChange) >= percentDiffForMaxScale ? 255
558                        : Math.abs(percentChange) * colorPerPercent);
559                    if (greenForIncrease
560                            && newValue.doubleValue() > oldValue.doubleValue()
561                            || !greenForIncrease && newValue.doubleValue()
562                            < oldValue.doubleValue()) {
563                        plot.setSectionPaint(key, new Color(0, (int) shade, 0));
564                    }
565                    else {
566                        plot.setSectionPaint(key, new Color((int) shade, 0, 0));
567                    }
568                    if (showDifference) {
569                        series.setValue(key + " (" + (percentChange >= 0 ? "+" : "")
570                                + NumberFormat.getPercentInstance().format(
571                                percentChange / 100.0) + ")", newValue);
572                    }
573                }
574            }
575    
576            if (showDifference) {
577                plot.setDataset(series);
578            }
579    
580            JFreeChart chart =  new JFreeChart(title,
581                    JFreeChart.DEFAULT_TITLE_FONT, plot, legend);
582    
583            if (subTitle) {
584                TextTitle subtitle = new TextTitle("Bright " + (greenForIncrease 
585                        ? "red" : "green") + "=change >=-" + percentDiffForMaxScale
586                        + "%, Bright " + (!greenForIncrease ? "red" : "green")
587                        + "=change >=+" + percentDiffForMaxScale + "%",
588                        new Font("SansSerif", Font.PLAIN, 10));
589                chart.addSubtitle(subtitle);
590            }
591            currentTheme.apply(chart);
592            return chart;
593        }
