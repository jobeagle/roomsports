695        /**
696         * Initialises the renderer and returns a state object that should be
697         * passed to all subsequent calls to the drawItem() method.  Here we
698         * calculate the Java2D y-coordinate for zero, since all the bars have
699         * their bases fixed at zero.
700         *
701         * @param g2  the graphics device.
702         * @param dataArea  the area inside the axes.
703         * @param plot  the plot.
704         * @param dataset  the data.
705         * @param info  an optional info collection object to return data back to
706         *              the caller.
707         *
708         * @return A state object.
709         */
710        public XYItemRendererState initialise(Graphics2D g2, Rectangle2D dataArea,
711                XYPlot plot, XYDataset dataset, PlotRenderingInfo info) {
712    
713            XYBarRendererState state = new XYBarRendererState(info);
714            ValueAxis rangeAxis = plot.getRangeAxisForDataset(plot.indexOf(
715                    dataset));
716            state.setG2Base(rangeAxis.valueToJava2D(this.base, dataArea,
717                    plot.getRangeAxisEdge()));
718            return state;
719    
720        }

722        /**
723         * Returns a default legend item for the specified series.  Subclasses
724         * should override this method to generate customised items.
725         *
726         * @param datasetIndex  the dataset index (zero-based).
727         * @param series  the series index (zero-based).
728         *
729         * @return A legend item for the series.
730         */
731        public LegendItem getLegendItem(int datasetIndex, int series) {
732            XYPlot xyplot = getPlot();
733            if (xyplot == null) {
734                return null;
735            }
736            XYDataset dataset = xyplot.getDataset(datasetIndex);
737            if (dataset == null) {
738                return null;
739            }
740            LegendItem result = null;
741            XYSeriesLabelGenerator lg = getLegendItemLabelGenerator();
742            String label = lg.generateLabel(dataset, series);
743            String description = label;
744            String toolTipText = null;
745            if (getLegendItemToolTipGenerator() != null) {
746                toolTipText = getLegendItemToolTipGenerator().generateLabel(
747                        dataset, series);
748            }
749            String urlText = null;
750            if (getLegendItemURLGenerator() != null) {
751                urlText = getLegendItemURLGenerator().generateLabel(dataset, 
752                        series);
753            }
754            Shape shape = this.legendBar;
755            Paint paint = lookupSeriesPaint(series);
756            Paint outlinePaint = lookupSeriesOutlinePaint(series);
757            Stroke outlineStroke = lookupSeriesOutlineStroke(series);
758            if (this.drawBarOutline) {
759                result = new LegendItem(label, description, toolTipText,
760                        urlText, shape, paint, outlineStroke, outlinePaint);
761            }
762            else {
763                result = new LegendItem(label, description, toolTipText, urlText, 
764                        shape, paint);
765            }
766            result.setLabelFont(lookupLegendTextFont(series));
767            Paint labelPaint = lookupLegendTextPaint(series);
768            if (labelPaint != null) {
769                result.setLabelPaint(labelPaint);
770            }
771            result.setDataset(dataset);
772            result.setDatasetIndex(datasetIndex);
773            result.setSeriesKey(dataset.getSeriesKey(series));
774            result.setSeriesIndex(series);
775            if (getGradientPaintTransformer() != null) {
776                result.setFillPaintTransformer(getGradientPaintTransformer());
777            }
778            return result;
779        }

781        /**
782         * Draws the visual representation of a single data item.
783         *
784         * @param g2  the graphics device.
785         * @param state  the renderer state.
786         * @param dataArea  the area within which the plot is being drawn.
787         * @param info  collects information about the drawing.
788         * @param plot  the plot (can be used to obtain standard color
789         *              information etc).
790         * @param domainAxis  the domain axis.
791         * @param rangeAxis  the range axis.
792         * @param dataset  the dataset.
793         * @param series  the series index (zero-based).
794         * @param item  the item index (zero-based).
795         * @param crosshairState  crosshair information for the plot
796         *                        (<code>null</code> permitted).
797         * @param pass  the pass index.
798         */
799        public void drawItem(Graphics2D g2,
800                             XYItemRendererState state,
801                             Rectangle2D dataArea,
802                             PlotRenderingInfo info,
803                             XYPlot plot,
804                             ValueAxis domainAxis,
805                             ValueAxis rangeAxis,
806                             XYDataset dataset,
807                             int series,
808                             int item,
809                             CrosshairState crosshairState,
810                             int pass) {
811    
812            if (!getItemVisible(series, item)) {
813                return;
814            }
815            IntervalXYDataset intervalDataset = (IntervalXYDataset) dataset;
816    
817            double value0;
818            double value1;
819            if (this.useYInterval) {
820                value0 = intervalDataset.getStartYValue(series, item);
821                value1 = intervalDataset.getEndYValue(series, item);
822            }
823            else {
824                value0 = this.base;
825                value1 = intervalDataset.getYValue(series, item);
826            }
827            if (Double.isNaN(value0) || Double.isNaN(value1)) {
828                return;
829            }
830            if (value0 <= value1) {
831                if (!rangeAxis.getRange().intersects(value0, value1)) {
832                    return;
833                }
834            }
835            else {
836                if (!rangeAxis.getRange().intersects(value1, value0)) {
837                    return;
838                }
839            }
840    
841            double translatedValue0 = rangeAxis.valueToJava2D(value0, dataArea,
842                    plot.getRangeAxisEdge());
843            double translatedValue1 = rangeAxis.valueToJava2D(value1, dataArea,
844                    plot.getRangeAxisEdge());
845            double bottom = Math.min(translatedValue0, translatedValue1);
846            double top = Math.max(translatedValue0, translatedValue1);
847    
848            double startX = intervalDataset.getStartXValue(series, item);
849            if (Double.isNaN(startX)) {
850                return;
851            }
852            double endX = intervalDataset.getEndXValue(series, item);
853            if (Double.isNaN(endX)) {
854                return;
855            }
856            if (startX <= endX) {
857                if (!domainAxis.getRange().intersects(startX, endX)) {
858                    return;
859                }
860            }
861            else {
862                if (!domainAxis.getRange().intersects(endX, startX)) {
863                    return;
864                }
865            }
866    
867            // is there an alignment adjustment to be made?
868            if (this.barAlignmentFactor >= 0.0 && this.barAlignmentFactor <= 1.0) {
869                double x = intervalDataset.getXValue(series, item);
870                double interval = endX - startX;
871                startX = x - interval * this.barAlignmentFactor;
872                endX = startX + interval;
873            }
874    
875            RectangleEdge location = plot.getDomainAxisEdge();
876            double translatedStartX = domainAxis.valueToJava2D(startX, dataArea,
877                    location);
878            double translatedEndX = domainAxis.valueToJava2D(endX, dataArea,
879                    location);
880    
881            double translatedWidth = Math.max(1, Math.abs(translatedEndX
882                    - translatedStartX));
883    
884            double left = Math.min(translatedStartX, translatedEndX);
885            if (getMargin() > 0.0) {
886                double cut = translatedWidth * getMargin();
887                translatedWidth = translatedWidth - cut;
888                left = left + cut / 2;
889            }
890    
891            Rectangle2D bar = null;
892            PlotOrientation orientation = plot.getOrientation();
893            if (orientation == PlotOrientation.HORIZONTAL) {
894                // clip left and right bounds to data area
895                bottom = Math.max(bottom, dataArea.getMinX());
896                top = Math.min(top, dataArea.getMaxX());
897                bar = new Rectangle2D.Double(
898                    bottom, left, top - bottom, translatedWidth);
899            }
900            else if (orientation == PlotOrientation.VERTICAL) {
901                // clip top and bottom bounds to data area
902                bottom = Math.max(bottom, dataArea.getMinY());
903                top = Math.min(top, dataArea.getMaxY());
904                bar = new Rectangle2D.Double(left, bottom, translatedWidth,
905                        top - bottom);
906            }
907    
908            boolean positive = (value1 > 0.0);
909            boolean inverted = rangeAxis.isInverted();
910            RectangleEdge barBase;
911            if (orientation == PlotOrientation.HORIZONTAL) {
912                if (positive && inverted || !positive && !inverted) {
913                    barBase = RectangleEdge.RIGHT;
914                }
915                else {
916                    barBase = RectangleEdge.LEFT;
917                }
918            }
919            else {
920                if (positive && !inverted || !positive && inverted) {
921                    barBase = RectangleEdge.BOTTOM;
922                }
923                else {
924                    barBase = RectangleEdge.TOP;
925                }
926            }
927            if (getShadowsVisible()) {
928                this.barPainter.paintBarShadow(g2, this, series, item, bar, barBase,
929                    !this.useYInterval);
930            }
931            this.barPainter.paintBar(g2, this, series, item, bar, barBase);
932    
933            if (isItemLabelVisible(series, item)) {
934                XYItemLabelGenerator generator = getItemLabelGenerator(series,
935                        item);
936                drawItemLabel(g2, dataset, series, item, plot, generator, bar,
937                        value1 < 0.0);
938            }
939    
940            // update the crosshair point
941            double x1 = (startX + endX) / 2.0;
942            double y1 = dataset.getYValue(series, item);
943            double transX1 = domainAxis.valueToJava2D(x1, dataArea, location);
944            double transY1 = rangeAxis.valueToJava2D(y1, dataArea,
945                    plot.getRangeAxisEdge());
946            int domainAxisIndex = plot.getDomainAxisIndex(domainAxis);
947            int rangeAxisIndex = plot.getRangeAxisIndex(rangeAxis);
948            updateCrosshairValues(crosshairState, x1, y1, domainAxisIndex,
949                    rangeAxisIndex, transX1, transY1, plot.getOrientation());
950    
951            EntityCollection entities = state.getEntityCollection();
952            if (entities != null) {
953                addEntity(entities, bar, dataset, series, item, 0.0, 0.0);
954            }
955    
956        }

958        /**
959         * Draws an item label.  This method is provided as an alternative to
960         * {@link #drawItemLabel(Graphics2D, PlotOrientation, XYDataset, int, int,
961         * double, double, boolean)} so that the bar can be used to calculate the
962         * label anchor point.
963         *
964         * @param g2  the graphics device.
965         * @param dataset  the dataset.
966         * @param series  the series index.
967         * @param item  the item index.
968         * @param plot  the plot.
969         * @param generator  the label generator (<code>null</code> permitted, in
970         *         which case the method does nothing, just returns).
971         * @param bar  the bar.
972         * @param negative  a flag indicating a negative value.
973         */
974        protected void drawItemLabel(Graphics2D g2, XYDataset dataset,
975                int series, int item, XYPlot plot, XYItemLabelGenerator generator,
976                Rectangle2D bar, boolean negative) {
977    
978            if (generator == null) {
979                return;  // nothing to do
980            }
981            String label = generator.generateLabel(dataset, series, item);
982            if (label == null) {
983                return;  // nothing to do
984            }
985    
986            Font labelFont = getItemLabelFont(series, item);
987            g2.setFont(labelFont);
988            Paint paint = getItemLabelPaint(series, item);
989            g2.setPaint(paint);
990    
991            // find out where to place the label...
992            ItemLabelPosition position = null;
993            if (!negative) {
994                position = getPositiveItemLabelPosition(series, item);
995            }
996            else {
997                position = getNegativeItemLabelPosition(series, item);
998            }
999    
1000            // work out the label anchor point...
1001            Point2D anchorPoint = calculateLabelAnchorPoint(
1002                    position.getItemLabelAnchor(), bar, plot.getOrientation());
1003    
1004            if (isInternalAnchor(position.getItemLabelAnchor())) {
1005                Shape bounds = TextUtilities.calculateRotatedStringBounds(label,
1006                        g2, (float) anchorPoint.getX(), (float) anchorPoint.getY(),
1007                        position.getTextAnchor(), position.getAngle(),
1008                        position.getRotationAnchor());
1009    
1010                if (bounds != null) {
1011                    if (!bar.contains(bounds.getBounds2D())) {
1012                        if (!negative) {
1013                            position = getPositiveItemLabelPositionFallback();
1014                        }
1015                        else {
1016                            position = getNegativeItemLabelPositionFallback();
1017                        }
1018                        if (position != null) {
1019                            anchorPoint = calculateLabelAnchorPoint(
1020                                    position.getItemLabelAnchor(), bar,
1021                                    plot.getOrientation());
1022                        }
1023                    }
1024                }
1025    
1026            }
1027    
1028            if (position != null) {
1029                TextUtilities.drawRotatedString(label, g2,
1030                        (float) anchorPoint.getX(), (float) anchorPoint.getY(),
1031                        position.getTextAnchor(), position.getAngle(),
1032                        position.getRotationAnchor());
1033            }
1034        }

1047    
1048            Point2D result = null;
1049            double offset = getItemLabelAnchorOffset();
1050            double x0 = bar.getX() - offset;
1051            double x1 = bar.getX();
1052            double x2 = bar.getX() + offset;
1053            double x3 = bar.getCenterX();
1054            double x4 = bar.getMaxX() - offset;
1055            double x5 = bar.getMaxX();
1056            double x6 = bar.getMaxX() + offset;
1057    
1058            double y0 = bar.getMaxY() + offset;
1059            double y1 = bar.getMaxY();
1060            double y2 = bar.getMaxY() - offset;
1061            double y3 = bar.getCenterY();
1062            double y4 = bar.getMinY() + offset;
1063            double y5 = bar.getMinY();
1064            double y6 = bar.getMinY() - offset;
1065    
1066            if (anchor == ItemLabelAnchor.CENTER) {
1067                result = new Point2D.Double(x3, y3);
1068            }
1069            else if (anchor == ItemLabelAnchor.INSIDE1) {
1070                result = new Point2D.Double(x4, y4);
1071            }
1072            else if (anchor == ItemLabelAnchor.INSIDE2) {
1073                result = new Point2D.Double(x4, y4);
1074            }
1075            else if (anchor == ItemLabelAnchor.INSIDE3) {
1076                result = new Point2D.Double(x4, y3);
1077            }
1078            else if (anchor == ItemLabelAnchor.INSIDE4) {
1079                result = new Point2D.Double(x4, y2);
1080            }
1081            else if (anchor == ItemLabelAnchor.INSIDE5) {
1082                result = new Point2D.Double(x4, y2);
1083            }
1084            else if (anchor == ItemLabelAnchor.INSIDE6) {
1085                result = new Point2D.Double(x3, y2);
1086            }
1087            else if (anchor == ItemLabelAnchor.INSIDE7) {
1088                result = new Point2D.Double(x2, y2);
1089            }
1090            else if (anchor == ItemLabelAnchor.INSIDE8) {
1091                result = new Point2D.Double(x2, y2);
1092            }
1093            else if (anchor == ItemLabelAnchor.INSIDE9) {
1094                result = new Point2D.Double(x2, y3);
1095            }
1096            else if (anchor == ItemLabelAnchor.INSIDE10) {
1097                result = new Point2D.Double(x2, y4);
1098            }
1099            else if (anchor == ItemLabelAnchor.INSIDE11) {
1100                result = new Point2D.Double(x2, y4);
1101            }
1102            else if (anchor == ItemLabelAnchor.INSIDE12) {
1103                result = new Point2D.Double(x3, y4);
1104            }
1105            else if (anchor == ItemLabelAnchor.OUTSIDE1) {
1106                result = new Point2D.Double(x5, y6);
1107            }

1109                result = new Point2D.Double(x6, y5);
1110            }
1111            else if (anchor == ItemLabelAnchor.OUTSIDE3) {
1112                result = new Point2D.Double(x6, y3);
1113            }
1114            else if (anchor == ItemLabelAnchor.OUTSIDE4) {
1115                result = new Point2D.Double(x6, y1);
1116            }
1117            else if (anchor == ItemLabelAnchor.OUTSIDE5) {
1118                result = new Point2D.Double(x5, y0);
1119            }
1120            else if (anchor == ItemLabelAnchor.OUTSIDE6) {
1121                result = new Point2D.Double(x3, y0);
1122            }
1123            else if (anchor == ItemLabelAnchor.OUTSIDE7) {
1124                result = new Point2D.Double(x1, y0);
1125            }
1126            else if (anchor == ItemLabelAnchor.OUTSIDE8) {
1127                result = new Point2D.Double(x0, y1);
1128            }

1130                result = new Point2D.Double(x0, y3);

1218         *
1219         * @return A boolean.

1264            }

1266                return false;
1267            }
