149    /**
150     * Standard item renderer for an {@link XYPlot}.  This class can draw (a)
151     * shapes at each point, or (b) lines between points, or (c) both shapes and
152     * lines.
153     * <P>
154     * This renderer has been retained for historical reasons and, in general, you
155     * should use the {@link XYLineAndShapeRenderer} class instead.
156     */

259        /**
260         * Constructs a new renderer.  To specify the type of renderer, use one of
261         * the constants: {@link #SHAPES}, {@link #LINES} or
262         * {@link #SHAPES_AND_LINES}.
263         *
264         * @param type  the type of renderer.
265         * @param toolTipGenerator  the item label generator (<code>null</code>
266         *                          permitted).
267         * @param urlGenerator  the URL generator.
268         */
269        public StandardXYItemRenderer(int type,
270                                      XYToolTipGenerator toolTipGenerator,
271                                      XYURLGenerator urlGenerator) {
272    
273            super();
274            setBaseToolTipGenerator(toolTipGenerator);
275            setURLGenerator(urlGenerator);
276            if ((type & SHAPES) != 0) {
277                this.baseShapesVisible = true;
278            }
279            if ((type & LINES) != 0) {
280                this.plotLines = true;
281            }
282            if ((type & IMAGES) != 0) {
283                this.plotImages = true;
284            }
285            if ((type & DISCONTINUOUS) != 0) {
286                this.plotDiscontinuous = true;
287            }
288    
289            this.shapesFilled = null;
290            this.seriesShapesFilled = new BooleanList();
291            this.baseShapesFilled = true;
292            this.legendLine = new Line2D.Double(-7.0, 0.0, 7.0, 0.0);
293            this.drawSeriesLineAsPath = false;
294        }

324        /**
325         * Returns the flag used to control whether or not the shape for an item is
326         * filled.
327         * <p>
328         * The default implementation passes control to the
329         * <code>getSeriesShapesFilled</code> method.  You can override this method
330         * if you require different behaviour.
331         *
332         * @param series  the series index (zero-based).
333         * @param item  the item index (zero-based).
334         *
335         * @return A boolean.
336         *
337         * @see #getSeriesShapesFilled(int)
338         */
339        public boolean getItemShapeFilled(int series, int item) {
340            // return the overall setting, if there is one...
341            if (this.shapesFilled != null) {
342                return this.shapesFilled.booleanValue();
343            }
344    
345            // otherwise look up the paint table
346            Boolean flag = this.seriesShapesFilled.getBoolean(series);
347            if (flag != null) {
348                return flag.booleanValue();
349            }
350            else {
351                return this.baseShapesFilled;
352            }
353        }

636        /**
637         * Returns a legend item for a series.
638         *
639         * @param datasetIndex  the dataset index (zero-based).
640         * @param series  the series index (zero-based).
641         *
642         * @return A legend item for the series.
643         */
644        public LegendItem getLegendItem(int datasetIndex, int series) {
645            XYPlot plot = getPlot();
646            if (plot == null) {
647                return null;
648            }
649            LegendItem result = null;
650            XYDataset dataset = plot.getDataset(datasetIndex);
651            if (dataset != null) {
652                if (getItemVisible(series, 0)) {
653                    String label = getLegendItemLabelGenerator().generateLabel(
654                            dataset, series);
655                    String description = label;
656                    String toolTipText = null;
657                    if (getLegendItemToolTipGenerator() != null) {
658                        toolTipText = getLegendItemToolTipGenerator().generateLabel(
659                                dataset, series);
660                    }
661                    String urlText = null;
662                    if (getLegendItemURLGenerator() != null) {
663                        urlText = getLegendItemURLGenerator().generateLabel(
664                                dataset, series);
665                    }
666                    Shape shape = lookupLegendShape(series);
667                    boolean shapeFilled = getItemShapeFilled(series, 0);
668                    Paint paint = lookupSeriesPaint(series);
669                    Paint linePaint = paint;
670                    Stroke lineStroke = lookupSeriesStroke(series);
671                    result = new LegendItem(label, description, toolTipText,
672                            urlText, this.baseShapesVisible, shape, shapeFilled,
673                            paint, !shapeFilled, paint, lineStroke,
674                            this.plotLines, this.legendLine, lineStroke, linePaint);
675                    result.setLabelFont(lookupLegendTextFont(series));
676                    Paint labelPaint = lookupLegendTextPaint(series);
677                    if (labelPaint != null) {
678                        result.setLabelPaint(labelPaint);
679                    }
680                    result.setDataset(dataset);
681                    result.setDatasetIndex(datasetIndex);
682                    result.setSeriesKey(dataset.getSeriesKey(series));
683                    result.setSeriesIndex(series);
684                }
685            }
686            return result;
687        }

785        /**
786         * Draws the visual representation of a single data item.
787         *
788         * @param g2  the graphics device.
789         * @param state  the renderer state.
790         * @param dataArea  the area within which the data is being drawn.
791         * @param info  collects information about the drawing.
792         * @param plot  the plot (can be used to obtain standard color information
793         *              etc).
794         * @param domainAxis  the domain axis.
795         * @param rangeAxis  the range axis.
796         * @param dataset  the dataset.
797         * @param series  the series index (zero-based).
798         * @param item  the item index (zero-based).
799         * @param crosshairState  crosshair information for the plot
800         *                        (<code>null</code> permitted).
801         * @param pass  the pass index.
802         */
803        public void drawItem(Graphics2D g2,
804                             XYItemRendererState state,
805                             Rectangle2D dataArea,
806                             PlotRenderingInfo info,
807                             XYPlot plot,
808                             ValueAxis domainAxis,
809                             ValueAxis rangeAxis,
810                             XYDataset dataset,
811                             int series,
812                             int item,
813                             CrosshairState crosshairState,
814                             int pass) {
815    
816            boolean itemVisible = getItemVisible(series, item);
817    
818            // setup for collecting optional entity info...
819            Shape entityArea = null;
820            EntityCollection entities = null;
821            if (info != null) {
822                entities = info.getOwner().getEntityCollection();
823            }
824    
825            PlotOrientation orientation = plot.getOrientation();
826            Paint paint = getItemPaint(series, item);
827            Stroke seriesStroke = getItemStroke(series, item);
828            g2.setPaint(paint);
829            g2.setStroke(seriesStroke);
830    
831            // get the data point...
832            double x1 = dataset.getXValue(series, item);
833            double y1 = dataset.getYValue(series, item);
834            if (Double.isNaN(x1) || Double.isNaN(y1)) {
835                itemVisible = false;
836            }
837    
838            RectangleEdge xAxisLocation = plot.getDomainAxisEdge();
839            RectangleEdge yAxisLocation = plot.getRangeAxisEdge();
840            double transX1 = domainAxis.valueToJava2D(x1, dataArea, xAxisLocation);
841            double transY1 = rangeAxis.valueToJava2D(y1, dataArea, yAxisLocation);
842    
843            if (getPlotLines()) {
844                if (this.drawSeriesLineAsPath) {
845                    State s = (State) state;
846                    if (s.getSeriesIndex() != series) {
847                        // we are starting a new series path
848                        s.seriesPath.reset();
849                        s.lastPointGood = false;
850                        s.setSeriesIndex(series);
851                    }
852    
853                    // update path to reflect latest point
854                    if (itemVisible && !Double.isNaN(transX1)
855                            && !Double.isNaN(transY1)) {
856                        float x = (float) transX1;
857                        float y = (float) transY1;
858                        if (orientation == PlotOrientation.HORIZONTAL) {
859                            x = (float) transY1;
860                            y = (float) transX1;
861                        }
862                        if (s.isLastPointGood()) {
863                            // TODO: check threshold
864                            s.seriesPath.lineTo(x, y);
865                        }
866                        else {
867                            s.seriesPath.moveTo(x, y);
868                        }
869                        s.setLastPointGood(true);
870                    }
871                    else {
872                        s.setLastPointGood(false);
873                    }
874                    if (item == dataset.getItemCount(series) - 1) {
875                        if (s.seriesIndex == series) {
876                            // draw path
877                            g2.setStroke(lookupSeriesStroke(series));
878                            g2.setPaint(lookupSeriesPaint(series));
879                            g2.draw(s.seriesPath);
880                        }
881                    }
882                }
883    
884                else if (item != 0 && itemVisible) {
885                    // get the previous data point...
886                    double x0 = dataset.getXValue(series, item - 1);
887                    double y0 = dataset.getYValue(series, item - 1);
888                    if (!Double.isNaN(x0) && !Double.isNaN(y0)) {
889                        boolean drawLine = true;
890                        if (getPlotDiscontinuous()) {
891                            // only draw a line if the gap between the current and
892                            // previous data point is within the threshold
893                            int numX = dataset.getItemCount(series);
894                            double minX = dataset.getXValue(series, 0);
895                            double maxX = dataset.getXValue(series, numX - 1);
896                            if (this.gapThresholdType == UnitType.ABSOLUTE) {
897                                drawLine = Math.abs(x1 - x0) <= this.gapThreshold;
898                            }
899                            else {
900                                drawLine = Math.abs(x1 - x0) <= ((maxX - minX)
901                                    / numX * getGapThreshold());
902                            }
903                        }
904                        if (drawLine) {
905                            double transX0 = domainAxis.valueToJava2D(x0, dataArea,
906                                    xAxisLocation);
907                            double transY0 = rangeAxis.valueToJava2D(y0, dataArea,
908                                    yAxisLocation);
909    
910                            // only draw if we have good values
911                            if (Double.isNaN(transX0) || Double.isNaN(transY0)
912                                || Double.isNaN(transX1) || Double.isNaN(transY1)) {
913                                return;
914                            }
915    
916                            if (orientation == PlotOrientation.HORIZONTAL) {
917                                state.workingLine.setLine(transY0, transX0,
918                                        transY1, transX1);
919                            }
920                            else if (orientation == PlotOrientation.VERTICAL) {
921                                state.workingLine.setLine(transX0, transY0,
922                                        transX1, transY1);
923                            }
924    
925                            if (state.workingLine.intersects(dataArea)) {
926                                g2.draw(state.workingLine);
927                            }
928                        }
929                    }
930                }
931            }
932    
933            // we needed to get this far even for invisible items, to ensure that
934            // seriesPath updates happened, but now there is nothing more we need
935            // to do for non-visible items...
936            if (!itemVisible) {
937                return;
938            }
939    
940            if (getBaseShapesVisible()) {
941    
942                Shape shape = getItemShape(series, item);
943                if (orientation == PlotOrientation.HORIZONTAL) {
944                    shape = ShapeUtilities.createTranslatedShape(shape, transY1,
945                            transX1);
946                }
947                else if (orientation == PlotOrientation.VERTICAL) {
948                    shape = ShapeUtilities.createTranslatedShape(shape, transX1,
949                            transY1);
950                }
951                if (shape.intersects(dataArea)) {
952                    if (getItemShapeFilled(series, item)) {
953                        g2.fill(shape);
954                    }
955                    else {
956                        g2.draw(shape);
957                    }
958                }
959                entityArea = shape;
960    
961            }
962    
963            if (getPlotImages()) {
964                Image image = getImage(plot, series, item, transX1, transY1);
965                if (image != null) {
966                    Point hotspot = getImageHotspot(plot, series, item, transX1,
967                            transY1, image);
968                    g2.drawImage(image, (int) (transX1 - hotspot.getX()),
969                            (int) (transY1 - hotspot.getY()), null);
970                    entityArea = new Rectangle2D.Double(transX1 - hotspot.getX(),
971                            transY1 - hotspot.getY(), image.getWidth(null),
972                            image.getHeight(null));
973                }
974    
975            }
976    
977            double xx = transX1;
978            double yy = transY1;
979            if (orientation == PlotOrientation.HORIZONTAL) {
980                xx = transY1;
981                yy = transX1;
982            }
983    
984            // draw the item label if there is one...
985            if (isItemLabelVisible(series, item)) {
986                drawItemLabel(g2, orientation, dataset, series, item, xx, yy,
987                        (y1 < 0.0));
988            }
989    
990            int domainAxisIndex = plot.getDomainAxisIndex(domainAxis);
991            int rangeAxisIndex = plot.getRangeAxisIndex(rangeAxis);
992            updateCrosshairValues(crosshairState, x1, y1, domainAxisIndex,
993                    rangeAxisIndex, transX1, transY1, orientation);
994    
995            // add an entity for the item...
996            if (entities != null && isPointInRect(dataArea, xx, yy)) {
997                addEntity(entities, entityArea, dataset, series, item, xx, yy);
998            }
999    
1000        }

1002        /**
1003         * Tests this renderer for equality with another object.
1004         *
1005         * @param obj  the object (<code>null</code> permitted).
1006         *
1007         * @return A boolean.
1008         */
1009        public boolean equals(Object obj) {
1010    
1011            if (obj == this) {
1012                return true;
1013            }
1014            if (!(obj instanceof StandardXYItemRenderer)) {
1015                return false;
1016            }
1017            StandardXYItemRenderer that = (StandardXYItemRenderer) obj;
1018            if (this.baseShapesVisible != that.baseShapesVisible) {
1019                return false;
1020            }
1021            if (this.plotLines != that.plotLines) {
1022                return false;
1023            }
1024            if (this.plotImages != that.plotImages) {
1025                return false;
1026            }
1027            if (this.plotDiscontinuous != that.plotDiscontinuous) {
1028                return false;
1029            }
1030            if (this.gapThresholdType != that.gapThresholdType) {
1031                return false;
1032            }
1033            if (this.gapThreshold != that.gapThreshold) {
1034                return false;
1035            }
1036            if (!ObjectUtilities.equal(this.shapesFilled, that.shapesFilled)) {
1037                return false;
1038            }
1039            if (!this.seriesShapesFilled.equals(that.seriesShapesFilled)) {
1040                return false;
1041            }
1042            if (this.baseShapesFilled != that.baseShapesFilled) {
1043                return false;
1044            }
1045            if (this.drawSeriesLineAsPath != that.drawSeriesLineAsPath) {
1046                return false;
1047            }
1048            if (!ShapeUtilities.equal(this.legendLine, that.legendLine)) {
1049                return false;
1050            }
1051            return super.equals(obj);
1052    
1053        }
