105    /**
106     * A renderer that connects data points with lines and/or draws shapes at each
107     * data point.  This renderer is designed for use with the {@link XYPlot}
108     * class.  The example shown here is generated by
109     * the <code>XYLineAndShapeRendererDemo2.java</code> program included in the
110     * JFreeChart demo collection:
111     * <br><br>
112     * <img src="../../../../../images/XYLineAndShapeRendererSample.png"
113     * alt="XYLineAndShapeRendererSample.png" />
114     *
115     */
116    public class XYLineAndShapeRenderer extends AbstractXYItemRenderer
117            implements XYItemRenderer, Cloneable, PublicCloneable, Serializable {

194        /**
195         * Creates a new renderer with both lines and shapes visible.
196         */
197        public XYLineAndShapeRenderer() {
198            this(true, true);
199        }

201        /**
202         * Creates a new renderer.
203         *
204         * @param lines  lines visible?
205         * @param shapes  shapes visible?
206         */
207        public XYLineAndShapeRenderer(boolean lines, boolean shapes) {
208            this.linesVisible = null;
209            this.seriesLinesVisible = new BooleanList();
210            this.baseLinesVisible = lines;
211            this.legendLine = new Line2D.Double(-7.0, 0.0, 7.0, 0.0);
212    
213            this.shapesVisible = null;
214            this.seriesShapesVisible = new BooleanList();
215            this.baseShapesVisible = shapes;
216    
217            this.shapesFilled = null;
218            this.useFillPaint = false;     // use item paint for fills by default
219            this.seriesShapesFilled = new BooleanList();
220            this.baseShapesFilled = true;
221    
222            this.drawOutlines = true;
223            this.useOutlinePaint = false;  // use item paint for outlines by
224                                           // default, not outline paint
225    
226            this.drawSeriesLineAsPath = false;
227        }

868        /**
869         * Draws the visual representation of a single data item.
870         *
871         * @param g2  the graphics device.
872         * @param state  the renderer state.
873         * @param dataArea  the area within which the data is being drawn.
874         * @param info  collects information about the drawing.
875         * @param plot  the plot (can be used to obtain standard color
876         *              information etc).
877         * @param domainAxis  the domain axis.
878         * @param rangeAxis  the range axis.
879         * @param dataset  the dataset.
880         * @param series  the series index (zero-based).
881         * @param item  the item index (zero-based).
882         * @param crosshairState  crosshair information for the plot
883         *                        (<code>null</code> permitted).
884         * @param pass  the pass index.
885         */
886        public void drawItem(Graphics2D g2, XYItemRendererState state,
887            Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
888            ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
889            int series, int item, CrosshairState crosshairState, int pass) {
890    
891            // do nothing if item is not visible
892            if (!getItemVisible(series, item)) {
893                return;
894            }
895    
896            // first pass draws the background (lines, for instance)
897            if (isLinePass(pass)) {
898                if (getItemLineVisible(series, item)) {
899                    if (this.drawSeriesLineAsPath) {
900                        drawPrimaryLineAsPath(state, g2, plot, dataset, pass,
901                                series, item, domainAxis, rangeAxis, dataArea);
902                    }
903                    else {
904                        drawPrimaryLine(state, g2, plot, dataset, pass, series,
905                                item, domainAxis, rangeAxis, dataArea);
906                    }
907                }
908            }
909            // second pass adds shapes where the items are ..
910            else if (isItemPass(pass)) {
911    
912                // setup for collecting optional entity info...
913                EntityCollection entities = null;
914                if (info != null && info.getOwner() != null) {
915                    entities = info.getOwner().getEntityCollection();
916                }
917    
918                drawSecondaryPass(g2, plot, dataset, pass, series, item,
919                        domainAxis, dataArea, rangeAxis, crosshairState, entities);
920            }
921        }

947        /**
948         * Draws the item (first pass). This method draws the lines
949         * connecting the items.
950         *
951         * @param g2  the graphics device.
952         * @param state  the renderer state.
953         * @param dataArea  the area within which the data is being drawn.
954         * @param plot  the plot (can be used to obtain standard color
955         *              information etc).
956         * @param domainAxis  the domain axis.
957         * @param rangeAxis  the range axis.
958         * @param dataset  the dataset.
959         * @param pass  the pass.
960         * @param series  the series index (zero-based).
961         * @param item  the item index (zero-based).
962         */
963        protected void drawPrimaryLine(XYItemRendererState state,
964                                       Graphics2D g2,
965                                       XYPlot plot,
966                                       XYDataset dataset,
967                                       int pass,
968                                       int series,
969                                       int item,
970                                       ValueAxis domainAxis,
971                                       ValueAxis rangeAxis,
972                                       Rectangle2D dataArea) {
973            if (item == 0) {
974                return;
975            }
976    
977            // get the data point...
978            double x1 = dataset.getXValue(series, item);
979            double y1 = dataset.getYValue(series, item);
980            if (Double.isNaN(y1) || Double.isNaN(x1)) {
981                return;
982            }
983    
984            double x0 = dataset.getXValue(series, item - 1);
985            double y0 = dataset.getYValue(series, item - 1);
986            if (Double.isNaN(y0) || Double.isNaN(x0)) {
987                return;
988            }
989    
990            RectangleEdge xAxisLocation = plot.getDomainAxisEdge();
991            RectangleEdge yAxisLocation = plot.getRangeAxisEdge();
992    
993            double transX0 = domainAxis.valueToJava2D(x0, dataArea, xAxisLocation);
994            double transY0 = rangeAxis.valueToJava2D(y0, dataArea, yAxisLocation);
995    
996            double transX1 = domainAxis.valueToJava2D(x1, dataArea, xAxisLocation);
997            double transY1 = rangeAxis.valueToJava2D(y1, dataArea, yAxisLocation);
998    
999            // only draw if we have good values
1000            if (Double.isNaN(transX0) || Double.isNaN(transY0)
1001                || Double.isNaN(transX1) || Double.isNaN(transY1)) {
1002                return;
1003            }
1004    
1005            PlotOrientation orientation = plot.getOrientation();
1006            boolean visible = false;
1007            if (orientation == PlotOrientation.HORIZONTAL) {
1008                state.workingLine.setLine(transY0, transX0, transY1, transX1);
1009            }
1010            else if (orientation == PlotOrientation.VERTICAL) {
1011                state.workingLine.setLine(transX0, transY0, transX1, transY1);
1012            }
1013            visible = LineUtilities.clipLine(state.workingLine, dataArea);
1014            if (visible) {
1015                drawFirstPassShape(g2, pass, series, item, state.workingLine);
1016            }
1017        }

1019        /**

1025         * @param item  the item index.
1026         * @param shape  the shape.

1029                                          int item, Shape shape) {
1030            g2.setStroke(getItemStroke(series, item));
1031            g2.setPaint(getItemPaint(series, item));
1032            g2.draw(shape);
1033        }

1036        /**
1037         * Draws the item (first pass). This method draws the lines
1038         * connecting the items. Instead of drawing separate lines,
1039         * a GeneralPath is constructed and drawn at the end of
1040         * the series painting.
1041         *
1042         * @param g2  the graphics device.
1043         * @param state  the renderer state.
1044         * @param plot  the plot (can be used to obtain standard color information
1045         *              etc).
1046         * @param dataset  the dataset.
1047         * @param pass  the pass.
1048         * @param series  the series index (zero-based).
1049         * @param item  the item index (zero-based).
1050         * @param domainAxis  the domain axis.
1051         * @param rangeAxis  the range axis.
1052         * @param dataArea  the area within which the data is being drawn.
1053         */
1054        protected void drawPrimaryLineAsPath(XYItemRendererState state,
1055                                             Graphics2D g2, XYPlot plot,
1056                                             XYDataset dataset,
1057                                             int pass,
1058                                             int series,
1059                                             int item,
1060                                             ValueAxis domainAxis,
1061                                             ValueAxis rangeAxis,
1062                                             Rectangle2D dataArea) {
1063    
1064    
1065            RectangleEdge xAxisLocation = plot.getDomainAxisEdge();
1066            RectangleEdge yAxisLocation = plot.getRangeAxisEdge();
1067    
1068            // get the data point...
1069            double x1 = dataset.getXValue(series, item);
1070            double y1 = dataset.getYValue(series, item);
1071            double transX1 = domainAxis.valueToJava2D(x1, dataArea, xAxisLocation);
1072            double transY1 = rangeAxis.valueToJava2D(y1, dataArea, yAxisLocation);
1073    
1074            State s = (State) state;
1075            // update path to reflect latest point
1076            if (!Double.isNaN(transX1) && !Double.isNaN(transY1)) {
1077                float x = (float) transX1;
1078                float y = (float) transY1;
1079                PlotOrientation orientation = plot.getOrientation();
1080                if (orientation == PlotOrientation.HORIZONTAL) {
1081                    x = (float) transY1;
1082                    y = (float) transX1;
1083                }
1084                if (s.isLastPointGood()) {
1085                    s.seriesPath.lineTo(x, y);
1086                }
1087                else {
1088                    s.seriesPath.moveTo(x, y);
1089                }
1090                s.setLastPointGood(true);
1091            }
1092            else {
1093                s.setLastPointGood(false);
1094            }
1095            // if this is the last item, draw the path ...
1096            if (item == s.getLastItemIndex()) {
1097                // draw path
1098                drawFirstPassShape(g2, pass, series, item, s.seriesPath);
1099            }
1100        }

1102        /**
1103         * Draws the item shapes and adds chart entities (second pass). This method
1104         * draws the shapes which mark the item positions. If <code>entities</code>
1105         * is not <code>null</code> it will be populated with entity information
1106         * for points that fall within the data area.
1107         *
1108         * @param g2  the graphics device.
1109         * @param plot  the plot (can be used to obtain standard color
1110         *              information etc).
1111         * @param domainAxis  the domain axis.
1112         * @param dataArea  the area within which the data is being drawn.
1113         * @param rangeAxis  the range axis.
1114         * @param dataset  the dataset.
1115         * @param pass  the pass.
1116         * @param series  the series index (zero-based).
1117         * @param item  the item index (zero-based).
1118         * @param crosshairState  the crosshair state.
1119         * @param entities the entity collection.
1120         */
1121        protected void drawSecondaryPass(Graphics2D g2, XYPlot plot, 
1122                XYDataset dataset, int pass, int series, int item,
1123                ValueAxis domainAxis, Rectangle2D dataArea, ValueAxis rangeAxis,
1124                CrosshairState crosshairState, EntityCollection entities) {
1125    
1126            Shape entityArea = null;
1127    
1128            // get the data point...
1129            double x1 = dataset.getXValue(series, item);
1130            double y1 = dataset.getYValue(series, item);
1131            if (Double.isNaN(y1) || Double.isNaN(x1)) {
1132                return;
1133            }
1134    
1135            PlotOrientation orientation = plot.getOrientation();
1136            RectangleEdge xAxisLocation = plot.getDomainAxisEdge();
1137            RectangleEdge yAxisLocation = plot.getRangeAxisEdge();
1138            double transX1 = domainAxis.valueToJava2D(x1, dataArea, xAxisLocation);
1139            double transY1 = rangeAxis.valueToJava2D(y1, dataArea, yAxisLocation);
1140    
1141            if (getItemShapeVisible(series, item)) {
1142                Shape shape = getItemShape(series, item);
1143                if (orientation == PlotOrientation.HORIZONTAL) {
1144                    shape = ShapeUtilities.createTranslatedShape(shape, transY1,
1145                            transX1);
1146                }
1147                else if (orientation == PlotOrientation.VERTICAL) {
1148                    shape = ShapeUtilities.createTranslatedShape(shape, transX1,
1149                            transY1);
1150                }
1151                entityArea = shape;
1152                if (shape.intersects(dataArea)) {
1153                    if (getItemShapeFilled(series, item)) {
1154                        if (this.useFillPaint) {
1155                            g2.setPaint(getItemFillPaint(series, item));
1156                        }
1157                        else {
1158                            g2.setPaint(getItemPaint(series, item));
1159                        }
1160                        g2.fill(shape);
1161                    }
1162                    if (this.drawOutlines) {
1163                        if (getUseOutlinePaint()) {
1164                            g2.setPaint(getItemOutlinePaint(series, item));
1165                        }
1166                        else {
1167                            g2.setPaint(getItemPaint(series, item));
1168                        }
1169                        g2.setStroke(getItemOutlineStroke(series, item));
1170                        g2.draw(shape);
1171                    }
1172                }
1173            }
1174    
1175            double xx = transX1;
1176            double yy = transY1;
1177            if (orientation == PlotOrientation.HORIZONTAL) {
1178                xx = transY1;
1179                yy = transX1;
1180            }
1181    
1182            // draw the item label if there is one...
1183            if (isItemLabelVisible(series, item)) {
1184                drawItemLabel(g2, orientation, dataset, series, item, xx, yy,
1185                        (y1 < 0.0));
1186            }
1187    
1188            int domainAxisIndex = plot.getDomainAxisIndex(domainAxis);
1189            int rangeAxisIndex = plot.getRangeAxisIndex(rangeAxis);
1190            updateCrosshairValues(crosshairState, x1, y1, domainAxisIndex,
1191                    rangeAxisIndex, transX1, transY1, orientation);
1192    
1193            // add an entity for the item, but only if it falls within the data
1194            // area...
1195            if (entities != null && isPointInRect(dataArea, xx, yy)) {
1196                addEntity(entities, entityArea, dataset, series, item, xx, yy);
1197            }
1198        }

1201        /**
1202         * Returns a legend item for the specified series.
1203         *
1204         * @param datasetIndex  the dataset index (zero-based).
1205         * @param series  the series index (zero-based).
1206         *
1207         * @return A legend item for the series (possibly <code>null</code).
1208         */
1209        public LegendItem getLegendItem(int datasetIndex, int series) {
1210            XYPlot plot = getPlot();
1211            if (plot == null) {
1212                return null;
1213            }
1214    
1215            XYDataset dataset = plot.getDataset(datasetIndex);
1216            if (dataset == null) {
1217                return null;
1218            }
1219    
1220            if (!getItemVisible(series, 0)) {
1221                return null;
1222            }
1223            String label = getLegendItemLabelGenerator().generateLabel(dataset,
1224                    series);
1225            String description = label;
1226            String toolTipText = null;
1227            if (getLegendItemToolTipGenerator() != null) {
1228                toolTipText = getLegendItemToolTipGenerator().generateLabel(
1229                        dataset, series);
1230            }
1231            String urlText = null;
1232            if (getLegendItemURLGenerator() != null) {
1233                urlText = getLegendItemURLGenerator().generateLabel(dataset,
1234                        series);
1235            }
1236            boolean shapeIsVisible = getItemShapeVisible(series, 0);
1237            Shape shape = lookupLegendShape(series);
1238            boolean shapeIsFilled = getItemShapeFilled(series, 0);
1239            Paint fillPaint = (this.useFillPaint ? lookupSeriesFillPaint(series)
1240                    : lookupSeriesPaint(series));
1241            boolean shapeOutlineVisible = this.drawOutlines;
1242            Paint outlinePaint = (this.useOutlinePaint ? lookupSeriesOutlinePaint(
1243                    series) : lookupSeriesPaint(series));
1244            Stroke outlineStroke = lookupSeriesOutlineStroke(series);
1245            boolean lineVisible = getItemLineVisible(series, 0);
1246            Stroke lineStroke = lookupSeriesStroke(series);
1247            Paint linePaint = lookupSeriesPaint(series);
1248            LegendItem result = new LegendItem(label, description, toolTipText,
1249                    urlText, shapeIsVisible, shape, shapeIsFilled, fillPaint,
1250                    shapeOutlineVisible, outlinePaint, outlineStroke, lineVisible,
1251                    this.legendLine, lineStroke, linePaint);
1252            result.setLabelFont(lookupLegendTextFont(series));
1253            Paint labelPaint = lookupLegendTextPaint(series);
1254            if (labelPaint != null) {
1255                result.setLabelPaint(labelPaint);
1256            }
1257            result.setSeriesKey(dataset.getSeriesKey(series));
1258            result.setSeriesIndex(series);
1259            result.setDataset(dataset);
1260            result.setDatasetIndex(datasetIndex);
1261    
1262            return result;
1263        }
