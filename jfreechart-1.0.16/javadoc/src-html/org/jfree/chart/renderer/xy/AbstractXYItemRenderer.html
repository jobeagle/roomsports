311        /**
312         * Returns the label generator for a data item.  This implementation simply
313         * passes control to the {@link #getSeriesItemLabelGenerator(int)} method.
314         * If, for some reason, you want a different generator for individual
315         * items, you can override this method.
316         *
317         * @param series  the series index (zero based).
318         * @param item  the item index (zero based).
319         *
320         * @return The generator (possibly <code>null</code>).
321         */
322        @Override
323        public XYItemLabelGenerator getItemLabelGenerator(int series, int item) {
324            // return the generator for ALL series, if there is one...
325            if (this.itemLabelGenerator != null) {
326                return this.itemLabelGenerator;
327            }
328    
329            // otherwise look up the generator table
330            XYItemLabelGenerator generator
331                = (XYItemLabelGenerator) this.itemLabelGeneratorList.get(series);
332            if (generator == null) {
333                generator = this.baseItemLabelGenerator;
334            }
335            return generator;
336        }

503        /**
504         * Adds an annotation to the specified layer and sends a
505         * {@link RendererChangeEvent} to all registered listeners.
506         *
507         * @param annotation  the annotation (<code>null</code> not permitted).
508         * @param layer  the layer (<code>null</code> not permitted).
509         */
510        @Override
511        public void addAnnotation(XYAnnotation annotation, Layer layer) {
512            ParamChecks.nullNotPermitted(annotation, "annotation");
513            if (layer.equals(Layer.FOREGROUND)) {
514                this.foregroundAnnotations.add(annotation);
515                annotation.addChangeListener(this);
516                fireChangeEvent();
517            }
518            else if (layer.equals(Layer.BACKGROUND)) {
519                this.backgroundAnnotations.add(annotation);
520                annotation.addChangeListener(this);
521                fireChangeEvent();
522            }
523            else {
524                // should never get here
525                throw new RuntimeException("Unknown layer.");
526            }
527        }
528        /**
529         * Removes the specified annotation and sends a {@link RendererChangeEvent}
530         * to all registered listeners.
531         *
532         * @param annotation  the annotation to remove (<code>null</code> not
533         *                    permitted).
534         *
535         * @return A boolean to indicate whether or not the annotation was
536         *         successfully removed.
537         */
538        @Override
539        public boolean removeAnnotation(XYAnnotation annotation) {
540            boolean removed = this.foregroundAnnotations.remove(annotation);
541            removed = removed & this.backgroundAnnotations.remove(annotation);
542            annotation.removeChangeListener(this);
543            fireChangeEvent();
544            return removed;
545        }

547        /**
548         * Removes all annotations and sends a {@link RendererChangeEvent}
549         * to all registered listeners.
550         */
551        @Override
552        public void removeAnnotations() {
553            for(int i = 0; i < this.foregroundAnnotations.size(); i++){
554                XYAnnotation annotation 
555                        = (XYAnnotation) this.foregroundAnnotations.get(i);
556                annotation.removeChangeListener(this);
557            }
558             for(int i = 0; i < this.backgroundAnnotations.size(); i++){
559                XYAnnotation annotation 
560                        = (XYAnnotation) this.backgroundAnnotations.get(i);
561                annotation.removeChangeListener(this);
562            }
563            this.foregroundAnnotations.clear();
564            this.backgroundAnnotations.clear();
565            fireChangeEvent();
566        }

1324            else if (marker instanceof IntervalMarker) {
1325                IntervalMarker im = (IntervalMarker) marker;
1326                double start = im.getStartValue();
1327                double end = im.getEndValue();
1328                Range range = rangeAxis.getRange();
1329                if (!(range.intersects(start, end))) {
1330                    return;
1331                }
1332    
1333                double start2d = rangeAxis.valueToJava2D(start, dataArea,
1334                        plot.getRangeAxisEdge());
1335                double end2d = rangeAxis.valueToJava2D(end, dataArea,
1336                        plot.getRangeAxisEdge());
1337                double low = Math.min(start2d, end2d);
1338                double high = Math.max(start2d, end2d);
1339    
1340                PlotOrientation orientation = plot.getOrientation();
1341                Rectangle2D rect = null;
1342                if (orientation == PlotOrientation.HORIZONTAL) {
1343                    // clip left and right bounds to data area
1344                    low = Math.max(low, dataArea.getMinX());
1345                    high = Math.min(high, dataArea.getMaxX());
1346                    rect = new Rectangle2D.Double(low,
1347                            dataArea.getMinY(), high - low,
1348                            dataArea.getHeight());
1349                }
1350                else if (orientation == PlotOrientation.VERTICAL) {
1351                    // clip top and bottom bounds to data area
1352                    low = Math.max(low, dataArea.getMinY());
1353                    high = Math.min(high, dataArea.getMaxY());
1354                    rect = new Rectangle2D.Double(dataArea.getMinX(),
1355                            low, dataArea.getWidth(),
1356                            high - low);
1357                }
1358    
1359                final Composite originalComposite = g2.getComposite();
1360                g2.setComposite(AlphaComposite.getInstance(
1361                        AlphaComposite.SRC_OVER, marker.getAlpha()));
1362                Paint p = marker.getPaint();
1363                if (p instanceof GradientPaint) {
1364                    GradientPaint gp = (GradientPaint) p;
1365                    GradientPaintTransformer t = im.getGradientPaintTransformer();
1366                    if (t != null) {
1367                        gp = t.transform(gp, rect);
1368                    }
1369                    g2.setPaint(gp);
1370                }
1371                else {
1372                    g2.setPaint(p);
1373                }
1374                g2.fill(rect);
1375    
1376                // now draw the outlines, if visible...
1377                if (im.getOutlinePaint() != null && im.getOutlineStroke() != null) {
1378                    if (orientation == PlotOrientation.VERTICAL) {
1379                        Line2D line = new Line2D.Double();
1380                        double x0 = dataArea.getMinX();
1381                        double x1 = dataArea.getMaxX();
1382                        g2.setPaint(im.getOutlinePaint());
1383                        g2.setStroke(im.getOutlineStroke());
1384                        if (range.contains(start)) {
1385                            line.setLine(x0, start2d, x1, start2d);
1386                            g2.draw(line);
1387                        }
1388                        if (range.contains(end)) {
1389                            line.setLine(x0, end2d, x1, end2d);
1390                            g2.draw(line);
1391                        }
1392                    }
1393                    else { // PlotOrientation.HORIZONTAL
1394                        Line2D line = new Line2D.Double();
1395                        double y0 = dataArea.getMinY();
1396                        double y1 = dataArea.getMaxY();
1397                        g2.setPaint(im.getOutlinePaint());
1398                        g2.setStroke(im.getOutlineStroke());
1399                        if (range.contains(start)) {
1400                            line.setLine(start2d, y0, start2d, y1);
1401                            g2.draw(line);
1402                        }
1403                        if (range.contains(end)) {
1404                            line.setLine(end2d, y0, end2d, y1);
1405                            g2.draw(line);
1406                        }
1407                    }
1408                }
1409    
1410                String label = marker.getLabel();
1411                RectangleAnchor anchor = marker.getLabelAnchor();
1412                if (label != null) {
1413                    Font labelFont = marker.getLabelFont();
1414                    g2.setFont(labelFont);
1415                    g2.setPaint(marker.getLabelPaint());
1416                    Point2D coordinates = calculateRangeMarkerTextAnchorPoint(
1417                            g2, orientation, dataArea, rect,
1418                            marker.getLabelOffset(), marker.getLabelOffsetType(),
1419                            anchor);
1420                    TextUtilities.drawAlignedString(label, g2,
1421                            (float) coordinates.getX(), (float) coordinates.getY(),
1422                            marker.getLabelTextAnchor());
1423                }
1424                g2.setComposite(originalComposite);
1425            }
1426        }
