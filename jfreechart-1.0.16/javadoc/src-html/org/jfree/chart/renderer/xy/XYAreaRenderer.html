122    /**
123     * Area item renderer for an {@link XYPlot}.  This class can draw (a) shapes at
124     * each point, or (b) lines between points, or (c) both shapes and lines,
125     * or (d) filled areas, or (e) filled areas and shapes. The example shown here
126     * is generated by the <code>XYAreaRendererDemo1.java</code> program included
127     * in the JFreeChart demo collection:
128     * <br><br>
129     * <img src="../../../../../images/XYAreaRendererSample.png"
130     * alt="XYAreaRendererSample.png" />
131     */

231        /**
232         * Constructs a new renderer.  To specify the type of renderer, use one of
233         * the constants: <code>SHAPES</code>, <code>LINES</code>,
234         * <code>SHAPES_AND_LINES</code>, <code>AREA</code> or
235         * <code>AREA_AND_SHAPES</code>.
236         *
237         * @param type  the type of renderer.
238         * @param toolTipGenerator  the tool tip generator to use
239         *                          (<code>null</code> permitted).
240         * @param urlGenerator  the URL generator (<code>null</code> permitted).
241         */
242        public XYAreaRenderer(int type, XYToolTipGenerator toolTipGenerator,
243                              XYURLGenerator urlGenerator) {
244    
245            super();
246            setBaseToolTipGenerator(toolTipGenerator);
247            setURLGenerator(urlGenerator);
248    
249            if (type == SHAPES) {
250                this.plotShapes = true;
251            }
252            if (type == LINES) {
253                this.plotLines = true;
254            }
255            if (type == SHAPES_AND_LINES) {
256                this.plotShapes = true;
257                this.plotLines = true;
258            }
259            if (type == AREA) {
260                this.plotArea = true;
261            }
262            if (type == AREA_AND_SHAPES) {
263                this.plotArea = true;
264                this.plotShapes = true;
265            }
266            this.showOutline = false;
267            GeneralPath area = new GeneralPath();
268            area.moveTo(0.0f, -4.0f);
269            area.lineTo(3.0f, -2.0f);
270            area.lineTo(4.0f, 4.0f);
271            area.lineTo(-4.0f, 4.0f);
272            area.lineTo(-3.0f, -2.0f);
273            area.closePath();
274            this.legendArea = area;
275            this.useFillPaint = false;
276            this.gradientTransformer = new StandardGradientPaintTransformer();
277        }

426        /**
427         * Returns a default legend item for the specified series.  Subclasses
428         * should override this method to generate customised items.
429         *
430         * @param datasetIndex  the dataset index (zero-based).
431         * @param series  the series index (zero-based).
432         *
433         * @return A legend item for the series.
434         */
435        public LegendItem getLegendItem(int datasetIndex, int series) {
436            LegendItem result = null;
437            XYPlot xyplot = getPlot();
438            if (xyplot != null) {
439                XYDataset dataset = xyplot.getDataset(datasetIndex);
440                if (dataset != null) {
441                    XYSeriesLabelGenerator lg = getLegendItemLabelGenerator();
442                    String label = lg.generateLabel(dataset, series);
443                    String description = label;
444                    String toolTipText = null;
445                    if (getLegendItemToolTipGenerator() != null) {
446                        toolTipText = getLegendItemToolTipGenerator().generateLabel(
447                                dataset, series);
448                    }
449                    String urlText = null;
450                    if (getLegendItemURLGenerator() != null) {
451                        urlText = getLegendItemURLGenerator().generateLabel(
452                                dataset, series);
453                    }
454                    Paint paint = lookupSeriesPaint(series);
455                    result = new LegendItem(label, description, toolTipText,
456                            urlText, this.legendArea, paint);
457                    result.setLabelFont(lookupLegendTextFont(series));
458                    Paint labelPaint = lookupLegendTextPaint(series);
459                    if (labelPaint != null) {
460                        result.setLabelPaint(labelPaint);
461                    }
462                    result.setDataset(dataset);
463                    result.setDatasetIndex(datasetIndex);
464                    result.setSeriesKey(dataset.getSeriesKey(series));
465                    result.setSeriesIndex(series);
466                }
467            }
468            return result;
469        }

471        /**
472         * Draws the visual representation of a single data item.
473         *
474         * @param g2  the graphics device.
475         * @param state  the renderer state.
476         * @param dataArea  the area within which the data is being drawn.
477         * @param info  collects information about the drawing.
478         * @param plot  the plot (can be used to obtain standard color information
479         *              etc).
480         * @param domainAxis  the domain axis.
481         * @param rangeAxis  the range axis.
482         * @param dataset  the dataset.
483         * @param series  the series index (zero-based).
484         * @param item  the item index (zero-based).
485         * @param crosshairState  crosshair information for the plot
486         *                        (<code>null</code> permitted).
487         * @param pass  the pass index.
488         */
489        public void drawItem(Graphics2D g2, XYItemRendererState state,
490                Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
491                ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
492                int series, int item, CrosshairState crosshairState, int pass) {
493    
494            if (!getItemVisible(series, item)) {
495                return;
496            }
497            XYAreaRendererState areaState = (XYAreaRendererState) state;
498    
499            // get the data point...
500            double x1 = dataset.getXValue(series, item);
501            double y1 = dataset.getYValue(series, item);
502            if (Double.isNaN(y1)) {
503                y1 = 0.0;
504            }
505            double transX1 = domainAxis.valueToJava2D(x1, dataArea,
506                    plot.getDomainAxisEdge());
507            double transY1 = rangeAxis.valueToJava2D(y1, dataArea,
508                    plot.getRangeAxisEdge());
509    
510            // get the previous point and the next point so we can calculate a
511            // "hot spot" for the area (used by the chart entity)...
512            int itemCount = dataset.getItemCount(series);
513            double x0 = dataset.getXValue(series, Math.max(item - 1, 0));
514            double y0 = dataset.getYValue(series, Math.max(item - 1, 0));
515            if (Double.isNaN(y0)) {
516                y0 = 0.0;
517            }
518            double transX0 = domainAxis.valueToJava2D(x0, dataArea,
519                    plot.getDomainAxisEdge());
520            double transY0 = rangeAxis.valueToJava2D(y0, dataArea,
521                    plot.getRangeAxisEdge());
522    
523            double x2 = dataset.getXValue(series, Math.min(item + 1,
524                    itemCount - 1));
525            double y2 = dataset.getYValue(series, Math.min(item + 1,
526                    itemCount - 1));
527            if (Double.isNaN(y2)) {
528                y2 = 0.0;
529            }
530            double transX2 = domainAxis.valueToJava2D(x2, dataArea,
531                    plot.getDomainAxisEdge());
532            double transY2 = rangeAxis.valueToJava2D(y2, dataArea,
533                    plot.getRangeAxisEdge());
534    
535            double transZero = rangeAxis.valueToJava2D(0.0, dataArea,
536                    plot.getRangeAxisEdge());
537            GeneralPath hotspot = new GeneralPath();
538            if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
539                moveTo(hotspot, transZero, ((transX0 + transX1) / 2.0));
540                lineTo(hotspot, ((transY0 + transY1) / 2.0), 
541                                ((transX0 + transX1) / 2.0));
542                lineTo(hotspot, transY1, transX1);
543                lineTo(hotspot, ((transY1 + transY2) / 2.0), 
544                                ((transX1 + transX2) / 2.0));
545                lineTo(hotspot, transZero, ((transX1 + transX2) / 2.0));
546            }
547            else {  // vertical orientation
548                moveTo(hotspot, ((transX0 + transX1) / 2.0), transZero);
549                lineTo(hotspot, ((transX0 + transX1) / 2.0),
550                                ((transY0 + transY1) / 2.0));
551                lineTo(hotspot, transX1, transY1);
552                lineTo(hotspot, ((transX1 + transX2) / 2.0),
553                                ((transY1 + transY2) / 2.0));
554                lineTo(hotspot, ((transX1 + transX2) / 2.0), transZero);
555            }
556            hotspot.closePath();
557    
558            if (item == 0) {  // create a new area polygon for the series
559                areaState.area = new GeneralPath();
560                // the first point is (x, 0)
561                double zero = rangeAxis.valueToJava2D(0.0, dataArea,
562                        plot.getRangeAxisEdge());
563                if (plot.getOrientation() == PlotOrientation.VERTICAL) {
564                    moveTo(areaState.area, transX1, zero);
565                }
566                else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
567                    moveTo(areaState.area, zero, transX1);
568                }
569            }
570    
571            // Add each point to Area (x, y)
572            if (plot.getOrientation() == PlotOrientation.VERTICAL) {
573                lineTo(areaState.area, transX1, transY1);
574            }
575            else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
576                lineTo(areaState.area, transY1, transX1);
577            }
578    
579            PlotOrientation orientation = plot.getOrientation();
580            Paint paint = getItemPaint(series, item);
581            Stroke stroke = getItemStroke(series, item);
582            g2.setPaint(paint);
583            g2.setStroke(stroke);
584    
585            Shape shape = null;
586            if (getPlotShapes()) {
587                shape = getItemShape(series, item);
588                if (orientation == PlotOrientation.VERTICAL) {
589                    shape = ShapeUtilities.createTranslatedShape(shape, transX1,
590                            transY1);
591                }
592                else if (orientation == PlotOrientation.HORIZONTAL) {
593                    shape = ShapeUtilities.createTranslatedShape(shape, transY1,
594                            transX1);
595                }
596                g2.draw(shape);
597            }
598    
599            if (getPlotLines()) {
600                if (item > 0) {
601                    if (plot.getOrientation() == PlotOrientation.VERTICAL) {
602                        areaState.line.setLine(transX0, transY0, transX1, transY1);
603                    }
604                    else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
605                        areaState.line.setLine(transY0, transX0, transY1, transX1);
606                    }
607                    g2.draw(areaState.line);
608                }
609            }
610    
611            // Check if the item is the last item for the series.
612            // and number of items > 0.  We can't draw an area for a single point.
613            if (getPlotArea() && item > 0 && item == (itemCount - 1)) {
614    
615                if (orientation == PlotOrientation.VERTICAL) {
616                    // Add the last point (x,0)
617                    lineTo(areaState.area, transX1, transZero);
618                    areaState.area.closePath();
619                }
620                else if (orientation == PlotOrientation.HORIZONTAL) {
621                    // Add the last point (x,0)
622                    lineTo(areaState.area, transZero, transX1);
623                    areaState.area.closePath();
624                }
625    
626                if (this.useFillPaint) {
627                    paint = lookupSeriesFillPaint(series);
628                }
629                if (paint instanceof GradientPaint) {
630                    GradientPaint gp = (GradientPaint) paint;
631                    GradientPaint adjGP = this.gradientTransformer.transform(gp,
632                            dataArea);
633                    g2.setPaint(adjGP);
634                }
635                g2.fill(areaState.area);
636    
637                // draw an outline around the Area.
638                if (isOutline()) {
639                    Shape area = areaState.area;
640    
641                    // Java2D has some issues drawing dashed lines around "large"
642                    // geometrical shapes - for example, see bug 6620013 in the
643                    // Java bug database.  So, we'll check if the outline is
644                    // dashed and, if it is, do our own clipping before drawing
645                    // the outline...
646                    Stroke outlineStroke = lookupSeriesOutlineStroke(series);
647                    if (outlineStroke instanceof BasicStroke) {
648                        BasicStroke bs = (BasicStroke) outlineStroke;
649                        if (bs.getDashArray() != null) {
650                            Area poly = new Area(areaState.area);
651                            // we make the clip region slightly larger than the
652                            // dataArea so that the clipped edges don't show lines
653                            // on the chart
654                            Area clip = new Area(new Rectangle2D.Double(
655                                    dataArea.getX() - 5.0, dataArea.getY() - 5.0,
656                                    dataArea.getWidth() + 10.0,
657                                    dataArea.getHeight() + 10.0));
658                            poly.intersect(clip);
659                            area = poly;
660                        }
661                    } // end of workaround
662    
663                    g2.setStroke(outlineStroke);
664                    g2.setPaint(lookupSeriesOutlinePaint(series));
665                    g2.draw(area);
666                }
667            }
668    
669            int domainAxisIndex = plot.getDomainAxisIndex(domainAxis);
670            int rangeAxisIndex = plot.getRangeAxisIndex(rangeAxis);
671            updateCrosshairValues(crosshairState, x1, y1, domainAxisIndex,
672                    rangeAxisIndex, transX1, transY1, orientation);
673    
674            // collect entity and tool tip information...
675            EntityCollection entities = state.getEntityCollection();
676            if (entities != null) {
677                addEntity(entities, hotspot, dataset, series, item, 0.0, 0.0);
678            }
679    
680        }

726                return false;
727            }

729                return false;
