042    package org.jfree.chart.renderer;
043    
044    import java.awt.Paint;
045    
046    import org.jfree.chart.renderer.xy.XYBlockRenderer;
047    
048    /**
049     * A source for <code>Paint</code> instances, used by the
050     * {@link XYBlockRenderer}.
051     * <br><br>
052     * NOTE: Classes that implement this interface should also implement
053     * <code>PublicCloneable</code> and <code>Serializable</code>, so
054     * that any renderer (or other object instance) that references an instance of
055     * this interface can still be cloned or serialized.
056     *
057     * @since 1.0.4
058     */
059    public interface PaintScale {
060    
061        /**
062         * Returns the lower bound for the scale.
063         *
064         * @return The lower bound.
065         *
066         * @see #getUpperBound()
067         */
068        public double getLowerBound();
069    
070        /**
071         * Returns the upper bound for the scale.
072         *
073         * @return The upper bound.
074         *
075         * @see #getLowerBound()
076         */
077        public double getUpperBound();
078    
079        /**
080         * Returns a <code>Paint</code> instance for the specified value.
081         *
082         * @param value  the value.
083         *
084         * @return A <code>Paint</code> instance (never <code>null</code>).
085         */
086        public Paint getPaint(double value);
087    
088    }
