572        /**
573         * Initialises the renderer and returns a state object that will be used
574         * for the remainder of the drawing process for a single chart.  The state
575         * object allows for the fact that the renderer may be used simultaneously
576         * by multiple threads (each thread will work with a separate state object).
577         *
578         * @param g2  the graphics device.
579         * @param dataArea  the data area.
580         * @param plot  the plot.
581         * @param rendererIndex  the renderer index.
582         * @param info  an object for returning information about the structure of
583         *              the plot (<code>null</code> permitted).
584         *
585         * @return The renderer state.
586         */
587        @Override
588        public CategoryItemRendererState initialise(Graphics2D g2,
589                Rectangle2D dataArea, CategoryPlot plot, int rendererIndex,
590                PlotRenderingInfo info) {
591    
592            setPlot(plot);
593            CategoryDataset data = plot.getDataset(rendererIndex);
594            if (data != null) {
595                this.rowCount = data.getRowCount();
596                this.columnCount = data.getColumnCount();
597            }
598            else {
599                this.rowCount = 0;
600                this.columnCount = 0;
601            }
602            CategoryItemRendererState state = createState(info);
603            int[] visibleSeriesTemp = new int[this.rowCount];
604            int visibleSeriesCount = 0;
605            for (int row = 0; row < this.rowCount; row++) {
606                if (isSeriesVisible(row)) {
607                    visibleSeriesTemp[visibleSeriesCount] = row;
608                    visibleSeriesCount++;
609                }
610            }
611            int[] visibleSeries = new int[visibleSeriesCount];
612            System.arraycopy(visibleSeriesTemp, 0, visibleSeries, 0,
613                    visibleSeriesCount);
614            state.setVisibleSeriesArray(visibleSeries);
615            return state;
616        }

632        /**
633         * Returns the range of values the renderer requires to display all the
634         * items from the specified dataset.
635         *
636         * @param dataset  the dataset (<code>null</code> permitted).
637         * @param includeInterval  include the y-interval if the dataset has one.
638         *
639         * @return The range (<code>null</code> if the dataset is <code>null</code>
640         *         or empty).
641         *
642         * @since 1.0.13
643         */
644        protected Range findRangeBounds(CategoryDataset dataset,
645                boolean includeInterval) {
646            if (dataset == null) {
647                return null;
648            }
649            if (getDataBoundsIncludesVisibleSeriesOnly()) {
650                List visibleSeriesKeys = new ArrayList();
651                int seriesCount = dataset.getRowCount();
652                for (int s = 0; s < seriesCount; s++) {
653                    if (isSeriesVisible(s)) {
654                        visibleSeriesKeys.add(dataset.getRowKey(s));
655                    }
656                }
657                return DatasetUtilities.findRangeBounds(dataset,
658                        visibleSeriesKeys, includeInterval);
659            }
660            else {
661                return DatasetUtilities.findRangeBounds(dataset, includeInterval);
662            }
663        }

878            }

880            final Composite savedComposite = g2.getComposite();
881            g2.setComposite(AlphaComposite.getInstance(
882                    AlphaComposite.SRC_OVER, marker.getAlpha()));
883    
884            PlotOrientation orientation = plot.getOrientation();
885            Rectangle2D bounds;
886            if (marker.getDrawAsLine()) {

900                g2.setStroke(marker.getStroke());
901                g2.draw(line);
902                bounds = line.getBounds2D();
903            }
904            else {
905                double v0 = axis.getCategoryStart(columnIndex,
906                        dataset.getColumnCount(), dataArea,

917                    area = new Rectangle2D.Double(v0, dataArea.getMinY(),
918                            (v1 - v0), dataArea.getHeight());
919                }

922                bounds = area;
923            }
924

926            RectangleAnchor anchor = marker.getLabelAnchor();

932                        g2, orientation, dataArea, bounds, marker.getLabelOffset(),
933                        marker.getLabelOffsetType(), anchor);
934                TextUtilities.drawAlignedString(label, g2,
935                        (float) coordinates.getX(), (float) coordinates.getY(),
936                        marker.getLabelTextAnchor());

942         * Draws a marker for the range axis.
943         *
944         * @param g2  the graphics device (not <code>null</code>).
945         * @param plot  the plot (not <code>null</code>).
946         * @param axis  the range axis (not <code>null</code>).

949         *
950         * @see #drawDomainMarker(Graphics2D, CategoryPlot, CategoryAxis,

952         */
953        @Override
954        public void drawRangeMarker(Graphics2D g2, CategoryPlot plot,
955                ValueAxis axis, Marker marker, Rectangle2D dataArea) {
956    
957            if (marker instanceof ValueMarker) {
958                ValueMarker vm = (ValueMarker) marker;
959                double value = vm.getValue();
960                Range range = axis.getRange();
961    
962                if (!range.contains(value)) {
963                    return;
964                }
965    
966                final Composite savedComposite = g2.getComposite();
967                g2.setComposite(AlphaComposite.getInstance(
968                        AlphaComposite.SRC_OVER, marker.getAlpha()));
969    
970                PlotOrientation orientation = plot.getOrientation();
971                double v = axis.valueToJava2D(value, dataArea,
972                        plot.getRangeAxisEdge());
973                Line2D line = null;
974                if (orientation == PlotOrientation.HORIZONTAL) {
975                    line = new Line2D.Double(v, dataArea.getMinY(), v,
976                            dataArea.getMaxY());
977                }
978                else if (orientation == PlotOrientation.VERTICAL) {
979                    line = new Line2D.Double(dataArea.getMinX(), v,
980                            dataArea.getMaxX(), v);
981                }
982    
983                g2.setPaint(marker.getPaint());
984                g2.setStroke(marker.getStroke());
985                g2.draw(line);
986    
987                String label = marker.getLabel();
988                RectangleAnchor anchor = marker.getLabelAnchor();
989                if (label != null) {
990                    Font labelFont = marker.getLabelFont();
991                    g2.setFont(labelFont);
992                    g2.setPaint(marker.getLabelPaint());
993                    Point2D coordinates = calculateRangeMarkerTextAnchorPoint(
994                            g2, orientation, dataArea, line.getBounds2D(),
995                            marker.getLabelOffset(), LengthAdjustmentType.EXPAND,
996                            anchor);
997                    TextUtilities.drawAlignedString(label, g2,
998                            (float) coordinates.getX(), (float) coordinates.getY(),
999                            marker.getLabelTextAnchor());
1000                }
1001                g2.setComposite(savedComposite);
1002            }
1003            else if (marker instanceof IntervalMarker) {
1004                IntervalMarker im = (IntervalMarker) marker;
1005                double start = im.getStartValue();
1006                double end = im.getEndValue();
1007                Range range = axis.getRange();
1008                if (!(range.intersects(start, end))) {
1009                    return;
1010                }
1011    
1012                final Composite savedComposite = g2.getComposite();
1013                g2.setComposite(AlphaComposite.getInstance(
1014                        AlphaComposite.SRC_OVER, marker.getAlpha()));
1015    
1016                double start2d = axis.valueToJava2D(start, dataArea,
1017                        plot.getRangeAxisEdge());
1018                double end2d = axis.valueToJava2D(end, dataArea,
1019                        plot.getRangeAxisEdge());
1020                double low = Math.min(start2d, end2d);
1021                double high = Math.max(start2d, end2d);
1022    
1023                PlotOrientation orientation = plot.getOrientation();
1024                Rectangle2D rect = null;
1025                if (orientation == PlotOrientation.HORIZONTAL) {
1026                    // clip left and right bounds to data area
1027                    low = Math.max(low, dataArea.getMinX());
1028                    high = Math.min(high, dataArea.getMaxX());
1029                    rect = new Rectangle2D.Double(low,
1030                            dataArea.getMinY(), high - low,
1031                            dataArea.getHeight());
1032                }
1033                else if (orientation == PlotOrientation.VERTICAL) {
1034                    // clip top and bottom bounds to data area
1035                    low = Math.max(low, dataArea.getMinY());
1036                    high = Math.min(high, dataArea.getMaxY());
1037                    rect = new Rectangle2D.Double(dataArea.getMinX(),
1038                            low, dataArea.getWidth(),
1039                            high - low);
1040                }
1041                Paint p = marker.getPaint();
1042                if (p instanceof GradientPaint) {
1043                    GradientPaint gp = (GradientPaint) p;
1044                    GradientPaintTransformer t = im.getGradientPaintTransformer();
1045                    if (t != null) {
1046                        gp = t.transform(gp, rect);
1047                    }
1048                    g2.setPaint(gp);
1049                }
1050                else {
1051                    g2.setPaint(p);
1052                }
1053                g2.fill(rect);
1054    
1055                // now draw the outlines, if visible...
1056                if (im.getOutlinePaint() != null && im.getOutlineStroke() != null) {
1057                    if (orientation == PlotOrientation.VERTICAL) {
1058                        Line2D line = new Line2D.Double();
1059                        double x0 = dataArea.getMinX();
1060                        double x1 = dataArea.getMaxX();
1061                        g2.setPaint(im.getOutlinePaint());
1062                        g2.setStroke(im.getOutlineStroke());
1063                        if (range.contains(start)) {
1064                            line.setLine(x0, start2d, x1, start2d);
1065                            g2.draw(line);
1066                        }
1067                        if (range.contains(end)) {
1068                            line.setLine(x0, end2d, x1, end2d);
1069                            g2.draw(line);
1070                        }
1071                    }
1072                    else { // PlotOrientation.HORIZONTAL
1073                        Line2D line = new Line2D.Double();
1074                        double y0 = dataArea.getMinY();
1075                        double y1 = dataArea.getMaxY();
1076                        g2.setPaint(im.getOutlinePaint());
1077                        g2.setStroke(im.getOutlineStroke());
1078                        if (range.contains(start)) {
1079                            line.setLine(start2d, y0, start2d, y1);
1080                            g2.draw(line);
1081                        }
1082                        if (range.contains(end)) {
1083                            line.setLine(end2d, y0, end2d, y1);
1084                            g2.draw(line);
1085                        }
1086                    }
1087                }
1088    
1089                String label = marker.getLabel();
1090                RectangleAnchor anchor = marker.getLabelAnchor();
1091                if (label != null) {
1092                    Font labelFont = marker.getLabelFont();
1093                    g2.setFont(labelFont);
1094                    g2.setPaint(marker.getLabelPaint());
1095                    Point2D coordinates = calculateRangeMarkerTextAnchorPoint(
1096                            g2, orientation, dataArea, rect,
1097                            marker.getLabelOffset(), marker.getLabelOffsetType(),
1098                            anchor);
1099                    TextUtilities.drawAlignedString(label, g2,
1100                            (float) coordinates.getX(), (float) coordinates.getY(),
1101                            marker.getLabelTextAnchor());
1102                }
1103                g2.setComposite(savedComposite);
1104            }
1105        }

1563            LegendItemCollection result = new LegendItemCollection();
1564            if (this.plot == null) {
1565                return result;
1566            }
1567            int index = this.plot.getIndexOf(this);
1568            CategoryDataset dataset = this.plot.getDataset(index);
1569            if (dataset == null) {
1570                return result;
1571            }
1572            int seriesCount = dataset.getRowCount();
1573            if (plot.getRowRenderingOrder().equals(SortOrder.ASCENDING)) {
1574                for (int i = 0; i < seriesCount; i++) {
1575                    if (isSeriesVisibleInLegend(i)) {
1576                        LegendItem item = getLegendItem(index, i);
1577                        if (item != null) {
1578                            result.add(item);
1579                        }
1580                    }
1581                }
1582            }
1583            else {
1584                for (int i = seriesCount - 1; i >= 0; i--) {
1585                    if (isSeriesVisibleInLegend(i)) {
1586                        LegendItem item = getLegendItem(index, i);
1587                        if (item != null) {
1588                            result.add(item);

1736            CategoryToolTipGenerator generator = getToolTipGenerator(row, column);
1737            if (generator != null) {
