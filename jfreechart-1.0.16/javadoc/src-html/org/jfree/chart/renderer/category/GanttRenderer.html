229         * Draws the bar for a single (series, category) data item.
230         *
231         * @param g2  the graphics device.
232         * @param state  the renderer state.
233         * @param dataArea  the data area.
234         * @param plot  the plot.
235         * @param domainAxis  the domain axis.
236         * @param rangeAxis  the range axis.

238         * @param row  the row index (zero-based).
239         * @param column  the column index (zero-based).
240         * @param pass  the pass index.

242        public void drawItem(Graphics2D g2,
243                             CategoryItemRendererState state,
244                             Rectangle2D dataArea,
245                             CategoryPlot plot,
246                             CategoryAxis domainAxis,
247                             ValueAxis rangeAxis,
248                             CategoryDataset dataset,
249                             int row,
250                             int column,
251                             int pass) {
252    
253             if (dataset instanceof GanttCategoryDataset) {
254                 GanttCategoryDataset gcd = (GanttCategoryDataset) dataset;
255                 drawTasks(g2, state, dataArea, plot, domainAxis, rangeAxis, gcd,
256                         row, column);
257             }
258             else {  // let the superclass handle it...
259                 super.drawItem(g2, state, dataArea, plot, domainAxis, rangeAxis,
260                         dataset, row, column, pass);
261             }
262    
263         }

265        /**
266         * Draws the tasks/subtasks for one item.
267         *
268         * @param g2  the graphics device.
269         * @param state  the renderer state.
270         * @param dataArea  the data plot area.
271         * @param plot  the plot.
272         * @param domainAxis  the domain axis.
273         * @param rangeAxis  the range axis.
274         * @param dataset  the data.
275         * @param row  the row index (zero-based).
276         * @param column  the column index (zero-based).
277         */
278        protected void drawTasks(Graphics2D g2,
279                                 CategoryItemRendererState state,
280                                 Rectangle2D dataArea,
281                                 CategoryPlot plot,
282                                 CategoryAxis domainAxis,
283                                 ValueAxis rangeAxis,
284                                 GanttCategoryDataset dataset,
285                                 int row,
286                                 int column) {
287    
288            int count = dataset.getSubIntervalCount(row, column);
289            if (count == 0) {
290                drawTask(g2, state, dataArea, plot, domainAxis, rangeAxis,
291                        dataset, row, column);
292            }
293    
294            PlotOrientation orientation = plot.getOrientation();
295            for (int subinterval = 0; subinterval < count; subinterval++) {
296    
297                RectangleEdge rangeAxisLocation = plot.getRangeAxisEdge();
298    
299                // value 0
300                Number value0 = dataset.getStartValue(row, column, subinterval);
301                if (value0 == null) {
302                    return;
303                }
304                double translatedValue0 = rangeAxis.valueToJava2D(
305                        value0.doubleValue(), dataArea, rangeAxisLocation);
306    
307                // value 1
308                Number value1 = dataset.getEndValue(row, column, subinterval);
309                if (value1 == null) {
310                    return;
311                }
312                double translatedValue1 = rangeAxis.valueToJava2D(
313                        value1.doubleValue(), dataArea, rangeAxisLocation);
314    
315                if (translatedValue1 < translatedValue0) {
316                    double temp = translatedValue1;
317                    translatedValue1 = translatedValue0;
318                    translatedValue0 = temp;
319                }
320    
321                double rectStart = calculateBarW0(plot, plot.getOrientation(),
322                        dataArea, domainAxis, state, row, column);
323                double rectLength = Math.abs(translatedValue1 - translatedValue0);
324                double rectBreadth = state.getBarWidth();
325    
326                // DRAW THE BARS...
327                Rectangle2D bar = null;
328                RectangleEdge barBase = null;
329                if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
330                    bar = new Rectangle2D.Double(translatedValue0, rectStart,
331                            rectLength, rectBreadth);
332                    barBase = RectangleEdge.LEFT;
333                }
334                else if (plot.getOrientation() == PlotOrientation.VERTICAL) {
335                    bar = new Rectangle2D.Double(rectStart, translatedValue0,
336                            rectBreadth, rectLength);
337                    barBase = RectangleEdge.BOTTOM;
338                }
339    
340                Rectangle2D completeBar = null;
341                Rectangle2D incompleteBar = null;
342                Number percent = dataset.getPercentComplete(row, column,
343                        subinterval);
344                double start = getStartPercent();
345                double end = getEndPercent();
346                if (percent != null) {
347                    double p = percent.doubleValue();
348                    if (orientation == PlotOrientation.HORIZONTAL) {
349                        completeBar = new Rectangle2D.Double(translatedValue0,
350                                rectStart + start * rectBreadth, rectLength * p,
351                                rectBreadth * (end - start));
352                        incompleteBar = new Rectangle2D.Double(translatedValue0
353                                + rectLength * p, rectStart + start * rectBreadth,
354                                rectLength * (1 - p), rectBreadth * (end - start));
355                    }
356                    else if (orientation == PlotOrientation.VERTICAL) {
357                        completeBar = new Rectangle2D.Double(rectStart + start
358                                * rectBreadth, translatedValue0 + rectLength
359                                * (1 - p), rectBreadth * (end - start),
360                                rectLength * p);
361                        incompleteBar = new Rectangle2D.Double(rectStart + start
362                                * rectBreadth, translatedValue0, rectBreadth
363                                * (end - start), rectLength * (1 - p));
364                    }
365    
366                }
367    
368                if (getShadowsVisible()) {
369                    getBarPainter().paintBarShadow(g2, this, row, column, bar,
370                            barBase, true);
371                }
372                getBarPainter().paintBar(g2, this, row, column, bar, barBase);
373    
374                if (completeBar != null) {
375                    g2.setPaint(getCompletePaint());
376                    g2.fill(completeBar);
377                }
378                if (incompleteBar != null) {
379                    g2.setPaint(getIncompletePaint());
380                    g2.fill(incompleteBar);
381                }
382                if (isDrawBarOutline()
383                        && state.getBarWidth() > BAR_OUTLINE_WIDTH_THRESHOLD) {
384                    g2.setStroke(getItemStroke(row, column));
385                    g2.setPaint(getItemOutlinePaint(row, column));
386                    g2.draw(bar);
387                }
388    
389                if (subinterval == count - 1) {
390                    // submit the current data point as a crosshair candidate
391                    int datasetIndex = plot.indexOf(dataset);
392                    Comparable columnKey = dataset.getColumnKey(column);
393                    Comparable rowKey = dataset.getRowKey(row);
394                    double xx = domainAxis.getCategorySeriesMiddle(columnKey,
395                            rowKey, dataset, getItemMargin(), dataArea,
396                            plot.getDomainAxisEdge());
397                    updateCrosshairValues(state.getCrosshairState(),
398                            dataset.getRowKey(row), dataset.getColumnKey(column),
399                            value1.doubleValue(), datasetIndex, xx,
400                            translatedValue1, orientation);
401    
402                }
403                // collect entity and tool tip information...
404                if (state.getInfo() != null) {
405                    EntityCollection entities = state.getEntityCollection();
406                    if (entities != null) {
407                        addItemEntity(entities, dataset, row, column, bar);
408                    }
409                }
410            }
411        }

413        /**
414         * Draws a single task.
415         *
416         * @param g2  the graphics device.
417         * @param state  the renderer state.
418         * @param dataArea  the data plot area.
419         * @param plot  the plot.
420         * @param domainAxis  the domain axis.
421         * @param rangeAxis  the range axis.
422         * @param dataset  the data.
423         * @param row  the row index (zero-based).
424         * @param column  the column index (zero-based).
425         */
426        protected void drawTask(Graphics2D g2,
427                                CategoryItemRendererState state,
428                                Rectangle2D dataArea,
429                                CategoryPlot plot,
430                                CategoryAxis domainAxis,
431                                ValueAxis rangeAxis,
432                                GanttCategoryDataset dataset,
433                                int row,
434                                int column) {
435    
436            PlotOrientation orientation = plot.getOrientation();
437            RectangleEdge rangeAxisLocation = plot.getRangeAxisEdge();
438    
439            // Y0
440            Number value0 = dataset.getEndValue(row, column);
441            if (value0 == null) {
442                return;
443            }
444            double java2dValue0 = rangeAxis.valueToJava2D(value0.doubleValue(),
445                    dataArea, rangeAxisLocation);
446    
447            // Y1
448            Number value1 = dataset.getStartValue(row, column);
449            if (value1 == null) {
450                return;
451            }
452            double java2dValue1 = rangeAxis.valueToJava2D(value1.doubleValue(),
453                    dataArea, rangeAxisLocation);
454    
455            if (java2dValue1 < java2dValue0) {
456                double temp = java2dValue1;
457                java2dValue1 = java2dValue0;
458                java2dValue0 = temp;
459                value1 = value0;
460            }
461    
462            double rectStart = calculateBarW0(plot, orientation, dataArea,
463                    domainAxis, state, row, column);
464            double rectBreadth = state.getBarWidth();
465            double rectLength = Math.abs(java2dValue1 - java2dValue0);
466    
467            Rectangle2D bar = null;
468            RectangleEdge barBase = null;
469            if (orientation == PlotOrientation.HORIZONTAL) {
470                bar = new Rectangle2D.Double(java2dValue0, rectStart, rectLength,
471                        rectBreadth);
472                barBase = RectangleEdge.LEFT;
473            }
474            else if (orientation == PlotOrientation.VERTICAL) {
475                bar = new Rectangle2D.Double(rectStart, java2dValue1, rectBreadth,
476                        rectLength);
477                barBase = RectangleEdge.BOTTOM;
478            }
479    
480            Rectangle2D completeBar = null;
481            Rectangle2D incompleteBar = null;
482            Number percent = dataset.getPercentComplete(row, column);
483            double start = getStartPercent();
484            double end = getEndPercent();
485            if (percent != null) {
486                double p = percent.doubleValue();
487                if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
488                    completeBar = new Rectangle2D.Double(java2dValue0,
489                            rectStart + start * rectBreadth, rectLength * p,
490                            rectBreadth * (end - start));
491                    incompleteBar = new Rectangle2D.Double(java2dValue0
492                            + rectLength * p, rectStart + start * rectBreadth,
493                            rectLength * (1 - p), rectBreadth * (end - start));
494                }
495                else if (plot.getOrientation() == PlotOrientation.VERTICAL) {
496                    completeBar = new Rectangle2D.Double(rectStart + start
497                            * rectBreadth, java2dValue1 + rectLength * (1 - p),
498                            rectBreadth * (end - start), rectLength * p);
499                    incompleteBar = new Rectangle2D.Double(rectStart + start
500                            * rectBreadth, java2dValue1, rectBreadth * (end
501                            - start), rectLength * (1 - p));
502                }
503    
504            }
505    
506            if (getShadowsVisible()) {
507                getBarPainter().paintBarShadow(g2, this, row, column, bar,
508                        barBase, true);
509            }
510            getBarPainter().paintBar(g2, this, row, column, bar, barBase);
511    
512            if (completeBar != null) {
513                g2.setPaint(getCompletePaint());
514                g2.fill(completeBar);
515            }
516            if (incompleteBar != null) {
517                g2.setPaint(getIncompletePaint());
518                g2.fill(incompleteBar);
519            }
520    
521            // draw the outline...
522            if (isDrawBarOutline()
523                    && state.getBarWidth() > BAR_OUTLINE_WIDTH_THRESHOLD) {
524                Stroke stroke = getItemOutlineStroke(row, column);
525                Paint paint = getItemOutlinePaint(row, column);
526                if (stroke != null && paint != null) {
527                    g2.setStroke(stroke);
528                    g2.setPaint(paint);
529                    g2.draw(bar);
530                }
531            }
532    
533            CategoryItemLabelGenerator generator = getItemLabelGenerator(row,
534                    column);
535            if (generator != null && isItemLabelVisible(row, column)) {
536                drawItemLabel(g2, dataset, row, column, plot, generator, bar,
537                        false);
538            }
539    
540            // submit the current data point as a crosshair candidate
541            int datasetIndex = plot.indexOf(dataset);
542            Comparable columnKey = dataset.getColumnKey(column);
543            Comparable rowKey = dataset.getRowKey(row);
544            double xx = domainAxis.getCategorySeriesMiddle(columnKey, rowKey,
545                    dataset, getItemMargin(), dataArea, plot.getDomainAxisEdge());
546            updateCrosshairValues(state.getCrosshairState(),
547                    dataset.getRowKey(row), dataset.getColumnKey(column),
548                    value1.doubleValue(), datasetIndex, xx, java2dValue1,
549                    orientation);
550    
551            // collect entity and tool tip information...
552            EntityCollection entities = state.getEntityCollection();
553            if (entities != null) {
554                addItemEntity(entities, dataset, row, column, bar);
555            }
556        }

558        /**
559         * Returns the Java2D coordinate for the middle of the specified data item.
560         *
561         * @param rowKey  the row key.
562         * @param columnKey  the column key.
563         * @param dataset  the dataset.
564         * @param axis  the axis.
565         * @param area  the drawing area.
566         * @param edge  the edge along which the axis lies.
567         *
568         * @return The Java2D coordinate.
569         *
570         * @since 1.0.11

572        public double getItemMiddle(Comparable rowKey, Comparable columnKey,
573                CategoryDataset dataset, CategoryAxis axis, Rectangle2D area,
574                RectangleEdge edge) {
575            return axis.getCategorySeriesMiddle(columnKey, rowKey, dataset,
576                    getItemMargin(), area, edge);
577        }

579        /**
580         * Tests this renderer for equality with an arbitrary object.
581         *
582         * @param obj  the object (<code>null</code> permitted).
583         *
584         * @return A boolean.
585         */
586        public boolean equals(Object obj) {
587            if (obj == this) {
588                return true;
589            }
590            if (!(obj instanceof GanttRenderer)) {
591                return false;
592            }
593            GanttRenderer that = (GanttRenderer) obj;
594            if (!PaintUtilities.equal(this.completePaint, that.completePaint)) {
595                return false;
596            }
597            if (!PaintUtilities.equal(this.incompletePaint, that.incompletePaint)) {
598                return false;
599            }
600            if (this.startPercent != that.startPercent) {
601                return false;
602            }
603            if (this.endPercent != that.endPercent) {
604                return false;
605            }
606            return super.equals(obj);
607        }

609        /**
610         * Provides serialization support.
611         *
612         * @param stream  the output stream.
613         *
614         * @throws IOException  if there is an I/O error.
615         */

617            stream.defaultWriteObject();
618            SerialUtilities.writePaint(this.completePaint, stream);
619            SerialUtilities.writePaint(this.incompletePaint, stream);
620        }

624         *
625         * @param stream  the input stream.
626         *
627         * @throws IOException  if there is an I/O error.
628         * @throws ClassNotFoundException  if there is a classpath problem.
629         */
630        private void readObject(ObjectInputStream stream)
631            throws IOException, ClassNotFoundException {
632            stream.defaultReadObject();
633            this.completePaint = SerialUtilities.readPaint(stream);
634            this.incompletePaint = SerialUtilities.readPaint(stream);
635        }
