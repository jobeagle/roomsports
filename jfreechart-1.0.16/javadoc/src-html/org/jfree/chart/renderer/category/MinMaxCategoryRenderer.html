094     * Renderer for drawing min max plot. This renderer draws all the series under

096     * a line from the maximum value to the minimum value. For use with the
097     * {@link CategoryPlot} class. The example shown here is generated by
098     * the <code>MinMaxCategoryPlotDemo1.java</code> program included in the
099     * JFreeChart Demo Collection:
100     * <br><br>
101     * <img src="../../../../../images/MinMaxCategoryRendererSample.png"
102     * alt="MinMaxCategoryRendererSample.png" />

309        /**
310         * Draw a single data item.
311         *
312         * @param g2  the graphics device.
313         * @param state  the renderer state.
314         * @param dataArea  the area in which the data is drawn.
315         * @param plot  the plot.
316         * @param domainAxis  the domain axis.
317         * @param rangeAxis  the range axis.
318         * @param dataset  the dataset.
319         * @param row  the row index (zero-based).
320         * @param column  the column index (zero-based).
321         * @param pass  the pass index.
322         */
323        public void drawItem(Graphics2D g2, CategoryItemRendererState state,
324                Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
325                ValueAxis rangeAxis, CategoryDataset dataset, int row, int column,
326                int pass) {
327    
328            // first check the number we are plotting...
329            Number value = dataset.getValue(row, column);
330            if (value != null) {
331                // current data point...
332                double x1 = domainAxis.getCategoryMiddle(column, getColumnCount(),
333                        dataArea, plot.getDomainAxisEdge());
334                double y1 = rangeAxis.valueToJava2D(value.doubleValue(), dataArea,
335                        plot.getRangeAxisEdge());
336                Shape hotspot = new Rectangle2D.Double(x1 - 4, y1 - 4, 8.0, 8.0);
337    
338                g2.setPaint(getItemPaint(row, column));
339                g2.setStroke(getItemStroke(row, column));
340    
341                PlotOrientation orient = plot.getOrientation();
342                if (orient == PlotOrientation.VERTICAL) {
343                    this.objectIcon.paintIcon(null, g2, (int) x1, (int) y1);
344                }
345                else {
346                    this.objectIcon.paintIcon(null, g2, (int) y1, (int) x1);
347                }
348    
349                if (this.lastCategory == column) {
350                    if (this.min > value.doubleValue()) {
351                        this.min = value.doubleValue();
352                    }
353                    if (this.max < value.doubleValue()) {
354                        this.max = value.doubleValue();
355                    }
356    
357                    // last series, so we are ready to draw the min and max
358                    if (dataset.getRowCount() - 1 == row) {
359                        g2.setPaint(this.groupPaint);
360                        g2.setStroke(this.groupStroke);
361                        double minY = rangeAxis.valueToJava2D(this.min, dataArea,
362                                plot.getRangeAxisEdge());
363                        double maxY = rangeAxis.valueToJava2D(this.max, dataArea,
364                                plot.getRangeAxisEdge());
365    
366                        if (orient == PlotOrientation.VERTICAL) {
367                            g2.draw(new Line2D.Double(x1, minY, x1, maxY));
368                            this.minIcon.paintIcon(null, g2, (int) x1, (int) minY);
369                            this.maxIcon.paintIcon(null, g2, (int) x1, (int) maxY);
370                        }
371                        else {
372                            g2.draw(new Line2D.Double(minY, x1, maxY, x1));
373                            this.minIcon.paintIcon(null, g2, (int) minY, (int) x1);
374                            this.maxIcon.paintIcon(null, g2, (int) maxY, (int) x1);
375                        }
376                    }
377                }
378                else {  // reset the min and max
379                    this.lastCategory = column;
380                    this.min = value.doubleValue();
381                    this.max = value.doubleValue();
382                }
383    
384                // connect to the previous point
385                if (this.plotLines) {
386                    if (column != 0) {
387                        Number previousValue = dataset.getValue(row, column - 1);
388                        if (previousValue != null) {
389                            // previous data point...
390                            double previous = previousValue.doubleValue();
391                            double x0 = domainAxis.getCategoryMiddle(column - 1,
392                                    getColumnCount(), dataArea,
393                                    plot.getDomainAxisEdge());
394                            double y0 = rangeAxis.valueToJava2D(previous, dataArea,
395                                    plot.getRangeAxisEdge());
396                            g2.setPaint(getItemPaint(row, column));
397                            g2.setStroke(getItemStroke(row, column));
398                            Line2D line;
399                            if (orient == PlotOrientation.VERTICAL) {
400                                line = new Line2D.Double(x0, y0, x1, y1);
401                            }
402                            else {
403                                line = new Line2D.Double(y0, x0, y1, x1);
404                            }
405                            g2.draw(line);
406                        }
407                    }
408                }
409    
410                // add an item entity, if this information is being collected
411                EntityCollection entities = state.getEntityCollection();
412                if (entities != null) {
413                    addItemEntity(entities, dataset, row, column, hotspot);
414                }
415            }
416        }

418        /**
419         * Tests this instance for equality with an arbitrary object.  The icon
420         * fields are NOT included in the test, so this implementation is a little
421         * weak.
422         *
423         * @param obj  the object (<code>null</code> permitted).
424         *
425         * @return A boolean.
426         *
427         * @since 1.0.7
428         */
429        public boolean equals(Object obj) {
430            if (obj == this) {
431                return true;
432            }
433            if (!(obj instanceof MinMaxCategoryRenderer)) {
434                return false;
435            }
436            MinMaxCategoryRenderer that = (MinMaxCategoryRenderer) obj;
437            if (this.plotLines != that.plotLines) {
438                return false;
439            }
440            if (!PaintUtilities.equal(this.groupPaint, that.groupPaint)) {
441                return false;
442            }
443            if (!this.groupStroke.equals(that.groupStroke)) {
444                return false;
445            }
446            return super.equals(obj);
447        }

449        /**
450         * Returns an icon.
451         *
452         * @param shape  the shape.
453         * @param fillPaint  the fill paint.
454         * @param outlinePaint  the outline paint.
455         *
456         * @return The icon.
457         */
458        private Icon getIcon(Shape shape, final Paint fillPaint,
459                            final Paint outlinePaint) {
460    
461          final int width = shape.getBounds().width;
462          final int height = shape.getBounds().height;
463          final GeneralPath path = new GeneralPath(shape);
464          return new Icon() {
465              public void paintIcon(Component c, Graphics g, int x, int y) {
466                  Graphics2D g2 = (Graphics2D) g;
467                  path.transform(AffineTransform.getTranslateInstance(x, y));
468                  if (fillPaint != null) {
469                      g2.setPaint(fillPaint);
470                      g2.fill(path);
471                  }
472                  if (outlinePaint != null) {
473                      g2.setPaint(outlinePaint);
474                      g2.draw(path);
475                  }
476                  path.transform(AffineTransform.getTranslateInstance(-x, -y));
477            }
478    
479            public int getIconWidth() {
480                return width;
481            }
482    
483            public int getIconHeight() {
484                return height;
485            }
486    
487          };
488        }
489    
490        /**
491         * Returns an icon from a shape.
492         *
493         * @param shape  the shape.
494         * @param fill  the fill flag.
495         * @param outline  the outline flag.
496         *
497         * @return The icon.
498         */
499        private Icon getIcon(Shape shape, final boolean fill,
500                final boolean outline) {
501            final int width = shape.getBounds().width;
502            final int height = shape.getBounds().height;
503            final GeneralPath path = new GeneralPath(shape);
504            return new Icon() {
505                public void paintIcon(Component c, Graphics g, int x, int y) {
506                    Graphics2D g2 = (Graphics2D) g;
507                    path.transform(AffineTransform.getTranslateInstance(x, y));
508                    if (fill) {
509                        g2.fill(path);
510                    }
511                    if (outline) {
512                        g2.draw(path);
513                    }
514                    path.transform(AffineTransform.getTranslateInstance(-x, -y));
515                }
516    
517                public int getIconWidth() {
518                    return width;
519                }
520    
521                public int getIconHeight() {
522                    return height;
523                }
524            };
525        }

527        /**
528         * Provides serialization support.
529         *
530         * @param stream  the output stream.
531         *
532         * @throws IOException  if there is an I/O error.
533         */
534        private void writeObject(ObjectOutputStream stream) throws IOException {
535            stream.defaultWriteObject();
536            SerialUtilities.writeStroke(this.groupStroke, stream);
537            SerialUtilities.writePaint(this.groupPaint, stream);
538        }
539    
540        /**
541         * Provides serialization support.
542         *
543         * @param stream  the input stream.
544         *
545         * @throws IOException  if there is an I/O error.
546         * @throws ClassNotFoundException  if there is a classpath problem.
547         */
548        private void readObject(ObjectInputStream stream)
549            throws IOException, ClassNotFoundException {
550            stream.defaultReadObject();
551            this.groupStroke = SerialUtilities.readStroke(stream);
552            this.groupPaint = SerialUtilities.readPaint(stream);
553    
554            this.minIcon = getIcon(new Arc2D.Double(-4, -4, 8, 8, 0, 360,
555                    Arc2D.OPEN), null, Color.black);
556            this.maxIcon = getIcon(new Arc2D.Double(-4, -4, 8, 8, 0, 360,
557                    Arc2D.OPEN), null, Color.black);
558            this.objectIcon = getIcon(new Line2D.Double(-4, 0, 4, 0), false, true);
559        }
