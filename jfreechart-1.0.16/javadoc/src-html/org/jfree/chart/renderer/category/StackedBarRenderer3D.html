269        /**
270         * Calculates the bar width and stores it in the renderer state.
271         *
272         * @param plot  the plot.
273         * @param dataArea  the data area.
274         * @param rendererIndex  the renderer index.
275         * @param state  the renderer state.
276         */
277        protected void calculateBarWidth(CategoryPlot plot,
278                                         Rectangle2D dataArea,
279                                         int rendererIndex,
280                                         CategoryItemRendererState state) {
281    
282            // calculate the bar width
283            CategoryAxis domainAxis = getDomainAxis(plot, rendererIndex);
284            CategoryDataset data = plot.getDataset(rendererIndex);
285            if (data != null) {
286                PlotOrientation orientation = plot.getOrientation();
287                double space = 0.0;
288                if (orientation == PlotOrientation.HORIZONTAL) {
289                    space = dataArea.getHeight();
290                }
291                else if (orientation == PlotOrientation.VERTICAL) {
292                    space = dataArea.getWidth();
293                }
294                double maxWidth = space * getMaximumBarWidth();
295                int columns = data.getColumnCount();
296                double categoryMargin = 0.0;
297                if (columns > 1) {
298                    categoryMargin = domainAxis.getCategoryMargin();
299                }
300    
301                double used = space * (1 - domainAxis.getLowerMargin()
302                                         - domainAxis.getUpperMargin()
303                                         - categoryMargin);
304                if (columns > 0) {
305                    state.setBarWidth(Math.min(used / columns, maxWidth));
306                }
307                else {
308                    state.setBarWidth(Math.min(used, maxWidth));
309                }
310            }
311    
312        }

314        /**
315         * Returns a list containing the stacked values for the specified series
316         * in the given dataset, plus the supplied base value.
317         *
318         * @param dataset  the dataset (<code>null</code> not permitted).
319         * @param category  the category key (<code>null</code> not permitted).
320         * @param base  the base value.
321         * @param asPercentages  a flag that controls whether the values in the
322         *     list are converted to percentages of the total.
323         *
324         * @return The value list.
325         *
326         * @since 1.0.4
327         *
328         * @deprecated As of 1.0.13, use {@link #createStackedValueList(
329         *     CategoryDataset, Comparable, int[], double, boolean)}.
330         */
331        protected List createStackedValueList(CategoryDataset dataset,
332                Comparable category, double base, boolean asPercentages) {
333            int[] rows = new int[dataset.getRowCount()];
334            for (int i = 0; i < rows.length; i++) {
335                rows[i] = i;
336            }
337            return createStackedValueList(dataset, category, rows, base,
338                    asPercentages);
339        }
340    
341        /**
342         * Returns a list containing the stacked values for the specified series
343         * in the given dataset, plus the supplied base value.
344         *
345         * @param dataset  the dataset (<code>null</code> not permitted).
346         * @param category  the category key (<code>null</code> not permitted).
347         * @param includedRows  the included rows.
348         * @param base  the base value.
349         * @param asPercentages  a flag that controls whether the values in the
350         *     list are converted to percentages of the total.
351         *
352         * @return The value list.
353         *
354         * @since 1.0.13
355         */
356        protected List createStackedValueList(CategoryDataset dataset,
357                Comparable category, int[] includedRows, double base,
358                boolean asPercentages) {
359    
360            List result = new ArrayList();
361            double posBase = base;
362            double negBase = base;
363            double total = 0.0;
364            if (asPercentages) {
365                total = DataUtilities.calculateColumnTotal(dataset,
366                        dataset.getColumnIndex(category), includedRows);
367            }
368    
369            int baseIndex = -1;
370            int rowCount = includedRows.length;
371            for (int i = 0; i < rowCount; i++) {
372                int r = includedRows[i];
373                Number n = dataset.getValue(dataset.getRowKey(r), category);
374                if (n == null) {
375                    continue;
376                }
377                double v = n.doubleValue();
378                if (asPercentages) {
379                    v = v / total;
380                }
381                if ((v > 0.0) || (!this.ignoreZeroValues && v >= 0.0)) {
382                    if (baseIndex < 0) {
383                        result.add(new Object[] {null, new Double(base)});
384                        baseIndex = 0;
385                    }
386                    posBase = posBase + v;
387                    result.add(new Object[] {new Integer(r), new Double(posBase)});
388                }
389                else if (v < 0.0) {
390                    if (baseIndex < 0) {
391                        result.add(new Object[] {null, new Double(base)});
392                        baseIndex = 0;
393                    }
394                    negBase = negBase + v; // '+' because v is negative
395                    result.add(0, new Object[] {new Integer(-r - 1),
396                            new Double(negBase)});
397                    baseIndex++;
398                }
399            }
400            return result;
401    
402        }

404        /**
405         * Draws the visual representation of one data item from the chart (in
406         * fact, this method does nothing until it reaches the last item for each
407         * category, at which point it draws all the items for that category).
408         *
409         * @param g2  the graphics device.
410         * @param state  the renderer state.
411         * @param dataArea  the plot area.
412         * @param plot  the plot.
413         * @param domainAxis  the domain (category) axis.
414         * @param rangeAxis  the range (value) axis.
415         * @param dataset  the data.
416         * @param row  the row index (zero-based).
417         * @param column  the column index (zero-based).
418         * @param pass  the pass index.
419         */
420        public void drawItem(Graphics2D g2, CategoryItemRendererState state,
421                Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
422                ValueAxis rangeAxis, CategoryDataset dataset, int row, int column,
423                int pass) {
424    
425            // wait till we are at the last item for the row then draw the
426            // whole stack at once
427            if (row < dataset.getRowCount() - 1) {
428                return;
429            }
430            Comparable category = dataset.getColumnKey(column);
431    
432            List values = createStackedValueList(dataset,
433                    dataset.getColumnKey(column), state.getVisibleSeriesArray(),
434                    getBase(), this.renderAsPercentages);
435    
436            Rectangle2D adjusted = new Rectangle2D.Double(dataArea.getX(),
437                    dataArea.getY() + getYOffset(),
438                    dataArea.getWidth() - getXOffset(),
439                    dataArea.getHeight() - getYOffset());
440    
441    
442            PlotOrientation orientation = plot.getOrientation();
443    
444            // handle rendering separately for the two plot orientations...
445            if (orientation == PlotOrientation.HORIZONTAL) {
446                drawStackHorizontal(values, category, g2, state, adjusted, plot,
447                        domainAxis, rangeAxis, dataset);
448            }
449            else {
450                drawStackVertical(values, category, g2, state, adjusted, plot,
451                        domainAxis, rangeAxis, dataset);
452            }
453    
454        }

456        /**
457         * Draws a stack of bars for one category, with a horizontal orientation.
458         *
459         * @param values  the value list.
460         * @param category  the category.
461         * @param g2  the graphics device.
462         * @param state  the state.
463         * @param dataArea  the data area (adjusted for the 3D effect).
464         * @param plot  the plot.
465         * @param domainAxis  the domain axis.
466         * @param rangeAxis  the range axis.
467         * @param dataset  the dataset.
468         *
469         * @since 1.0.4
470         */
471        protected void drawStackHorizontal(List values, Comparable category,
472                Graphics2D g2, CategoryItemRendererState state,
473                Rectangle2D dataArea, CategoryPlot plot,
474                CategoryAxis domainAxis, ValueAxis rangeAxis,
475                CategoryDataset dataset) {
476    
477            int column = dataset.getColumnIndex(category);
478            double barX0 = domainAxis.getCategoryMiddle(column,
479                    dataset.getColumnCount(), dataArea, plot.getDomainAxisEdge())
480                    - state.getBarWidth() / 2.0;
481            double barW = state.getBarWidth();
482    
483            // a list to store the series index and bar region, so we can draw
484            // all the labels at the end...
485            List itemLabelList = new ArrayList();
486    
487            // draw the blocks
488            boolean inverted = rangeAxis.isInverted();
489            int blockCount = values.size() - 1;
490            for (int k = 0; k < blockCount; k++) {
491                int index = (inverted ? blockCount - k - 1 : k);
492                Object[] prev = (Object[]) values.get(index);
493                Object[] curr = (Object[]) values.get(index + 1);
494                int series = 0;
495                if (curr[0] == null) {
496                    series = -((Integer) prev[0]).intValue() - 1;
497                }
498                else {
499                    series = ((Integer) curr[0]).intValue();
500                    if (series < 0) {
501                        series = -((Integer) prev[0]).intValue() - 1;
502                    }
503                }
504                double v0 = ((Double) prev[1]).doubleValue();
505                double vv0 = rangeAxis.valueToJava2D(v0, dataArea,
506                        plot.getRangeAxisEdge());
507    
508                double v1 = ((Double) curr[1]).doubleValue();
509                double vv1 = rangeAxis.valueToJava2D(v1, dataArea,
510                        plot.getRangeAxisEdge());
511    
512                Shape[] faces = createHorizontalBlock(barX0, barW, vv0, vv1,
513                        inverted);
514                Paint fillPaint = getItemPaint(series, column);
515                Paint fillPaintDark = PaintAlpha.darker(fillPaint);
516                boolean drawOutlines = isDrawBarOutline();
517                Paint outlinePaint = fillPaint;
518                if (drawOutlines) {
519                    outlinePaint = getItemOutlinePaint(series, column);
520                    g2.setStroke(getItemOutlineStroke(series, column));
521                }
522                for (int f = 0; f < 6; f++) {
523                    if (f == 5) {
524                        g2.setPaint(fillPaint);
525                    }
526                    else {
527                        g2.setPaint(fillPaintDark);
528                    }
529                    g2.fill(faces[f]);
530                    if (drawOutlines) {
531                        g2.setPaint(outlinePaint);
532                        g2.draw(faces[f]);
533                    }
534                }
535    
536                itemLabelList.add(new Object[] {new Integer(series),
537                        faces[5].getBounds2D(),
538                        BooleanUtilities.valueOf(v0 < getBase())});
539    
540                // add an item entity, if this information is being collected
541                EntityCollection entities = state.getEntityCollection();
542                if (entities != null) {
543                    addItemEntity(entities, dataset, series, column, faces[5]);
544                }
545    
546            }
547    
548            for (int i = 0; i < itemLabelList.size(); i++) {
549                Object[] record = (Object[]) itemLabelList.get(i);
550                int series = ((Integer) record[0]).intValue();
551                Rectangle2D bar = (Rectangle2D) record[1];
552                boolean neg = ((Boolean) record[2]).booleanValue();
553                CategoryItemLabelGenerator generator
554                        = getItemLabelGenerator(series, column);
555                if (generator != null && isItemLabelVisible(series, column)) {
556                    drawItemLabel(g2, dataset, series, column, plot, generator,
557                            bar, neg);
558                }
559    
560            }
561        }

563        /**
564         * Creates an array of shapes representing the six sides of a block in a
565         * horizontal stack.
566         *
567         * @param x0  left edge of bar (in Java2D space).
568         * @param width  the width of the bar (in Java2D units).
569         * @param y0  the base of the block (in Java2D space).
570         * @param y1  the top of the block (in Java2D space).
571         * @param inverted  a flag indicating whether or not the block is inverted
572         *     (this changes the order of the faces of the block).
573         *
574         * @return The sides of the block.
575         */
576        private Shape[] createHorizontalBlock(double x0, double width, double y0,
577                double y1, boolean inverted) {
578            Shape[] result = new Shape[6];
579            Point2D p00 = new Point2D.Double(y0, x0);
580            Point2D p01 = new Point2D.Double(y0, x0 + width);
581            Point2D p02 = new Point2D.Double(p01.getX() + getXOffset(),
582                    p01.getY() - getYOffset());
583            Point2D p03 = new Point2D.Double(p00.getX() + getXOffset(),
584                    p00.getY() - getYOffset());
585    
586            Point2D p0 = new Point2D.Double(y1, x0);
587            Point2D p1 = new Point2D.Double(y1, x0 + width);
588            Point2D p2 = new Point2D.Double(p1.getX() + getXOffset(),
589                    p1.getY() - getYOffset());
590            Point2D p3 = new Point2D.Double(p0.getX() + getXOffset(),
591                    p0.getY() - getYOffset());
592    
593            GeneralPath bottom = new GeneralPath();
594            bottom.moveTo((float) p1.getX(), (float) p1.getY());
595            bottom.lineTo((float) p01.getX(), (float) p01.getY());
596            bottom.lineTo((float) p02.getX(), (float) p02.getY());
597            bottom.lineTo((float) p2.getX(), (float) p2.getY());
598            bottom.closePath();
599    
600            GeneralPath top = new GeneralPath();
601            top.moveTo((float) p0.getX(), (float) p0.getY());
602            top.lineTo((float) p00.getX(), (float) p00.getY());
603            top.lineTo((float) p03.getX(), (float) p03.getY());
604            top.lineTo((float) p3.getX(), (float) p3.getY());
605            top.closePath();
606    
607            GeneralPath back = new GeneralPath();
608            back.moveTo((float) p2.getX(), (float) p2.getY());
609            back.lineTo((float) p02.getX(), (float) p02.getY());
610            back.lineTo((float) p03.getX(), (float) p03.getY());
611            back.lineTo((float) p3.getX(), (float) p3.getY());
612            back.closePath();
613    
614            GeneralPath front = new GeneralPath();
615            front.moveTo((float) p0.getX(), (float) p0.getY());
616            front.lineTo((float) p1.getX(), (float) p1.getY());
617            front.lineTo((float) p01.getX(), (float) p01.getY());
618            front.lineTo((float) p00.getX(), (float) p00.getY());
619            front.closePath();
620    
621            GeneralPath left = new GeneralPath();
622            left.moveTo((float) p0.getX(), (float) p0.getY());
623            left.lineTo((float) p1.getX(), (float) p1.getY());
624            left.lineTo((float) p2.getX(), (float) p2.getY());
625            left.lineTo((float) p3.getX(), (float) p3.getY());
626            left.closePath();
627    
628            GeneralPath right = new GeneralPath();
629            right.moveTo((float) p00.getX(), (float) p00.getY());
630            right.lineTo((float) p01.getX(), (float) p01.getY());
631            right.lineTo((float) p02.getX(), (float) p02.getY());
632            right.lineTo((float) p03.getX(), (float) p03.getY());
633            right.closePath();
634            result[0] = bottom;
635            result[1] = back;
636            if (inverted) {
637                result[2] = right;
638                result[3] = left;
639            }
640            else {
641                result[2] = left;
642                result[3] = right;
643            }
644            result[4] = top;
645            result[5] = front;
646            return result;
647        }

649        /**
650         * Draws a stack of bars for one category, with a vertical orientation.
651         *
652         * @param values  the value list.
653         * @param category  the category.
654         * @param g2  the graphics device.
655         * @param state  the state.
656         * @param dataArea  the data area (adjusted for the 3D effect).
657         * @param plot  the plot.
658         * @param domainAxis  the domain axis.
659         * @param rangeAxis  the range axis.
660         * @param dataset  the dataset.
661         *
662         * @since 1.0.4
663         */
664        protected void drawStackVertical(List values, Comparable category,
665                Graphics2D g2, CategoryItemRendererState state,
666                Rectangle2D dataArea, CategoryPlot plot,
667                CategoryAxis domainAxis, ValueAxis rangeAxis,
668                CategoryDataset dataset) {
669    
670            int column = dataset.getColumnIndex(category);
671            double barX0 = domainAxis.getCategoryMiddle(column,
672                    dataset.getColumnCount(), dataArea, plot.getDomainAxisEdge())
673                    - state.getBarWidth() / 2.0;
674            double barW = state.getBarWidth();
675    
676            // a list to store the series index and bar region, so we can draw
677            // all the labels at the end...
678            List itemLabelList = new ArrayList();
679    
680            // draw the blocks
681            boolean inverted = rangeAxis.isInverted();
682            int blockCount = values.size() - 1;
683            for (int k = 0; k < blockCount; k++) {
684                int index = (inverted ? blockCount - k - 1 : k);
685                Object[] prev = (Object[]) values.get(index);
686                Object[] curr = (Object[]) values.get(index + 1);
687                int series = 0;
688                if (curr[0] == null) {
689                    series = -((Integer) prev[0]).intValue() - 1;
690                }
691                else {
692                    series = ((Integer) curr[0]).intValue();
693                    if (series < 0) {
694                        series = -((Integer) prev[0]).intValue() - 1;
695                    }
696                }
697                double v0 = ((Double) prev[1]).doubleValue();
698                double vv0 = rangeAxis.valueToJava2D(v0, dataArea,
699                        plot.getRangeAxisEdge());
700    
701                double v1 = ((Double) curr[1]).doubleValue();
702                double vv1 = rangeAxis.valueToJava2D(v1, dataArea,
703                        plot.getRangeAxisEdge());
704    
705                Shape[] faces = createVerticalBlock(barX0, barW, vv0, vv1,
706                        inverted);
707                Paint fillPaint = getItemPaint(series, column);
708                Paint fillPaintDark = PaintAlpha.darker(fillPaint);
709                boolean drawOutlines = isDrawBarOutline();
710                Paint outlinePaint = fillPaint;
711                if (drawOutlines) {
712                    outlinePaint = getItemOutlinePaint(series, column);
713                    g2.setStroke(getItemOutlineStroke(series, column));
714                }
715    
716                for (int f = 0; f < 6; f++) {
717                    if (f == 5) {
718                        g2.setPaint(fillPaint);
719                    }
720                    else {
721                        g2.setPaint(fillPaintDark);
722                    }
723                    g2.fill(faces[f]);
724                    if (drawOutlines) {
725                        g2.setPaint(outlinePaint);
726                        g2.draw(faces[f]);
727                    }
728                }
729    
730                itemLabelList.add(new Object[] {new Integer(series),
731                        faces[5].getBounds2D(),
732                        BooleanUtilities.valueOf(v0 < getBase())});
733    
734                // add an item entity, if this information is being collected
735                EntityCollection entities = state.getEntityCollection();
736                if (entities != null) {
737                    addItemEntity(entities, dataset, series, column, faces[5]);
738                }
739    
740            }
741    
742            for (int i = 0; i < itemLabelList.size(); i++) {
743                Object[] record = (Object[]) itemLabelList.get(i);
744                int series = ((Integer) record[0]).intValue();
745                Rectangle2D bar = (Rectangle2D) record[1];
746                boolean neg = ((Boolean) record[2]).booleanValue();
747                CategoryItemLabelGenerator generator
748                        = getItemLabelGenerator(series, column);
749                if (generator != null && isItemLabelVisible(series, column)) {
750                    drawItemLabel(g2, dataset, series, column, plot, generator,
751                            bar, neg);
752                }
753    
754            }
755        }

757        /**
758         * Creates an array of shapes representing the six sides of a block in a
759         * vertical stack.
760         *
761         * @param x0  left edge of bar (in Java2D space).
762         * @param width  the width of the bar (in Java2D units).
763         * @param y0  the base of the block (in Java2D space).
764         * @param y1  the top of the block (in Java2D space).
765         * @param inverted  a flag indicating whether or not the block is inverted
766         *     (this changes the order of the faces of the block).
767         *
768         * @return The sides of the block.
769         */
770        private Shape[] createVerticalBlock(double x0, double width, double y0,
771                double y1, boolean inverted) {
772            Shape[] result = new Shape[6];
773            Point2D p00 = new Point2D.Double(x0, y0);
774            Point2D p01 = new Point2D.Double(x0 + width, y0);
775            Point2D p02 = new Point2D.Double(p01.getX() + getXOffset(),
776                    p01.getY() - getYOffset());
777            Point2D p03 = new Point2D.Double(p00.getX() + getXOffset(),
778                    p00.getY() - getYOffset());
779    
780    
781            Point2D p0 = new Point2D.Double(x0, y1);
782            Point2D p1 = new Point2D.Double(x0 + width, y1);
783            Point2D p2 = new Point2D.Double(p1.getX() + getXOffset(),
784                    p1.getY() - getYOffset());
785            Point2D p3 = new Point2D.Double(p0.getX() + getXOffset(),
786                    p0.getY() - getYOffset());
787    
788            GeneralPath right = new GeneralPath();
789            right.moveTo((float) p1.getX(), (float) p1.getY());
790            right.lineTo((float) p01.getX(), (float) p01.getY());
791            right.lineTo((float) p02.getX(), (float) p02.getY());
792            right.lineTo((float) p2.getX(), (float) p2.getY());
793            right.closePath();
794    
795            GeneralPath left = new GeneralPath();
796            left.moveTo((float) p0.getX(), (float) p0.getY());
797            left.lineTo((float) p00.getX(), (float) p00.getY());
798            left.lineTo((float) p03.getX(), (float) p03.getY());
799            left.lineTo((float) p3.getX(), (float) p3.getY());
800            left.closePath();
801    
802            GeneralPath back = new GeneralPath();
803            back.moveTo((float) p2.getX(), (float) p2.getY());
804            back.lineTo((float) p02.getX(), (float) p02.getY());
805            back.lineTo((float) p03.getX(), (float) p03.getY());
806            back.lineTo((float) p3.getX(), (float) p3.getY());
807            back.closePath();
808    
809            GeneralPath front = new GeneralPath();
810            front.moveTo((float) p0.getX(), (float) p0.getY());
811            front.lineTo((float) p1.getX(), (float) p1.getY());
812            front.lineTo((float) p01.getX(), (float) p01.getY());
813            front.lineTo((float) p00.getX(), (float) p00.getY());
814            front.closePath();
815    
816            GeneralPath top = new GeneralPath();
817            top.moveTo((float) p0.getX(), (float) p0.getY());
818            top.lineTo((float) p1.getX(), (float) p1.getY());
819            top.lineTo((float) p2.getX(), (float) p2.getY());
820            top.lineTo((float) p3.getX(), (float) p3.getY());
821            top.closePath();
822    
823            GeneralPath bottom = new GeneralPath();
824            bottom.moveTo((float) p00.getX(), (float) p00.getY());
825            bottom.lineTo((float) p01.getX(), (float) p01.getY());
826            bottom.lineTo((float) p02.getX(), (float) p02.getY());
827            bottom.lineTo((float) p03.getX(), (float) p03.getY());
828            bottom.closePath();
829    
830            result[0] = bottom;
831            result[1] = back;
832            result[2] = left;
833            result[3] = right;
834            result[4] = top;
835            result[5] = front;
836            if (inverted) {
837                result[0] = top;
838                result[4] = bottom;
839            }
840            return result;
841        }

843        /**
844         * Tests this renderer for equality with an arbitrary object.
845         *
846         * @param obj  the object (<code>null</code> permitted).
847         *
848         * @return A boolean.
849         */
850        public boolean equals(Object obj) {
851            if (obj == this) {
852                return true;
853            }
854            if (!(obj instanceof StackedBarRenderer3D)) {
855                return false;
856            }
857            StackedBarRenderer3D that = (StackedBarRenderer3D) obj;
858            if (this.renderAsPercentages != that.getRenderAsPercentages()) {
859                return false;
860            }
861            if (this.ignoreZeroValues != that.ignoreZeroValues) {
862                return false;
863            }
864            return super.equals(obj);
865        }
