090    /**
091     * A renderer that handles the drawing of waterfall bar charts, for use with
092     * the {@link CategoryPlot} class.  Some quirks to note:
093     * <ul>
094     * <li>the value in the last category of the dataset should be (redundantly)
095     *   specified as the sum of the items in the preceding categories - otherwise
096     *   the final bar in the plot will be incorrectly plotted;</li>
097     * <li>the bar colors are defined using special methods in this class - the
098     *   inherited methods (for example,
099     *   {@link AbstractRenderer#setSeriesPaint(int, Paint)}) are ignored;</li>
100     * </ul>
101     * The example shown here is generated by the
102     * <code>WaterfallChartDemo1.java</code> program included in the JFreeChart
103     * Demo Collection:
104     * <br><br>
105     * <img src="../../../../../images/WaterfallBarRendererSample.png"
106     * alt="WaterfallBarRendererSample.png" />
107     */
108    public class WaterfallBarRenderer extends BarRenderer {

139        /**
140         * Constructs a new waterfall renderer.
141         *
142         * @param firstBarPaint  the color of the first bar (<code>null</code> not
143         *                       permitted).
144         * @param positiveBarPaint  the color for bars with positive values
145         *                          (<code>null</code> not permitted).
146         * @param negativeBarPaint  the color for bars with negative values
147         *                          (<code>null</code> not permitted).
148         * @param lastBarPaint  the color of the last bar (<code>null</code> not
149         *                      permitted).
150         */
151        public WaterfallBarRenderer(Paint firstBarPaint, Paint positiveBarPaint,
152                Paint negativeBarPaint, Paint lastBarPaint) {
153            super();
154            ParamChecks.nullNotPermitted(firstBarPaint, "firstBarPaint");
155            ParamChecks.nullNotPermitted(positiveBarPaint, "positiveBarPaint");
156            ParamChecks.nullNotPermitted(negativeBarPaint, "negativeBarPaint");
157            ParamChecks.nullNotPermitted(lastBarPaint, "lastBarPaint");
158            this.firstBarPaint = firstBarPaint;
159            this.lastBarPaint = lastBarPaint;
160            this.positiveBarPaint = positiveBarPaint;
161            this.negativeBarPaint = negativeBarPaint;
162            setGradientPaintTransformer(new StandardGradientPaintTransformer(
163                    GradientPaintTransformType.CENTER_VERTICAL));
164            setMinimumBarLength(1.0);
165        }

250        /**
251         * Returns the range of values the renderer requires to display all the
252         * items from the specified dataset.
253         *
254         * @param dataset  the dataset (<code>null</code> not permitted).
255         *
256         * @return The range (or <code>null</code> if the dataset is empty).
257         */
258        public Range findRangeBounds(CategoryDataset dataset) {
259            if (dataset == null) {
260                return null;
261            }
262            boolean allItemsNull = true; // we'll set this to false if there is at
263                                         // least one non-null data item...
264            double minimum = 0.0;
265            double maximum = 0.0;
266            int columnCount = dataset.getColumnCount();
267            for (int row = 0; row < dataset.getRowCount(); row++) {
268                double runningTotal = 0.0;
269                for (int column = 0; column <= columnCount - 1; column++) {
270                    Number n = dataset.getValue(row, column);
271                    if (n != null) {
272                        allItemsNull = false;
273                        double value = n.doubleValue();
274                        if (column == columnCount - 1) {
275                            // treat the last column value as an absolute
276                            runningTotal = value;
277                        }
278                        else {
279                            runningTotal = runningTotal + value;
280                        }
281                        minimum = Math.min(minimum, runningTotal);
282                        maximum = Math.max(maximum, runningTotal);
283                    }
284                }
285    
286            }
287            if (!allItemsNull) {
288                return new Range(minimum, maximum);
289            }
290            else {
291                return null;
292            }
293    
294        }

296        /**
297         * Draws the bar for a single (series, category) data item.
298         *
299         * @param g2  the graphics device.
300         * @param state  the renderer state.
301         * @param dataArea  the data area.
302         * @param plot  the plot.
303         * @param domainAxis  the domain axis.
304         * @param rangeAxis  the range axis.
305         * @param dataset  the dataset.
306         * @param row  the row index (zero-based).
307         * @param column  the column index (zero-based).
308         * @param pass  the pass index.
309         */
310        public void drawItem(Graphics2D g2,
311                             CategoryItemRendererState state,
312                             Rectangle2D dataArea,
313                             CategoryPlot plot,
314                             CategoryAxis domainAxis,
315                             ValueAxis rangeAxis,
316                             CategoryDataset dataset,
317                             int row,
318                             int column,
319                             int pass) {
320    
321            double previous = state.getSeriesRunningTotal();
322            if (column == dataset.getColumnCount() - 1) {
323                previous = 0.0;
324            }
325            double current = 0.0;
326            Number n = dataset.getValue(row, column);
327            if (n != null) {
328                current = previous + n.doubleValue();
329            }
330            state.setSeriesRunningTotal(current);
331    
332            int categoryCount = getColumnCount();
333            PlotOrientation orientation = plot.getOrientation();
334    
335            double rectX = 0.0;
336            double rectY = 0.0;
337    
338            RectangleEdge rangeAxisLocation = plot.getRangeAxisEdge();
339    
340            // Y0
341            double j2dy0 = rangeAxis.valueToJava2D(previous, dataArea,
342                    rangeAxisLocation);
343    
344            // Y1
345            double j2dy1 = rangeAxis.valueToJava2D(current, dataArea,
346                    rangeAxisLocation);
347    
348            double valDiff = current - previous;
349            if (j2dy1 < j2dy0) {
350                double temp = j2dy1;
351                j2dy1 = j2dy0;
352                j2dy0 = temp;
353            }
354    
355            // BAR WIDTH
356            double rectWidth = state.getBarWidth();
357    
358            // BAR HEIGHT
359            double rectHeight = Math.max(getMinimumBarLength(),
360                    Math.abs(j2dy1 - j2dy0));
361    
362            Comparable seriesKey = dataset.getRowKey(row);
363            Comparable categoryKey = dataset.getColumnKey(column);
364            if (orientation == PlotOrientation.HORIZONTAL) {
365                rectY = domainAxis.getCategorySeriesMiddle(categoryKey, seriesKey,
366                        dataset, getItemMargin(), dataArea, RectangleEdge.LEFT);
367    
368                rectX = j2dy0;
369                rectHeight = state.getBarWidth();
370                rectY = rectY - rectHeight / 2.0;
371                rectWidth = Math.max(getMinimumBarLength(),
372                        Math.abs(j2dy1 - j2dy0));
373    
374            }
375            else if (orientation == PlotOrientation.VERTICAL) {
376                rectX = domainAxis.getCategorySeriesMiddle(categoryKey, seriesKey,
377                        dataset, getItemMargin(), dataArea, RectangleEdge.TOP);
378                rectX = rectX - rectWidth / 2.0;
379                rectY = j2dy0;
380            }
381            Rectangle2D bar = new Rectangle2D.Double(rectX, rectY, rectWidth,
382                    rectHeight);
383            Paint seriesPaint;
384            if (column == 0) {
385                seriesPaint = getFirstBarPaint();
386            }
387            else if (column == categoryCount - 1) {
388                seriesPaint = getLastBarPaint();
389            }
390            else {
391                if (valDiff < 0.0) {
392                    seriesPaint = getNegativeBarPaint();
393                }
394                else if (valDiff > 0.0) {
395                    seriesPaint = getPositiveBarPaint();
396                }
397                else {
398                    seriesPaint = getLastBarPaint();
399                }
400            }
401            if (getGradientPaintTransformer() != null
402                    && seriesPaint instanceof GradientPaint) {
403                GradientPaint gp = (GradientPaint) seriesPaint;
404                seriesPaint = getGradientPaintTransformer().transform(gp, bar);
405            }
406            g2.setPaint(seriesPaint);
407            g2.fill(bar);
408    
409            // draw the outline...
410            if (isDrawBarOutline()
411                    && state.getBarWidth() > BAR_OUTLINE_WIDTH_THRESHOLD) {
412                Stroke stroke = getItemOutlineStroke(row, column);
413                Paint paint = getItemOutlinePaint(row, column);
414                if (stroke != null && paint != null) {
415                    g2.setStroke(stroke);
416                    g2.setPaint(paint);
417                    g2.draw(bar);
418                }
419            }
420    
421            CategoryItemLabelGenerator generator
422                = getItemLabelGenerator(row, column);
423            if (generator != null && isItemLabelVisible(row, column)) {
424                drawItemLabel(g2, dataset, row, column, plot, generator, bar,
425                        (valDiff < 0.0));
426            }
427    
428            // add an item entity, if this information is being collected
429            EntityCollection entities = state.getEntityCollection();
430            if (entities != null) {
431                addItemEntity(entities, dataset, row, column, bar);
432            }
433    
434        }

436        /**
437         * Tests an object for equality with this instance.
438         *
439         * @param obj  the object (<code>null</code> permitted).
440         *
441         * @return A boolean.
442         */
443        public boolean equals(Object obj) {
444    
445            if (obj == this) {
446                return true;
447            }
448            if (!super.equals(obj)) {
449                return false;
450            }
451            if (!(obj instanceof WaterfallBarRenderer)) {
452                return false;
453            }
454            WaterfallBarRenderer that = (WaterfallBarRenderer) obj;
455            if (!PaintUtilities.equal(this.firstBarPaint, that.firstBarPaint)) {
456                return false;
457            }
458            if (!PaintUtilities.equal(this.lastBarPaint, that.lastBarPaint)) {
459                return false;
460            }
461            if (!PaintUtilities.equal(this.positiveBarPaint,
462                    that.positiveBarPaint)) {
463                return false;
464            }
465            if (!PaintUtilities.equal(this.negativeBarPaint,
466                    that.negativeBarPaint)) {
467                return false;
468            }
469            return true;
470    
471        }
