285        /**
286         * Draws the background for the plot.
287         *
288         * @param g2  the graphics device.
289         * @param plot  the plot.
290         * @param dataArea  the area inside the axes.
291         */
292        public void drawBackground(Graphics2D g2, CategoryPlot plot,
293                                   Rectangle2D dataArea) {
294    
295            float x0 = (float) dataArea.getX();
296            float x1 = x0 + (float) Math.abs(this.xOffset);
297            float x3 = (float) dataArea.getMaxX();
298            float x2 = x3 - (float) Math.abs(this.xOffset);
299    
300            float y0 = (float) dataArea.getMaxY();
301            float y1 = y0 - (float) Math.abs(this.yOffset);
302            float y3 = (float) dataArea.getMinY();
303            float y2 = y3 + (float) Math.abs(this.yOffset);
304    
305            GeneralPath clip = new GeneralPath();
306            clip.moveTo(x0, y0);
307            clip.lineTo(x0, y2);
308            clip.lineTo(x1, y3);
309            clip.lineTo(x3, y3);
310            clip.lineTo(x3, y1);
311            clip.lineTo(x2, y0);
312            clip.closePath();
313    
314            Composite originalComposite = g2.getComposite();
315            g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER,
316                    plot.getBackgroundAlpha()));
317    
318            // fill background...
319            Paint backgroundPaint = plot.getBackgroundPaint();
320            if (backgroundPaint != null) {
321                g2.setPaint(backgroundPaint);
322                g2.fill(clip);
323            }
324    
325            GeneralPath leftWall = new GeneralPath();
326            leftWall.moveTo(x0, y0);
327            leftWall.lineTo(x0, y2);
328            leftWall.lineTo(x1, y3);
329            leftWall.lineTo(x1, y1);
330            leftWall.closePath();
331            g2.setPaint(getWallPaint());
332            g2.fill(leftWall);
333    
334            GeneralPath bottomWall = new GeneralPath();
335            bottomWall.moveTo(x0, y0);
336            bottomWall.lineTo(x1, y1);
337            bottomWall.lineTo(x3, y1);
338            bottomWall.lineTo(x2, y0);
339            bottomWall.closePath();
340            g2.setPaint(getWallPaint());
341            g2.fill(bottomWall);
342    
343            // highlight the background corners...
344            g2.setPaint(Color.lightGray);
345            Line2D corner = new Line2D.Double(x0, y0, x1, y1);
346            g2.draw(corner);
347            corner.setLine(x1, y1, x1, y3);
348            g2.draw(corner);
349            corner.setLine(x1, y1, x3, y1);
350            g2.draw(corner);
351    
352            // draw background image, if there is one...
353            Image backgroundImage = plot.getBackgroundImage();
354            if (backgroundImage != null) {
355                Rectangle2D adjusted = new Rectangle2D.Double(dataArea.getX()
356                        + getXOffset(), dataArea.getY(),
357                        dataArea.getWidth() - getXOffset(),
358                        dataArea.getHeight() - getYOffset());
359                plot.drawBackgroundImage(g2, adjusted);
360            }
361    
362            g2.setComposite(originalComposite);
363    
364        }

367         * Draws the outline for the plot.
368         *
369         * @param g2  the graphics device.
370         * @param plot  the plot.
371         * @param dataArea  the area inside the axes.
372         */
373        public void drawOutline(Graphics2D g2, CategoryPlot plot,
374                                Rectangle2D dataArea) {
375    
376            float x0 = (float) dataArea.getX();
377            float x1 = x0 + (float) Math.abs(this.xOffset);
378            float x3 = (float) dataArea.getMaxX();
379            float x2 = x3 - (float) Math.abs(this.xOffset);
380    
381            float y0 = (float) dataArea.getMaxY();
382            float y1 = y0 - (float) Math.abs(this.yOffset);
383            float y3 = (float) dataArea.getMinY();
384            float y2 = y3 + (float) Math.abs(this.yOffset);
385    
386            GeneralPath clip = new GeneralPath();
387            clip.moveTo(x0, y0);
388            clip.lineTo(x0, y2);
389            clip.lineTo(x1, y3);
390            clip.lineTo(x3, y3);
391            clip.lineTo(x3, y1);
392            clip.lineTo(x2, y0);
393            clip.closePath();
394    
395            // put an outline around the data area...
396            Stroke outlineStroke = plot.getOutlineStroke();
397            Paint outlinePaint = plot.getOutlinePaint();
398            if ((outlineStroke != null) && (outlinePaint != null)) {
399                g2.setStroke(outlineStroke);
400                g2.setPaint(outlinePaint);
401                g2.draw(clip);
402            }
403    
404        }

406        /**
407         * Draws a grid line against the domain axis.
408         *
409         * @param g2  the graphics device.
410         * @param plot  the plot.
411         * @param dataArea  the area for plotting data (not yet adjusted for any
412         *                  3D effect).
413         * @param value  the Java2D value at which the grid line should be drawn.
414         *
415         */
416        public void drawDomainGridline(Graphics2D g2,
417                                       CategoryPlot plot,
418                                       Rectangle2D dataArea,
419                                       double value) {
420    
421            Line2D line1 = null;
422            Line2D line2 = null;
423            PlotOrientation orientation = plot.getOrientation();
424            if (orientation == PlotOrientation.HORIZONTAL) {
425                double y0 = value;
426                double y1 = value - getYOffset();
427                double x0 = dataArea.getMinX();
428                double x1 = x0 + getXOffset();
429                double x2 = dataArea.getMaxX();
430                line1 = new Line2D.Double(x0, y0, x1, y1);
431                line2 = new Line2D.Double(x1, y1, x2, y1);
432            }
433            else if (orientation == PlotOrientation.VERTICAL) {
434                double x0 = value;
435                double x1 = value + getXOffset();
436                double y0 = dataArea.getMaxY();
437                double y1 = y0 - getYOffset();
438                double y2 = dataArea.getMinY();
439                line1 = new Line2D.Double(x0, y0, x1, y1);
440                line2 = new Line2D.Double(x1, y1, x1, y2);
441            }
442            Paint paint = plot.getDomainGridlinePaint();
443            Stroke stroke = plot.getDomainGridlineStroke();
444            g2.setPaint(paint != null ? paint : Plot.DEFAULT_OUTLINE_PAINT);
445            g2.setStroke(stroke != null ? stroke : Plot.DEFAULT_OUTLINE_STROKE);
446            g2.draw(line1);
447            g2.draw(line2);
448    
449        }

565        /**
566         * Draws a range marker.
567         *
568         * @param g2  the graphics device.
569         * @param plot  the plot.
570         * @param axis  the value axis.
571         * @param marker  the marker.
572         * @param dataArea  the area for plotting data (not including 3D effect).
573         */
574        public void drawRangeMarker(Graphics2D g2,
575                                    CategoryPlot plot,
576                                    ValueAxis axis,
577                                    Marker marker,
578                                    Rectangle2D dataArea) {
579    
580    
581            Rectangle2D adjusted = new Rectangle2D.Double(dataArea.getX(),
582                    dataArea.getY() + getYOffset(), dataArea.getWidth()
583                    - getXOffset(), dataArea.getHeight() - getYOffset());
584            if (marker instanceof ValueMarker) {
585                ValueMarker vm = (ValueMarker) marker;
586                double value = vm.getValue();
587                Range range = axis.getRange();
588                if (!range.contains(value)) {
589                    return;
590                }
591    
592                GeneralPath path = null;
593                PlotOrientation orientation = plot.getOrientation();
594                if (orientation == PlotOrientation.HORIZONTAL) {
595                    float x = (float) axis.valueToJava2D(value, adjusted,
596                            plot.getRangeAxisEdge());
597                    float y = (float) adjusted.getMaxY();
598                    path = new GeneralPath();
599                    path.moveTo(x, y);
600                    path.lineTo((float) (x + getXOffset()),
601                            y - (float) getYOffset());
602                    path.lineTo((float) (x + getXOffset()),
603                            (float) (adjusted.getMinY() - getYOffset()));
604                    path.lineTo(x, (float) adjusted.getMinY());
605                    path.closePath();
606                }
607                else if (orientation == PlotOrientation.VERTICAL) {
608                    float y = (float) axis.valueToJava2D(value, adjusted,
609                            plot.getRangeAxisEdge());
610                    float x = (float) dataArea.getX();
611                    path = new GeneralPath();
612                    path.moveTo(x, y);
613                    path.lineTo(x + (float) this.xOffset, y - (float) this.yOffset);
614                    path.lineTo((float) (adjusted.getMaxX() + this.xOffset),
615                            y - (float) this.yOffset);
616                    path.lineTo((float) (adjusted.getMaxX()), y);
617                    path.closePath();
618                }
619                g2.setPaint(marker.getPaint());
620                g2.fill(path);
621                g2.setPaint(marker.getOutlinePaint());
622                g2.draw(path);
623    
624                String label = marker.getLabel();
625                RectangleAnchor anchor = marker.getLabelAnchor();
626                if (label != null) {
627                    Font labelFont = marker.getLabelFont();
628                    g2.setFont(labelFont);
629                    g2.setPaint(marker.getLabelPaint());
630                    Point2D coordinates = calculateRangeMarkerTextAnchorPoint(
631                            g2, orientation, dataArea, path.getBounds2D(),
632                            marker.getLabelOffset(), LengthAdjustmentType.EXPAND,
633                            anchor);
634                    TextUtilities.drawAlignedString(label, g2,
635                            (float) coordinates.getX(), (float) coordinates.getY(),
636                            marker.getLabelTextAnchor());
637                }
638    
639            }
640            else {
641                super.drawRangeMarker(g2, plot, axis, marker, adjusted);
642                // TODO: draw the interval marker with a 3D effect
643            }
644        }

646        /**
647         * Draws a 3D bar to represent one data item.
648         *
649         * @param g2  the graphics device.
650         * @param state  the renderer state.
651         * @param dataArea  the area for plotting the data.
652         * @param plot  the plot.
653         * @param domainAxis  the domain axis.
654         * @param rangeAxis  the range axis.
655         * @param dataset  the dataset.
656         * @param row  the row index (zero-based).
657         * @param column  the column index (zero-based).
658         * @param pass  the pass index.
659         */
660        public void drawItem(Graphics2D g2,
661                             CategoryItemRendererState state,
662                             Rectangle2D dataArea,
663                             CategoryPlot plot,
664                             CategoryAxis domainAxis,
665                             ValueAxis rangeAxis,
666                             CategoryDataset dataset,
667                             int row,
668                             int column,
669                             int pass) {
670    
671            // check the value we are plotting...
672            Number dataValue = dataset.getValue(row, column);
673            if (dataValue == null) {
674                return;
675            }
676    
677            double value = dataValue.doubleValue();
678    
679            Rectangle2D adjusted = new Rectangle2D.Double(dataArea.getX(),
680                    dataArea.getY() + getYOffset(),
681                    dataArea.getWidth() - getXOffset(),
682                    dataArea.getHeight() - getYOffset());
683    
684            PlotOrientation orientation = plot.getOrientation();
685    
686            double barW0 = calculateBarW0(plot, orientation, adjusted, domainAxis,
687                    state, row, column);
688            double[] barL0L1 = calculateBarL0L1(value);
689            if (barL0L1 == null) {
690                return;  // the bar is not visible
691            }
692    
693            RectangleEdge edge = plot.getRangeAxisEdge();
694            double transL0 = rangeAxis.valueToJava2D(barL0L1[0], adjusted, edge);
695            double transL1 = rangeAxis.valueToJava2D(barL0L1[1], adjusted, edge);
696            double barL0 = Math.min(transL0, transL1);
697            double barLength = Math.abs(transL1 - transL0);
698    
699            // draw the bar...
700            Rectangle2D bar = null;
701            if (orientation == PlotOrientation.HORIZONTAL) {
702                bar = new Rectangle2D.Double(barL0, barW0, barLength,
703                        state.getBarWidth());
704            }
705            else {
706                bar = new Rectangle2D.Double(barW0, barL0, state.getBarWidth(),
707                        barLength);
708            }
709            Paint itemPaint = getItemPaint(row, column);
710            g2.setPaint(itemPaint);
711            g2.fill(bar);
712    
713            double x0 = bar.getMinX();
714            double x1 = x0 + getXOffset();
715            double x2 = bar.getMaxX();
716            double x3 = x2 + getXOffset();
717    
718            double y0 = bar.getMinY() - getYOffset();
719            double y1 = bar.getMinY();
720            double y2 = bar.getMaxY() - getYOffset();
721            double y3 = bar.getMaxY();
722    
723            GeneralPath bar3dRight = null;
724            GeneralPath bar3dTop = null;
725            if (barLength > 0.0) {
726                bar3dRight = new GeneralPath();
727                bar3dRight.moveTo((float) x2, (float) y3);
728                bar3dRight.lineTo((float) x2, (float) y1);
729                bar3dRight.lineTo((float) x3, (float) y0);
730                bar3dRight.lineTo((float) x3, (float) y2);
731                bar3dRight.closePath();
732    
733                g2.setPaint(PaintAlpha.darker(itemPaint));
734                g2.fill(bar3dRight);
735            }
736    
737            bar3dTop = new GeneralPath();
738            bar3dTop.moveTo((float) x0, (float) y1);
739            bar3dTop.lineTo((float) x1, (float) y0);
740            bar3dTop.lineTo((float) x3, (float) y0);
741            bar3dTop.lineTo((float) x2, (float) y1);
742            bar3dTop.closePath();
743            g2.fill(bar3dTop);
744    
745            if (isDrawBarOutline()
746                    && state.getBarWidth() > BAR_OUTLINE_WIDTH_THRESHOLD) {
747                g2.setStroke(getItemOutlineStroke(row, column));
748                g2.setPaint(getItemOutlinePaint(row, column));
749                g2.draw(bar);
750                if (bar3dRight != null) {
751                    g2.draw(bar3dRight);
752                }
753                g2.draw(bar3dTop);
754            }
755    
756            CategoryItemLabelGenerator generator
757                = getItemLabelGenerator(row, column);
758            if (generator != null && isItemLabelVisible(row, column)) {
759                drawItemLabel(g2, dataset, row, column, plot, generator, bar,
760                        (value < 0.0));
761            }
762    
763            // add an item entity, if this information is being collected
764            EntityCollection entities = state.getEntityCollection();
765            if (entities != null) {
766                GeneralPath barOutline = new GeneralPath();
767                barOutline.moveTo((float) x0, (float) y3);
768                barOutline.lineTo((float) x0, (float) y1);
769                barOutline.lineTo((float) x1, (float) y0);
770                barOutline.lineTo((float) x3, (float) y0);
771                barOutline.lineTo((float) x3, (float) y2);
772                barOutline.lineTo((float) x2, (float) y3);
773                barOutline.closePath();
774                addItemEntity(entities, dataset, row, column, barOutline);
775            }
776    
777        }

779        /**
780         * Tests this renderer for equality with an arbitrary object.
781         *
782         * @param obj  the object (<code>null</code> permitted).
783         *
784         * @return A boolean.
785         */
786        public boolean equals(Object obj) {
787            if (obj == this) {
788                return true;
789            }
790            if (!(obj instanceof BarRenderer3D)) {
791                return false;
792            }
793            BarRenderer3D that = (BarRenderer3D) obj;
794            if (this.xOffset != that.xOffset) {
795                return false;
796            }
797            if (this.yOffset != that.yOffset) {
798                return false;
799            }
800            if (!PaintUtilities.equal(this.wallPaint, that.wallPaint)) {
801                return false;
802            }
803            return super.equals(obj);
804        }
