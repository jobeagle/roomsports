061    package org.jfree.data;
062    
063    import java.io.Serializable;
064    import java.util.ArrayList;
065    import java.util.Arrays;
066    import java.util.Comparator;
067    import java.util.HashMap;
068    import java.util.List;
069    import org.jfree.chart.util.ParamChecks;
070    
071    import org.jfree.util.PublicCloneable;
072    import org.jfree.util.SortOrder;
073    
074    /**
075     * An ordered list of (key, value) items.  This class provides a default
076     * implementation of the {@link KeyedValues} interface.
077     */
078    public class DefaultKeyedValues implements KeyedValues, Cloneable,
079            PublicCloneable, Serializable {
080    
081        /** For serialization. */
082        private static final long serialVersionUID = 8468154364608194797L;
083    
084        /** Storage for the keys. */
085        private ArrayList keys;
086    
087        /** Storage for the values. */
088        private ArrayList values;
089    
090        /**
091         * Contains (key, Integer) mappings, where the Integer is the index for
092         * the key in the list.
093         */
094        private HashMap indexMap;
095    
096      /**
097         * Creates a new collection (initially empty).
098         */
099        public DefaultKeyedValues() {
100            this.keys = new ArrayList();
101            this.values = new ArrayList();
102            this.indexMap = new HashMap();
103        }
104    
105        /**
106         * Returns the number of items (values) in the collection.
107         *
108         * @return The item count.
109         */
110        public int getItemCount() {
111            return this.indexMap.size();
112        }
113    
114        /**
115         * Returns a value.
116         *
117         * @param item  the item of interest (zero-based index).
118         *
119         * @return The value (possibly <code>null</code>).
120         *
121         * @throws IndexOutOfBoundsException if <code>item</code> is out of bounds.
122         */
123        public Number getValue(int item) {
124            return (Number) this.values.get(item);
125        }
126    
127        /**
128         * Returns a key.
129         *
130         * @param index  the item index (zero-based).
131         *
132         * @return The row key.
133         *
134         * @throws IndexOutOfBoundsException if <code>item</code> is out of bounds.
135         */
136        public Comparable getKey(int index) {
137            return (Comparable) this.keys.get(index);
138        }
139    
140        /**
141         * Returns the index for a given key.
142         *
143         * @param key  the key (<code>null</code> not permitted).
144         *
145         * @return The index, or <code>-1</code> if the key is not recognised.
146         *
147         * @throws IllegalArgumentException if <code>key</code> is
148         *     <code>null</code>.
149         */
150        public int getIndex(Comparable key) {
151            ParamChecks.nullNotPermitted(key, "key");
152            final Integer i = (Integer) this.indexMap.get(key);
153            if (i == null) {
154                return -1;  // key not found
155            }
156            return i.intValue();
157        }
158    
159        /**
160         * Returns the keys for the values in the collection.
161         *
162         * @return The keys (never <code>null</code>).
163         */
164        public List getKeys() {
165            return (List) this.keys.clone();
166        }
167    
168        /**
169         * Returns the value for a given key.
170         *
171         * @param key  the key (<code>null</code> not permitted).
172         *
173         * @return The value (possibly <code>null</code>).
174         *
175         * @throws UnknownKeyException if the key is not recognised.
176         *
177         * @see #getValue(int)
178         */
179        public Number getValue(Comparable key) {
180            int index = getIndex(key);
181            if (index < 0) {
182                throw new UnknownKeyException("Key not found: " + key);
183            }
184            return getValue(index);
185        }
186    
187        /**
188         * Updates an existing value, or adds a new value to the collection.
189         *
190         * @param key  the key (<code>null</code> not permitted).
191         * @param value  the value.
192         *
193         * @see #addValue(Comparable, Number)
194         */
195        public void addValue(Comparable key, double value) {
196            addValue(key, new Double(value));
197        }
198    
199        /**
200         * Adds a new value to the collection, or updates an existing value.
201         * This method passes control directly to the
202         * {@link #setValue(Comparable, Number)} method.
203         *
204         * @param key  the key (<code>null</code> not permitted).
205         * @param value  the value (<code>null</code> permitted).
206         */
207        public void addValue(Comparable key, Number value) {
208            setValue(key, value);
209        }
210    
211        /**
212         * Updates an existing value, or adds a new value to the collection.
213         *
214         * @param key  the key (<code>null</code> not permitted).
215         * @param value  the value.
216         */
217        public void setValue(Comparable key, double value) {
218            setValue(key, new Double(value));
219        }
220    
221        /**
222         * Updates an existing value, or adds a new value to the collection.
223         *
224         * @param key  the key (<code>null</code> not permitted).
225         * @param value  the value (<code>null</code> permitted).
226         */
227        public void setValue(Comparable key, Number value) {
228            ParamChecks.nullNotPermitted(key, "key");
229            int keyIndex = getIndex(key);
230            if (keyIndex >= 0) {
231                this.keys.set(keyIndex, key);
232                this.values.set(keyIndex, value);
233            }
234            else {
235                this.keys.add(key);
236                this.values.add(value);
237                this.indexMap.put(key, new Integer(this.keys.size() - 1));
238            }
239        }
240    
241        /**
242         * Inserts a new value at the specified position in the dataset or, if
243         * there is an existing item with the specified key, updates the value
244         * for that item and moves it to the specified position.
245         *
246         * @param position  the position (in the range 0 to getItemCount()).
247         * @param key  the key (<code>null</code> not permitted).
248         * @param value  the value.
249         *
250         * @since 1.0.6
251         */
252        public void insertValue(int position, Comparable key, double value) {
253            insertValue(position, key, new Double(value));
254        }
255    
256        /**
257         * Inserts a new value at the specified position in the dataset or, if
258         * there is an existing item with the specified key, updates the value
259         * for that item and moves it to the specified position.
260         *
261         * @param position  the position (in the range 0 to getItemCount()).
262         * @param key  the key (<code>null</code> not permitted).
263         * @param value  the value (<code>null</code> permitted).
264         *
265         * @since 1.0.6
266         */
267        public void insertValue(int position, Comparable key, Number value) {
268            if (position < 0 || position > getItemCount()) {
269                throw new IllegalArgumentException("'position' out of bounds.");
270            }
271            ParamChecks.nullNotPermitted(key, "key");
272            int pos = getIndex(key);
273            if (pos == position) {
274                this.keys.set(pos, key);
275                this.values.set(pos, value);
276            }
277            else {
278                if (pos >= 0) {
279                    this.keys.remove(pos);
280                    this.values.remove(pos);
281                }
282    
283                this.keys.add(position, key);
284                this.values.add(position, value);
285                rebuildIndex();
286            }
287        }
288    
289        /**
290         * Rebuilds the key to indexed-position mapping after an positioned insert
291         * or a remove operation.
292         */
293        private void rebuildIndex () {
294            this.indexMap.clear();
295            for (int i = 0; i < this.keys.size(); i++) {
296                final Object key = this.keys.get(i);
297                this.indexMap.put(key, new Integer(i));
298            }
299        }
300    
301        /**
302         * Removes a value from the collection.
303         *
304         * @param index  the index of the item to remove (in the range
305         *     <code>0</code> to <code>getItemCount() - 1</code>).
306         *
307         * @throws IndexOutOfBoundsException if <code>index</code> is not within
308         *     the specified range.
309         */
310        public void removeValue(int index) {
311            this.keys.remove(index);
312            this.values.remove(index);
313            rebuildIndex();
314        }
315    
316        /**
317         * Removes a value from the collection.
318         *
319         * @param key  the item key (<code>null</code> not permitted).
320         *
321         * @throws IllegalArgumentException if <code>key</code> is
322         *     <code>null</code>.
323         * @throws UnknownKeyException if <code>key</code> is not recognised.
324         */
325        public void removeValue(Comparable key) {
326            int index = getIndex(key);
327            if (index < 0) {
328                throw new UnknownKeyException("The key (" + key
329                        + ") is not recognised.");
330            }
331            removeValue(index);
332        }
333    
334        /**
335         * Clears all values from the collection.
336         *
337         * @since 1.0.2
338         */
339        public void clear() {
340            this.keys.clear();
341            this.values.clear();
342            this.indexMap.clear();
343        }
344    
345        /**
346         * Sorts the items in the list by key.
347         *
348         * @param order  the sort order (<code>null</code> not permitted).
349         */
350        public void sortByKeys(SortOrder order) {
351            final int size = this.keys.size();
352            final DefaultKeyedValue[] data = new DefaultKeyedValue[size];
353    
354            for (int i = 0; i < size; i++) {
355                data[i] = new DefaultKeyedValue((Comparable) this.keys.get(i),
356                        (Number) this.values.get(i));
357            }
358    
359            Comparator comparator = new KeyedValueComparator(
360                    KeyedValueComparatorType.BY_KEY, order);
361            Arrays.sort(data, comparator);
362            clear();
363    
364            for (int i = 0; i < data.length; i++) {
365                final DefaultKeyedValue value = data[i];
366                addValue(value.getKey(), value.getValue());
367            }
368        }
369    
370        /**
371         * Sorts the items in the list by value.  If the list contains
372         * <code>null</code> values, they will sort to the end of the list,
373         * irrespective of the sort order.
374         *
375         * @param order  the sort order (<code>null</code> not permitted).
376         */
377        public void sortByValues(SortOrder order) {
378            final int size = this.keys.size();
379            final DefaultKeyedValue[] data = new DefaultKeyedValue[size];
380            for (int i = 0; i < size; i++) {
381                data[i] = new DefaultKeyedValue((Comparable) this.keys.get(i),
382                        (Number) this.values.get(i));
383            }
384    
385            Comparator comparator = new KeyedValueComparator(
386                    KeyedValueComparatorType.BY_VALUE, order);
387            Arrays.sort(data, comparator);
388    
389            clear();
390            for (int i = 0; i < data.length; i++) {
391                final DefaultKeyedValue value = data[i];
392                addValue(value.getKey(), value.getValue());
393            }
394        }
395    
396        /**
397         * Tests if this object is equal to another.
398         *
399         * @param obj  the object (<code>null</code> permitted).
400         *
401         * @return A boolean.
402         */
403        public boolean equals(Object obj) {
404            if (obj == this) {
405                return true;
406            }
407    
408            if (!(obj instanceof KeyedValues)) {
409                return false;
410            }
411    
412            KeyedValues that = (KeyedValues) obj;
413            int count = getItemCount();
414            if (count != that.getItemCount()) {
415                return false;
416            }
417    
418            for (int i = 0; i < count; i++) {
419                Comparable k1 = getKey(i);
420                Comparable k2 = that.getKey(i);
421                if (!k1.equals(k2)) {
422                    return false;
423                }
424                Number v1 = getValue(i);
425                Number v2 = that.getValue(i);
426                if (v1 == null) {
427                    if (v2 != null) {
428                        return false;
429                    }
430                }
431                else {
432                    if (!v1.equals(v2)) {
433                        return false;
434                    }
435                }
436            }
437            return true;
438        }
439    
440        /**
441         * Returns a hash code.
442         *
443         * @return A hash code.
444         */
445        public int hashCode() {
446            return (this.keys != null ? this.keys.hashCode() : 0);
447        }
448    
449        /**
450         * Returns a clone.
451         *
452         * @return A clone.
453         *
454         * @throws CloneNotSupportedException  this class will not throw this
455         *         exception, but subclasses might.
456         */
457        public Object clone() throws CloneNotSupportedException {
458            DefaultKeyedValues clone = (DefaultKeyedValues) super.clone();
459            clone.keys = (ArrayList) this.keys.clone();
460            clone.values = (ArrayList) this.values.clone();
461            clone.indexMap = (HashMap) this.indexMap.clone();
462            return clone;
463        }
