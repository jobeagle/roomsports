041    package org.jfree.data.xml;
042    
043    import java.util.Stack;
044    
045    import org.xml.sax.SAXException;
046    import org.xml.sax.helpers.DefaultHandler;
047    
048    /**
049     * A SAX handler that delegates work to sub-handlers.
050     */
051    public class RootHandler extends DefaultHandler implements DatasetTags {
052    
053        /** The sub-handlers. */
054        private Stack subHandlers;
055    
056        /**
057         * Creates a new handler.
058         */
059        public RootHandler() {
060            this.subHandlers = new Stack();
061        }
062    
063        /**
064         * Returns the stack of sub handlers.
065         *
066         * @return The sub-handler stack.
067         */
068        public Stack getSubHandlers() {
069            return this.subHandlers;
070        }
071    
072        /**
073         * Receives some (or all) of the text in the current element.
074         *
075         * @param ch  character buffer.
076         * @param start  the start index.
077         * @param length  the length of the valid character data.
078         *
079         * @throws SAXException for errors.
080         */
081        public void characters(char[] ch, int start, int length)
082            throws SAXException {
083            DefaultHandler handler = getCurrentHandler();
084            if (handler != this) {
085                handler.characters(ch, start, length);
086            }
087        }
088    
089        /**
090         * Returns the handler at the top of the stack.
091         *
092         * @return The handler.
093         */
094        public DefaultHandler getCurrentHandler() {
095            DefaultHandler result = this;
096            if (this.subHandlers != null) {
097                if (this.subHandlers.size() > 0) {
098                    Object top = this.subHandlers.peek();
099                    if (top != null) {
100                        result = (DefaultHandler) top;
101                    }
102                }
103            }
104            return result;
105        }
106    
107        /**
108         * Pushes a sub-handler onto the stack.
109         *
110         * @param subhandler  the sub-handler.
111         */
112        public void pushSubHandler(DefaultHandler subhandler) {
113            this.subHandlers.push(subhandler);
114        }
115    
116        /**
117         * Pops a sub-handler from the stack.
118         *
119         * @return The sub-handler.
120         */
121        public DefaultHandler popSubHandler() {
122            return (DefaultHandler) this.subHandlers.pop();
123        }
124    
125    }
