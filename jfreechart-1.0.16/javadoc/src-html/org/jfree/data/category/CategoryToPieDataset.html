050    package org.jfree.data.category;
051    
052    import java.util.Collections;
053    import java.util.List;
054    import org.jfree.chart.util.ParamChecks;
055    
056    import org.jfree.data.general.AbstractDataset;
057    import org.jfree.data.general.DatasetChangeEvent;
058    import org.jfree.data.general.DatasetChangeListener;
059    import org.jfree.data.general.PieDataset;
060    import org.jfree.util.TableOrder;
061    
062    /**
063     * A {@link PieDataset} implementation that obtains its data from one row or
064     * column of a {@link CategoryDataset}.
065     */
066    public class CategoryToPieDataset extends AbstractDataset
067            implements PieDataset, DatasetChangeListener {
068    
069        /** For serialization. */
070        static final long serialVersionUID = 5516396319762189617L;
071    
072        /** The source. */
073        private CategoryDataset source;
074    
075        /** The extract type. */
076        private TableOrder extract;
077    
078        /** The row or column index. */
079        private int index;
080    
081        /**
082         * An adaptor class that converts any {@link CategoryDataset} into a
083         * {@link PieDataset}, by taking the values from a single row or column.
084         * <p>
085         * If <code>source</code> is <code>null</code>, the created dataset will
086         * be empty.
087         *
088         * @param source  the source dataset (<code>null</code> permitted).
089         * @param extract  extract data from rows or columns? (<code>null</code>
090         *                 not permitted).
091         * @param index  the row or column index.
092         */
093        public CategoryToPieDataset(CategoryDataset source, TableOrder extract,
094                int index) {
095            ParamChecks.nullNotPermitted(extract, "extract");
096            this.source = source;
097            if (this.source != null) {
098                this.source.addChangeListener(this);
099            }
100            this.extract = extract;
101            this.index = index;
102        }
103    
104        /**
105         * Returns the underlying dataset.
106         *
107         * @return The underlying dataset (possibly <code>null</code>).
108         *
109         * @since 1.0.2
110         */
111        public CategoryDataset getUnderlyingDataset() {
112            return this.source;
113        }
114    
115        /**
116         * Returns the extract type, which determines whether data is read from
117         * one row or one column of the underlying dataset.
118         *
119         * @return The extract type.
120         *
121         * @since 1.0.2
122         */
123        public TableOrder getExtractType() {
124            return this.extract;
125        }
126    
127        /**
128         * Returns the index of the row or column from which to extract the data.
129         *
130         * @return The extract index.
131         *
132         * @since 1.0.2
133         */
134        public int getExtractIndex() {
135            return this.index;
136        }
137    
138        /**
139         * Returns the number of items (values) in the collection.  If the
140         * underlying dataset is <code>null</code>, this method returns zero.
141         *
142         * @return The item count.
143         */
144        public int getItemCount() {
145            int result = 0;
146            if (this.source != null) {
147                if (this.extract == TableOrder.BY_ROW) {
148                    result = this.source.getColumnCount();
149                }
150                else if (this.extract == TableOrder.BY_COLUMN) {
151                    result = this.source.getRowCount();
152                }
153            }
154            return result;
155        }
156    
157        /**
158         * Returns a value from the dataset.
159         *
160         * @param item  the item index (zero-based).
161         *
162         * @return The value (possibly <code>null</code>).
163         *
164         * @throws IndexOutOfBoundsException if <code>item</code> is not in the
165         *     range <code>0</code> to <code>getItemCount() - 1</code>.
166         */
167        public Number getValue(int item) {
168            Number result = null;
169            if (item < 0 || item >= getItemCount()) {
170                // this will include the case where the underlying dataset is null
171                throw new IndexOutOfBoundsException(
172                        "The 'item' index is out of bounds.");
173            }
174            if (this.extract == TableOrder.BY_ROW) {
175                result = this.source.getValue(this.index, item);
176            }
177            else if (this.extract == TableOrder.BY_COLUMN) {
178                result = this.source.getValue(item, this.index);
179            }
180            return result;
181        }
182    
183        /**
184         * Returns the key at the specified index.
185         *
186         * @param index  the item index (in the range <code>0</code> to
187         *     <code>getItemCount() - 1</code>).
188         *
189         * @return The key.
190         *
191         * @throws IndexOutOfBoundsException if <code>index</code> is not in the
192         *     specified range.
193         */
194        public Comparable getKey(int index) {
195            Comparable result = null;
196            if (index < 0 || index >= getItemCount()) {
197                // this includes the case where the underlying dataset is null
198                throw new IndexOutOfBoundsException("Invalid 'index': " + index);
199            }
200            if (this.extract == TableOrder.BY_ROW) {
201                result = this.source.getColumnKey(index);
202            }
203            else if (this.extract == TableOrder.BY_COLUMN) {
204                result = this.source.getRowKey(index);
205            }
206            return result;
207        }
208    
209        /**
210         * Returns the index for a given key, or <code>-1</code> if there is no
211         * such key.
212         *
213         * @param key  the key.
214         *
215         * @return The index for the key, or <code>-1</code>.
216         */
217        public int getIndex(Comparable key) {
218            int result = -1;
219            if (this.source != null) {
220                if (this.extract == TableOrder.BY_ROW) {
221                    result = this.source.getColumnIndex(key);
222                }
223                else if (this.extract == TableOrder.BY_COLUMN) {
224                    result = this.source.getRowIndex(key);
225                }
226            }
227            return result;
228        }
229    
230        /**
231         * Returns the keys for the dataset.
232         * <p>
233         * If the underlying dataset is <code>null</code>, this method returns an
234         * empty list.
235         *
236         * @return The keys.
237         */
238        public List getKeys() {
239            List result = Collections.EMPTY_LIST;
240            if (this.source != null) {
241                if (this.extract == TableOrder.BY_ROW) {
242                    result = this.source.getColumnKeys();
243                }
244                else if (this.extract == TableOrder.BY_COLUMN) {
245                    result = this.source.getRowKeys();
246                }
247            }
248            return result;
249        }
250    
251        /**
252         * Returns the value for a given key.  If the key is not recognised, the
253         * method should return <code>null</code> (but note that <code>null</code>
254         * can be associated with a valid key also).
255         *
256         * @param key  the key.
257         *
258         * @return The value (possibly <code>null</code>).
259         */
260        public Number getValue(Comparable key) {
261            Number result = null;
262            int keyIndex = getIndex(key);
263            if (keyIndex != -1) {
264                if (this.extract == TableOrder.BY_ROW) {
265                    result = this.source.getValue(this.index, keyIndex);
266                }
267                else if (this.extract == TableOrder.BY_COLUMN) {
268                    result = this.source.getValue(keyIndex, this.index);
269                }
270            }
271            return result;
272        }
273    
274        /**
275         * Sends a {@link DatasetChangeEvent} to all registered listeners, with
276         * this (not the underlying) dataset as the source.
277         *
278         * @param event  the event (ignored, a new event with this dataset as the
279         *     source is sent to the listeners).
280         */
281        public void datasetChanged(DatasetChangeEvent event) {
282            fireDatasetChanged();
283        }
284    
285        /**
286         * Tests this dataset for equality with an arbitrary object, returning
287         * <code>true</code> if <code>obj</code> is a dataset containing the same
288         * keys and values in the same order as this dataset.
289         *
290         * @param obj  the object to test (<code>null</code> permitted).
291         *
292         * @return A boolean.
293         */
294        public boolean equals(Object obj) {
295            if (obj == this) {
296                return true;
297            }
298            if (!(obj instanceof PieDataset)) {
299                return false;
300            }
301            PieDataset that = (PieDataset) obj;
302            int count = getItemCount();
303            if (that.getItemCount() != count) {
304                return false;
305            }
306            for (int i = 0; i < count; i++) {
307                Comparable k1 = getKey(i);
308                Comparable k2 = that.getKey(i);
309                if (!k1.equals(k2)) {
310                    return false;
311                }
312    
313                Number v1 = getValue(i);
314                Number v2 = that.getValue(i);
315                if (v1 == null) {
316                    if (v2 != null) {
317                        return false;
318                    }
319                }
320                else {
321                    if (!v1.equals(v2)) {
322                        return false;
323                    }
324                }
325            }
326            return true;
327        }
