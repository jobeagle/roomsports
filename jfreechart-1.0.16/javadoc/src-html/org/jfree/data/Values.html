044    package org.jfree.data;
045    
046    /**
047     * An interface through which (single-dimension) data values can be accessed.
048     */
049    public interface Values {
050    
051        /**
052         * Returns the number of items (values) in the collection.
053         *
054         * @return The item count (possibly zero).
055         */
056        public int getItemCount();
057    
058        /**
059         * Returns the value with the specified index.
060         *
061         * @param index  the item index (in the range <code>0</code> to
062         *     <code>getItemCount() - 1</code>).
063         *
064         * @return The value (possibly <code>null</code>).
065         *
066         * @throws IndexOutOfBoundsException if <code>index</code> is not in the
067         *     specified range.
068         */
069        public Number getValue(int index);
070    
071    }
