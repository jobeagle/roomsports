070    /**
071     * A data structure that stores zero, one or many values, where each value
072     * is associated with two keys (a 'row' key and a 'column' key).  The keys
073     * should be (a) instances of {@link Comparable} and (b) immutable.
074     */
075    public class DefaultKeyedValues2D implements KeyedValues2D, PublicCloneable,
076            Cloneable, Serializable {
077    
078        /** For serialization. */
079        private static final long serialVersionUID = -5514169970951994748L;
080    
081        /** The row keys. */
082        private List rowKeys;
083    
084        /** The column keys. */
085        private List columnKeys;
086    
087        /** The row data. */
088        private List rows;
089    
090        /** If the row keys should be sorted by their comparable order. */
091        private boolean sortRowKeys;
092    
093        /**
094         * Creates a new instance (initially empty).
095         */
096        public DefaultKeyedValues2D() {
097            this(false);
098        }
099    
100        /**
101         * Creates a new instance (initially empty).
102         *
103         * @param sortRowKeys  if the row keys should be sorted.
104         */
105        public DefaultKeyedValues2D(boolean sortRowKeys) {
106            this.rowKeys = new java.util.ArrayList();
107            this.columnKeys = new java.util.ArrayList();
108            this.rows = new java.util.ArrayList();
109            this.sortRowKeys = sortRowKeys;
110        }
111    
112        /**
113         * Returns the row count.
114         *
115         * @return The row count.
116         *
117         * @see #getColumnCount()
118         */
119        public int getRowCount() {
120            return this.rowKeys.size();
121        }
122    
123        /**
124         * Returns the column count.
125         *
126         * @return The column count.
127         *
128         * @see #getRowCount()
129         */
130        public int getColumnCount() {
131            return this.columnKeys.size();
132        }
133    
134        /**
135         * Returns the value for a given row and column.
136         *
137         * @param row  the row index.
138         * @param column  the column index.
139         *
140         * @return The value.
141         *
142         * @see #getValue(Comparable, Comparable)
143         */
144        public Number getValue(int row, int column) {
145            Number result = null;
146            DefaultKeyedValues rowData = (DefaultKeyedValues) this.rows.get(row);
147            if (rowData != null) {
148                Comparable columnKey = (Comparable) this.columnKeys.get(column);
149                // the row may not have an entry for this key, in which case the
150                // return value is null
151                int index = rowData.getIndex(columnKey);
152                if (index >= 0) {
153                    result = rowData.getValue(index);
154                }
155            }
156            return result;
157        }

159        /**
160         * Returns the key for a given row.
161         *
162         * @param row  the row index (in the range 0 to {@link #getRowCount()} - 1).
163         *
164         * @return The row key.
165         *
166         * @see #getRowIndex(Comparable)
167         * @see #getColumnKey(int)
168         */
169        public Comparable getRowKey(int row) {
170            return (Comparable) this.rowKeys.get(row);
171        }
172    
173        /**
174         * Returns the row index for a given key.
175         *
176         * @param key  the key (<code>null</code> not permitted).
177         *
178         * @return The row index.
179         *
180         * @see #getRowKey(int)
181         * @see #getColumnIndex(Comparable)
182         */
183        public int getRowIndex(Comparable key) {
184            ParamChecks.nullNotPermitted(key, "key");
185            if (this.sortRowKeys) {
186                return Collections.binarySearch(this.rowKeys, key);
187            }
188            else {
189                return this.rowKeys.indexOf(key);
190            }
191        }

242            return Collections.unmodifiableList(this.columnKeys);

245        /**
246         * Returns the value for the given row and column keys.  This method will
247         * throw an {@link UnknownKeyException} if either key is not defined in the
248         * data structure.
249         *
250         * @param rowKey  the row key (<code>null</code> not permitted).
251         * @param columnKey  the column key (<code>null</code> not permitted).
252         *
253         * @return The value (possibly <code>null</code>).
254         *
255         * @see #addValue(Number, Comparable, Comparable)
256         * @see #removeValue(Comparable, Comparable)
257         */
258        public Number getValue(Comparable rowKey, Comparable columnKey) {
259            ParamChecks.nullNotPermitted(rowKey, "rowKey");
260            ParamChecks.nullNotPermitted(columnKey, "columnKey");
261    
262            // check that the column key is defined in the 2D structure
263            if (!(this.columnKeys.contains(columnKey))) {
264                throw new UnknownKeyException("Unrecognised columnKey: "
265                        + columnKey);
266            }
267    
268            // now fetch the row data - need to bear in mind that the row
269            // structure may not have an entry for the column key, but that we
270            // have already checked that the key is valid for the 2D structure
271            int row = getRowIndex(rowKey);
272            if (row >= 0) {
273                DefaultKeyedValues rowData
274                    = (DefaultKeyedValues) this.rows.get(row);
275                int col = rowData.getIndex(columnKey);
276                return (col >= 0 ? rowData.getValue(col) : null);
277            }
278            else {
279                throw new UnknownKeyException("Unrecognised rowKey: " + rowKey);
280            }
281        }

283        /**
284         * Adds a value to the table.  Performs the same function as
285         * #setValue(Number, Comparable, Comparable).
286         *
287         * @param value  the value (<code>null</code> permitted).
288         * @param rowKey  the row key (<code>null</code> not permitted).
289         * @param columnKey  the column key (<code>null</code> not permitted).
290         *
291         * @see #setValue(Number, Comparable, Comparable)
292         * @see #removeValue(Comparable, Comparable)
293         */
294        public void addValue(Number value, Comparable rowKey,
295                             Comparable columnKey) {
296            // defer argument checking
297            setValue(value, rowKey, columnKey);
298        }
299    
300        /**
301         * Adds or updates a value.
302         *
303         * @param value  the value (<code>null</code> permitted).
304         * @param rowKey  the row key (<code>null</code> not permitted).
305         * @param columnKey  the column key (<code>null</code> not permitted).
306         *
307         * @see #addValue(Number, Comparable, Comparable)
308         * @see #removeValue(Comparable, Comparable)
309         */
310        public void setValue(Number value, Comparable rowKey,
311                             Comparable columnKey) {
312    
313            DefaultKeyedValues row;
314            int rowIndex = getRowIndex(rowKey);
315    
316            if (rowIndex >= 0) {
317                row = (DefaultKeyedValues) this.rows.get(rowIndex);
318            }
319            else {
320                row = new DefaultKeyedValues();
321                if (this.sortRowKeys) {
322                    rowIndex = -rowIndex - 1;
323                    this.rowKeys.add(rowIndex, rowKey);
324                    this.rows.add(rowIndex, row);
325                }
326                else {
327                    this.rowKeys.add(rowKey);
328                    this.rows.add(row);
329                }
330            }
331            row.setValue(columnKey, value);
332    
333            int columnIndex = this.columnKeys.indexOf(columnKey);
334            if (columnIndex < 0) {
335                this.columnKeys.add(columnKey);
336            }
337        }
338    
339        /**
340         * Removes a value from the table by setting it to <code>null</code>.  If
341         * all the values in the specified row and/or column are now
342         * <code>null</code>, the row and/or column is removed from the table.
343         *
344         * @param rowKey  the row key (<code>null</code> not permitted).
345         * @param columnKey  the column key (<code>null</code> not permitted).
346         *
347         * @see #addValue(Number, Comparable, Comparable)
348         */
349        public void removeValue(Comparable rowKey, Comparable columnKey) {
350            setValue(null, rowKey, columnKey);
351    
352            // 1. check whether the row is now empty.
353            boolean allNull = true;
354            int rowIndex = getRowIndex(rowKey);
355            DefaultKeyedValues row = (DefaultKeyedValues) this.rows.get(rowIndex);
356    
357            for (int item = 0, itemCount = row.getItemCount(); item < itemCount;
358                 item++) {
359                if (row.getValue(item) != null) {
360                    allNull = false;
361                    break;
362                }
363            }
364    
365            if (allNull) {
366                this.rowKeys.remove(rowIndex);
367                this.rows.remove(rowIndex);
368            }
369    
370            // 2. check whether the column is now empty.
371            allNull = true;
372            //int columnIndex = getColumnIndex(columnKey);
373    
374            for (int item = 0, itemCount = this.rows.size(); item < itemCount;
375                 item++) {
376                row = (DefaultKeyedValues) this.rows.get(item);
377                int columnIndex = row.getIndex(columnKey);
378                if (columnIndex >= 0 && row.getValue(columnIndex) != null) {
379                    allNull = false;
380                    break;
381                }
382            }
383    
384            if (allNull) {
385                for (int item = 0, itemCount = this.rows.size(); item < itemCount;
386                     item++) {
387                    row = (DefaultKeyedValues) this.rows.get(item);
388                    int columnIndex = row.getIndex(columnKey);
389                    if (columnIndex >= 0) {
390                        row.removeValue(columnIndex);
391                    }
392                }
393                this.columnKeys.remove(columnKey);
394            }
395        }

397        /**
398         * Removes a row.
399         *
400         * @param rowIndex  the row index.
401         *
402         * @see #removeRow(Comparable)
403         * @see #removeColumn(int)
404         */
405        public void removeRow(int rowIndex) {
406            this.rowKeys.remove(rowIndex);
407            this.rows.remove(rowIndex);
408        }
409    
410        /**
411         * Removes a row from the table.
412         *
413         * @param rowKey  the row key (<code>null</code> not permitted).
414         *
415         * @see #removeRow(int)
416         * @see #removeColumn(Comparable)
417         *
418         * @throws UnknownKeyException if <code>rowKey</code> is not defined in the
419         *         table.
420         */
421        public void removeRow(Comparable rowKey) {
422            ParamChecks.nullNotPermitted(rowKey, "rowKey");
423            int index = getRowIndex(rowKey);
424            if (index >= 0) {
425                removeRow(index);
426            }
427            else {
428                throw new UnknownKeyException("Unknown key: " + rowKey);
429            }
430        }

432        /**
433         * Removes a column.
434         *
435         * @param columnIndex  the column index.
436         *
437         * @see #removeColumn(Comparable)
438         * @see #removeRow(int)
439         */
440        public void removeColumn(int columnIndex) {
441            Comparable columnKey = getColumnKey(columnIndex);
442            removeColumn(columnKey);
443        }
444    
445        /**
446         * Removes a column from the table.
447         *
448         * @param columnKey  the column key (<code>null</code> not permitted).
449         *
450         * @throws UnknownKeyException if the table does not contain a column with
451         *     the specified key.
452         * @throws IllegalArgumentException if <code>columnKey</code> is
453         *     <code>null</code>.
454         *
455         * @see #removeColumn(int)
456         * @see #removeRow(Comparable)
457         */
458        public void removeColumn(Comparable columnKey) {
459            ParamChecks.nullNotPermitted(columnKey, "columnKey");
460            if (!this.columnKeys.contains(columnKey)) {
461                throw new UnknownKeyException("Unknown key: " + columnKey);
462            }
463            Iterator iterator = this.rows.iterator();
464            while (iterator.hasNext()) {
465                DefaultKeyedValues rowData = (DefaultKeyedValues) iterator.next();
466                int index = rowData.getIndex(columnKey);
467                if (index >= 0) {
468                    rowData.removeValue(columnKey);
469                }
470            }
471            this.columnKeys.remove(columnKey);
472        }

474        /**
475         * Clears all the data and associated keys.
476         */
477        public void clear() {
478            this.rowKeys.clear();
479            this.columnKeys.clear();
480            this.rows.clear();
481        }

483        /**
484         * Tests if this object is equal to another.
485         *
486         * @param o  the other object (<code>null</code> permitted).
487         *
488         * @return A boolean.
489         */
490        public boolean equals(Object o) {
491    
492            if (o == null) {
493                return false;
494            }
495            if (o == this) {
496                return true;
497            }
498    
499            if (!(o instanceof KeyedValues2D)) {
500                return false;
501            }
502            KeyedValues2D kv2D = (KeyedValues2D) o;
503            if (!getRowKeys().equals(kv2D.getRowKeys())) {
504                return false;
505            }
506            if (!getColumnKeys().equals(kv2D.getColumnKeys())) {
507                return false;
508            }
509            int rowCount = getRowCount();
510            if (rowCount != kv2D.getRowCount()) {
511                return false;
512            }
513    
514            int colCount = getColumnCount();
515            if (colCount != kv2D.getColumnCount()) {
516                return false;
517            }
518    
519            for (int r = 0; r < rowCount; r++) {
520                for (int c = 0; c < colCount; c++) {
521                    Number v1 = getValue(r, c);
522                    Number v2 = kv2D.getValue(r, c);
523                    if (v1 == null) {
524                        if (v2 != null) {
525                            return false;
526                        }
527                    }
528                    else {
529                        if (!v1.equals(v2)) {
530                            return false;
531                        }
532                    }
533                }
534            }
535            return true;
536        }

538        /**
539         * Returns a hash code.
540         *
541         * @return A hash code.
542         */
543        public int hashCode() {
544            int result;
545            result = this.rowKeys.hashCode();
546            result = 29 * result + this.columnKeys.hashCode();
547            result = 29 * result + this.rows.hashCode();
548            return result;
549        }

551        /**
552         * Returns a clone.
553         *
554         * @return A clone.
555         *
556         * @throws CloneNotSupportedException  this class will not throw this
557         *         exception, but subclasses (if any) might.
558         */
559        public Object clone() throws CloneNotSupportedException {
560            DefaultKeyedValues2D clone = (DefaultKeyedValues2D) super.clone();
561            // for the keys, a shallow copy should be fine because keys
562            // should be immutable...
563            clone.columnKeys = new java.util.ArrayList(this.columnKeys);
564            clone.rowKeys = new java.util.ArrayList(this.rowKeys);
565    
566            // but the row data requires a deep copy
567            clone.rows = (List) ObjectUtilities.deepClone(this.rows);
568            return clone;
569        }
