043    package org.jfree.data;
044    
045    import java.io.Serializable;
046    import java.util.Collections;
047    import java.util.List;
048    import org.jfree.chart.util.ParamChecks;
049    
050    import org.jfree.data.general.Series;
051    import org.jfree.data.general.SeriesChangeEvent;
052    import org.jfree.data.general.SeriesException;
053    import org.jfree.util.ObjectUtilities;
054    
055    /**
056     * A (possibly ordered) list of (Comparable, Object) data items.
057     *
058     * @since 1.0.3
059     */
060    public class ComparableObjectSeries extends Series
061            implements Cloneable, Serializable {
062    
063        /** Storage for the data items in the series. */
064        protected List data;
065    
066        /** The maximum number of items for the series. */
067        private int maximumItemCount = Integer.MAX_VALUE;
068    
069        /** A flag that controls whether the items are automatically sorted. */
070        private boolean autoSort;
071    
072        /** A flag that controls whether or not duplicate x-values are allowed. */
073        private boolean allowDuplicateXValues;
074    
075        /**
076         * Creates a new empty series.  By default, items added to the series will
077         * be sorted into ascending order by x-value, and duplicate x-values will
078         * be allowed (these defaults can be modified with another constructor.
079         *
080         * @param key  the series key (<code>null</code> not permitted).
081         */
082        public ComparableObjectSeries(Comparable key) {
083            this(key, true, true);
084        }
085    
086        /**
087         * Constructs a new series that contains no data.  You can specify
088         * whether or not duplicate x-values are allowed for the series.
089         *
090         * @param key  the series key (<code>null</code> not permitted).
091         * @param autoSort  a flag that controls whether or not the items in the
092         *                  series are sorted.
093         * @param allowDuplicateXValues  a flag that controls whether duplicate
094         *                               x-values are allowed.
095         */
096        public ComparableObjectSeries(Comparable key, boolean autoSort,
097                boolean allowDuplicateXValues) {
098            super(key);
099            this.data = new java.util.ArrayList();
100            this.autoSort = autoSort;
101            this.allowDuplicateXValues = allowDuplicateXValues;
102        }
103    
104        /**
105         * Returns the flag that controls whether the items in the series are
106         * automatically sorted.  There is no setter for this flag, it must be
107         * defined in the series constructor.
108         *
109         * @return A boolean.
110         */
111        public boolean getAutoSort() {
112            return this.autoSort;
113        }
114    
115        /**
116         * Returns a flag that controls whether duplicate x-values are allowed.
117         * This flag can only be set in the constructor.
118         *
119         * @return A boolean.
120         */
121        public boolean getAllowDuplicateXValues() {
122            return this.allowDuplicateXValues;
123        }
124    
125        /**
126         * Returns the number of items in the series.
127         *
128         * @return The item count.
129         */
130        public int getItemCount() {
131            return this.data.size();
132        }
133    
134        /**
135         * Returns the maximum number of items that will be retained in the series.
136         * The default value is <code>Integer.MAX_VALUE</code>.
137         *
138         * @return The maximum item count.
139         * @see #setMaximumItemCount(int)
140         */
141        public int getMaximumItemCount() {
142            return this.maximumItemCount;
143        }
144    
145        /**
146         * Sets the maximum number of items that will be retained in the series.
147         * If you add a new item to the series such that the number of items will
148         * exceed the maximum item count, then the first element in the series is
149         * automatically removed, ensuring that the maximum item count is not
150         * exceeded.
151         * <p>
152         * Typically this value is set before the series is populated with data,
153         * but if it is applied later, it may cause some items to be removed from
154         * the series (in which case a {@link SeriesChangeEvent} will be sent to
155         * all registered listeners.
156         *
157         * @param maximum  the maximum number of items for the series.
158         */
159        public void setMaximumItemCount(int maximum) {
160            this.maximumItemCount = maximum;
161            boolean dataRemoved = false;
162            while (this.data.size() > maximum) {
163                this.data.remove(0);
164                dataRemoved = true;
165            }
166            if (dataRemoved) {
167                fireSeriesChanged();
168            }
169        }
170    
171        /**
172         * Adds new data to the series and sends a {@link SeriesChangeEvent} to
173         * all registered listeners.
174         * <P>
175         * Throws an exception if the x-value is a duplicate AND the
176         * allowDuplicateXValues flag is false.
177         *
178         * @param x  the x-value (<code>null</code> not permitted).
179         * @param y  the y-value (<code>null</code> permitted).
180         */
181        protected void add(Comparable x, Object y) {
182            // argument checking delegated...
183            add(x, y, true);
184        }
185    
186        /**
187         * Adds new data to the series and, if requested, sends a
188         * {@link SeriesChangeEvent} to all registered listeners.
189         * <P>
190         * Throws an exception if the x-value is a duplicate AND the
191         * allowDuplicateXValues flag is false.
192         *
193         * @param x  the x-value (<code>null</code> not permitted).
194         * @param y  the y-value (<code>null</code> permitted).
195         * @param notify  a flag the controls whether or not a
196         *                {@link SeriesChangeEvent} is sent to all registered
197         *                listeners.
198         */
199        protected void add(Comparable x, Object y, boolean notify) {
200            // delegate argument checking to XYDataItem...
201            ComparableObjectItem item = new ComparableObjectItem(x, y);
202            add(item, notify);
203        }
204    
205        /**
206         * Adds a data item to the series and, if requested, sends a
207         * {@link SeriesChangeEvent} to all registered listeners.
208         *
209         * @param item  the (x, y) item (<code>null</code> not permitted).
210         * @param notify  a flag that controls whether or not a
211         *                {@link SeriesChangeEvent} is sent to all registered
212         *                listeners.
213         */
214        protected void add(ComparableObjectItem item, boolean notify) {
215    
216            ParamChecks.nullNotPermitted(item, "item");
217            if (this.autoSort) {
218                int index = Collections.binarySearch(this.data, item);
219                if (index < 0) {
220                    this.data.add(-index - 1, item);
221                }
222                else {
223                    if (this.allowDuplicateXValues) {
224                        // need to make sure we are adding *after* any duplicates
225                        int size = this.data.size();
226                        while (index < size
227                               && item.compareTo(this.data.get(index)) == 0) {
228                            index++;
229                        }
230                        if (index < this.data.size()) {
231                            this.data.add(index, item);
232                        }
233                        else {
234                            this.data.add(item);
235                        }
236                    }
237                    else {
238                        throw new SeriesException("X-value already exists.");
239                    }
240                }
241            }
242            else {
243                if (!this.allowDuplicateXValues) {
244                    // can't allow duplicate values, so we need to check whether
245                    // there is an item with the given x-value already
246                    int index = indexOf(item.getComparable());
247                    if (index >= 0) {
248                        throw new SeriesException("X-value already exists.");
249                    }
250                }
251                this.data.add(item);
252            }
253            if (getItemCount() > this.maximumItemCount) {
254                this.data.remove(0);
255            }
256            if (notify) {
257                fireSeriesChanged();
258            }
259        }
260    
261        /**
262         * Returns the index of the item with the specified x-value, or a negative
263         * index if the series does not contain an item with that x-value.  Be
264         * aware that for an unsorted series, the index is found by iterating
265         * through all items in the series.
266         *
267         * @param x  the x-value (<code>null</code> not permitted).
268         *
269         * @return The index.
270         */
271        public int indexOf(Comparable x) {
272            if (this.autoSort) {
273                return Collections.binarySearch(this.data, new ComparableObjectItem(
274                        x, null));
275            }
276            else {
277                for (int i = 0; i < this.data.size(); i++) {
278                    ComparableObjectItem item = (ComparableObjectItem)
279                            this.data.get(i);
280                    if (item.getComparable().equals(x)) {
281                        return i;
282                    }
283                }
284                return -1;
285            }
286        }
287    
288        /**
289         * Updates an item in the series.
290         *
291         * @param x  the x-value (<code>null</code> not permitted).
292         * @param y  the y-value (<code>null</code> permitted).
293         *
294         * @throws SeriesException if there is no existing item with the specified
295         *         x-value.
296         */
297        protected void update(Comparable x, Object y) {
298            int index = indexOf(x);
299            if (index < 0) {
300                throw new SeriesException("No observation for x = " + x);
301            }
302            else {
303                ComparableObjectItem item = getDataItem(index);
304                item.setObject(y);
305                fireSeriesChanged();
306            }
307        }
308    
309        /**
310         * Updates the value of an item in the series and sends a
311         * {@link SeriesChangeEvent} to all registered listeners.
312         *
313         * @param index  the item (zero based index).
314         * @param y  the new value (<code>null</code> permitted).
315         */
316        protected void updateByIndex(int index, Object y) {
317            ComparableObjectItem item = getDataItem(index);
318            item.setObject(y);
319            fireSeriesChanged();
320        }
321    
322        /**
323         * Return the data item with the specified index.
324         *
325         * @param index  the index.
326         *
327         * @return The data item with the specified index.
328         */
329        protected ComparableObjectItem getDataItem(int index) {
330            return (ComparableObjectItem) this.data.get(index);
331        }
332    
333        /**
334         * Deletes a range of items from the series and sends a
335         * {@link SeriesChangeEvent} to all registered listeners.
336         *
337         * @param start  the start index (zero-based).
338         * @param end  the end index (zero-based).
339         */
340        protected void delete(int start, int end) {
341            for (int i = start; i <= end; i++) {
342                this.data.remove(start);
343            }
344            fireSeriesChanged();
345        }
346    
347        /**
348         * Removes all data items from the series and, unless the series is
349         * already empty, sends a {@link SeriesChangeEvent} to all registered
350         * listeners.
351         */
352        public void clear() {
353            if (this.data.size() > 0) {
354                this.data.clear();
355                fireSeriesChanged();
356            }
357        }
358    
359        /**
360         * Removes the item at the specified index and sends a
361         * {@link SeriesChangeEvent} to all registered listeners.
362         *
363         * @param index  the index.
364         *
365         * @return The item removed.
366         */
367        protected ComparableObjectItem remove(int index) {
368            ComparableObjectItem result = (ComparableObjectItem) this.data.remove(
369                    index);
370            fireSeriesChanged();
371            return result;
372        }
373    
374        /**
375         * Removes the item with the specified x-value and sends a
376         * {@link SeriesChangeEvent} to all registered listeners.
377         *
378         * @param x  the x-value.
379    
380         * @return The item removed.
381         */
382        public ComparableObjectItem remove(Comparable x) {
383            return remove(indexOf(x));
384        }
385    
386        /**
387         * Tests this series for equality with an arbitrary object.
388         *
389         * @param obj  the object to test against for equality
390         *             (<code>null</code> permitted).
391         *
392         * @return A boolean.
393         */
394        public boolean equals(Object obj) {
395            if (obj == this) {
396                return true;
397            }
398            if (!(obj instanceof ComparableObjectSeries)) {
399                return false;
400            }
401            if (!super.equals(obj)) {
402                return false;
403            }
404            ComparableObjectSeries that = (ComparableObjectSeries) obj;
405            if (this.maximumItemCount != that.maximumItemCount) {
406                return false;
407            }
408            if (this.autoSort != that.autoSort) {
409                return false;
410            }
411            if (this.allowDuplicateXValues != that.allowDuplicateXValues) {
412                return false;
413            }
414            if (!ObjectUtilities.equal(this.data, that.data)) {
415                return false;
416            }
417            return true;
418        }
419    
420        /**
421         * Returns a hash code.
422         *
423         * @return A hash code.
424         */
425        public int hashCode() {
426            int result = super.hashCode();
427            // it is too slow to look at every data item, so let's just look at
428            // the first, middle and last items...
429            int count = getItemCount();
430            if (count > 0) {
431                ComparableObjectItem item = getDataItem(0);
432                result = 29 * result + item.hashCode();
433            }
434            if (count > 1) {
435                ComparableObjectItem item = getDataItem(count - 1);
436                result = 29 * result + item.hashCode();
437            }
438            if (count > 2) {
439                ComparableObjectItem item = getDataItem(count / 2);
440                result = 29 * result + item.hashCode();
441            }
442            result = 29 * result + this.maximumItemCount;
443            result = 29 * result + (this.autoSort ? 1 : 0);
444            result = 29 * result + (this.allowDuplicateXValues ? 1 : 0);
445            return result;
446        }
