064    /**
065     * A collection of {@link TimePeriodValues} objects.
066     * <P>
067     * This class implements the {@link org.jfree.data.xy.XYDataset} interface, as
068     * well as the extended {@link IntervalXYDataset} interface.  This makes it a
069     * convenient dataset for use with the {@link org.jfree.chart.plot.XYPlot}
070     * class.
071     */
072    public class TimePeriodValuesCollection extends AbstractIntervalXYDataset
073            implements IntervalXYDataset, DomainInfo, Serializable {
074    
075        /** For serialization. */
076        private static final long serialVersionUID = -3077934065236454199L;
077    
078        /** Storage for the time series. */
079        private List data;
080    
081        /**
082         * The position within a time period to return as the x-value (START,
083         * MIDDLE or END).
084         */
085        private TimePeriodAnchor xPosition;
086    
087        /**
088         * A flag that indicates that the domain is 'points in time'.  If this
089         * flag is true, only the x-value is used to determine the range of values
090         * in the domain, the start and end x-values are ignored.
091         */
092        private boolean domainIsPointsInTime;
093    
094        /**
095         * Constructs an empty dataset.
096         */
097        public TimePeriodValuesCollection() {
098            this((TimePeriodValues) null);
099        }
100    
101        /**
102         * Constructs a dataset containing a single series.  Additional series can
103         * be added.
104         *
105         * @param series  the series (<code>null</code> ignored).
106         */
107        public TimePeriodValuesCollection(TimePeriodValues series) {
108            this.data = new java.util.ArrayList();
109            this.xPosition = TimePeriodAnchor.MIDDLE;
110            this.domainIsPointsInTime = false;
111            if (series != null) {
112                this.data.add(series);
113                series.addChangeListener(this);
114            }
115        }
116    
117        /**
118         * Returns the position of the X value within each time period.
119         *
120         * @return The position (never <code>null</code>).
121         *
122         * @see #setXPosition(TimePeriodAnchor)
123         */
124        public TimePeriodAnchor getXPosition() {
125            return this.xPosition;
126        }
127    
128        /**
129         * Sets the position of the x axis within each time period.
130         *
131         * @param position  the position (<code>null</code> not permitted).
132         *
133         * @see #getXPosition()
134         */
135        public void setXPosition(TimePeriodAnchor position) {
136            ParamChecks.nullNotPermitted(position, "position");
137            this.xPosition = position;
138        }
139    
140        /**
141         * Returns the number of series in the collection.
142         *
143         * @return The series count.
144         */
145        public int getSeriesCount() {
146            return this.data.size();
147        }
148    
149        /**
150         * Returns a series.
151         *
152         * @param series  the index of the series (zero-based).
153         *
154         * @return The series.
155         */
156        public TimePeriodValues getSeries(int series) {
157            if ((series < 0) || (series >= getSeriesCount())) {
158                throw new IllegalArgumentException("Index 'series' out of range.");
159            }
160            return (TimePeriodValues) this.data.get(series);
161        }
162    
163        /**
164         * Returns the key for a series.
165         *
166         * @param series  the index of the series (zero-based).
167         *
168         * @return The key for a series.
169         */
170        public Comparable getSeriesKey(int series) {
171            // defer argument checking
172            return getSeries(series).getKey();
173        }
174    
175        /**
176         * Adds a series to the collection.  A
177         * {@link org.jfree.data.general.DatasetChangeEvent} is sent to all
178         * registered listeners.
179         *
180         * @param series  the time series.
181         */
182        public void addSeries(TimePeriodValues series) {
183            ParamChecks.nullNotPermitted(series, "series");
184            this.data.add(series);
185            series.addChangeListener(this);
186            fireDatasetChanged();
187        }
188    
189        /**
190         * Removes the specified series from the collection.
191         *
192         * @param series  the series to remove (<code>null</code> not permitted).
193         */
194        public void removeSeries(TimePeriodValues series) {
195            ParamChecks.nullNotPermitted(series, "series");
196            this.data.remove(series);
197            series.removeChangeListener(this);
198            fireDatasetChanged();
199    
200        }
201    
202        /**
203         * Removes a series from the collection.
204         *
205         * @param index  the series index (zero-based).
206         */
207        public void removeSeries(int index) {
208            TimePeriodValues series = getSeries(index);
209            if (series != null) {
210                removeSeries(series);
211            }
212        }
213    
214        /**
215         * Returns the number of items in the specified series.
216         * <P>
217         * This method is provided for convenience.
218         *
219         * @param series  the index of the series of interest (zero-based).
220         *
221         * @return The number of items in the specified series.
222         */
223        public int getItemCount(int series) {
224            return getSeries(series).getItemCount();
225        }
226    
227        /**
228         * Returns the x-value for the specified series and item.
229         *
230         * @param series  the series (zero-based index).
231         * @param item  the item (zero-based index).
232         *
233         * @return The x-value for the specified series and item.
234         */
235        public Number getX(int series, int item) {
236            TimePeriodValues ts = (TimePeriodValues) this.data.get(series);
237            TimePeriodValue dp = ts.getDataItem(item);
238            TimePeriod period = dp.getPeriod();
239            return new Long(getX(period));
240        }
241    
242        /**
243         * Returns the x-value for a time period.
244         *
245         * @param period  the time period.
246         *
247         * @return The x-value.
248         */
249        private long getX(TimePeriod period) {
250    
251            if (this.xPosition == TimePeriodAnchor.START) {
252                return period.getStart().getTime();
253            }
254            else if (this.xPosition == TimePeriodAnchor.MIDDLE) {
255                return period.getStart().getTime()
256                    / 2 + period.getEnd().getTime() / 2;
257            }
258            else if (this.xPosition == TimePeriodAnchor.END) {
259                return period.getEnd().getTime();
260            }
261            else {
262                throw new IllegalStateException("TimePeriodAnchor unknown.");
263            }
264    
265        }
266    
267        /**
268         * Returns the starting X value for the specified series and item.
269         *
270         * @param series  the series (zero-based index).
271         * @param item  the item (zero-based index).
272         *
273         * @return The starting X value for the specified series and item.
274         */
275        public Number getStartX(int series, int item) {
276            TimePeriodValues ts = (TimePeriodValues) this.data.get(series);
277            TimePeriodValue dp = ts.getDataItem(item);
278            return new Long(dp.getPeriod().getStart().getTime());
279        }
280

282         * Returns the ending X value for the specified series and item.
283         *
284         * @param series  the series (zero-based index).
285         * @param item  the item (zero-based index).
286         *
287         * @return The ending X value for the specified series and item.
288         */
289        public Number getEndX(int series, int item) {
290            TimePeriodValues ts = (TimePeriodValues) this.data.get(series);
291            TimePeriodValue dp = ts.getDataItem(item);
292            return new Long(dp.getPeriod().getEnd().getTime());
293        }
294

296         * Returns the y-value for the specified series and item.
297         *
298         * @param series  the series (zero-based index).
299         * @param item  the item (zero-based index).
300         *
301         * @return The y-value for the specified series and item.

303        public Number getY(int series, int item) {
304            TimePeriodValues ts = (TimePeriodValues) this.data.get(series);
305            TimePeriodValue dp = ts.getDataItem(item);

307        }

315         * @return The starting Y value for the specified series and item.

322         * Returns the ending Y value for the specified series and item.

324         * @param series  the series (zero-based index).
325         * @param item  the item (zero-based index).
326         *
327         * @return The ending Y value for the specified series and item.
328         */
329        public Number getEndY(int series, int item) {

331        }

333        /**
334         * Returns the minimum x-value in the dataset.
335         *
336         * @param includeInterval  a flag that determines whether or not the
337         *                         x-interval is taken into account.
338         *
339         * @return The minimum value.
340         */
341        public double getDomainLowerBound(boolean includeInterval) {
342            double result = Double.NaN;
343            Range r = getDomainBounds(includeInterval);
344            if (r != null) {
345                result = r.getLowerBound();
346            }
347            return result;
348        }
349    
350        /**
351         * Returns the maximum x-value in the dataset.
352         *
353         * @param includeInterval  a flag that determines whether or not the
354         *                         x-interval is taken into account.
355         *
356         * @return The maximum value.
357         */
358        public double getDomainUpperBound(boolean includeInterval) {
359            double result = Double.NaN;
360            Range r = getDomainBounds(includeInterval);
361            if (r != null) {
362                result = r.getUpperBound();
363            }
364            return result;
365        }
366    
367        /**
368         * Returns the range of the values in this dataset's domain.
369         *
370         * @param includeInterval  a flag that determines whether or not the
371         *                         x-interval is taken into account.
372         *
373         * @return The range.
374         */
375        public Range getDomainBounds(boolean includeInterval) {
376            boolean interval = includeInterval || this.domainIsPointsInTime;
377            Range result = null;
378            Range temp = null;
379            Iterator iterator = this.data.iterator();
380            while (iterator.hasNext()) {
381                TimePeriodValues series = (TimePeriodValues) iterator.next();
382                int count = series.getItemCount();
383                if (count > 0) {
384                    TimePeriod start = series.getTimePeriod(
385                            series.getMinStartIndex());
386                    TimePeriod end = series.getTimePeriod(series.getMaxEndIndex());
387                    if (!interval) {
388                        if (this.xPosition == TimePeriodAnchor.START) {
389                            TimePeriod maxStart = series.getTimePeriod(
390                                    series.getMaxStartIndex());
391                            temp = new Range(start.getStart().getTime(),
392                                    maxStart.getStart().getTime());
393                        }
394                        else if (this.xPosition == TimePeriodAnchor.MIDDLE) {
395                            TimePeriod minMiddle = series.getTimePeriod(
396                                    series.getMinMiddleIndex());
397                            long s1 = minMiddle.getStart().getTime();
398                            long e1 = minMiddle.getEnd().getTime();
399                            TimePeriod maxMiddle = series.getTimePeriod(
400                                    series.getMaxMiddleIndex());
401                            long s2 = maxMiddle.getStart().getTime();
402                            long e2 = maxMiddle.getEnd().getTime();
403                            temp = new Range(s1 + (e1 - s1) / 2,
404                                    s2 + (e2 - s2) / 2);
405                        }
406                        else if (this.xPosition == TimePeriodAnchor.END) {
407                            TimePeriod minEnd = series.getTimePeriod(
408                                    series.getMinEndIndex());
409                            temp = new Range(minEnd.getEnd().getTime(),
410                                    end.getEnd().getTime());
411                        }
412                    }
413                    else {
414                        temp = new Range(start.getStart().getTime(),
415                                end.getEnd().getTime());
416                    }
417                    result = Range.combine(result, temp);
418                }
419            }
420            return result;
421        }

423        /**
424         * Tests this instance for equality with an arbitrary object.
425         *
426         * @param obj  the object (<code>null</code> permitted).
427         *
428         * @return A boolean.
429         */
430        public boolean equals(Object obj) {
431            if (obj == this) {
432                return true;
433            }
434            if (!(obj instanceof TimePeriodValuesCollection)) {
435                return false;
436            }
437            TimePeriodValuesCollection that = (TimePeriodValuesCollection) obj;
438            if (this.domainIsPointsInTime != that.domainIsPointsInTime) {
439                return false;
440            }
441            if (this.xPosition != that.xPosition) {
442                return false;
443            }
444            if (!ObjectUtilities.equal(this.data, that.data)) {
445                return false;
446            }
447            return true;
448        }

461         * @deprecated This flag is no longer used by JFreeChart (as of version
462         *     1.0.3).

465            return this.domainIsPointsInTime;
466        }

475         *     1.0.3).
476         */

478            this.domainIsPointsInTime = flag;
