109    /**
110     * Represents a sequence of zero or more data items in the form (period, value)
111     * where 'period' is some instance of a subclass of {@link RegularTimePeriod}.
112     * The time series will ensure that (a) all data items have the same type of
113     * period (for example, {@link Day}) and (b) that each period appears at
114     * most one time in the series.
115     */

282         * Sets the maximum number of items that will be retained in the series.
283         * If you add a new item to the series such that the number of items will
284         * exceed the maximum item count, then the FIRST element in the series is
285         * automatically removed, ensuring that the maximum item count is not
286         * exceeded.
287         *
288         * @param maximum  the maximum (requires >= 0).
289         *
290         * @see #getMaximumItemCount()
291         */
292        public void setMaximumItemCount(int maximum) {
293            if (maximum < 0) {
294                throw new IllegalArgumentException("Negative 'maximum' argument.");
295            }
296            this.maximumItemCount = maximum;
297            int count = this.data.size();
298            if (count > maximum) {
299                delete(0, count - maximum - 1);
300            }
301        }

558        /**
559         * Adds a data item to the series and sends a {@link SeriesChangeEvent} to
560         * all registered listeners.
561         *
562         * @param item  the (timeperiod, value) pair (<code>null</code> not
563         *              permitted).
564         * @param notify  notify listeners?
565         */
566        public void add(TimeSeriesDataItem item, boolean notify) {
567            ParamChecks.nullNotPermitted(item, "item");
568            item = (TimeSeriesDataItem) item.clone();
569            Class c = item.getPeriod().getClass();
570            if (this.timePeriodClass == null) {
571                this.timePeriodClass = c;
572            }
573            else if (!this.timePeriodClass.equals(c)) {
574                StringBuffer b = new StringBuffer();
575                b.append("You are trying to add data where the time period class ");
576                b.append("is ");
577                b.append(item.getPeriod().getClass().getName());
578                b.append(", but the TimeSeries is expecting an instance of ");
579                b.append(this.timePeriodClass.getName());
580                b.append(".");
581                throw new SeriesException(b.toString());
582            }
583    
584            // make the change (if it's not a duplicate time period)...
585            boolean added = false;
586            int count = getItemCount();
587            if (count == 0) {
588                this.data.add(item);
589                added = true;
590            }
591            else {
592                RegularTimePeriod last = getTimePeriod(getItemCount() - 1);
593                if (item.getPeriod().compareTo(last) > 0) {
594                    this.data.add(item);
595                    added = true;
596                }
597                else {
598                    int index = Collections.binarySearch(this.data, item);
599                    if (index < 0) {
600                        this.data.add(-index - 1, item);
601                        added = true;
602                    }
603                    else {
604                        StringBuffer b = new StringBuffer();
605                        b.append("You are attempting to add an observation for ");
606                        b.append("the time period ");
607                        b.append(item.getPeriod().toString());
608                        b.append(" but the series already contains an observation");
609                        b.append(" for that time period. Duplicates are not ");
610                        b.append("permitted.  Try using the addOrUpdate() method.");
611                        throw new SeriesException(b.toString());
612                    }
613                }
614            }
615            if (added) {
616                updateBoundsForAddedItem(item);
617                // check if this addition will exceed the maximum item count...
618                if (getItemCount() > this.maximumItemCount) {
619                    TimeSeriesDataItem d = (TimeSeriesDataItem) this.data.remove(0);
620                    updateBoundsForRemovedItem(d);
621                }
622    
623                removeAgedItems(false);  // remove old items if necessary, but
624                                         // don't notify anyone, because that
625                                         // happens next anyway...
626                if (notify) {
627                    fireSeriesChanged();
628                }
629            }
630    
631        }

707            TimeSeriesDataItem temp = new TimeSeriesDataItem(period, value);
708            int index = Collections.binarySearch(this.data, temp);
709            if (index < 0) {
710                throw new SeriesException("There is no existing value for the "
711                        + "specified 'period'.");
712            }
713            update(index, value);
714        }

717         * Updates (changes) the value of a data item.
718         *
719         * @param index  the index of the data item.
720         * @param value  the new value (<code>null</code> permitted).
721         */
722        public void update(int index, Number value) {
723            TimeSeriesDataItem item = (TimeSeriesDataItem) this.data.get(index);
724            boolean iterate = false;
725            Number oldYN = item.getValue();
726            if (oldYN != null) {
727                double oldY = oldYN.doubleValue();
728                if (!Double.isNaN(oldY)) {
729                    iterate = oldY <= this.minY || oldY >= this.maxY;
730                }
731            }
732            item.setValue(value);
733            if (iterate) {
734                findBoundsByIteration();
735            }
736            else if (value != null) {
737                double yy = value.doubleValue();
738                this.minY = minIgnoreNaN(this.minY, yy);
739                this.maxY = maxIgnoreNaN(this.maxY, yy);
740            }
741            fireSeriesChanged();
742        }

798        /**
799         * Adds or updates an item in the times series and sends a
800         * {@link SeriesChangeEvent} to all registered listeners.
801         *
802         * @param item  the data item (<code>null</code> not permitted).
803         *
804         * @return A copy of the overwritten data item, or <code>null</code> if no
805         *         item was overwritten.
806         *
807         * @since 1.0.14
808         */
809        public TimeSeriesDataItem addOrUpdate(TimeSeriesDataItem item) {
810    
811            ParamChecks.nullNotPermitted(item, "item");
812            Class periodClass = item.getPeriod().getClass();
813            if (this.timePeriodClass == null) {
814                this.timePeriodClass = periodClass;
815            }
816            else if (!this.timePeriodClass.equals(periodClass)) {
817                String msg = "You are trying to add data where the time "
818                        + "period class is " + periodClass.getName()
819                        + ", but the TimeSeries is expecting an instance of "
820                        + this.timePeriodClass.getName() + ".";
821                throw new SeriesException(msg);
822            }
823            TimeSeriesDataItem overwritten = null;
824            int index = Collections.binarySearch(this.data, item);
825            if (index >= 0) {
826                TimeSeriesDataItem existing
827                        = (TimeSeriesDataItem) this.data.get(index);
828                overwritten = (TimeSeriesDataItem) existing.clone();
829                // figure out if we need to iterate through all the y-values
830                // to find the revised minY / maxY
831                boolean iterate = false;
832                Number oldYN = existing.getValue();
833                double oldY = oldYN != null ? oldYN.doubleValue() : Double.NaN;
834                if (!Double.isNaN(oldY)) {
835                    iterate = oldY <= this.minY || oldY >= this.maxY;
836                }
837                existing.setValue(item.getValue());
838                if (iterate) {
839                    findBoundsByIteration();
840                }
841                else if (item.getValue() != null) {
842                    double yy = item.getValue().doubleValue();
843                    this.minY = minIgnoreNaN(this.minY, yy);
844                    this.maxY = maxIgnoreNaN(this.maxY, yy);
845                }
846            }
847            else {
848                item = (TimeSeriesDataItem) item.clone();
849                this.data.add(-index - 1, item);
850                updateBoundsForAddedItem(item);
851    
852                // check if this addition will exceed the maximum item count...
853                if (getItemCount() > this.maximumItemCount) {
854                    TimeSeriesDataItem d = (TimeSeriesDataItem) this.data.remove(0);
855                    updateBoundsForRemovedItem(d);
856                }
857            }
858            removeAgedItems(false);  // remove old items if necessary, but
859                                     // don't notify anyone, because that
860                                     // happens next anyway...
861            fireSeriesChanged();
862            return overwritten;
863    
864        }

866        /**
867         * Age items in the series.  Ensure that the timespan from the youngest to
868         * the oldest record in the series does not exceed maximumItemAge time
869         * periods.  Oldest items will be removed if required.
870         *
871         * @param notify  controls whether or not a {@link SeriesChangeEvent} is
872         *                sent to registered listeners IF any items are removed.
873         */
874        public void removeAgedItems(boolean notify) {
875            // check if there are any values earlier than specified by the history
876            // count...
877            if (getItemCount() > 1) {
878                long latest = getTimePeriod(getItemCount() - 1).getSerialIndex();
879                boolean removed = false;
880                while ((latest - getTimePeriod(0).getSerialIndex())
881                        > this.maximumItemAge) {
882                    this.data.remove(0);
883                    removed = true;
884                }
885                if (removed) {
886                    findBoundsByIteration();
887                    if (notify) {
888                        fireSeriesChanged();
889                    }
890                }
891            }
892        }

894        /**
895         * Age items in the series.  Ensure that the timespan from the supplied
896         * time to the oldest record in the series does not exceed history count.
897         * oldest items will be removed if required.
898         *
899         * @param latest  the time to be compared against when aging data
900         *     (specified in milliseconds).
901         * @param notify  controls whether or not a {@link SeriesChangeEvent} is
902         *                sent to registered listeners IF any items are removed.
903         */
904        public void removeAgedItems(long latest, boolean notify) {
905            if (this.data.isEmpty()) {
906                return;  // nothing to do
907            }
908            // find the serial index of the period specified by 'latest'
909            long index = Long.MAX_VALUE;
910            try {
911                Method m = RegularTimePeriod.class.getDeclaredMethod(
912                        "createInstance", new Class[] {Class.class, Date.class,
913                        TimeZone.class});
914                RegularTimePeriod newest = (RegularTimePeriod) m.invoke(
915                        this.timePeriodClass, new Object[] {this.timePeriodClass,
916                                new Date(latest), TimeZone.getDefault()});
917                index = newest.getSerialIndex();
918            }
919            catch (NoSuchMethodException e) {
920                e.printStackTrace();
921            }
922            catch (IllegalAccessException e) {
923                e.printStackTrace();
924            }
925            catch (InvocationTargetException e) {
926                e.printStackTrace();
927            }
928    
929            // check if there are any values earlier than specified by the history
930            // count...
931            boolean removed = false;
932            while (getItemCount() > 0 && (index
933                    - getTimePeriod(0).getSerialIndex()) > this.maximumItemAge) {
934                this.data.remove(0);
935                removed = true;
936            }
937            if (removed) {
938                findBoundsByIteration();
939                if (notify) {
940                    fireSeriesChanged();
941                }
942            }
943        }

1036        /**
1037         * Creates a new timeseries by copying a subset of the data in this time
1038         * series.
1039         *
1040         * @param start  the index of the first time period to copy.
1041         * @param end  the index of the last time period to copy.
1042         *
1043         * @return A series containing a copy of this times series from start until
1044         *         end.
1045         *
1046         * @throws CloneNotSupportedException if there is a cloning problem.
1047         */
1048        public TimeSeries createCopy(int start, int end)
1049                throws CloneNotSupportedException {
1050            if (start < 0) {
1051                throw new IllegalArgumentException("Requires start >= 0.");
1052            }
1053            if (end < start) {
1054                throw new IllegalArgumentException("Requires start <= end.");
1055            }
1056            TimeSeries copy = (TimeSeries) super.clone();
1057            copy.minY = Double.NaN;
1058            copy.maxY = Double.NaN;
1059            copy.data = new java.util.ArrayList();
1060            if (this.data.size() > 0) {
1061                for (int index = start; index <= end; index++) {
1062                    TimeSeriesDataItem item
1063                            = (TimeSeriesDataItem) this.data.get(index);
1064                    TimeSeriesDataItem clone = (TimeSeriesDataItem) item.clone();
1065                    try {
1066                        copy.add(clone);
1067                    }
1068                    catch (SeriesException e) {
1069                        e.printStackTrace();
1070                    }
1071                }
1072            }
1073            return copy;
1074        }

1076        /**
1077         * Creates a new timeseries by copying a subset of the data in this time
1078         * series.
1079         *
1080         * @param start  the first time period to copy (<code>null</code> not
1081         *         permitted).
1082         * @param end  the last time period to copy (<code>null</code> not
1083         *         permitted).
1084         *
1085         * @return A time series containing a copy of this time series from start
1086         *         until end.
1087         *
1088         * @throws CloneNotSupportedException if there is a cloning problem.
1089         */
1090        public TimeSeries createCopy(RegularTimePeriod start, RegularTimePeriod end)
1091            throws CloneNotSupportedException {
1092    
1093            ParamChecks.nullNotPermitted(start, "start");
1094            ParamChecks.nullNotPermitted(end, "end");
1095            if (start.compareTo(end) > 0) {
1096                throw new IllegalArgumentException(
1097                        "Requires start on or before end.");
1098            }
1099            boolean emptyRange = false;
1100            int startIndex = getIndex(start);
1101            if (startIndex < 0) {
1102                startIndex = -(startIndex + 1);
1103                if (startIndex == this.data.size()) {
1104                    emptyRange = true;  // start is after last data item
1105                }
1106            }
1107            int endIndex = getIndex(end);
1108            if (endIndex < 0) {             // end period is not in original series
1109                endIndex = -(endIndex + 1); // this is first item AFTER end period
1110                endIndex = endIndex - 1;    // so this is last item BEFORE end
1111            }
1112            if ((endIndex < 0)  || (endIndex < startIndex)) {
1113                emptyRange = true;
1114            }
1115            if (emptyRange) {
1116                TimeSeries copy = (TimeSeries) super.clone();
1117                copy.data = new java.util.ArrayList();
1118                return copy;
1119            }
1120            return createCopy(startIndex, endIndex);
1121        }

1155            }

1159            }

1166        /**
1167         * Returns a hash code value for the object.
1168         *
1169         * @return The hashcode
1170         */
1171        public int hashCode() {
1172            int result = super.hashCode();
1173            result = 29 * result + (this.domain != null ? this.domain.hashCode()
1174                    : 0);
1175            result = 29 * result + (this.range != null ? this.range.hashCode() : 0);
1176            result = 29 * result + (this.timePeriodClass != null
1177                    ? this.timePeriodClass.hashCode() : 0);
1178            // it is too slow to look at every data item, so let's just look at
1179            // the first, middle and last items...
1180            int count = getItemCount();
1181            if (count > 0) {
1182                TimeSeriesDataItem item = getRawDataItem(0);
1183                result = 29 * result + item.hashCode();
1184            }
1185            if (count > 1) {
1186                TimeSeriesDataItem item = getRawDataItem(count - 1);
1187                result = 29 * result + item.hashCode();
1188            }
1189            if (count > 2) {
1190                TimeSeriesDataItem item = getRawDataItem(count / 2);
1191                result = 29 * result + item.hashCode();
1192            }
1193            result = 29 * result + this.maximumItemCount;
1194            result = 29 * result + (int) this.maximumItemAge;
1195            return result;
1196        }

1251         * A function to find the minimum of two values, but ignoring any
1252         * Double.NaN values.
1253         *
1254         * @param a  the first value.
1255         * @param b  the second value.
1256         *
1257         * @return The minimum of the two values.

1264                return a;
1265            }
