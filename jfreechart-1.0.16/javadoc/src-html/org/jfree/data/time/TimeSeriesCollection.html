111    /**
112     * A collection of time series objects.  This class implements the
113     * {@link XYDataset} interface, as well as the extended
114     * {@link IntervalXYDataset} interface.  This makes it a convenient dataset for
115     * use with the {@link org.jfree.chart.plot.XYPlot} class.
116     */
117    public class TimeSeriesCollection extends AbstractIntervalXYDataset
118            implements XYDataset, IntervalXYDataset, DomainInfo, XYDomainInfo,

175         * Constructs a dataset containing a single series (more can be added),
176         * tied to a specific timezone.
177         *
178         * @param series  a series to add to the collection (<code>null</code>
179         *                permitted).
180         * @param zone  the timezone (<code>null</code> permitted, will use
181         *              <code>TimeZone.getDefault()</code> in that case).
182         */
183        public TimeSeriesCollection(TimeSeries series, TimeZone zone) {
184            // FIXME:  need a locale as well as a timezone
185            if (zone == null) {
186                zone = TimeZone.getDefault();
187            }
188            this.workingCalendar = Calendar.getInstance(zone);
189            this.data = new ArrayList();
190            if (series != null) {
191                this.data.add(series);
192                series.addChangeListener(this);
193            }
194            this.xPosition = TimePeriodAnchor.START;
195            this.domainIsPointsInTime = true;
196    
197        }

311         * Returns the series with the specified key, or <code>null</code> if
312         * there is no such series.
313         *
314         * @param key  the series key (<code>null</code> permitted).
315         *
316         * @return The series with the given key.
317         */
318        public TimeSeries getSeries(Comparable key) {
319            TimeSeries result = null;
320            Iterator iterator = this.data.iterator();
321            while (iterator.hasNext()) {
322                TimeSeries series = (TimeSeries) iterator.next();
323                Comparable k = series.getKey();
324                if (k != null && k.equals(key)) {
325                    result = series;
326                }
327            }
328            return result;
329        }

386        public void removeAllSeries() {
387    
388            // deregister the collection as a change listener to each series in the
389            // collection
390            for (int i = 0; i < this.data.size(); i++) {
391                TimeSeries series = (TimeSeries) this.data.get(i);

396            this.data.clear();
397            fireDatasetChanged();

441        /**
442         * Returns the x-value for a time period.
443         *
444         * @param period  the time period (<code>null</code> not permitted).
445         *
446         * @return The x-value.
447         */
448        protected synchronized long getX(RegularTimePeriod period) {
449            long result = 0L;
450            if (this.xPosition == TimePeriodAnchor.START) {
451                result = period.getFirstMillisecond(this.workingCalendar);
452            }
453            else if (this.xPosition == TimePeriodAnchor.MIDDLE) {
454                result = period.getMiddleMillisecond(this.workingCalendar);
455            }
456            else if (this.xPosition == TimePeriodAnchor.END) {
457                result = period.getLastMillisecond(this.workingCalendar);
458            }
459            return result;
460        }

528        /**
529         * Returns the indices of the two data items surrounding a particular
530         * millisecond value.
531         *
532         * @param series  the series index.
533         * @param milliseconds  the time.
534         *
535         * @return An array containing the (two) indices of the items surrounding
536         *         the time.
537         */
538        public int[] getSurroundingItems(int series, long milliseconds) {
539            int[] result = new int[] {-1, -1};
540            TimeSeries timeSeries = getSeries(series);
541            for (int i = 0; i < timeSeries.getItemCount(); i++) {
542                Number x = getX(series, i);
543                long m = x.longValue();
544                if (m <= milliseconds) {
545                    result[0] = i;
546                }
547                if (m >= milliseconds) {
548                    result[1] = i;
549                    break;
550                }
551            }
552            return result;
553        }

589        /**
590         * Returns the range of the values in this dataset's domain.
591         *
592         * @param includeInterval  a flag that determines whether or not the
593         *                         x-interval is taken into account.
594         *
595         * @return The range.
596         */
597        public Range getDomainBounds(boolean includeInterval) {
598            Range result = null;
599            Iterator iterator = this.data.iterator();
600            while (iterator.hasNext()) {
601                TimeSeries series = (TimeSeries) iterator.next();
602                int count = series.getItemCount();
603                if (count > 0) {
604                    RegularTimePeriod start = series.getTimePeriod(0);
605                    RegularTimePeriod end = series.getTimePeriod(count - 1);
606                    Range temp;
607                    if (!includeInterval) {
608                        temp = new Range(getX(start), getX(end));
609                    }
610                    else {
611                        temp = new Range(
612                                start.getFirstMillisecond(this.workingCalendar),
613                                end.getLastMillisecond(this.workingCalendar));
614                    }
615                    result = Range.combine(result, temp);
616                }
617            }
618            return result;
619        }

621        /**
622         * Returns the bounds of the domain values for the specified series.
623         *
624         * @param visibleSeriesKeys  a list of keys for the visible series.
625         * @param includeInterval  include the x-interval?
626         *
627         * @return A range.
628         *
629         * @since 1.0.13
630         */
631        public Range getDomainBounds(List visibleSeriesKeys,
632                boolean includeInterval) {
633            Range result = null;
634            Iterator iterator = visibleSeriesKeys.iterator();
635            while (iterator.hasNext()) {
636                Comparable seriesKey = (Comparable) iterator.next();
637                TimeSeries series = getSeries(seriesKey);
638                int count = series.getItemCount();
639                if (count > 0) {
640                    RegularTimePeriod start = series.getTimePeriod(0);
641                    RegularTimePeriod end = series.getTimePeriod(count - 1);
642                    Range temp;
643                    if (!includeInterval) {
644                        temp = new Range(getX(start), getX(end));
645                    }
646                    else {
647                        temp = new Range(
648                                start.getFirstMillisecond(this.workingCalendar),
649                                end.getLastMillisecond(this.workingCalendar));
650                    }
651                    result = Range.combine(result, temp);
652                }
653            }
654            return result;
655        }
