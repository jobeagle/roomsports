050    package org.jfree.data.xy;
051    
052    import java.util.Arrays;
053    import java.util.Date;
054    import org.jfree.chart.util.ParamChecks;
055    
056    import org.jfree.util.PublicCloneable;
057    
058    /**
059     * A simple implementation of the {@link OHLCDataset} interface.  See also
060     * the {@link DefaultOHLCDataset} class, which provides another implementation
061     * that is very similar.
062     */
063    public class DefaultHighLowDataset extends AbstractXYDataset
064            implements OHLCDataset, PublicCloneable {
065    
066        /** The series key. */
067        private Comparable seriesKey;
068    
069        /** Storage for the dates. */
070        private Date[] date;
071    
072        /** Storage for the high values. */
073        private Number[] high;
074    
075        /** Storage for the low values. */
076        private Number[] low;
077    
078        /** Storage for the open values. */
079        private Number[] open;
080    
081        /** Storage for the close values. */
082        private Number[] close;
083    
084        /** Storage for the volume values. */
085        private Number[] volume;
086    
087        /**
088         * Constructs a new high/low/open/close dataset.
089         * <p>
090         * The current implementation allows only one series in the dataset.
091         * This may be extended in a future version.
092         *
093         * @param seriesKey  the key for the series (<code>null</code> not
094         *     permitted).
095         * @param date  the dates (<code>null</code> not permitted).
096         * @param high  the high values (<code>null</code> not permitted).
097         * @param low  the low values (<code>null</code> not permitted).
098         * @param open  the open values (<code>null</code> not permitted).
099         * @param close  the close values (<code>null</code> not permitted).
100         * @param volume  the volume values (<code>null</code> not permitted).
101         */
102        public DefaultHighLowDataset(Comparable seriesKey, Date[] date,
103                double[] high, double[] low, double[] open, double[] close,
104                double[] volume) {
105    
106            ParamChecks.nullNotPermitted(seriesKey, "seriesKey");
107            ParamChecks.nullNotPermitted(date, "date");
108            this.seriesKey = seriesKey;
109            this.date = date;
110            this.high = createNumberArray(high);
111            this.low = createNumberArray(low);
112            this.open = createNumberArray(open);
113            this.close = createNumberArray(close);
114            this.volume = createNumberArray(volume);
115    
116        }
117    
118        /**
119         * Returns the key for the series stored in this dataset.
120         *
121         * @param series  the index of the series (ignored, this dataset supports
122         *     only one series and this method always returns the key for series 0).
123         *
124         * @return The series key (never <code>null</code>).
125         */
126        public Comparable getSeriesKey(int series) {
127            return this.seriesKey;
128        }
129    
130        /**
131         * Returns the x-value for one item in a series.  The value returned is a
132         * <code>Long</code> instance generated from the underlying
133         * <code>Date</code> object.  To avoid generating a new object instance,
134         * you might prefer to call {@link #getXValue(int, int)}.
135         *
136         * @param series  the series (zero-based index).
137         * @param item  the item (zero-based index).
138         *
139         * @return The x-value.
140         *
141         * @see #getXValue(int, int)
142         * @see #getXDate(int, int)
143         */
144        public Number getX(int series, int item) {
145            return new Long(this.date[item].getTime());
146        }
147    
148        /**
149         * Returns the x-value for one item in a series, as a Date.
150         * <p>
151         * This method is provided for convenience only.
152         *
153         * @param series  the series (zero-based index).
154         * @param item  the item (zero-based index).
155         *
156         * @return The x-value as a Date.
157         *
158         * @see #getX(int, int)
159         */
160        public Date getXDate(int series, int item) {
161            return this.date[item];
162        }
163    
164        /**
165         * Returns the y-value for one item in a series.
166         * <p>
167         * This method (from the {@link XYDataset} interface) is mapped to the
168         * {@link #getCloseValue(int, int)} method.
169         *
170         * @param series  the series (zero-based index).
171         * @param item  the item (zero-based index).
172         *
173         * @return The y-value.
174         *
175         * @see #getYValue(int, int)
176         */
177        public Number getY(int series, int item) {
178            return getClose(series, item);
179        }
180    
181        /**
182         * Returns the high-value for one item in a series.
183         *
184         * @param series  the series (zero-based index).
185         * @param item  the item (zero-based index).
186         *
187         * @return The high-value.
188         *
189         * @see #getHighValue(int, int)
190         */
191        public Number getHigh(int series, int item) {
192            return this.high[item];
193        }
194    
195        /**
196         * Returns the high-value (as a double primitive) for an item within a
197         * series.
198         *
199         * @param series  the series (zero-based index).
200         * @param item  the item (zero-based index).
201         *
202         * @return The high-value.
203         *
204         * @see #getHigh(int, int)
205         */
206        public double getHighValue(int series, int item) {
207            double result = Double.NaN;
208            Number high = getHigh(series, item);
209            if (high != null) {
210                result = high.doubleValue();
211            }
212            return result;
213        }
214    
215        /**
216         * Returns the low-value for one item in a series.
217         *
218         * @param series  the series (zero-based index).
219         * @param item  the item (zero-based index).
220         *
221         * @return The low-value.
222         *
223         * @see #getLowValue(int, int)
224         */
225        public Number getLow(int series, int item) {
226            return this.low[item];
227        }
228    
229        /**
230         * Returns the low-value (as a double primitive) for an item within a
231         * series.
232         *
233         * @param series  the series (zero-based index).
234         * @param item  the item (zero-based index).
235         *
236         * @return The low-value.
237         *
238         * @see #getLow(int, int)
239         */
240        public double getLowValue(int series, int item) {
241            double result = Double.NaN;
242            Number low = getLow(series, item);
243            if (low != null) {
244                result = low.doubleValue();
245            }
246            return result;
247        }
248    
249        /**
250         * Returns the open-value for one item in a series.
251         *
252         * @param series  the series (zero-based index).
253         * @param item  the item (zero-based index).
254         *
255         * @return The open-value.
256         *
257         * @see #getOpenValue(int, int)
258         */
259        public Number getOpen(int series, int item) {
260            return this.open[item];
261        }
262    
263        /**
264         * Returns the open-value (as a double primitive) for an item within a
265         * series.
266         *
267         * @param series  the series (zero-based index).
268         * @param item  the item (zero-based index).
269         *
270         * @return The open-value.
271         *
272         * @see #getOpen(int, int)
273         */
274        public double getOpenValue(int series, int item) {
275            double result = Double.NaN;
276            Number open = getOpen(series, item);
277            if (open != null) {
278                result = open.doubleValue();
279            }
280            return result;
281        }
282    
283        /**
284         * Returns the close-value for one item in a series.
285         *
286         * @param series  the series (zero-based index).
287         * @param item  the item (zero-based index).
288         *
289         * @return The close-value.
290         *
291         * @see #getCloseValue(int, int)
292         */
293        public Number getClose(int series, int item) {
294            return this.close[item];
295        }
296    
297        /**
298         * Returns the close-value (as a double primitive) for an item within a
299         * series.
300         *
301         * @param series  the series (zero-based index).
302         * @param item  the item (zero-based index).
303         *
304         * @return The close-value.
305         *
306         * @see #getClose(int, int)
307         */
308        public double getCloseValue(int series, int item) {
309            double result = Double.NaN;
310            Number close = getClose(series, item);
311            if (close != null) {
312                result = close.doubleValue();
313            }
314            return result;
315        }
316    
317        /**
318         * Returns the volume-value for one item in a series.
319         *
320         * @param series  the series (zero-based index).
321         * @param item  the item (zero-based index).
322         *
323         * @return The volume-value.
324         *
325         * @see #getVolumeValue(int, int)
326         */
327        public Number getVolume(int series, int item) {
328            return this.volume[item];
329        }
330    
331        /**
332         * Returns the volume-value (as a double primitive) for an item within a
333         * series.
334         *
335         * @param series  the series (zero-based index).
336         * @param item  the item (zero-based index).
337         *
338         * @return The volume-value.
339         *
340         * @see #getVolume(int, int)
341         */
342        public double getVolumeValue(int series, int item) {
343            double result = Double.NaN;
344            Number volume = getVolume(series, item);
345            if (volume != null) {
346                result = volume.doubleValue();
347            }
348            return result;
349        }
350    
351        /**
352         * Returns the number of series in the dataset.
353         * <p>
354         * This implementation only allows one series.
355         *
356         * @return The number of series.
357         */
358        public int getSeriesCount() {
359            return 1;
360        }
361    
362        /**
363         * Returns the number of items in the specified series.
364         *
365         * @param series  the index (zero-based) of the series.
366         *
367         * @return The number of items in the specified series.
368         */
369        public int getItemCount(int series) {
370            return this.date.length;
371        }
372    
373        /**
374         * Tests this dataset for equality with an arbitrary instance.
375         *
376         * @param obj  the object (<code>null</code> permitted).
377         *
378         * @return A boolean.
379         */
380        public boolean equals(Object obj) {
381            if (obj == this) {
382                return true;
383            }
384            if (!(obj instanceof DefaultHighLowDataset)) {
385                return false;
386            }
387            DefaultHighLowDataset that = (DefaultHighLowDataset) obj;
388            if (!this.seriesKey.equals(that.seriesKey)) {
389                return false;
390            }
391            if (!Arrays.equals(this.date, that.date)) {
392                return false;
393            }
394            if (!Arrays.equals(this.open, that.open)) {
395                return false;
396            }
397            if (!Arrays.equals(this.high, that.high)) {
398                return false;
399            }
400            if (!Arrays.equals(this.low, that.low)) {
401                return false;
402            }
403            if (!Arrays.equals(this.close, that.close)) {
404                return false;
405            }
406            if (!Arrays.equals(this.volume, that.volume)) {
407                return false;
408            }
409            return true;
410        }
411    
412        /**
413         * Constructs an array of Number objects from an array of doubles.
414         *
415         * @param data  the double values to convert (<code>null</code> not
416         *     permitted).
417         *
418         * @return The data as an array of Number objects.
419         */
420        public static Number[] createNumberArray(double[] data) {
421            Number[] result = new Number[data.length];
422            for (int i = 0; i < data.length; i++) {
423                result[i] = new Double(data[i]);
424            }
425            return result;
426        }
