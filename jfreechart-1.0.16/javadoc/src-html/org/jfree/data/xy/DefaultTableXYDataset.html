080    /**
081     * An {@link XYDataset} where every series shares the same x-values (required
082     * for generating stacked area charts).
083     */
084    public class DefaultTableXYDataset extends AbstractIntervalXYDataset
085            implements TableXYDataset, IntervalXYDataset, DomainInfo,
086                       PublicCloneable {

088        /**
089         * Storage for the data - this list will contain zero, one or many
090         * XYSeries objects.
091         */
092        private List data = null;
093    
094        /** Storage for the x values. */
095        private HashSet xPoints = null;

097        /** A flag that controls whether or not events are propogated. */
098        private boolean propagateEvents = true;
099    
100        /** A flag that controls auto pruning. */
101        private boolean autoPrune = false;

110            this(false);

113        /**
114         * Creates a new empty dataset.
115         *
116         * @param autoPrune  a flag that controls whether or not x-values are
117         *                   removed whenever the corresponding y-values are all
118         *                   <code>null</code>.
119         */
120        public DefaultTableXYDataset(boolean autoPrune) {
121            this.autoPrune = autoPrune;
122            this.data = new ArrayList();
123            this.xPoints = new HashSet();
124            this.intervalDelegate = new IntervalXYDelegate(this, false);
125            addChangeListener(this.intervalDelegate);
126        }

128        /**
129         * Returns the flag that controls whether or not x-values are removed from
130         * the dataset when the corresponding y-values are all <code>null</code>.
131         *
132         * @return A boolean.
133         */
134        public boolean isAutoPrune() {
135            return this.autoPrune;
136        }

138        /**
139         * Adds a series to the collection and sends a {@link DatasetChangeEvent}
140         * to all registered listeners.  The series should be configured to NOT
141         * allow duplicate x-values.
142         *
143         * @param series  the series (<code>null</code> not permitted).
144         */
145        public void addSeries(XYSeries series) {
146            ParamChecks.nullNotPermitted(series, "series");
147            if (series.getAllowDuplicateXValues()) {
148                throw new IllegalArgumentException(
149                    "Cannot accept XYSeries that allow duplicate values. "
150                    + "Use XYSeries(seriesName, <sort>, false) constructor."
151                );
152            }
153            updateXPoints(series);
154            this.data.add(series);
155            series.addChangeListener(this);
156            fireDatasetChanged();
157        }
158    
159        /**
160         * Adds any unique x-values from 'series' to the dataset, and also adds any
161         * x-values that are in the dataset but not in 'series' to the series.
162         *
163         * @param series  the series (<code>null</code> not permitted).
164         */
165        private void updateXPoints(XYSeries series) {
166            ParamChecks.nullNotPermitted(series, "series");
167            HashSet seriesXPoints = new HashSet();
168            boolean savedState = this.propagateEvents;
169            this.propagateEvents = false;
170            for (int itemNo = 0; itemNo < series.getItemCount(); itemNo++) {
171                Number xValue = series.getX(itemNo);
172                seriesXPoints.add(xValue);
173                if (!this.xPoints.contains(xValue)) {
174                    this.xPoints.add(xValue);
175                    int seriesCount = this.data.size();
176                    for (int seriesNo = 0; seriesNo < seriesCount; seriesNo++) {
177                        XYSeries dataSeries = (XYSeries) this.data.get(seriesNo);
178                        if (!dataSeries.equals(series)) {
179                            dataSeries.add(xValue, null);
180                        }
181                    }
182                }
183            }
184            Iterator iterator = this.xPoints.iterator();
185            while (iterator.hasNext()) {
186                Number xPoint = (Number) iterator.next();
187                if (!seriesXPoints.contains(xPoint)) {
188                    series.add(xPoint, null);
189                }
190            }
191            this.propagateEvents = savedState;
192        }
193    
194        /**
195         * Updates the x-values for all the series in the dataset.
196         */
197        public void updateXPoints() {
198            this.propagateEvents = false;
199            for (int s = 0; s < this.data.size(); s++) {
200                updateXPoints((XYSeries) this.data.get(s));
201            }
202            if (this.autoPrune) {
203                prune();
204            }
205            this.propagateEvents = true;
206        }

345        /**
346         * Removes all the series from the collection and sends a
347         * {@link DatasetChangeEvent} to all registered listeners.
348         */
349        public void removeAllSeries() {
350    
351            // Unregister the collection as a change listener to each series in
352            // the collection.
353            for (int i = 0; i < this.data.size(); i++) {
354                XYSeries series = (XYSeries) this.data.get(i);
355                series.removeChangeListener(this);
356            }
357    
358            // Remove all the series from the collection and notify listeners.
359            this.data.clear();
360            this.xPoints.clear();
361            fireDatasetChanged();
362        }
363    
364        /**
365         * Removes a series from the collection and sends a
366         * {@link DatasetChangeEvent} to all registered listeners.
367         *
368         * @param series  the series (<code>null</code> not permitted).
369         */
370        public void removeSeries(XYSeries series) {
371            ParamChecks.nullNotPermitted(series, "series");
372            if (this.data.contains(series)) {
373                series.removeChangeListener(this);
374                this.data.remove(series);
375                if (this.data.isEmpty()) {
376                    this.xPoints.clear();
377                }
378                fireDatasetChanged();
379            }
380        }
381    
382        /**
383         * Removes a series from the collection and sends a
384         * {@link DatasetChangeEvent} to all registered listeners.
385         *
386         * @param series  the series (zero based index).
387         */
388        public void removeSeries(int series) {
389    
390            // check arguments...
391            if ((series < 0) || (series > getSeriesCount())) {
392                throw new IllegalArgumentException("Index outside valid range.");
393            }
394    
395            // fetch the series, remove the change listener, then remove the series.
396            XYSeries s = (XYSeries) this.data.get(series);
397            s.removeChangeListener(this);
398            this.data.remove(series);
399            if (this.data.isEmpty()) {
400                this.xPoints.clear();
401            }
402            else if (this.autoPrune) {
403                prune();
404            }
405            fireDatasetChanged();
406    
407        }
408    
409        /**
410         * Removes the items from all series for a given x value.
411         *
412         * @param x  the x-value.
413         */
414        public void removeAllValuesForX(Number x) {
415            ParamChecks.nullNotPermitted(x, "x");
416            boolean savedState = this.propagateEvents;
417            this.propagateEvents = false;
418            for (int s = 0; s < this.data.size(); s++) {
419                XYSeries series = (XYSeries) this.data.get(s);
420                series.remove(x);
421            }
422            this.propagateEvents = savedState;
423            this.xPoints.remove(x);
424            fireDatasetChanged();
425        }
426    
427        /**
428         * Returns <code>true</code> if all the y-values for the specified x-value
429         * are <code>null</code> and <code>false</code> otherwise.
430         *
431         * @param x  the x-value.
432         *
433         * @return A boolean.
434         */
435        protected boolean canPrune(Number x) {
436            for (int s = 0; s < this.data.size(); s++) {
437                XYSeries series = (XYSeries) this.data.get(s);
438                if (series.getY(series.indexOf(x)) != null) {
439                    return false;
440                }
441            }
442            return true;
443        }
444    
445        /**
446         * Removes all x-values for which all the y-values are <code>null</code>.
447         */
448        public void prune() {
449            HashSet hs = (HashSet) this.xPoints.clone();
450            Iterator iterator = hs.iterator();
451            while (iterator.hasNext()) {
452                Number x = (Number) iterator.next();
453                if (canPrune(x)) {
454                    removeAllValuesForX(x);
455                }
456            }
457        }

459        /**
460         * This method receives notification when a series belonging to the dataset
461         * changes.  It responds by updating the x-points for the entire dataset
462         * and sending a {@link DatasetChangeEvent} to all registered listeners.
463         *
464         * @param event  information about the change.
465         */
466        public void seriesChanged(SeriesChangeEvent event) {
467            if (this.propagateEvents) {
468                updateXPoints();
469                fireDatasetChanged();
470            }
471        }

473        /**
474         * Tests this collection for equality with an arbitrary object.
475         *
476         * @param obj  the object (<code>null</code> permitted).
477         *
478         * @return A boolean.
479         */
480        public boolean equals(Object obj) {
481            if (obj == this) {
482                return true;
483            }
484            if (!(obj instanceof DefaultTableXYDataset)) {
485                return false;
486            }
487            DefaultTableXYDataset that = (DefaultTableXYDataset) obj;
488            if (this.autoPrune != that.autoPrune) {
489                return false;
490            }
491            if (this.propagateEvents != that.propagateEvents) {
492                return false;
493            }
494            if (!this.intervalDelegate.equals(that.intervalDelegate)) {
495                return false;
496            }
497            if (!ObjectUtilities.equal(this.data, that.data)) {
498                return false;
499            }
500            return true;
501        }
502    
503        /**
504         * Returns a hash code.
505         *
506         * @return A hash code.
507         */
508        public int hashCode() {
509            int result;
510            result = (this.data != null ? this.data.hashCode() : 0);
511            result = 29 * result
512                     + (this.xPoints != null ? this.xPoints.hashCode() : 0);
513            result = 29 * result + (this.propagateEvents ? 1 : 0);
514            result = 29 * result + (this.autoPrune ? 1 : 0);
515            return result;
516        }

518        /**
519         * Returns an independent copy of this dataset.
520         *
521         * @return A clone.
522         *
523         * @throws CloneNotSupportedException if there is some reason that cloning
524         *     cannot be performed.
525         */
526        public Object clone() throws CloneNotSupportedException {
527            DefaultTableXYDataset clone = (DefaultTableXYDataset) super.clone();
528            int seriesCount = this.data.size();
529            clone.data = new java.util.ArrayList(seriesCount);
530            for (int i = 0; i < seriesCount; i++) {
531                XYSeries series = (XYSeries) this.data.get(i);
532                clone.data.add(series.clone());
533            }
534    
535            clone.intervalDelegate = new IntervalXYDelegate(clone);
536            // need to configure the intervalDelegate to match the original
537            clone.intervalDelegate.setFixedIntervalWidth(getIntervalWidth());
538            clone.intervalDelegate.setAutoWidth(isAutoWidth());
539            clone.intervalDelegate.setIntervalPositionFactor(
540                    getIntervalPositionFactor());
541            clone.updateXPoints();
542            return clone;
543        }
