094    /**
095     * Represents a sequence of zero or more data items in the form (x, y).  By
096     * default, items in the series will be sorted into ascending order by x-value,
097     * and duplicate x-values are permitted.  Both the sorting and duplicate
098     * defaults can be changed in the constructor.  Y-values can be
099     * <code>null</code> to represent missing values.
100     */

242        /**
243         * Updates the cached values for the minimum and maximum data values.
244         *
245         * @param item  the item added (<code>null</code> not permitted).
246         *
247         * @since 1.0.13
248         */
249        private void updateBoundsForAddedItem(XYDataItem item) {
250            double x = item.getXValue();
251            this.minX = minIgnoreNaN(this.minX, x);
252            this.maxX = maxIgnoreNaN(this.maxX, x);
253            if (item.getY() != null) {
254                double y = item.getYValue();
255                this.minY = minIgnoreNaN(this.minY, y);
256                this.maxY = maxIgnoreNaN(this.maxY, y);
257            }
258        }
259    
260        /**
261         * Updates the cached values for the minimum and maximum data values on
262         * the basis that the specified item has just been removed.
263         *
264         * @param item  the item added (<code>null</code> not permitted).
265         *
266         * @since 1.0.13
267         */
268        private void updateBoundsForRemovedItem(XYDataItem item) {
269            boolean itemContributesToXBounds = false;
270            boolean itemContributesToYBounds = false;
271            double x = item.getXValue();
272            if (!Double.isNaN(x)) {
273                if (x <= this.minX || x >= this.maxX) {
274                    itemContributesToXBounds = true;
275                }
276            }
277            if (item.getY() != null) {
278                double y = item.getYValue();
279                if (!Double.isNaN(y)) {
280                    if (y <= this.minY || y >= this.maxY) {
281                        itemContributesToYBounds = true;
282                    }
283                }
284            }
285            if (itemContributesToYBounds) {
286                findBoundsByIteration();
287            }
288            else if (itemContributesToXBounds) {
289                if (getAutoSort()) {
290                    this.minX = getX(0).doubleValue();
291                    this.maxX = getX(getItemCount() - 1).doubleValue();
292                }
293                else {
294                    findBoundsByIteration();
295                }
296            }
297        }
298    
299        /**
300         * Finds the bounds of the x and y values for the series, by iterating
301         * through all the data items.
302         *
303         * @since 1.0.13
304         */
305        private void findBoundsByIteration() {
306            this.minX = Double.NaN;
307            this.maxX = Double.NaN;
308            this.minY = Double.NaN;
309            this.maxY = Double.NaN;
310            Iterator iterator = this.data.iterator();
311            while (iterator.hasNext()) {
312                XYDataItem item = (XYDataItem) iterator.next();
313                updateBoundsForAddedItem(item);
314            }
315        }

371        /**
372         * Sets the maximum number of items that will be retained in the series.
373         * If you add a new item to the series such that the number of items will
374         * exceed the maximum item count, then the first element in the series is
375         * automatically removed, ensuring that the maximum item count is not
376         * exceeded.
377         * <p>
378         * Typically this value is set before the series is populated with data,
379         * but if it is applied later, it may cause some items to be removed from
380         * the series (in which case a {@link SeriesChangeEvent} will be sent to
381         * all registered listeners).
382         *
383         * @param maximum  the maximum number of items for the series.
384         */
385        public void setMaximumItemCount(int maximum) {
386            this.maximumItemCount = maximum;
387            int remove = this.data.size() - maximum;
388            if (remove > 0) {
389                this.data.subList(0, remove).clear();
390                findBoundsByIteration();
391                fireSeriesChanged();
392            }
393        }

496        /**
497         * Adds a data item to the series and, if requested, sends a
498         * {@link SeriesChangeEvent} to all registered listeners.
499         *
500         * @param item  the (x, y) item (<code>null</code> not permitted).
501         * @param notify  a flag that controls whether or not a
502         *                {@link SeriesChangeEvent} is sent to all registered
503         *                listeners.
504         */
505        public void add(XYDataItem item, boolean notify) {
506            ParamChecks.nullNotPermitted(item, "item");
507            item = (XYDataItem) item.clone();
508            if (this.autoSort) {
509                int index = Collections.binarySearch(this.data, item);
510                if (index < 0) {
511                    this.data.add(-index - 1, item);
512                }
513                else {
514                    if (this.allowDuplicateXValues) {
515                        // need to make sure we are adding *after* any duplicates
516                        int size = this.data.size();
517                        while (index < size && item.compareTo(
518                                this.data.get(index)) == 0) {
519                            index++;
520                        }
521                        if (index < this.data.size()) {
522                            this.data.add(index, item);
523                        }
524                        else {
525                            this.data.add(item);
526                        }
527                    }
528                    else {
529                        throw new SeriesException("X-value already exists.");
530                    }
531                }
532            }
533            else {
534                if (!this.allowDuplicateXValues) {
535                    // can't allow duplicate values, so we need to check whether
536                    // there is an item with the given x-value already
537                    int index = indexOf(item.getX());
538                    if (index >= 0) {
539                        throw new SeriesException("X-value already exists.");
540                    }
541                }
542                this.data.add(item);
543            }
544            updateBoundsForAddedItem(item);
545            if (getItemCount() > this.maximumItemCount) {
546                XYDataItem removed = (XYDataItem) this.data.remove(0);
547                updateBoundsForRemovedItem(removed);
548            }
549            if (notify) {
550                fireSeriesChanged();
551            }
552        }

658        /**
659         * Updates the value of an item in the series and sends a
660         * {@link SeriesChangeEvent} to all registered listeners.
661         *
662         * @param index  the item (zero based index).
663         * @param y  the new value (<code>null</code> permitted).
664         *
665         * @deprecated Renamed {@link #updateByIndex(int, Number)} to avoid
666         *         confusion with the {@link #update(Number, Number)} method.
667         */
668        public void update(int index, Number y) {
669            XYDataItem item = getRawDataItem(index);
670    
671            // figure out if we need to iterate through all the y-values
672            boolean iterate = false;
673            double oldY = item.getYValue();
674            if (!Double.isNaN(oldY)) {
675                iterate = oldY <= this.minY || oldY >= this.maxY;
676            }
677            item.setY(y);
678    
679            if (iterate) {
680                findBoundsByIteration();
681            }
682            else if (y != null) {
683                double yy = y.doubleValue();
684                this.minY = minIgnoreNaN(this.minY, yy);
685                this.maxY = maxIgnoreNaN(this.maxY, yy);
686            }
687            fireSeriesChanged();
688        }

690        /**
691         * A function to find the minimum of two values, but ignoring any
692         * Double.NaN values.
693         *
694         * @param a  the first value.
695         * @param b  the second value.
696         *
697         * @return The minimum of the two values.
698         */
699        private double minIgnoreNaN(double a, double b) {
700            if (Double.isNaN(a)) {
701                return b;
702            }
703            if (Double.isNaN(b)) {
704                return a;
705            }
706            return Math.min(a, b);
707        }

709        /**
710         * A function to find the maximum of two values, but ignoring any
711         * Double.NaN values.
712         *
713         * @param a  the first value.
714         * @param b  the second value.
715         *
716         * @return The maximum of the two values.
717         */
718        private double maxIgnoreNaN(double a, double b) {
719            if (Double.isNaN(a)) {
720                return b;
721            }
722            if (Double.isNaN(b)) {
723                return a;
724            }
725            return Math.max(a, b);
726        }

788        /**
789         * Adds or updates an item in the series and sends a
790         * {@link SeriesChangeEvent} to all registered listeners.
791         *
792         * @param item  the data item (<code>null</code> not permitted).
793         *
794         * @return A copy of the overwritten data item, or <code>null</code> if no
795         *         item was overwritten.
796         *
797         * @since 1.0.14
798         */
799        public XYDataItem addOrUpdate(XYDataItem item) {
800            ParamChecks.nullNotPermitted(item, "item");
801            if (this.allowDuplicateXValues) {
802                add(item);
803                return null;
804            }
805    
806            // if we get to here, we know that duplicate X values are not permitted
807            XYDataItem overwritten = null;
808            int index = indexOf(item.getX());
809            if (index >= 0) {
810                XYDataItem existing = (XYDataItem) this.data.get(index);
811                overwritten = (XYDataItem) existing.clone();
812                // figure out if we need to iterate through all the y-values
813                boolean iterate = false;
814                double oldY = existing.getYValue();
815                if (!Double.isNaN(oldY)) {
816                    iterate = oldY <= this.minY || oldY >= this.maxY;
817                }
818                existing.setY(item.getY());
819    
820                if (iterate) {
821                    findBoundsByIteration();
822                }
823                else if (item.getY() != null) {
824                    double yy = item.getY().doubleValue();
825                    this.minY = minIgnoreNaN(this.minY, yy);
826                    this.maxY = maxIgnoreNaN(this.maxY, yy);
827                }
828            }
829            else {
830                // if the series is sorted, the negative index is a result from
831                // Collections.binarySearch() and tells us where to insert the
832                // new item...otherwise it will be just -1 and we should just
833                // append the value to the list...
834                item = (XYDataItem) item.clone();
835                if (this.autoSort) {
836                    this.data.add(-index - 1, item);
837                }
838                else {
839                    this.data.add(item);
840                }
841                updateBoundsForAddedItem(item);
842    
843                // check if this addition will exceed the maximum item count...
844                if (getItemCount() > this.maximumItemCount) {
845                    XYDataItem removed = (XYDataItem) this.data.remove(0);
846                    updateBoundsForRemovedItem(removed);
847                }
848            }
849            fireSeriesChanged();
850            return overwritten;
851        }

853        /**
854         * Returns the index of the item with the specified x-value, or a negative
855         * index if the series does not contain an item with that x-value.  Be
856         * aware that for an unsorted series, the index is found by iterating
857         * through all items in the series.
858         *
859         * @param x  the x-value (<code>null</code> not permitted).
860         *
861         * @return The index.
862         */
863        public int indexOf(Number x) {
864            if (this.autoSort) {
865                return Collections.binarySearch(this.data, new XYDataItem(x, null));
866            }
867            else {
868                for (int i = 0; i < this.data.size(); i++) {
869                    XYDataItem item = (XYDataItem) this.data.get(i);
870                    if (item.getX().equals(x)) {
871                        return i;
872                    }
873                }
874                return -1;
875            }
876        }

879         * Returns a new array containing the x and y values from this series.
880         *
881         * @return A new array containing the x and y values from this series.
882         *
883         * @since 1.0.4
884         */
885        public double[][] toArray() {
886            int itemCount = getItemCount();
887            double[][] result = new double[2][itemCount];
888            for (int i = 0; i < itemCount; i++) {
889                result[0][i] = this.getX(i).doubleValue();
890                Number y = getY(i);
891                if (y != null) {
892                    result[1][i] = y.doubleValue();
893                }
894                else {
895                    result[1][i] = Double.NaN;
896                }
897            }
898            return result;
899        }

916         *
917         * @param start  the index of the first item to copy.
918         * @param end  the index of the last item to copy.
919         *
920         * @return A series containing a copy of this series from start until end.
921         *
922         * @throws CloneNotSupportedException if there is a cloning problem.
923         */
924        public XYSeries createCopy(int start, int end)
925                throws CloneNotSupportedException {
926    
927            XYSeries copy = (XYSeries) super.clone();
928            copy.data = new java.util.ArrayList();
929            if (this.data.size() > 0) {
930                for (int index = start; index <= end; index++) {
931                    XYDataItem item = (XYDataItem) this.data.get(index);
932                    XYDataItem clone = (XYDataItem) item.clone();
933                    try {
934                        copy.add(clone);
935                    }
936                    catch (SeriesException e) {
937                        System.err.println("Unable to add cloned data item.");
938                    }
939                }
940            }
941            return copy;

945        /**
946         * Tests this series for equality with an arbitrary object.
947         *
948         * @param obj  the object to test against for equality
949         *             (<code>null</code> permitted).
950         *
951         * @return A boolean.
952         */
953        public boolean equals(Object obj) {
954            if (obj == this) {
955                return true;
956            }
957            if (!(obj instanceof XYSeries)) {
958                return false;
959            }
960            if (!super.equals(obj)) {
961                return false;
962            }
963            XYSeries that = (XYSeries) obj;
964            if (this.maximumItemCount != that.maximumItemCount) {
965                return false;
966            }
967            if (this.autoSort != that.autoSort) {
968                return false;
969            }
970            if (this.allowDuplicateXValues != that.allowDuplicateXValues) {
971                return false;
972            }
973            if (!ObjectUtilities.equal(this.data, that.data)) {
974                return false;
975            }
976            return true;
977        }

979        /**
980         * Returns a hash code.
981         *
982         * @return A hash code.
983         */
984        public int hashCode() {
985            int result = super.hashCode();
986            // it is too slow to look at every data item, so let's just look at
987            // the first, middle and last items...
988            int count = getItemCount();
989            if (count > 0) {
990                XYDataItem item = getRawDataItem(0);
991                result = 29 * result + item.hashCode();
992            }
993            if (count > 1) {
994                XYDataItem item = getRawDataItem(count - 1);
995                result = 29 * result + item.hashCode();
996            }
997            if (count > 2) {
998                XYDataItem item = getRawDataItem(count / 2);
999                result = 29 * result + item.hashCode();
1000            }
1001            result = 29 * result + this.maximumItemCount;
1002            result = 29 * result + (this.autoSort ? 1 : 0);
1003            result = 29 * result + (this.allowDuplicateXValues ? 1 : 0);
1004            return result;
1005        }
